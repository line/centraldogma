17  package com.linecorp.centraldogma.common;
18  
19  import static java.util.Objects.requireNonNull;
20  
21  import java.io.IOException;
22  
23  import javax.annotation.Nullable;
24  
25  import com.fasterxml.jackson.core.JsonParseException;
26  import com.fasterxml.jackson.core.JsonParser;
27  import com.fasterxml.jackson.core.ObjectCodec;
28  import com.fasterxml.jackson.databind.DeserializationContext;
29  import com.fasterxml.jackson.databind.JsonNode;
30  import com.fasterxml.jackson.databind.deser.std.StdDeserializer;
31  
32  import com.linecorp.centraldogma.internal.Jackson;
33  
34  final class ChangeDeserializer extends StdDeserializer<Change<?>> {
35  
36      private static final long serialVersionUID = -1811286391194654116L;
37  
38      ChangeDeserializer() {
39          super(Change.class);
40      }
41  
42      @Override
43      public Change<?> deserialize(JsonParser jp, DeserializationContext ctxt)
44              throws IOException {
45          final ObjectCodec codec = jp.getCodec();
46          final JsonNode root = codec.readTree(jp);
47  
48          final JsonNode typeNode = root.get("type");
49          requireNonNull(typeNode, "type");
50          final ChangeType type = codec.treeToValue(typeNode, ChangeType.class);
51  
52          final JsonNode pathNode = root.get("path");
53          requireNonNull(pathNode, "path");
54          final String path = pathNode.asText();
55  
56          final JsonNode content = root.get("content");
57          final JsonNode rawContentNode = root.get("rawContent");
58          final String rawContent;
59          if (rawContentNode != null && !rawContentNode.isNull()) {
60              rawContent = rawContentNode.asText();
61          } else {
62              rawContent = null;
63          }
64          return deserialize(type, path, content, rawContent);
65      }
66  
67      private static Change<?> deserialize(ChangeType type, String path, @Nullable JsonNode content,
68                                           @Nullable String rawContent) {
69          requireNonNull(type, "type");
70          final Class<?> contentType = type.contentType();
71          if (contentType == Void.class) {
72              if ((content != null && !content.isNull()) || rawContent != null) {
73                  return rejectIncompatibleContent(content, Void.class);
74              }
75          } else if (type.contentType() == String.class) {
76              if (content != null && !content.isTextual()) {
77                  return rejectIncompatibleContent(content, String.class);
78              }
79          }
80  
81          if (type == ChangeType.REMOVE) {
82              return Change.ofRemoval(path);
83          }
84  
85          if (content == null && rawContent == null) {
86              throw new IllegalArgumentException(
87                      "content or rawContent is required for type: " + type);
88          }
89          if (content != null && rawContent != null) {
90              throw new IllegalArgumentException(
91                      "only one of content or rawContent must be set for type: " + type);
92          }
93  
94          final Change<?> result;
95          switch (type) {
96              case UPSERT_JSON:
97                  // Raw content is only used for JSON_UPSERT for now.
98                  if (rawContent != null) {
99                      result = Change.ofJsonUpsert(path, rawContent);
100                 } else {
101                     result = Change.ofJsonUpsert(path, content);
102                 }
103                 break;
104             case UPSERT_TEXT:
105                 if (rawContent != null) {
106                     result = Change.ofTextUpsert(path, rawContent);
107                 } else {
108                     result = Change.ofTextUpsert(path, content.asText());
109                 }
110                 break;
111             case UPSERT_YAML:
112                 if (rawContent == null) {
113                     throw new IllegalArgumentException("rawContent is required for YAML_UPSERT");
114                 }
115                 result = Change.ofYamlUpsert(path, rawContent);
116                 break;
117             case RENAME:
118                 assert content != null;
119                 result = Change.ofRename(path, content.asText());
120                 break;
121             case APPLY_JSON_PATCH:
122                 if (rawContent != null) {
123                     try {
124                         result = Change.ofJsonPatch(path, Jackson.readTree(rawContent));
125                     } catch (JsonParseException e) {
126                         throw new IllegalArgumentException("failed to parse JSON patch", e);
127                     }
128                 } else {
129                     result = Change.ofJsonPatch(path, content);
130                 }
131                 break;
132             case APPLY_TEXT_PATCH:
133                 if (rawContent != null) {
134                     result = Change.ofTextPatch(path, rawContent);
135                 } else {
136                     result = Change.ofTextPatch(path, content.asText());
137                 }
138                 break;
139             default:
140                 // Should never reach here
141                 throw new Error();
142         }
143 
144         return result;
145     }
146 
147     private static Change<?> rejectIncompatibleContent(@Nullable JsonNode content,
148                                                        Class<?> contentType) {
149         throw new IllegalArgumentException("incompatible content: " + content +
150                                            " (expected: " + contentType.getName() + ')');
151     }
152 }

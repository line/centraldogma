17  package com.linecorp.centraldogma.common;
18  
19  import static java.util.Objects.requireNonNull;
20  
21  import com.fasterxml.jackson.core.JsonProcessingException;
22  import com.fasterxml.jackson.databind.JsonNode;
23  
24  import com.linecorp.centraldogma.internal.Jackson;
25  
26  /**
27   * Roles of a user in a project.
28   */
29  public enum ProjectRole {
30      OWNER,
31      MEMBER,
32      GUEST;
33  
34      /**
35       * Returns a {@link ProjectRole} matched with the specified {@code str}.
36       *
37       * @throws IllegalArgumentException if there is no matched {@link ProjectRole}.
38       */
39      public static ProjectRole of(String str) {
40          requireNonNull(str, "str");
41          return valueOf(str.toUpperCase());
42      }
43  
44      /**
45       * Returns a {@link ProjectRole} matched with the specified {@link JsonNode}.
46       *
47       * @throws IllegalArgumentException if there is no matched {@link ProjectRole}.
48       */
49      public static ProjectRole of(JsonNode node) {
50          requireNonNull(node, "node");
51          try {
52              return Jackson.treeToValue(node, ProjectRole.class);
53          } catch (JsonProcessingException e) {
54              throw new IllegalArgumentException(e);
55          }
56      }
57  
58      /**
59       * Returns {@code true} if this {@link ProjectRole} has the specified {@link ProjectRole}.
60       */
61      public boolean has(ProjectRole other) {
62          requireNonNull(other, "other");
63          if (this == OWNER) {
64              return true;
65          }
66          if (this == MEMBER) {
67              return other != OWNER;
68          }
69          // this == GUEST
70          return this == other;
71      }
72  }

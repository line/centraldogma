35  package com.linecorp.centraldogma.common.jsonpatch;
36  
37  import static java.util.Objects.requireNonNull;
38  
39  import java.io.IOException;
40  
41  import com.fasterxml.jackson.annotation.JsonCreator;
42  import com.fasterxml.jackson.annotation.JsonProperty;
43  import com.fasterxml.jackson.core.JsonGenerator;
44  import com.fasterxml.jackson.core.JsonPointer;
45  import com.fasterxml.jackson.databind.JsonNode;
46  import com.fasterxml.jackson.databind.SerializerProvider;
47  import com.fasterxml.jackson.databind.jsontype.TypeSerializer;
48  import com.fasterxml.jackson.databind.node.ArrayNode;
49  import com.fasterxml.jackson.databind.node.MissingNode;
50  import com.fasterxml.jackson.databind.node.ObjectNode;
51  
52  /**
53   * JSON Path {@code remove} operation.
54   *
55   * <p>This operation only takes one pointer ({@code path}) as an argument. It
56   * is an error condition if no JSON value exists at that pointer.</p>
57   */
58  public final class RemoveOperation extends JsonPatchOperation {
59  
60      /**
61       * Creates a new instance.
62       */
63      @JsonCreator
64      RemoveOperation(@JsonProperty("path") final JsonPointer path) {
65          super("remove", path);
66      }
67  
68      @Override
69      public JsonNode apply(final JsonNode node) {
70          requireNonNull(node, "node");
71          final JsonPointer path = path();
72          if (path.toString().isEmpty()) {
73              return MissingNode.getInstance();
74          }
75          ensureExistence(node);
76  
77          final JsonNode parentNode = node.at(path.head());
78          final String raw = path.last().getMatchingProperty();
79          if (parentNode.isObject()) {
80              ((ObjectNode) parentNode).remove(raw);
81          } else {
82              ((ArrayNode) parentNode).remove(Integer.parseInt(raw));
83          }
84          return node;
85      }
86  
87      @Override
88      public void serialize(final JsonGenerator jgen,
89                            final SerializerProvider provider) throws IOException {
90          requireNonNull(jgen, "jgen");
91          jgen.writeStartObject();
92          jgen.writeStringField("op", "remove");
93          jgen.writeStringField("path", path().toString());
94          jgen.writeEndObject();
95      }
96  
97      @Override
98      public void serializeWithType(final JsonGenerator jgen,
99                                    final SerializerProvider provider, final TypeSerializer typeSer)
100             throws IOException {
101         serialize(jgen, provider);
102     }
103 
104     @Override
105     public String toString() {
106         return "op: " + op() + "; path: \"" + path() + '"';
107     }
108 }

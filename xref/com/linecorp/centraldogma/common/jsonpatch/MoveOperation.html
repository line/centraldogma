35  package com.linecorp.centraldogma.common.jsonpatch;
36  
37  import static java.util.Objects.requireNonNull;
38  
39  import com.fasterxml.jackson.annotation.JsonCreator;
40  import com.fasterxml.jackson.annotation.JsonProperty;
41  import com.fasterxml.jackson.core.JsonPointer;
42  import com.fasterxml.jackson.databind.JsonNode;
43  import com.fasterxml.jackson.databind.node.ArrayNode;
44  import com.fasterxml.jackson.databind.node.ObjectNode;
45  
46  /**
47   * JSON Patch {@code move} operation.
48   *
49   * <p>For this operation, {@code from} points to the value to move, and {@code
50   * path} points to the new location of the moved value.</p>
51   *
52   * <p>It is an error condition if {@code from} does not point to a JSON value.
53   * </p>
54   *
55   * <p>The specification adds another rule that the {@code from} path must not be
56   * an immediate parent of {@code path}. Unfortunately, that doesn't really work.
57   * Consider this patch:</p>
58   *
59   * <pre>
60   *     { "op": "move", "from": "/0", "path": "/0/x" }
61   * </pre>
62   *
63   * <p>Even though {@code /0} is an immediate parent of {@code /0/x}, when this
64   * patch is applied to:</p>
65   *
66   * <pre>
67   *     [ "victim", {} ]
68   * </pre>
69   *
70   * <p>it actually succeeds and results in the patched value:</p>
71   *
72   * <pre>
73   *     [ { "x": "victim" } ]
74   * </pre>
75   */
76  public final class MoveOperation extends DualPathOperation {
77  
78      /**
79       * Creates a new instance.
80       */
81      @JsonCreator
82      MoveOperation(@JsonProperty("from") final JsonPointer from,
83                    @JsonProperty("path") final JsonPointer path) {
84          super("move", from, path);
85      }
86  
87      @Override
88      public JsonNode apply(final JsonNode node) {
89          requireNonNull(node, "node");
90          final JsonPointer from = from();
91          final JsonPointer path = path();
92          if (from.equals(path)) {
93              return node;
94          }
95          if (node.at(from).isMissingNode()) {
96              throw new JsonPatchConflictException("non-existent source path: " + from);
97          }
98  
99          final JsonNode sourceParent = ensureSourceParent(node, from);
100 
101         // Remove
102         final String raw = from.last().getMatchingProperty();
103         final JsonNode source;
104         if (sourceParent.isObject()) {
105             source = ((ObjectNode) sourceParent).remove(raw);
106         } else {
107             source = ((ArrayNode) sourceParent).remove(Integer.parseInt(raw));
108         }
109 
110         // Add
111         if (path.toString().isEmpty()) {
112             return source;
113         }
114 
115         final JsonNode targetParent = ensureTargetParent(node, path);
116         return targetParent.isArray() ? AddOperation.addToArray(path, node, source)
117                                       : AddOperation.addToObject(path, node, source);
118     }
119 }

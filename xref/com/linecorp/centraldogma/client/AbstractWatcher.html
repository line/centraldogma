53  abstract class AbstractWatcher<T> implements Watcher<T> {
54  
55      private static final Logger logger = LoggerFactory.getLogger(AbstractWatcher.class);
56  
57      private enum State {
58          INIT,
59          STARTED,
60          STOPPED
61      }
62  
63      private final ScheduledExecutorService watchScheduler;
64      private final String projectName;
65      private final String repositoryName;
66      private final String pathPattern;
67      private final boolean errorOnEntryNotFound;
68      private final long delayOnSuccessMillis;
69      private final long initialDelayMillis;
70      private final long maxDelayMillis;
71      private final double multiplier;
72      private final double jitterRate;
73  
74      private final List<Map.Entry<BiConsumer<? super Revision, ? super T>, Executor>> updateListeners =
75              new CopyOnWriteArrayList<>();
76      private final AtomicReference<State> state = new AtomicReference<>(State.INIT);
77      private final CompletableFuture<Latest<T>> initialValueFuture = new CompletableFuture<>();
78  
79      @Nullable
80      private volatile Latest<T> latest;
81      @Nullable
82      private volatile ScheduledFuture<?> currentScheduleFuture;
83      @Nullable
84      private volatile CompletableFuture<?> currentWatchFuture;
85  
86      AbstractWatcher(ScheduledExecutorService watchScheduler, String projectName, String repositoryName,
87                      String pathPattern, boolean errorOnEntryNotFound, long delayOnSuccessMillis,
88                      long initialDelayMillis, long maxDelayMillis, double multiplier, double jitterRate) {
89          this.watchScheduler = watchScheduler;
90          this.projectName = projectName;
91          this.repositoryName = repositoryName;
92          this.pathPattern = pathPattern;
93          this.errorOnEntryNotFound = errorOnEntryNotFound;
94          this.delayOnSuccessMillis = delayOnSuccessMillis;
95          this.initialDelayMillis = initialDelayMillis;
96          this.maxDelayMillis = maxDelayMillis;
97          this.multiplier = multiplier;
98          this.jitterRate = jitterRate;
99      }

101     @Override
102     public ScheduledExecutorService watchScheduler() {
103         return watchScheduler;
104     }
105 
106     @Override
107     public CompletableFuture<Latest<T>> initialValueFuture() {
108         return initialValueFuture;
109     }
110 
111     @Override
112     public Latest<T> latest() {
113         final Latest<T> latest = this.latest;
114         if (latest == null) {
115             throw new IllegalStateException("value not available yet");
116         }
117         return latest;
118     }
119 
120     /**
121      * Starts to watch the file specified in the {@link Query} or the {@link PathPattern}
122      * given with the constructor.
123      */
124     void start() {
125         if (state.compareAndSet(State.INIT, State.STARTED)) {
126             scheduleWatch(0);
127         }
128     }
129 
130     @Override
131     public void close() {
132         state.set(State.STOPPED);
133         if (!initialValueFuture.isDone()) {
134             initialValueFuture.cancel(false);
135         }
136 
137         // Cancel any scheduled operations.
138         final ScheduledFuture<?> currentScheduleFuture = this.currentScheduleFuture;
139         if (currentScheduleFuture != null && !currentScheduleFuture.isDone()) {
140             currentScheduleFuture.cancel(false);
141         }
142         final CompletableFuture<?> currentWatchFuture = this.currentWatchFuture;
143         if (currentWatchFuture != null && !currentWatchFuture.isDone()) {
144             currentWatchFuture.cancel(false);
145         }
146     }
147 
148     private boolean isStopped() {
149         return state.get() == State.STOPPED;
150     }
151 
152     @Override
153     public void watch(BiConsumer<? super Revision, ? super T> listener) {
154         watch(listener, watchScheduler);
155     }
156 
157     @Override
158     public void watch(BiConsumer<? super Revision, ? super T> listener, Executor executor) {
159         requireNonNull(listener, "listener");
160         checkState(!isStopped(), "watcher closed");
161         updateListeners.add(new SimpleImmutableEntry<>(listener, executor));
162 
163         final Latest<T> latest = this.latest;
164         if (latest != null) {
165             // There's a chance that listener.accept(...) is called twice for the same value
166             // if this watch method is called:
167             // - after " this.latest = newLatest;" is invoked.
168             // - and before notifyListener() is called.
169             // However, it's such a rare case and we usually call `watch` method right after creating a Watcher,
170             // which means latest is probably not set yet, so we don't use a lock to guarantee
171             // the atomicity.
172             executor.execute(() -> listener.accept(latest.revision(), latest.value()));
173         }
174     }
175 
176     private void scheduleWatch(int numAttemptsSoFar) {
177         if (isStopped()) {
178             return;
179         }
180 
181         final long delay;
182         if (numAttemptsSoFar == 0) {
183             delay = latest != null ? delayOnSuccessMillis : 0;
184         } else {
185             delay = nextDelayMillis(numAttemptsSoFar);
186         }
187 
188         currentScheduleFuture = watchScheduler.schedule(() -> {
189             currentScheduleFuture = null;
190             doWatch(numAttemptsSoFar);
191         }, delay, TimeUnit.MILLISECONDS);
192     }
193 
194     private long nextDelayMillis(int numAttemptsSoFar) {
195         final long nextDelayMillis;
196         if (numAttemptsSoFar == 1) {
197             nextDelayMillis = initialDelayMillis;
198         } else {
199             nextDelayMillis =
200                     Math.min(saturatedMultiply(initialDelayMillis, Math.pow(multiplier, numAttemptsSoFar - 1)),
201                              maxDelayMillis);
202         }
203 
204         final long minJitter = (long) (nextDelayMillis * (1 - jitterRate));
205         final long maxJitter = (long) (nextDelayMillis * (1 + jitterRate));
206         final long bound = maxJitter - minJitter + 1;
207         final long millis = random(bound);
208         return Math.max(0, saturatedAdd(minJitter, millis));
209     }
210 
211     private static long saturatedMultiply(long left, double right) {
212         final double result = left * right;
213         return result >= Long.MAX_VALUE ? Long.MAX_VALUE : (long) result;
214     }
215 
216     private static long random(long bound) {
217         assert bound > 0;
218         final long mask = bound - 1;
219         final Random random = ThreadLocalRandom.current();
220         long result = random.nextLong();
221 
222         if ((bound & mask) == 0L) {
223             // power of two
224             result &= mask;
225         } else { // reject over-represented candidates
226             for (long u = result >>> 1; u + mask - (result = u % bound) < 0L; u = random.nextLong() >>> 1) {
227                 continue;
228             }
229         }
230 
231         return result;
232     }
233 
234     private void doWatch(int numAttemptsSoFar) {
235         if (isStopped()) {
236             return;
237         }
238 
239         final Latest<T> latest = this.latest;
240         final Revision lastKnownRevision = latest != null ? latest.revision() : Revision.INIT;
241         final CompletableFuture<Latest<T>> f = doWatch(lastKnownRevision);
242 
243         currentWatchFuture = f;
244         f.thenAccept(newLatest -> {
245              currentWatchFuture = null;
246              if (newLatest != null) {
247                  this.latest = newLatest;
248                  logger.debug("watcher noticed updated file {}/{}{}: rev={}",
249                               projectName, repositoryName, pathPattern, newLatest.revision());
250                  notifyListeners(newLatest);
251                  if (!initialValueFuture.isDone()) {
252                      initialValueFuture.complete(newLatest);
253                  }
254              }
255 
256              // Watch again for the next change.
257              scheduleWatch(0);
258          })
259          .exceptionally(thrown -> {
260              currentWatchFuture = null;
261              try {
262                  final Throwable cause = thrown instanceof CompletionException ? thrown.getCause() : thrown;
263                  boolean logged = false;
264                  if (cause instanceof CentralDogmaException) {
265                      if (cause instanceof EntryNotFoundException) {
266                          if (!initialValueFuture.isDone() && errorOnEntryNotFound) {
267                              initialValueFuture.completeExceptionally(thrown);
268                              close();
269                              return null;
270                          }
271                          logger.info("{}/{}{} does not exist yet; trying again",
272                                      projectName, repositoryName, pathPattern);
273                          logged = true;
274                      } else if (cause instanceof RepositoryNotFoundException) {
275                          logger.info("{}/{} does not exist yet; trying again",
276                                      projectName, repositoryName);
277                          logged = true;
278                      } else if (cause instanceof ShuttingDownException) {
279                          logger.info("Central Dogma is shutting down; trying again");
280                          logged = true;
281                      }
282                  }
283 
284                  if (cause instanceof CancellationException) {
285                      // Cancelled by close()
286                      return null;
287                  }
288 
289                  if (!logged) {
290                      logger.warn("Failed to watch a file ({}/{}{}) at Central Dogma; trying again",
291                                  projectName, repositoryName, pathPattern, cause);
292                  }
293 
294                  scheduleWatch(numAttemptsSoFar + 1);
295              } catch (Throwable t) {
296                  logger.error("Unexpected exception while watching a file at Central Dogma:", t);
297              }
298              return null;
299          });
300     }
301 
302     abstract CompletableFuture<Latest<T>> doWatch(Revision lastKnownRevision);
303 
304     private void notifyListeners(Latest<T> latest) {
305         if (isStopped()) {
306             // Do not notify after stopped.
307             return;
308         }
309 
310         for (Map.Entry<BiConsumer<? super Revision, ? super T>, Executor> entry : updateListeners) {
311             final BiConsumer<? super Revision, ? super T> listener = entry.getKey();
312             final Executor executor = entry.getValue();
313             executor.execute(() -> {
314                 try {
315                     listener.accept(latest.revision(), latest.value());
316                 } catch (Exception e) {
317                     logger.warn("Exception thrown for watcher ({}/{}{}): rev={}",
318                                 projectName, repositoryName, pathPattern, latest.revision(), e);
319                 }
320             });
321         }
322     }

324     @Override
325     public String toString() {
326         return MoreObjects.toStringHelper(this).omitNullValues()
327                           .add("watchScheduler", watchScheduler)
328                           .add("projectName", projectName)
329                           .add("repositoryName", repositoryName)
330                           .add("pathPattern", pathPattern)
331                           .add("errorOnEntryNotFound", errorOnEntryNotFound)
332                           .add("delayOnSuccessMillis", delayOnSuccessMillis)
333                           .add("initialDelayMillis", initialDelayMillis)
334                           .add("maxDelayMillis", maxDelayMillis)
335                           .add("multiplier", multiplier)
336                           .add("jitterRate", jitterRate)
337                           .add("latest", latest)
338                           .toString();
339     }

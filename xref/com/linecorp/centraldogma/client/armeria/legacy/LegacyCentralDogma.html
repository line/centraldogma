241     private static <T> Entry<T> toEntry(Query<T> query, Revision normRev, QueryType queryType, String content,
242                                         com.linecorp.centraldogma.internal.thrift.EntryType receivedEntryType) {
243         switch (queryType) {
244             case IDENTITY_TEXT:
245                 return entryAsText(query, normRev, content);
246             case IDENTITY_JSON:
247             case JSON_PATH:
248                 if (receivedEntryType != com.linecorp.centraldogma.internal.thrift.EntryType.JSON) {
249                     throw new CentralDogmaException("invalid entry type. entry type: " + receivedEntryType +
250                                                     " (expected: " + queryType + ')');
251                 }
252                 return entryAsJson(query, normRev, content);
253             case IDENTITY:
254                 switch (receivedEntryType) {
255                     case JSON:
256                         return entryAsJson(query, normRev, content);
257                     case TEXT:
258                         return entryAsText(query, normRev, content);
259                     case DIRECTORY:
260                         return unsafeCast(Entry.ofDirectory(normRev, query.path()));
261                 }
262         }
263         throw new Error(); // Should never reach here.
264     }

266     private static <T> Entry<T> entryAsJson(Query<T> query, Revision normRev, String content) {
267         try {
268             return unsafeCast(Entry.ofJson(normRev, query.path(), Jackson.readTree(content)));
269         } catch (IOException e) {
270             throw new CentralDogmaException("failed to parse the query result: " + query, e);
271         }
272     }

274     private static <T> Entry<T> entryAsText(Query<T> query, Revision normRev, String content) {
275         return unsafeCast(Entry.ofText(normRev, query.path(), content));

294     @Override
295     public <T> CompletableFuture<MergedEntry<T>> mergeFiles(String projectName, String repositoryName,
296                                                             Revision revision, MergeQuery<T> mergeQuery) {
297         validateProjectAndRepositoryName(projectName, repositoryName);
298         requireNonNull(revision, "revision");
299         requireNonNull(mergeQuery, "mergeQuery");
300         final CompletableFuture<com.linecorp.centraldogma.internal.thrift.MergedEntry> future =
301                 run(callback -> {
302                     client.mergeFiles(projectName, repositoryName,
303                                       RevisionConverter.TO_DATA.convert(revision),
304                                       MergeQueryConverter.TO_DATA.convert(mergeQuery),
305                                       callback);
306                 });
307         return future.thenApply(entry -> {
308             final EntryType entryType = EntryConverter.convertEntryType(entry.getType());
309             assert entryType != null;
310             switch (entryType) {
311                 case JSON:
312                     try {
313                         @SuppressWarnings("unchecked")
314                         final MergedEntry<T> converted = (MergedEntry<T>) MergedEntry.of(
315                                 RevisionConverter.TO_MODEL.convert(entry.revision),
316                                 entryType, Jackson.readTree(entry.content), entry.paths);
317                         return converted;
318                     } catch (IOException e) {
319                         throw new CentralDogmaException(
320                                 "failed to parse the content: " + entry.content, e);
321                     }
322                 default:
323                     throw new Error("unsupported entry type: " + entryType);
324             }
325         });

348     @Override
349     public <T> CompletableFuture<Change<T>> getDiff(String projectName, String repositoryName,
350                                                     Revision from, Revision to, Query<T> query) {
351         validateProjectAndRepositoryName(projectName, repositoryName);
352         requireNonNull(from, "from");
353         requireNonNull(to, "to");
354         requireNonNull(query, "query");
355         final CompletableFuture<DiffFileResult> future = run(callback -> {
356             client.diffFile(projectName, repositoryName,
357                             RevisionConverter.TO_DATA.convert(from),
358                             RevisionConverter.TO_DATA.convert(to),
359                             QueryConverter.TO_DATA.convert(query), callback);
360         });
361         return future.thenApply(r -> {
362             if (r == null) {
363                 return null;
364             }
365 
366             final Change<T> converted;
367             switch (r.getType()) {
368                 case UPSERT_JSON:
369                     converted = unsafeCast(Change.ofJsonUpsert(query.path(), r.getContent()));
370                     break;
371                 case UPSERT_TEXT:
372                     converted = unsafeCast(Change.ofTextUpsert(query.path(), r.getContent()));
373                     break;
374                 case REMOVE:
375                     converted = unsafeCast(Change.ofRemoval(query.path()));
376                     break;
377                 case RENAME:
378                     converted = unsafeCast(Change.ofRename(query.path(), r.getContent()));
379                     break;
380                 case APPLY_JSON_PATCH:
381                     converted = unsafeCast(Change.ofJsonPatch(query.path(), r.getContent()));
382                     break;
383                 case APPLY_TEXT_PATCH:
384                     converted = unsafeCast(Change.ofTextPatch(query.path(), r.getContent()));
385                     break;
386                 default:
387                     throw new Error("unknown change type: " + r.getType());
388             }
389 
390             return converted;
391         });
392     }

534     @Nullable
535     private static <T, U, V extends Collection<T>> V convertToCollection(
536             @Nullable Iterable<U> c, Function<U, T> mapper, Collector<T, ?, V> collector) {
537         if (c == null) {
538             return null;
539         }
540 
541         final Stream<U> stream;
542         if (c instanceof Collection) {
543             stream = ((Collection<U>) c).stream();
544         } else {
545             stream = StreamSupport.stream(c.spliterator(), false);
546         }
547 
548         return stream.map(mapper).collect(collector);
549     }

555         if (c == null) {
556             return null;
557         }
558         return c.stream().map(entryMapper).collect(toImmutableMap(keyMapper, valueMapper));

567     private static <T> CompletableFuture<T> run(ThriftCall<T> call) {
568         final ThriftFuture<T> future = new ThriftFuture<>();
569         try {
570             call.apply(future);
571             return future.exceptionally(cause -> Exceptions.throwUnsafely(convertCause(cause)));
572         } catch (Exception e) {
573             return CompletableFutures.exceptionallyCompletedFuture(convertCause(e));
574         }
575     }
576 
577     private static Throwable convertCause(Throwable cause) {
578         final Throwable peeledCause = Exceptions.peel(cause);
579         final Throwable convertedCause;
580 
581         if (peeledCause instanceof com.linecorp.centraldogma.internal.thrift.CentralDogmaException) {
582             final String message = peeledCause.getMessage();
583             switch (((com.linecorp.centraldogma.internal.thrift.CentralDogmaException) peeledCause)
584                     .getErrorCode()) {
585                 case UNIMPLEMENTED:
586                     convertedCause = new CentralDogmaException("unimplemented", false);
587                     break;
588                 case INTERNAL_SERVER_ERROR:
589                     convertedCause = new CentralDogmaException("internal server error", false);
590                     break;
591                 case BAD_REQUEST:
592                     convertedCause = new CentralDogmaException("bad request", false);
593                     break;
594                 case PROJECT_NOT_FOUND:
595                     convertedCause = new ProjectNotFoundException(message, false);
596                     break;
597                 case PROJECT_EXISTS:
598                     convertedCause = new ProjectExistsException(message, false);
599                     break;
600                 case REPOSITORY_NOT_FOUND:
601                     convertedCause = new RepositoryNotFoundException(message, false);
602                     break;
603                 case REPOSITORY_EXISTS:
604                     convertedCause = new RepositoryExistsException(message, false);
605                     break;
606                 case REVISION_NOT_FOUND:
607                     convertedCause = new RevisionNotFoundException(message, false);
608                     break;
609                 case REVISION_EXISTS:
610                     convertedCause = new ChangeConflictException(message, false);
611                     break;
612                 case ENTRY_NOT_FOUND:
613                     convertedCause = new EntryNotFoundException(message, false);
614                     break;
615                 case REDUNDANT_CHANGE:
616                     convertedCause = new RedundantChangeException(message, false);
617                     break;
618                 case CHANGE_CONFLICT:
619                     convertedCause = new ChangeConflictException(message, false);
620                     break;
621                 case QUERY_FAILURE:
622                     convertedCause = new QueryExecutionException(message, false);
623                     break;
624                 case SHUTTING_DOWN:
625                     convertedCause = new ShuttingDownException(message, false);
626                     break;
627                 default:
628                     throw new Error();
629             }
630         } else {
631             convertedCause = peeledCause;
632         }
633         return convertedCause;
634     }

637     public void close() {
638         endpointGroup.close();
639     }

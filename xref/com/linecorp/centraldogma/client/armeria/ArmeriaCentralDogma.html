489     private static Map<String, Entry<?>> getFiles(Revision normRev, AggregatedHttpResponse res) {
490         switch (res.status().code()) {
491             case 200:
492                 final JsonNode node = toJson(res, null);
493                 final ImmutableMap.Builder<String, Entry<?>> builder = ImmutableMap.builder();
494                 if (node.isObject()) { // Single entry
495                     final Entry<?> entry = toEntry(normRev, node, QueryType.IDENTITY);
496                     builder.put(entry.path(), entry);
497                 } else if (node.isArray()) { // Multiple entries
498                     node.forEach(e -> {
499                         final Entry<?> entry = toEntry(normRev, e, QueryType.IDENTITY);
500                         builder.put(entry.path(), entry);
501                     });
502                 } else {
503                     return rejectNeitherArrayNorObject(res);
504                 }
505                 return builder.build();
506             case 204:
507                 return ImmutableMap.of();
508         }
509 
510         return handleErrorResponse(res);
511     }

534     private static <T> MergedEntry<T> mergeFiles(AggregatedHttpResponse res) {
535         if (res.status().code() == 200) {
536             final JsonNode node = toJson(res, JsonNodeType.OBJECT);
537 
538             // Build path list.
539             final ImmutableList.Builder<String> pathsBuilder = ImmutableList.builder();
540             for (JsonNode path : getField(node, "paths")) {
541                 if (path.getNodeType() != JsonNodeType.STRING) {
542                     throw new CentralDogmaException("Received a merged entry with a non-string path: " + node);
543                 }
544                 pathsBuilder.add(path.asText());
545             }
546             final List<String> paths = pathsBuilder.build();
547             if (paths.isEmpty()) {
548                 throw new CentralDogmaException("Received a merged entry with empty paths: " + node);
549             }
550 
551             // Build the merged entry.
552             final Revision revision = new Revision(getField(node, "revision").asInt());
553             final EntryType type = EntryType.valueOf(getField(node, "type").asText());
554             final JsonNode content = getField(node, "content");
555             switch (type) {
556                 case JSON: {
557                     @SuppressWarnings("unchecked")
558                     final MergedEntry<T> cast =
559                             (MergedEntry<T>) MergedEntry.of(revision, type, content, paths);
560                     return cast;
561                 }
562                 case TEXT: {
563                     if (content.getNodeType() != JsonNodeType.STRING) {
564                         throw new CentralDogmaException(
565                                 "Received a TEXT merged entry whose content is not a string: " + node);
566                     }
567                     @SuppressWarnings("unchecked")
568                     final MergedEntry<T> cast =
569                             (MergedEntry<T>) MergedEntry.of(revision, type, content.asText(), paths);
570                     return cast;
571                 }
572                 default:
573                     throw new CentralDogmaException(
574                             "Received a merged entry whose type is neither JSON nor TEXT: " + node);
575             }
576         }
577 
578         return handleErrorResponse(res);
579     }

589         validateProjectAndRepositoryName(projectName, repositoryName);
590         checkArgument(maxCommits >= 0 && maxCommits <= HistoryConstants.MAX_MAX_COMMITS,

592                       maxCommits, HistoryConstants.MAX_MAX_COMMITS);
593         try {

595             path.append("/commits/").append(from.text());
596             path.append("?to=").append(to.text());
597             path.append("&path=").append(pathPattern.encoded());
598             if (maxCommits > 0) {
599                 path.append("&maxCommits=").append(maxCommits);
600             }
601 
602             return client.execute(headers(HttpMethod.GET, path.toString()))
603                          .aggregate()
604                          .thenApply(ArmeriaCentralDogma::getHistory);
605         } catch (Exception e) {
606             return exceptionallyCompletedFuture(e);
607         }
608     }

610     private static List<Commit> getHistory(AggregatedHttpResponse res) {
611         switch (res.status().code()) {
612             case 200:
613                 final JsonNode node = toJson(res, null);
614                 if (node.isObject()) {
615                     return ImmutableList.of(toCommit(node));
616                 } else if (node.isArray()) {
617                     return Streams.stream(node)
618                                   .map(ArmeriaCentralDogma::toCommit)

864     private <T> CompletableFuture<T> watch(Revision lastKnownRevision, long timeoutMillis,
865                                            String path, QueryType queryType,
866                                            BiFunction<AggregatedHttpResponse, QueryType, T> func,
867                                            boolean errorOnEntryNotFound) {
868         final RequestHeadersBuilder builder = headersBuilder(HttpMethod.GET, path);
869         builder.set(HttpHeaderNames.IF_NONE_MATCH, lastKnownRevision.text())
870                .set(HttpHeaderNames.PREFER, "wait=" + LongMath.saturatedAdd(timeoutMillis, 999) / 1000L +
871                                             ", notify-entry-not-found=" + errorOnEntryNotFound);
872 
873         try (SafeCloseable ignored = Clients.withContextCustomizer(ctx -> {
874             final long responseTimeoutMillis = ctx.responseTimeoutMillis();
875             final long adjustmentMillis = WatchTimeout.availableTimeout(timeoutMillis, responseTimeoutMillis);
876             if (responseTimeoutMillis > 0) {
877                 ctx.setResponseTimeoutMillis(TimeoutMode.EXTEND, adjustmentMillis);
878             } else {
879                 ctx.setResponseTimeoutMillis(adjustmentMillis);
880             }
881         })) {
882             return client.execute(builder.build()).aggregate()
883                          .handle((res, cause) -> {
884                              if (cause == null) {
885                                  return func.apply(res, queryType);
886                              }
887 
888                              if ((cause instanceof ClosedStreamException) &&
889                                  client.options().factory().isClosing()) {
890                                  // A user closed the client factory while watching.
891                                  return null;
892                              }
893 
894                              return Exceptions.throwUnsafely(cause);
895                          });
896         }
897     }

1019     private static <T> Entry<T> toEntry(Revision revision, JsonNode node, QueryType queryType) {
1020         final String entryPath = getField(node, "path").asText();
1021         final EntryType receivedEntryType = EntryType.valueOf(getField(node, "type").asText());
1022         switch (queryType) {
1023             case IDENTITY_TEXT:
1024                 return entryAsText(revision, node, entryPath);
1025             case IDENTITY_JSON:
1026             case JSON_PATH:
1027                 if (receivedEntryType != EntryType.JSON) {
1028                     throw new CentralDogmaException("invalid entry type. entry type: " + receivedEntryType +
1029                                                     " (expected: " + queryType + ')');
1030                 }
1031                 return entryAsJson(revision, node, entryPath);
1032             case IDENTITY:
1033                 switch (receivedEntryType) {
1034                     case JSON:
1035                         return entryAsJson(revision, node, entryPath);
1036                     case TEXT:
1037                         return entryAsText(revision, node, entryPath);
1038                     case DIRECTORY:
1039                         return unsafeCast(Entry.ofDirectory(revision, entryPath));
1040                 }
1041         }
1042         throw new Error(); // Should never reach here.
1043     }

1074     private static <T> Change<T> toChange(JsonNode node) {
1075         final String actualPath = getField(node, "path").asText();
1076         final ChangeType type = ChangeType.valueOf(getField(node, "type").asText());
1077         switch (type) {
1078             case UPSERT_JSON:
1079                 return unsafeCast(Change.ofJsonUpsert(actualPath, getField(node, "content")));
1080             case UPSERT_TEXT:
1081                 return unsafeCast(Change.ofTextUpsert(actualPath, getField(node, "content").asText()));
1082             case REMOVE:
1083                 return unsafeCast(Change.ofRemoval(actualPath));
1084             case RENAME:
1085                 return unsafeCast(Change.ofRename(actualPath, getField(node, "content").asText()));
1086             case APPLY_JSON_PATCH:
1087                 return unsafeCast(Change.ofJsonPatch(actualPath, getField(node, "content")));
1088             case APPLY_TEXT_PATCH:
1089                 return unsafeCast(Change.ofTextPatch(actualPath, getField(node, "content").asText()));
1090         }
1091 
1092         throw new Error(); // Never reaches here.
1093     }

1116     private static <T> T handleErrorResponse(AggregatedHttpResponse res) {
1117         final HttpStatus status = res.status();
1118         if (status.codeClass() != HttpStatusClass.SUCCESS) {
1119             final JsonNode node = toJson(res, JsonNodeType.OBJECT);
1120             final JsonNode exceptionNode = node.get("exception");
1121             final JsonNode messageNode = node.get("message");
1122 
1123             if (exceptionNode != null) {
1124                 final String typeName = exceptionNode.textValue();
1125                 if (typeName != null) {
1126                     final Function<String, CentralDogmaException> exceptionFactory =
1127                             EXCEPTION_FACTORIES.get(typeName);
1128                     if (exceptionFactory != null) {
1129                         throw exceptionFactory.apply(messageNode.textValue());
1130                     }
1131                 }
1132             }
1133         }
1134 
1135         throw new CentralDogmaException("unexpected response: " + res.headers() + ", " + res.contentUtf8());
1136     }

74  /**
75   * Builds a {@link CentralDogma} client based on an <a href="https://line.github.io/armeria/">Armeria</a>
76   * HTTP client.
77   * This client differs from {@link ArmeriaCentralDogma} in that making requests is done in two phases.
78   * <ol>
79   *     <li>
80   *         An xDS request is made to xDS servers to fetch a {@link Listener} resource which represents
81   *         how to connect to {@link CentralDogma} servers.
82   *     </li>
83   *     <li>
84   *         Actual {@link CentralDogma} client requests are made based on the watched Listener resource.
85   *     </li>
86   * </ol>
87   * Because connection to the actual {@link CentralDogma} is determined by the watched {@link Listener},
88   * {@link #hosts()} and {@link #isUseTls()} apply to connecting to the bootstrap only.
89   * However, because Armeria's xDS implementation isn't complete, the following parameters are applied
90   * to both xDS and Central Dogma requests.
91   * <ul>
92   *     <li>{@link #accessToken(String)}</li>
93   *     <li>{@link #clientConfigurator(ArmeriaClientConfigurator)}</li>
94   *     <li>{@link #clientFactory(ClientFactory)}</li>
95   * </ul>
96   *
97   * <p>Note that this module is considered experimental and subject to behavioral change.
98   */
99  @UnstableApi
100 public final class XdsCentralDogmaBuilder extends AbstractCentralDogmaBuilder<XdsCentralDogmaBuilder> {
101

103     @VisibleForTesting
104     static final String DEFAULT_LISTENER_NAME = "centraldogma-listener";

110     private Locality locality = Locality.getDefaultInstance();
111     // an empty string means no local cluster
112     private String serviceCluster = "";

196     /**
197      * Returns a newly-created {@link CentralDogma} instance.
198      */
199     public CentralDogma build() {
200         final XdsBootstrap xdsBootstrap = xdsBootstrap();
201         final EndpointGroup endpointGroup = XdsEndpointGroup.of(listenerName, xdsBootstrap);
202         final String scheme = "none+" + (isUseTls() ? "https" : "http");
203         final ClientBuilder builder =
204                 newClientBuilder(scheme, endpointGroup, cb -> cb.decorator(DecodingClient.newDecorator()), "/");
205         final int maxRetriesOnReplicationLag = maxNumRetriesOnReplicationLag();
206 
207         // TODO(ikhoon): Apply ExecutorServiceMetrics for the 'blockingTaskExecutor' once
208         //               https://github.com/line/centraldogma/pull/542 is merged.
209         final ScheduledExecutorService blockingTaskExecutor = this.blockingTaskExecutor;
210 
211         final CentralDogma dogma = new ArmeriaCentralDogma(blockingTaskExecutor,
212                                                            builder.build(WebClient.class),
213                                                            accessToken(),
214                                                            () -> {
215                                                                endpointGroup.close();
216                                                                xdsBootstrap.close();
217                                                            });
218         if (maxRetriesOnReplicationLag <= 0) {
219             return dogma;
220         } else {
221             return new ReplicationLagTolerantCentralDogma(
222                     blockingTaskExecutor, dogma, maxRetriesOnReplicationLag,
223                     retryIntervalOnReplicationLagMillis(),
224                     () -> {
225                         // FIXME(trustin): Note that this will always return `null` due to a known limitation
226                         //                 in Armeria: https://github.com/line/armeria/issues/760
227                         final ClientRequestContext ctx = ClientRequestContext.currentOrNull();
228                         return ctx != null ? ctx.remoteAddress() : null;
229                     });
230         }
231     }

233     private ClientBuilder newClientBuilder(String scheme, EndpointGroup endpointGroup,
234                                            Consumer<ClientBuilder> customizer, String path) {
235         final ClientBuilder builder = Clients.builder(scheme, endpointGroup, path);
236         customizer.accept(builder);
237         clientConfigurator.configure(builder);
238         builder.factory(clientFactory);
239         return builder;
240     }
241 
242     private boolean isUnresolved() {
243         final Set<InetSocketAddress> hosts = hosts();
244         checkState(!hosts.isEmpty(), "No hosts were added.");
245         final Map<Boolean, List<InetSocketAddress>> addrByUnresolved =
246                 hosts.stream().collect(Collectors.partitioningBy(InetSocketAddress::isUnresolved));
247         // Until multiple clusters are supported, restrict users to either use STATIC or DNS (but not both)
248         checkState(addrByUnresolved.get(true).isEmpty() ||
249                    addrByUnresolved.get(false).isEmpty(),
250                    "Cannot mix resolved and unresolved hosts (%s)", addrByUnresolved);
251         final InetSocketAddress firstHost = Iterables.get(hosts(), 0);
252         return firstHost.isUnresolved();
253     }
254 
255     private XdsBootstrap xdsBootstrap() {
256         final GrpcService grpcService = GrpcService
257                 .newBuilder()
258                 .setEnvoyGrpc(EnvoyGrpc.newBuilder()
259                                        .setClusterName(BOOTSTRAP_CLUSTER_NAME))
260                 .addInitialMetadata(HeaderValue.newBuilder()
261                                                .setKey(HttpHeaderNames.AUTHORIZATION.toString())
262                                                .setValue("Bearer " + accessToken()))
263                 .build();
264         final ApiConfigSource apiConfigSource = ApiConfigSource
265                 .newBuilder()
266                 .addGrpcServices(grpcService)
267                 .setApiType(ApiType.AGGREGATED_GRPC)
268                 .build();
269         final DynamicResources dynamicResources =
270                 DynamicResources.newBuilder().setAdsConfig(apiConfigSource).build();
271         final Bootstrap bootstrap =
272                 Bootstrap.newBuilder()
273                          .setDynamicResources(dynamicResources)
274                          .setNode(Node.newBuilder()
275                                       .setCluster(serviceCluster)
276                                       .setLocality(locality))
277                          .setStaticResources(StaticResources.newBuilder().addClusters(bootstrapCluster()))
278                          .build();
279         return xdsBootstrapFactory.apply(bootstrap);
280     }
281 
282     private Cluster bootstrapCluster() {
283         final boolean isUnresolved = isUnresolved();
284 
285         final Cluster.Builder clusterBuilder = Cluster.newBuilder();
286         if (isUnresolved) {
287             clusterBuilder.setType(DiscoveryType.STRICT_DNS);
288         } else {
289             clusterBuilder.setType(DiscoveryType.STATIC);
290         }
291 
292         final LocalityLbEndpoints.Builder localityLbEndpointsBuilder = LocalityLbEndpoints.newBuilder();
293         for (InetSocketAddress addr : hosts()) {
294             final LbEndpoint lbEndpoint = fromAddress(addr);
295             localityLbEndpointsBuilder.addLbEndpoints(lbEndpoint);
296         }
297         final ClusterLoadAssignment clusterLoadAssignment =
298                 ClusterLoadAssignment.newBuilder().addEndpoints(localityLbEndpointsBuilder.build()).build();
299 
300         if (isUseTls()) {
301             clusterBuilder.setTransportSocket(
302                     TransportSocket.newBuilder()
303                                    .setName("envoy.transport_sockets.tls")
304                                    .setTypedConfig(Any.pack(UpstreamTlsContext.getDefaultInstance())));
305         }
306 
307         clusterBuilder.setLoadAssignment(clusterLoadAssignment)
308                       .setName(BOOTSTRAP_CLUSTER_NAME);
309         return clusterBuilder.build();
310     }
311 
312     private static LbEndpoint fromAddress(InetSocketAddress addr) {
313         final String hostString = addr.getHostString();
314         final int port = addr.getPort();
315         final Address address = Address.newBuilder()
316                                        .setSocketAddress(SocketAddress.newBuilder()
317                                                                       .setAddress(hostString)
318                                                                       .setPortValue(port)).build();
319         return LbEndpoint.newBuilder()
320                          .setEndpoint(Endpoint.newBuilder()
321                                               .setAddress(address))
322                          .build();
323     }

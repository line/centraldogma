16  package com.linecorp.centraldogma.client;
17  
18  import static java.util.Objects.requireNonNull;
19  
20  import java.util.concurrent.CompletableFuture;
21  
22  import com.linecorp.centraldogma.common.Entry;
23  import com.linecorp.centraldogma.common.Query;
24  import com.linecorp.centraldogma.common.QueryType;
25  import com.linecorp.centraldogma.common.Revision;
26  
27  /**
28   * Prepares to send a {@link CentralDogma#getFile(String, String, Revision, Query)} request to the
29   * Central Dogma repository.
30   */
31  public final class FileRequest<T> extends AbstractFileRequest<FileRequest<T>> {
32  
33      private final Query<T> query;
34      private final CentralDogmaRepository centralDogmaRepo;
35  
36      FileRequest(CentralDogmaRepository centralDogmaRepo, Query<T> query) {
37          this.query = query;
38          this.centralDogmaRepo = centralDogmaRepo;
39      }
40  
41      /**
42       * Retrieves a file located at {@link Query#path()} at the {@link Revision#HEAD}.
43       *
44       * @return the {@link Entry} that is matched by the given {@link Query}
45       */
46      public CompletableFuture<Entry<T>> get() {
47          return get(Revision.HEAD);
48      }
49  
50      /**
51       * Retrieves a file located at {@link Query#path()} at the {@link Revision}.
52       *
53       * @return the {@link Entry} that is matched by the given {@link Query}
54       */
55      public CompletableFuture<Entry<T>> get(Revision revision) {
56          requireNonNull(revision, "revision");
57          if (viewRaw() && query.type() == QueryType.JSON_PATH) {
58              throw new IllegalStateException("JSON_PATH query cannot be used with raw view");
59          }
60          return centralDogmaRepo.centralDogma().getFile(centralDogmaRepo.projectName(),
61                                                         centralDogmaRepo.repositoryName(),
62                                                         revision, query, viewRaw());
63      }
64  }

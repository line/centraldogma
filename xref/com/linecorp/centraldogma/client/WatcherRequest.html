16  package com.linecorp.centraldogma.client;
17  
18  import static com.google.common.base.Preconditions.checkArgument;
19  import static com.google.common.base.Preconditions.checkState;
20  import static com.linecorp.centraldogma.internal.Util.validateStructuredFilePath;
21  import static java.util.Objects.requireNonNull;
22  
23  import java.time.Duration;
24  import java.util.concurrent.Executor;
25  import java.util.concurrent.ScheduledExecutorService;
26  import java.util.concurrent.TimeUnit;
27  import java.util.function.Function;
28  
29  import org.jspecify.annotations.Nullable;
30  
31  import com.linecorp.centraldogma.common.PathPattern;
32  import com.linecorp.centraldogma.common.Query;
33  
34  import io.micrometer.core.instrument.MeterRegistry;
35  
36  /**
37   * Prepares to create a {@link Watcher}.
38   */
39  public final class WatcherRequest<T> extends WatchOptions {
40  
41      private static final long DEFAULT_DELAY_ON_SUCCESS_MILLIS = TimeUnit.SECONDS.toMillis(1);
42      private static final long DEFAULT_MAX_DELAY_MILLIS = TimeUnit.MINUTES.toMillis(1);
43      private static final double DEFAULT_MULTIPLIER = 2.0;
44      private static final double DEFAULT_JITTER_RATE = 0.2;
45  
46      private final CentralDogmaRepository centralDogmaRepo;
47      @Nullable
48      private final Query<T> query;
49      @Nullable
50      private final PathPattern pathPattern;
51      private final ScheduledExecutorService blockingTaskExecutor;
52  
53      @Nullable
54      private Function<Object, ? extends T> mapper;
55      private Executor executor;
56  
57      @Nullable
58      private final MeterRegistry meterRegistry;
59  
60      private long delayOnSuccessMillis = DEFAULT_DELAY_ON_SUCCESS_MILLIS;
61      private long initialDelayMillis = DEFAULT_DELAY_ON_SUCCESS_MILLIS * 2;
62      private long maxDelayMillis = DEFAULT_MAX_DELAY_MILLIS;
63      private double multiplier = DEFAULT_MULTIPLIER;
64      private double jitterRate = DEFAULT_JITTER_RATE;
65      private boolean renderTemplate;
66      @Nullable
67      private String variableFile;
68  
69      WatcherRequest(CentralDogmaRepository centralDogmaRepo, Query<T> query,
70                     ScheduledExecutorService blockingTaskExecutor, @Nullable MeterRegistry meterRegistry) {
71          this(centralDogmaRepo, query, null, blockingTaskExecutor, meterRegistry);
72      }
73  
74      WatcherRequest(CentralDogmaRepository centralDogmaRepo, PathPattern pathPattern,
75                     ScheduledExecutorService blockingTaskExecutor, @Nullable MeterRegistry meterRegistry) {
76          this(centralDogmaRepo, null, pathPattern, blockingTaskExecutor, meterRegistry);
77      }
78  
79      private WatcherRequest(CentralDogmaRepository centralDogmaRepo, @Nullable Query<T> query,
80                             @Nullable PathPattern pathPattern, ScheduledExecutorService blockingTaskExecutor,
81                             @Nullable MeterRegistry meterRegistry) {
82          this.centralDogmaRepo = centralDogmaRepo;
83          this.query = query;
84          this.pathPattern = pathPattern;
85          this.blockingTaskExecutor = blockingTaskExecutor;
86          executor = blockingTaskExecutor;
87          this.meterRegistry = meterRegistry;
88      }
89  
90      /**
91       * Sets the {@link Function} to apply to the result of a watch request.
92       */
93      @SuppressWarnings("unchecked")
94      public <U> WatcherRequest<U> map(Function<? super T, ? extends U> mapper) {
95          if (this.mapper == null) {
96              this.mapper = (Function<Object, ? extends T>) mapper;
97          } else {
98              this.mapper = (Function<Object, ? extends T>) this.mapper.andThen(mapper);
99          }
100         return (WatcherRequest<U>) this;
101     }
102 
103     /**
104      * Sets the {@link Executor} to execute the {@link #map(Function)}.
105      */
106     public WatcherRequest<T> mapperExecutor(Executor executor) {
107         this.executor = executor;
108         return this;
109     }
110 
111     /**
112      * Sets the delay for sending the next watch request when the previous request succeeds.
113      */
114     public WatcherRequest<T> delayOnSuccess(Duration delayOnSuccess) {
115         requireNonNull(delayOnSuccess, "delayOnSuccess");
116         checkArgument(!delayOnSuccess.isNegative(),
117                       "delayOnSuccess: %s (expected: >= 0)", delayOnSuccess);
118         return delayOnSuccessMillis(delayOnSuccess.toMillis());
119     }
120 
121     /**
122      * Sets the delay in milliseconds for sending the next watch request when the previous request succeeds.
123      */
124     public WatcherRequest<T> delayOnSuccessMillis(long delayOnSuccessMillis) {
125         this.delayOnSuccessMillis = delayOnSuccessMillis;
126         checkArgument(delayOnSuccessMillis >= 0,
127                       "delayOnSuccessMillis: %s (expected: >= 0)", delayOnSuccessMillis);
128         return this;
129     }
130 
131     /**
132      * Sets the delays and multiplier which is used to calculate the delay
133      * for sending the next watch request when the previous request fails.
134      * Currently, it uses exponential backoff. File a feature request if you need another algorithm.
135      */
136     public WatcherRequest<T> backoffOnFailure(long initialDelayMillis, long maxDelayMillis,
137                                               double multiplier) {
138         checkArgument(initialDelayMillis >= 0, "initialDelayMillis: %s (expected: >= 0)", initialDelayMillis);
139         checkArgument(initialDelayMillis <= maxDelayMillis, "maxDelayMillis: %s (expected: >= %s)",
140                       maxDelayMillis, initialDelayMillis);
141         checkArgument(multiplier > 1.0, "multiplier: %s (expected: > 1.0)", multiplier);
142         this.initialDelayMillis = initialDelayMillis;
143         this.maxDelayMillis = maxDelayMillis;
144         this.multiplier = multiplier;
145         return this;
146     }
147 
148     /**
149      * Sets the jitter to apply the delay.
150      */
151     public WatcherRequest<T> jitterRate(double jitterRate) {
152         checkArgument(0.0 <= jitterRate && jitterRate <= 1.0,
153                       "jitterRate: %s (expected: >= 0.0 and <= 1.0)", jitterRate);
154         this.jitterRate = jitterRate;
155         return this;
156     }
157 
158     /**
159      * Sets whether to apply template processing to the file using the variables defined in
160      * the same repository and its parent project.
161      * By default, template processing is not applied.
162      *
163      * <p>This option is only valid when watching a single file via
164      * {@link CentralDogmaRepository#watcher(Query)}.
165      */
166     public WatcherRequest<T> renderTemplate(boolean renderTemplate) {
167         this.renderTemplate = renderTemplate;
168         variableFile = null;
169         return this;
170     }
171 
172     /**
173      * Applies template processing to the file using the specified variable file in the same repository.
174      * The variable file must be a JSON, JSON5 or YAML file and have an object at the top level (arrays or
175      * string are not allowed).
176      *
177      * <p>This option is only valid when watching a single file via
178      * {@link CentralDogmaRepository#watcher(Query)}.
179      */
180     public WatcherRequest<T> renderTemplate(String variableFile) {
181         validateStructuredFilePath(variableFile, "variableFile");
182         renderTemplate = true;
183         this.variableFile = variableFile;
184         return this;
185     }
186 
187     @Override
188     public WatcherRequest<T> timeout(Duration timeout) {
189         //noinspection unchecked
190         return (WatcherRequest<T>) super.timeout(timeout);
191     }
192 
193     @Override
194     public WatcherRequest<T> timeoutMillis(long timeoutMillis) {
195         //noinspection unchecked
196         return (WatcherRequest<T>) super.timeoutMillis(timeoutMillis);
197     }
198 
199     @Override
200     public WatcherRequest<T> errorOnEntryNotFound(boolean errorOnEntryNotFound) {
201         //noinspection unchecked
202         return (WatcherRequest<T>) super.errorOnEntryNotFound(errorOnEntryNotFound);
203     }
204 
205     /**
206      * Creates a new {@link Watcher} and starts to watch the target. The {@link Watcher} must be closed via
207      * {@link Watcher#close()} after use.
208      */
209     public Watcher<T> start() {
210         final String proName = centralDogmaRepo.projectName();
211         final String repoName = centralDogmaRepo.repositoryName();
212         final AbstractWatcher<T> watcher;
213         if (query != null) {
214             watcher = new FileWatcher<>(
215                     centralDogmaRepo.centralDogma(), blockingTaskExecutor, proName, repoName, query,
216                     timeoutMillis(), errorOnEntryNotFound(), mapper, executor, delayOnSuccessMillis,
217                     initialDelayMillis, maxDelayMillis, multiplier, jitterRate, meterRegistry,
218                     renderTemplate, variableFile);
219         } else {
220             checkState(!renderTemplate, "renderTemplate can't be set when watching multiple files.");
221             assert pathPattern != null;
222             watcher = new FilesWatcher<>(
223                     centralDogmaRepo.centralDogma(), blockingTaskExecutor, proName, repoName, pathPattern,
224                     timeoutMillis(), errorOnEntryNotFound(), mapper, executor, delayOnSuccessMillis,
225                     initialDelayMillis, maxDelayMillis, multiplier, jitterRate, meterRegistry);
226         }
227         watcher.start();
228         return watcher;
229     }

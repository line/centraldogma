199     @Override
200     public CompletableFuture<PushResult> importDir(@Nullable String projectName,
201                                                    @Nullable String repositoryName, Path dir,
202                                                    boolean createIfMissing) {
203         requireNonNull(dir, "dir");
204 
205         if (!dir.isAbsolute()) {
206             throw new IllegalArgumentException("dir must be an absolute path: " + dir);
207         }
208         if (!Files.exists(dir) || !Files.isDirectory(dir)) {
209             throw new IllegalArgumentException("dir must be an existing directory: " + dir);
210         }
211 
212         final List<String> resolvedNames = resolveProjectAndRepoNames(projectName, repositoryName, dir);
213         final String finalProjectName = resolvedNames.get(0);
214         final String finalRepositoryName = resolvedNames.get(1);
215 
216         final CompletableFuture<Void> ensureProjectRepo = ensureProjectAndRepo(finalProjectName,
217                                                                                finalRepositoryName,
218                                                                                createIfMissing);
219         return ensureProjectRepo.thenCompose(unused -> {
220             final CentralDogmaRepository repo = forRepo(finalProjectName, finalRepositoryName);
221             return repo.importDir(dir);
222         });
223     }

225     /**
226      * Normalizes the specified {@link Revision} only if it is a relative revision.
227      *
228      * @return the absolute {@link Revision}
229      */
230     protected final CompletableFuture<Revision> maybeNormalizeRevision(
231             String projectName, String repositoryName, Revision revision) {
232 
233         if (revision.isRelative()) {
234             return normalizeRevision(projectName, repositoryName, revision);
235         } else {
236             return CompletableFuture.completedFuture(revision);
237         }
238     }

240     private CompletableFuture<Void> ensureProjectAndRepo(String project, String repo, boolean createIfMissing) {
241         return listProjects().thenCompose(projects -> {
242             if (!projects.contains(project)) {
243                 if (!createIfMissing) {
244                     return exceptionallyCompletedFuture(
245                             new IllegalStateException("Project does not exist: " + project));
246                 }
247                 return createProject(project);
248             }
249             return CompletableFuture.completedFuture(null);
250         }).thenCompose(unused -> listRepositories(project).thenCompose(repos -> {
251             if (!repos.containsKey(repo)) {
252                 if (!createIfMissing) {
253                     return exceptionallyCompletedFuture(
254                             new IllegalStateException("Repository does not exist: " + project + '/' + repo));
255                 }
256                 return createRepository(project, repo).thenApply(unusedRepo -> null);
257             }
258             return CompletableFuture.completedFuture(null);
259         }));
260     }

262     /**
263      * Resolves project and repository names based on the provided parameters and directory path.
264      *
265      * @param projectName the provided project name, or {@code null}
266      * @param repositoryName the provided repository name, or {@code null}
267      * @param dir the directory path to extract names from
268      * @return a {@link List} where get(0) is the project name and get(1) is the repository name
269      * @throws IllegalArgumentException if projectName is null but repositoryName is not null
270      */
271     private List<String> resolveProjectAndRepoNames(@Nullable String projectName,
272                                                     @Nullable String repositoryName,
273                                                     Path dir) {
274         if (projectName == null) {
275             if (repositoryName != null) {
276                 throw new IllegalArgumentException(
277                         "projectName cannot be null when repositoryName is not null");
278             }
279             return extractProjectAndRepoFromPath(dir);
280         }
281 
282         final String repositoryName0 = repositoryName != null ? repositoryName
283                                                               : extractRepoNameFromPath(dir);
284         return Arrays.asList(projectName, repositoryName0);
285     }
286 
287     /**
288      * Extracts both project and repository names from the directory path.
289      * Uses the second-to-last component as project name and the last component as repository name.
290      */
291     private static List<String> extractProjectAndRepoFromPath(Path dir) {
292         if (dir.getNameCount() < 2) {
293             throw new IllegalArgumentException(
294                     "Directory path must have at least 2 components : " + dir);
295         }
296         final String project = dir.getName(dir.getNameCount() - 2).toString();
297         final String repo = dir.getName(dir.getNameCount() - 1).toString();
298         return Arrays.asList(project, repo);
299     }
300 
301     /**
302      * Extracts the repository name from the directory path.
303      * Uses the last component as repository name.
304      */
305     private static String extractRepoNameFromPath(Path dir) {
306         if (dir.getNameCount() < 1) {
307             throw new IllegalArgumentException(
308                     "Directory path must have at least 1 component when repositoryName is null: " + dir);
309         }
310         return dir.getName(dir.getNameCount() - 1).toString();
311     }

217     public static boolean isValidEmailAddress(String emailAddr) {
218         requireNonNull(emailAddr, "emailAddr");
219         if (EMAIL_PATTERN.matcher(emailAddr).matches()) {
220             return true;
221         }
222         // Try to check whether the domain part is IP address format.
223         final Matcher m = GENERAL_EMAIL_PATTERN.matcher(emailAddr);
224         if (m.matches()) {
225             final String domainPart = m.group(1);
226             return isValidIpV4Address(domainPart) ||
227                    isValidIpV6Address(domainPart);
228         }
229         return false;
230     }

245     public static List<String> stringToLines(String str) {
246         final BufferedReader reader = new BufferedReader(new StringReader(str));
247         final List<String> lines = new ArrayList<>(128);
248         try {
249             String line;
250             while ((line = reader.readLine()) != null) {
251                 lines.add(line);
252             }
253         } catch (IOException ignored) {
254             // Should never happen.
255         }
256         return lines;
257     }

277     /**
278      * Returns the simplified and (optionally) decapitalized name of the specified type.
279      */
280     public static String simpleTypeName(Class<?> clazz, boolean decapitalize) {
281         if (clazz == null) {
282             return "null";
283         }
284 
285         String className = clazz.getName();
286         final int lastDotIdx = className.lastIndexOf('.');
287         if (lastDotIdx >= 0) {
288             className = className.substring(lastDotIdx + 1);
289         }
290 
291         if (!decapitalize) {
292             return className;
293         }
294 
295         final StringBuilder buf = new StringBuilder(className.length());
296         boolean lowercase = true;
297         for (int i = 0; i < className.length(); i++) {
298             final char c1 = className.charAt(i);
299             final char c2;
300             if (lowercase) {
301                 c2 = Character.toLowerCase(c1);
302                 if (c1 == c2) {
303                     lowercase = false;
304                 }
305             } else {
306                 c2 = c1;
307             }
308             buf.append(c2);
309         }
310 
311         return buf.toString();
312     }

323      * Makes sure the specified {@code values} and all its elements are not {@code null}.
324      */
325     public static <T> Iterable<T> requireNonNullElements(Iterable<T> values, String name) {
326         requireNonNull(values, name);
327 
328         int i = 0;
329         for (T v : values) {
330             if (v == null) {
331                 throw new NullPointerException(name + '[' + i + ']');
332             }
333             i++;
334         }
335 
336         return values;
337     }

343     @SuppressWarnings("DuplicateBooleanBranch")
344     private static boolean isValidIpV4Address(String ip, int from, int toExcluded) {
345         final int len = toExcluded - from;
346         int i;
347         return len <= 15 && len >= 7 &&
348                (i = ip.indexOf('.', from + 1)) > 0 && isValidIpV4Word(ip, from, i) &&
349                (i = ip.indexOf('.', from = i + 2)) > 0 && isValidIpV4Word(ip, from - 1, i) &&
350                (i = ip.indexOf('.', from = i + 2)) > 0 && isValidIpV4Word(ip, from - 1, i) &&
351                isValidIpV4Word(ip, i + 1, toExcluded);
352     }
353 
354     private static boolean isValidIpV4Word(CharSequence word, int from, int toExclusive) {
355         final int len = toExclusive - from;
356         final char c0;
357         final char c1;
358         final char c2;
359         if (len < 1 || len > 3 || (c0 = word.charAt(from)) < '0') {
360             return false;
361         }
362         if (len == 3) {
363             return (c1 = word.charAt(from + 1)) >= '0' &&
364                    (c2 = word.charAt(from + 2)) >= '0' &&
365                    (c0 <= '1' && c1 <= '9' && c2 <= '9' ||
366                     c0 == '2' && c1 <= '5' && (c2 <= '5' || c1 < '5' && c2 <= '9'));
367         }
368         return c0 <= '9' && (len == 1 || isValidNumericChar(word.charAt(from + 1)));
369     }

371     private static boolean isValidIpV6Address(String ip) {
372         int end = ip.length();
373         if (end < 2) {
374             return false;
375         }
376 
377         // strip "[]"
378         int start;
379         char c = ip.charAt(0);
380         if (c == '[') {
381             end--;
382             if (ip.charAt(end) != ']') {
383                 // must have a close ]
384                 return false;
385             }
386             start = 1;
387             c = ip.charAt(1);
388         } else {
389             start = 0;
390         }
391 
392         int colons;
393         int compressBegin;
394         if (c == ':') {
395             // an IPv6 address can start with "::" or with a number
396             if (ip.charAt(start + 1) != ':') {
397                 return false;
398             }
399             colons = 2;
400             compressBegin = start;
401             start += 2;
402         } else {
403             colons = 0;
404             compressBegin = -1;
405         }
406 
407         int wordLen = 0;
408         loop:
409         for (int i = start; i < end; i++) {
410             c = ip.charAt(i);
411             if (isValidHexChar(c)) {
412                 if (wordLen < 4) {
413                     wordLen++;
414                     continue;
415                 }
416                 return false;
417             }
418 
419             switch (c) {
420                 case ':':
421                     if (colons > 7) {
422                         return false;
423                     }
424                     if (ip.charAt(i - 1) == ':') {
425                         if (compressBegin >= 0) {
426                             return false;
427                         }
428                         compressBegin = i - 1;
429                     } else {
430                         wordLen = 0;
431                     }
432                     colons++;
433                     break;
434                 case '.':
435                     // case for the last 32-bits represented as IPv4 x:x:x:x:x:x:d.d.d.d
436 
437                     // check a normal case (6 single colons)
438                     if (compressBegin < 0 && colons != 6 ||
439                         // a special case ::1:2:3:4:5:d.d.d.d allows 7 colons with an
440                         // IPv4 ending, otherwise 7 :'s is bad
441                         colons == 7 && compressBegin >= start || colons > 7) {
442                         return false;
443                     }
444 
445                     // Verify this address is of the correct structure to contain an IPv4 address.
446                     // It must be IPv4-Mapped or IPv4-Compatible
447                     // (see  https://datatracker.ietf.org/doc/html/rfc4291#section-2.5.5).
448                     final int ipv4Start = i - wordLen;
449                     int j = ipv4Start - 2; // index of character before the previous ':'.
450                     if (isValidIPv4MappedChar(ip.charAt(j))) {
451                         if (!isValidIPv4MappedChar(ip.charAt(j - 1)) ||
452                             !isValidIPv4MappedChar(ip.charAt(j - 2)) ||
453                             !isValidIPv4MappedChar(ip.charAt(j - 3))) {
454                             return false;
455                         }
456                         j -= 5;
457                     }
458 
459                     for (; j >= start; --j) {
460                         final char tmpChar = ip.charAt(j);
461                         if (tmpChar != '0' && tmpChar != ':') {
462                             return false;
463                         }
464                     }
465 
466                     // 7 - is minimum IPv4 address length
467                     int ipv4End = ip.indexOf('%', ipv4Start + 7);
468                     if (ipv4End < 0) {
469                         ipv4End = end;
470                     }
471                     return isValidIpV4Address(ip, ipv4Start, ipv4End);
472                 case '%':
473                     // strip the interface name/index after the percent sign
474                     end = i;
475                     break loop;
476                 default:
477                     return false;
478             }
479         }
480 
481         // normal case without compression
482         if (compressBegin < 0) {
483             return colons == 7 && wordLen > 0;
484         }
485 
486         return compressBegin + 2 == end ||
487                // 8 colons is valid only if compression in start or end
488                wordLen > 0 && (colons < 8 || compressBegin <= start);
489     }

491     private static boolean isValidNumericChar(char c) {
492         return c >= '0' && c <= '9';
493     }

495     private static boolean isValidIPv4MappedChar(char c) {
496         return c == 'f' || c == 'F';
497     }

499     private static boolean isValidHexChar(char c) {
500         return c >= '0' && c <= '9' || c >= 'A' && c <= 'F' || c >= 'a' && c <= 'f';

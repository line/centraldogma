17  package com.linecorp.centraldogma.internal;
18  
19  import java.io.IOError;
20  import java.io.IOException;
21  import java.nio.charset.StandardCharsets;
22  import java.time.Instant;
23  
24  import com.fasterxml.jackson.core.JsonParseException;
25  import com.fasterxml.jackson.core.json.JsonReadFeature;
26  import com.fasterxml.jackson.databind.JsonNode;
27  import com.fasterxml.jackson.databind.ObjectMapper;
28  import com.fasterxml.jackson.databind.SerializationFeature;
29  import com.fasterxml.jackson.databind.module.SimpleModule;
30  import com.fasterxml.jackson.datatype.jsr310.deser.InstantDeserializer;
31  import com.fasterxml.jackson.datatype.jsr310.ser.InstantSerializer;
32  import com.google.common.base.Ascii;
33  
34  /**
35   * JSON5 parser that enables <a href="https://json5.org/">JSON5</a> features that Jackson supports.
36   * Example of supported JSON5 features:
37   * <pre>{@code
38   * {
39   *   // comments
40   *   unquoted: 'and you can quote me on that',
41   *   singleQuotes: 'I can use "double quotes" here',
42   *   lineBreaks: "Look, Mom! \
43   * No \\n's!",
44   *   leadingDecimalPoint: .8675309,
45   *   positiveSign: +1,
46   *   trailingComma: 'in objects', andIn: ['arrays',],
47   *   "backwardsCompatible": "with JSON",
48   * }
49   * }</pre>
50   *
51   * <p>The unsupported features are listed below:
52   * <ul>
53   *   <li>Hexadecimal integers are not supported. (e.g., 0xdecaf)</li>
54   *   <li>Trailing decimal points are not supported. (e.g., 8675309.)</li>
55   * </ul>
56   */
57  public final class Json5 {
58      private static final ObjectMapper mapper = new ObjectMapper();
59  
60      static {
61          // Sort the attributes when serialized via the mapper.
62          mapper.enable(SerializationFeature.ORDER_MAP_ENTRIES_BY_KEYS);
63  
64          mapper.enable(JsonReadFeature.ALLOW_SINGLE_QUOTES.mappedFeature(),
65                        JsonReadFeature.ALLOW_UNQUOTED_FIELD_NAMES.mappedFeature(),
66                        JsonReadFeature.ALLOW_JAVA_COMMENTS.mappedFeature(),
67                        JsonReadFeature.ALLOW_TRAILING_COMMA.mappedFeature(),
68                        JsonReadFeature.ALLOW_BACKSLASH_ESCAPING_ANY_CHARACTER.mappedFeature(),
69                        JsonReadFeature.ALLOW_NON_NUMERIC_NUMBERS.mappedFeature(),
70                        JsonReadFeature.ALLOW_LEADING_PLUS_SIGN_FOR_NUMBERS.mappedFeature(),
71                        JsonReadFeature.ALLOW_LEADING_DECIMAL_POINT_FOR_NUMBERS.mappedFeature());
72  
73          mapper.registerModules(new SimpleModule().addSerializer(Instant.class, InstantSerializer.INSTANCE)
74                                                   .addDeserializer(Instant.class, InstantDeserializer.INSTANT));
75      }
76  
77      public static boolean isJsonCompatible(String path) {
78          return isJson(path) || isJson5(path);
79      }
80  
81      public static boolean isJson5(String path) {
82          return Ascii.toLowerCase(path).endsWith(".json5");
83      }
84  
85      public static boolean isJson(String path) {
86          return Ascii.toLowerCase(path).endsWith(".json");
87      }
88  
89      public static JsonNode readTree(String data) throws JsonParseException {
90          try {
91              return mapper.readTree(data);
92          } catch (JsonParseException e) {
93              throw e;
94          } catch (IOException e) {
95              throw new IOError(e);
96          }
97      }
98  
99      public static JsonNode readTree(byte[] data) throws JsonParseException {
100         return readTree(new String(data, StandardCharsets.UTF_8));
101     }
102 
103     private Json5() {}
104 }

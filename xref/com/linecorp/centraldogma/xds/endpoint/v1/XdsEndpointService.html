61      private static final Pattern ENDPONT_NAME_PATTERN =
62              Pattern.compile("^groups/([^/]+)/endpoints/(" + RESOURCE_ID_PATTERN_STRING + ")$");

70          this.xdsResourceManager = xdsResourceManager;
71      }

73      @Override
74      public void createEndpoint(CreateEndpointRequest request,
75                                 StreamObserver<ClusterLoadAssignment> responseObserver) {
76          final String parent = request.getParent();
77          final String group = removePrefix("groups/", parent);
78          xdsResourceManager.checkGroup(group);
79  
80          final String endpointId = request.getEndpointId();
81          if (!RESOURCE_ID_PATTERN.matcher(endpointId).matches()) {
82              throw Status.INVALID_ARGUMENT.withDescription("Invalid endpoint_id: " + endpointId +
83                                                            " (expected: " + RESOURCE_ID_PATTERN + ')')
84                                           .asRuntimeException();
85          }
86  
87          final String clusterName = clusterName(parent, endpointId);
88          // Ignore the specified name in the endpoint and set the name
89          // with the format of "groups/{group}/clusters/{endpoint}".
90          // https://github.com/aip-dev/google.aip.dev/blob/master/aip/general/0133.md#user-specified-ids
91          final ClusterLoadAssignment endpoint = request.getEndpoint()
92                                                        .toBuilder()
93                                                        .setClusterName(clusterName)
94                                                        .build();
95          xdsResourceManager.push(responseObserver, group, clusterName, fileName(endpointId),
96                                  "Create endpoint: " + clusterName, endpoint, currentAuthor(), true);
97      }
98  
99      private static String clusterName(String parent, String endpointId) {
100         // Use /clusters/ instead of /endpoints/ for the cluster name.
101         // /endpoints/ will be used for the file name.
102         return parent + CLUSTERS_DIRECTORY + endpointId;
103     }
104 
105     @Override
106     public void updateEndpoint(UpdateEndpointRequest request,
107                                StreamObserver<ClusterLoadAssignment> responseObserver) {
108         final String endpointName = request.getEndpointName();
109         final Matcher matcher = checkEndpointName(endpointName);
110         final String group = matcher.group(1);
111         xdsResourceManager.checkGroup(group);
112 
113         final ClusterLoadAssignment endpoint = request.getEndpoint();
114         final String endpointId = matcher.group(2);
115         xdsResourceManager.update(responseObserver, group, endpointName,
116                                   fileName(endpointId), "Update endpoint: " + endpointName,
117                                   endpoint.toBuilder()
118                                           .setClusterName(clusterName("groups/" + group, endpointId))
119                                           .build(), currentAuthor());
120     }
121 
122     @Override
123     public void deleteEndpoint(DeleteEndpointRequest request, StreamObserver<Empty> responseObserver) {
124         final String endpointName = request.getName();
125         final Matcher matcher = checkEndpointName(endpointName);
126         final String group = matcher.group(1);
127         xdsResourceManager.checkGroup(group);
128         xdsResourceManager.delete(responseObserver, group, endpointName, fileName(matcher.group(2)),
129                                   "Delete endpoint: " + endpointName, currentAuthor());
130     }
131 
132     private static Matcher checkEndpointName(String endpointName) {
133         final Matcher matcher = ENDPONT_NAME_PATTERN.matcher(endpointName);
134         if (!matcher.matches()) {
135             throw Status.INVALID_ARGUMENT.withDescription("Invalid endpoint name: " + endpointName +
136                                                           " (expected: " + ENDPONT_NAME_PATTERN + ')')
137                                          .asRuntimeException();
138         }
139         return matcher;
140     }
141 
142     private static String fileName(String endpointId) {
143         return ENDPOINTS_DIRECTORY + endpointId + ".json";
144     }

146     @Override
147     public void registerLocalityLbEndpoint(RegisterLocalityLbEndpointRequest request,
148                                            StreamObserver<LocalityLbEndpoint> responseObserver) {
149         final LocalityLbEndpoint localityLbEndpoint = request.getLocalityLbEndpoint();
150         final String endpointName = request.getEndpointName();
151         handleRegisterOrDeregister(
152                 endpointName,
153                 localityLbEndpoint,
154                 localityLbEndpoint,
155                 responseObserver,
156                 "Register locality LB endpoint to " + endpointName + ". endpoint: " + localityLbEndpoint,
157                 true,
158                 cause -> {
159                     if (cause instanceof EntryNotFoundException) {
160                         responseObserver.onError(
161                                 Status.NOT_FOUND.withDescription(
162                                               "Endpoint does not exist: " + endpointName)
163                                                 .asRuntimeException());
164                     } else {
165                         responseObserver.onError(
166                                 Status.INTERNAL.withCause(cause).asRuntimeException());
167                     }
168                 }
169         );
170     }
171 
172     @Override
173     public void deregisterLocalityLbEndpoint(DeregisterLocalityLbEndpointRequest request,
174                                              StreamObserver<Empty> responseObserver) {
175         final String endpointName = request.getEndpointName();
176         final LocalityLbEndpoint localityLbEndpoint = request.getLocalityLbEndpoint();
177         handleRegisterOrDeregister(
178                 endpointName,
179                 localityLbEndpoint,
180                 Empty.getDefaultInstance(),
181                 responseObserver,
182                 "Deregister locality LB endpoint from " + endpointName + ". endpoint: " + localityLbEndpoint,
183                 false,
184                 cause -> {
185                     if (cause instanceof LocalityLbEndpointNotFoundException) {
186                         responseObserver.onError(
187                                 Status.NOT_FOUND.withDescription(
188                                               "Locality LB endpoint does not exist: " +
189                                               localityLbEndpoint)
190                                                 .asRuntimeException());
191                     } else if (cause instanceof EntryNotFoundException) {
192                         responseObserver.onError(
193                                 Status.NOT_FOUND.withDescription(
194                                               "Endpoint does not exist: " + endpointName)
195                                                 .asRuntimeException());
196                     } else {
197                         responseObserver.onError(
198                                 Status.INTERNAL.withCause(cause).asRuntimeException());
199                     }
200                 }
201         );
202     }

204     private <T> void handleRegisterOrDeregister(String endpointName,
205                                                 LocalityLbEndpoint localityLbEndpoint,
206                                                 T successResponse,
207                                                 StreamObserver<T> responseObserver,
208                                                 String commitMessage,
209                                                 boolean register,
210                                                 Consumer<Throwable> errorConsumer) {
211         final Matcher matcher = checkEndpointName(endpointName);
212         final String group = matcher.group(1);
213         xdsResourceManager.checkGroup(group);
214 
215         final String endpointId = matcher.group(2);
216         final String fileName = fileName(endpointId);
217         final Author author = currentAuthor();
218 
219         final ContentTransformer<JsonNode> transformer = new ContentTransformer<>(
220                 fileName, EntryType.JSON, new RegisterOrDeregisterTransformer(localityLbEndpoint, register));
221 
222         xdsResourceManager.updateOrDelete(
223                 responseObserver, group, endpointName, fileName, () -> xdsResourceManager
224                         .commandExecutor()
225                         .execute(Command.transform(
226                                 null, author, XDS_CENTRAL_DOGMA_PROJECT, group, Revision.HEAD,
227                                 commitMessage, "", Markup.PLAINTEXT, transformer))
228                         .handle((result, cause) -> {
229                             if (cause != null) {
230                                 final Throwable peeled = Exceptions.peel(cause);
231                                 errorConsumer.accept(peeled);
232                                 return null;
233                             }
234                             responseObserver.onNext(successResponse);
235                             responseObserver.onCompleted();
236                             return null;
237                         }));
238     }

240     private static class RegisterOrDeregisterTransformer implements BiFunction<Revision, JsonNode, JsonNode> {
241 
242         final LocalityLbEndpoint localityLbEndpoint;
243         private final boolean register;
244 
245         RegisterOrDeregisterTransformer(LocalityLbEndpoint localityLbEndpoint, boolean register) {
246             this.localityLbEndpoint = localityLbEndpoint;
247             this.register = register;
248         }
249 
250         @Override
251         public JsonNode apply(Revision revision, JsonNode oldJsonNode) {
252             if (oldJsonNode.isNull()) {
253                 throw new EntryNotFoundException();
254             }
255             final ClusterLoadAssignment.Builder clusterLoadAssignmentBuilder =
256                     toClusterLoadAssignmentBuilder(oldJsonNode);
257             final int localityIndex =
258                     findLocalityAndPriorityIndex(clusterLoadAssignmentBuilder, localityLbEndpoint);
259             if (localityIndex < 0) {
260                 if (register) {
261                     clusterLoadAssignmentBuilder.addEndpoints(
262                             LocalityLbEndpoints.newBuilder()
263                                                .setLocality(localityLbEndpoint.getLocality())
264                                                .setPriority(localityLbEndpoint.getPriority())
265                                                .addLbEndpoints(localityLbEndpoint.getLbEndpoint())
266                                                .build());
267                     return toJsonNode(clusterLoadAssignmentBuilder);
268                 }
269                 throw LocalityLbEndpointNotFoundException.INSTANCE;
270             }
271             final LocalityLbEndpoints targetLocalityLbEndpoints =
272                     clusterLoadAssignmentBuilder.getEndpoints(localityIndex);
273             // Remove from the endpoints. It will be added again.
274             clusterLoadAssignmentBuilder.removeEndpoints(localityIndex);
275 
276             final int lbEndpointIndex =
277                     findLbEndpointIndex(targetLocalityLbEndpoints, localityLbEndpoint);
278             final LocalityLbEndpoints.Builder targetLocalityLbEndpointsBuilder =
279                     targetLocalityLbEndpoints.toBuilder();
280             if (register) {
281                 if (lbEndpointIndex >= 0) {
282                     // When the endpoint whose address is the same as the registering endpoint exists,
283                     // remove it and add the registering endpoint because the other fields may be different.
284                     targetLocalityLbEndpointsBuilder.removeLbEndpoints(lbEndpointIndex);
285                 }
286                 targetLocalityLbEndpointsBuilder.addLbEndpoints(localityLbEndpoint.getLbEndpoint());
287                 clusterLoadAssignmentBuilder.addEndpoints(targetLocalityLbEndpointsBuilder.build());
288             } else {
289                 if (lbEndpointIndex < 0) {
290                     throw LocalityLbEndpointNotFoundException.INSTANCE;
291                 }
292                 targetLocalityLbEndpointsBuilder.removeLbEndpoints(lbEndpointIndex);
293                 if (targetLocalityLbEndpointsBuilder.getLbEndpointsCount() > 0) {
294                     clusterLoadAssignmentBuilder.addEndpoints(targetLocalityLbEndpointsBuilder.build());
295                 }
296             }
297             return toJsonNode(clusterLoadAssignmentBuilder);
298         }
299 
300         private static int findLbEndpointIndex(LocalityLbEndpoints targetLocalityLbEndpoints,
301                                                LocalityLbEndpoint localityLbEndpoint) {
302             int sameLbEndpointIndex = -1;
303             final List<LbEndpoint> lbEndpointsList = targetLocalityLbEndpoints.getLbEndpointsList();
304             for (int i = 0; i < lbEndpointsList.size(); i++) {
305                 final LbEndpoint lbEndpoint = lbEndpointsList.get(i);
306                 if (lbEndpoint.getEndpoint().getAddress().equals(
307                         localityLbEndpoint.getLbEndpoint().getEndpoint().getAddress())) {
308                     sameLbEndpointIndex = i;
309                     break;
310                 }
311             }
312             return sameLbEndpointIndex;
313         }
314 
315         /**
316          * Find the index of the {@link LocalityLbEndpoints} that has the same locality and priority as the
317          * specified {@link LocalityLbEndpoint}.
318          * If the locality and priority are not found, return -1.
319          */
320         private static int findLocalityAndPriorityIndex(Builder clusterLoadAssignmentBuilder,
321                                                         LocalityLbEndpoint localityLbEndpoint) {
322             int sameLocalityIndex = -1;
323 
324             final List<LocalityLbEndpoints> localityLbEndpointsList =
325                     clusterLoadAssignmentBuilder.getEndpointsList();
326             for (int i = 0; i < localityLbEndpointsList.size(); i++) {
327                 final LocalityLbEndpoints localityLbEndpoints = localityLbEndpointsList.get(i);
328                 if (localityLbEndpoints.getLocality().equals(localityLbEndpoint.getLocality()) &&
329                     localityLbEndpoints.getPriority() == localityLbEndpoint.getPriority()) {
330                     sameLocalityIndex = i;
331                     break;
332                 }
333             }
334             return sameLocalityIndex;
335         }
336 
337         private static ClusterLoadAssignment.Builder toClusterLoadAssignmentBuilder(JsonNode oldJsonNode) {
338             final Builder clusterLoadAssignmentBuilder =
339                     ClusterLoadAssignment.newBuilder();
340             try {
341                 JSON_MESSAGE_MARSHALLER.mergeValue(Jackson.writeValueAsString(oldJsonNode),
342                                                    clusterLoadAssignmentBuilder);
343             } catch (Throwable t) {
344                 // Should never reach here.
345                 throw new Error();
346             }
347             return clusterLoadAssignmentBuilder;
348         }
349 
350         private static JsonNode toJsonNode(ClusterLoadAssignment.Builder clusterLoadAssignmentBuilder) {
351             try {
352                 return Jackson.readTree(JSON_MESSAGE_MARSHALLER.writeValueAsString(
353                         clusterLoadAssignmentBuilder.build()));
354             } catch (IOException e) {
355                 // Should never reach here
356                 throw new Error(e);
357             }
358         }
359     }

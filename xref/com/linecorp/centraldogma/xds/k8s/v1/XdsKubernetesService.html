65      static final String K8S_ENDPOINT_AGGREGATORS_DIRECTORY = "/k8s/endpointAggregators/";

83      @Blocking
84      @Override
85      public void createKubernetesEndpointAggregator(
86              CreateKubernetesEndpointAggregatorRequest request,
87              StreamObserver<KubernetesEndpointAggregator> responseObserver) {
88          final String parent = request.getParent();
89          final String group = removePrefix("groups/", parent);
90          xdsResourceManager.checkGroup(group);
91          final String aggregatorId = request.getAggregatorId();
92          if (!RESOURCE_ID_PATTERN.matcher(aggregatorId).matches()) {
93              throw Status.INVALID_ARGUMENT.withDescription("Invalid aggregator_id: " + aggregatorId +
94                                                            " (expected: " + RESOURCE_ID_PATTERN + ')')
95                                           .asRuntimeException();
96          }
97  
98          final String kubernetesEndpointName = parent + K8S_ENDPOINT_AGGREGATORS_DIRECTORY + aggregatorId;
99          final String clusterName = parent + "/k8s/clusters/" + aggregatorId;
100         final KubernetesEndpointAggregator aggregator = request.getKubernetesEndpointAggregator().toBuilder()
101                                                                .setName(kubernetesEndpointName)
102                                                                .setClusterName(clusterName)
103                                                                .build();
104         final List<KubernetesLocalityLbEndpoints> kubernetesLocalityLbEndpointsList =
105                 aggregator.getLocalityLbEndpointsList();
106         if (kubernetesLocalityLbEndpointsList.isEmpty()) {
107             throw Status.INVALID_ARGUMENT.withDescription("kubernetes locality lb endpoints are empty.")
108                                          .asRuntimeException();
109         }
110         final Author author = currentAuthor();
111         validateKubernetesEndpointAndPush(
112                 responseObserver, kubernetesLocalityLbEndpointsList, () -> xdsResourceManager.push(
113                         responseObserver, group, kubernetesEndpointName,
114                         K8S_ENDPOINT_AGGREGATORS_DIRECTORY + aggregatorId + ".json",
115                         "Create kubernetes endpoint: " + kubernetesEndpointName, aggregator, author, true));
116     }

118     private void validateKubernetesEndpointAndPush(
119             StreamObserver<KubernetesEndpointAggregator> responseObserver,
120             List<KubernetesLocalityLbEndpoints> kubernetesLocalityLbEndpointsList, Runnable onSuccess) {
121         // Create a KubernetesEndpointGroup to check if the watcher is valid.
122         // We use KubernetesEndpointGroup for simplicity, but we will implement a custom implementation
123         // for better debugging and error handling in the future.
124         final ContextAwareBlockingTaskExecutor taskExecutor =
125                 ServiceRequestContext.current().blockingTaskExecutor();
126 
127         final ArrayList<CompletableFuture<Void>> futures = new ArrayList<>();
128         for (KubernetesLocalityLbEndpoints kubernetesLocalityLbEndpoints : kubernetesLocalityLbEndpointsList) {
129             final CompletableFuture<Void> future = new CompletableFuture<>();
130             futures.add(future);
131 
132             final ServiceEndpointWatcher watcher = kubernetesLocalityLbEndpoints.getWatcher();
133             final CompletableFuture<KubernetesEndpointGroup> endpointGroupFuture =
134                     createKubernetesEndpointGroup(watcher, xdsResourceManager.xdsProject().metaRepo(),
135                                                   taskExecutor);
136             endpointGroupFuture.handle((kubernetesEndpointGroup, cause) -> {
137                 if (cause != null) {
138                     cause = Exceptions.peel(cause);
139                     if (cause instanceof IllegalArgumentException || cause instanceof EntryNotFoundException) {
140                         future.completeExceptionally(Status.INVALID_ARGUMENT.withCause(cause)
141                                                                             .withDescription(cause.getMessage())
142                                                                             .asRuntimeException());
143                     } else {
144                         future.completeExceptionally(Status.INTERNAL.withCause(cause).asRuntimeException());
145                     }
146                     return null;
147                 }
148                 final AtomicBoolean completed = new AtomicBoolean();
149                 final CompletableFuture<List<Endpoint>> whenReady = kubernetesEndpointGroup.whenReady();
150                 // Use a schedule to time out the watcher creation until we implement a custom implementation.
151                 final ScheduledFuture<?> scheduledFuture = taskExecutor.schedule(() -> {
152                     if (!completed.compareAndSet(false, true)) {
153                         return;
154                     }
155                     kubernetesEndpointGroup.closeAsync();
156                     future.completeExceptionally(
157                             Status.INTERNAL.withDescription(
158                                     "Failed to retrieve k8s endpoints within 5 seconds. watcher: " +
159                                     watcher).asRuntimeException());
160                 }, 5, TimeUnit.SECONDS);
161 
162                 whenReady.handle((endpoints, cause1) -> {
163                     if (!completed.compareAndSet(false, true)) {
164                         return null;
165                     }
166                     scheduledFuture.cancel(false);
167                     kubernetesEndpointGroup.closeAsync();
168                     if (cause1 != null) {
169                         // Specific types.
170                         responseObserver.onError(Status.INTERNAL.withCause(cause1).asRuntimeException());
171                         return null;
172                     }
173                     logger.debug("Successfully retrieved k8s endpoints: {}, watcher: {}", endpoints, watcher);
174                     future.complete(null);
175                     return null;
176                 });
177                 return null;
178             });
179         }
180 
181         final CompletableFuture<Void> allOfFuture =
182                 CompletableFuture.allOf(futures.toArray(EMPTY_FUTURES));
183         allOfFuture.handle((unused, cause) -> {
184             if (cause != null) {
185                 responseObserver.onError(cause);
186             } else {
187                 onSuccess.run();
188             }
189             return null;
190         });
191     }

193     /**
194      * Creates a {@link KubernetesEndpointGroup} from the specified {@link ServiceEndpointWatcher}.
195      * This method must be executed in a blocking thread because
196      * {@link KubernetesEndpointGroupBuilder#build()} blocks the execution thread.
197      */
198     public static CompletableFuture<KubernetesEndpointGroup> createKubernetesEndpointGroup(
199             ServiceEndpointWatcher watcher, MetaRepository metaRepository, Executor executor) {
200         final Kubeconfig kubeconfig = watcher.getKubeconfig();
201         final String serviceName = watcher.getServiceName();
202 
203         return toConfig(kubeconfig, metaRepository).thenApplyAsync(config -> {
204             final KubernetesEndpointGroupBuilder kubernetesEndpointGroupBuilder =
205                     KubernetesEndpointGroup.builder(config).serviceName(serviceName);
206             if (!isNullOrEmpty(kubeconfig.getNamespace())) {
207                 kubernetesEndpointGroupBuilder.namespace(kubeconfig.getNamespace());
208             }
209             if (!isNullOrEmpty(watcher.getPortName())) {
210                 kubernetesEndpointGroupBuilder.portName(watcher.getPortName());
211             }
212             // This callback can be executed by an event loop from CachingRepository, so we should use the
213             // specified executor to avoid blocking the event loop below.
214             return kubernetesEndpointGroupBuilder.build();
215         }, executor);
216     }
217 
218     private static CompletableFuture<Config> toConfig(Kubeconfig kubeconfig, MetaRepository metaRepository) {
219         final ConfigBuilder configBuilder = new ConfigBuilder()
220                 .withMasterUrl(kubeconfig.getControlPlaneUrl())
221                 .withTrustCerts(kubeconfig.getTrustCerts());
222 
223         final String credentialId = kubeconfig.getCredentialId();
224         if (isNullOrEmpty(credentialId)) {
225             return CompletableFuture.completedFuture(configBuilder.build());
226         }
227 
228         return metaRepository.credential(credentialId)
229                              .thenApply(credential -> {
230                                  if (!(credential instanceof AccessTokenCredential)) {
231                                      throw new IllegalArgumentException(
232                                              "credential must be an access token: " + credential);
233                                  }
234 
235                                  return configBuilder.withOauthToken(
236                                          ((AccessTokenCredential) credential).accessToken()).build();
237                              });
238     }

240     @Blocking
241     @Override
242     public void updateKubernetesEndpointAggregator(
243             UpdateKubernetesEndpointAggregatorRequest request,
244             StreamObserver<KubernetesEndpointAggregator> responseObserver) {
245         final KubernetesEndpointAggregator aggregator = request.getKubernetesEndpointAggregator();
246         final String aggregatorName = aggregator.getName();
247         final String group = checkAggregatorName(aggregatorName).group(1);
248         xdsResourceManager.checkGroup(group);
249         final List<KubernetesLocalityLbEndpoints> kubernetesLocalityLbEndpointsList =
250                 aggregator.getLocalityLbEndpointsList();
251         if (kubernetesLocalityLbEndpointsList.isEmpty()) {
252             throw Status.INVALID_ARGUMENT.withDescription("kubernetes locality lb endpoints are empty.")
253                                          .asRuntimeException();
254         }
255 
256         // Update the cluster name just in case it's mistakenly set by the user.
257         final KubernetesEndpointAggregator aggregator0 = aggregator.toBuilder().setClusterName(
258                 AGGREGATORS_REPLCACE_PATTERN.matcher(aggregatorName).replaceFirst("/clusters/")).build();
259         final Author author = currentAuthor();
260         validateKubernetesEndpointAndPush(
261                 responseObserver, kubernetesLocalityLbEndpointsList, () -> xdsResourceManager.update(
262                 responseObserver, group, aggregatorName,
263                 "Update kubernetes endpoint aggregator: " + aggregatorName, aggregator0, author));
264     }
265 
266     private static Matcher checkAggregatorName(String aggregatorName) {
267         final Matcher matcher = K8S_ENDPOINT_AGGREGATORS_NAME_PATTERN.matcher(aggregatorName);
268         if (!matcher.matches()) {
269             throw Status.INVALID_ARGUMENT.withDescription(
270                                 "Invalid kubernetes endpoint aggregator name: " + aggregatorName +
271                                 " (expected: " + K8S_ENDPOINT_AGGREGATORS_NAME_PATTERN + ')')
272                                          .asRuntimeException();
273         }
274         return matcher;
275     }

277     @Override
278     public void deleteKubernetesEndpointAggregator(DeleteKubernetesEndpointAggregatorRequest request,
279                                                    StreamObserver<Empty> responseObserver) {
280         final String aggregatorName = request.getName();
281         final String group = checkAggregatorName(aggregatorName).group(1);
282         xdsResourceManager.checkGroup(group);
283         xdsResourceManager.delete(responseObserver, group, aggregatorName,
284                                   "Delete kubernetes endpoint aggregator: " + aggregatorName,
285                                   currentAuthor());
286     }

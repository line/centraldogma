133     void start(PluginInitContext pluginInitContext) {
134         init();
135         // No scoped client has connected yet, so this only sets the DEFAULT_GROUP snapshot.
136         updateAllSnapshots(null);
137         final CommandExecutor commandExecutor = pluginInitContext.commandExecutor();
138         final XdsClientStatusTracker clientStatusTracker = new XdsClientStatusTracker();
139         final V3DiscoveryServer server = new V3DiscoveryServer(clientStatusTracker, cache);
140         final GrpcServiceBuilder grpcServiceBuilder =
141                 GrpcService.builder()
142                            .addService(server.getClusterDiscoveryServiceImpl())
143                            .addService(server.getEndpointDiscoveryServiceImpl())
144                            .addService(server.getListenerDiscoveryServiceImpl())
145                            .addService(server.getRouteDiscoveryServiceImpl())
146                            .addService(server.getAggregatedDiscoveryServiceImpl());
147         final GrpcService grpcService = grpcServiceBuilder.build();
148         // Clients without an app id are served the full snapshot under DEFAULT_GROUP,
149         // preserving backward compatibility.
150         final MetadataService mds = new MetadataService(pluginInitContext.projectManager(), commandExecutor,
151                                                         pluginInitContext.internalProjectInitializer());
152         final ServerBuilder serverBuilder = pluginInitContext.serverBuilder();
153         serverBuilder.service(grpcService, optionalAppIdentityAuth(mds));
154         final XdsResourceManager xdsResourceManager = new XdsResourceManager(xdsProject(), commandExecutor);
155         final XdsEndpointService xdsEndpointService =
156                 new XdsEndpointService(xdsResourceManager, controlPlaneExecutor);
157         this.xdsEndpointService = xdsEndpointService;
158         final GrpcService xdsApplicationService =
159                 GrpcService.builder()
160                            .addService(new XdsGroupService(xdsProject(), commandExecutor, mds))
161                            .addService(new XdsListenerService(xdsResourceManager))
162                            .addService(new XdsRouteService(xdsResourceManager))
163                            .addService(new XdsClusterService(xdsResourceManager))
164                            .addService(xdsEndpointService)
165                            .addService(new XdsKubernetesService(xdsResourceManager))
166                            .exceptionHandler(new ControlPlaneExceptionHandlerFunction())
167                            .jsonMarshallerFactory(
168                                    serviceDescriptor -> {
169                                        // Use JSON_MESSAGE_MARSHALLER not to parse Envoy extensions twice.
170                                        final MessageMarshaller.Builder builder =
171                                                JSON_MESSAGE_MARSHALLER.toBuilder();
172                                        for (MethodDescriptor<?, ?> method : ImmutableList.copyOf(
173                                                serviceDescriptor.getMethods())) {
174                                            final Message reqPrototype =
175                                                    marshallerPrototype(method.getRequestMarshaller());
176                                            final Message resPrototype =
177                                                    marshallerPrototype(method.getResponseMarshaller());
178                                            if (reqPrototype != null) {
179                                                builder.register(reqPrototype);
180                                            }
181                                            if (resPrototype != null) {
182                                                builder.register(resPrototype);
183                                            }
184                                        }
185                                        return new DefaultJsonMarshaller(builder.build());
186                                    })
187                            .enableHttpJsonTranscoding(true).build();
188         // The global /api/v1/** route decorator in CentralDogma covers the HTTP-transcoded routes, but
189         // the native gRPC paths (e.g. /centraldogma.xds.*.v1.*Service/Method) bypass it. The per-service
190         // decorator ensures both paths require authentication.
191         serverBuilder.service(xdsApplicationService, pluginInitContext.authService());
192 
193         // Endpoints (EDS) are not access-controlled: any authenticated user can read the endpoints of every
194         // group regardless of its READ access.
195         serverBuilder.annotatedService()
196                      .pathPrefix("/api/v1")
197                      .build(new XdsEndpointReadService(xdsProject()));
198 
199         // System-administrator-only views of the control plane runtime state (connected clients and the served
200         // snapshot).
201         serverBuilder.annotatedService()
202                      .pathPrefix("/api/v1")
203                      .build(new XdsControlPlaneStatusService(clientStatusTracker, this));
204 
205         registerWebEnabledFlag(serverBuilder);
206     }

217     /**
218      * Builds a JSON view of a served snapshot, computed on the {@link #controlPlaneExecutor} because
219      * {@link #centralDogmaXdsResources} must only be read from that thread. The scope is, in order of
220      * precedence:
221      * <ul>
222      *   <li>{@code appId} set: the exact snapshot served to that application identity, read from the cache key
223      *       it is scoped under (the union of the groups it can read). {@code served} is {@code false} when the
224      *       app id has never connected, so nothing is cached for it.</li>
225      *   <li>{@code group} set: the snapshot scoped to that single group.</li>
226      *   <li>neither set: the full snapshot published under {@link #DEFAULT_GROUP} (what anonymous and
227      *       system-admin clients are served).</li>
228      * </ul>
229      * Each resource type is rendered as {@code {version, resources: {name: <proto-json>}}}.
230      */
231     CompletableFuture<XdsSnapshotDto> snapshot(@Nullable String group, @Nullable String appId) {
232         return CompletableFuture.supplyAsync(() -> {
233             final String resolvedAppId;
234             final List<String> readableGroups;
235             final Boolean served;
236             final String resolvedGroup;
237             final CentralDogmaSnapshot snapshot;
238             if (appId != null) {
239                 resolvedAppId = appId;
240                 resolvedGroup = null;
241                 final String key = appCacheKey(appId);
242                 final ScopedClient scopedClient = scopedClients.get(key);
243                 final Set<String> groups = scopedClient != null ? scopedClient.readableGroups : null;
244                 readableGroups = groups != null ? ImmutableList.copyOf(groups) : null;
245                 // The exact snapshot served to this app identity, scoped to the groups it can read.
246                 snapshot = (CentralDogmaSnapshot) cache.getSnapshot(key);
247                 served = snapshot != null;
248             } else if (group != null) {
249                 resolvedAppId = null;
250                 readableGroups = null;
251                 served = null;
252                 resolvedGroup = group;
253                 // snapshot(Set) is a pure read of the current state.
254                 snapshot = centralDogmaXdsResources.snapshot(ImmutableSet.of(group));
255             } else {
256                 resolvedAppId = null;
257                 readableGroups = null;
258                 served = null;
259                 resolvedGroup = null;
260                 // Reuse the full snapshot already published to the cache. The no-arg snapshot() must not be
261                 // called here because it has side effects (it resets the per-type "updated" flags, which would
262                 // make the next updateAllSnapshots() skip rebuilding the changed resources).
263                 snapshot = (CentralDogmaSnapshot) cache.getSnapshot(DEFAULT_GROUP);
264             }
265             final XdsResourceTypeDto listeners;
266             final XdsResourceTypeDto routes;
267             final XdsResourceTypeDto clusters;
268             final XdsResourceTypeDto endpoints;
269             if (snapshot != null) {
270                 listeners = toResourceTypeDto(snapshot.listenersVersion(), snapshot.listeners().resources());
271                 routes = toResourceTypeDto(snapshot.routesVersion(), snapshot.routes().resources());
272                 clusters = toResourceTypeDto(snapshot.clustersVersion(), snapshot.clusters().resources());
273                 endpoints = toResourceTypeDto(snapshot.endpointsVersion(), snapshot.endpoints().resources());
274             } else {
275                 listeners = null;
276                 routes = null;
277                 clusters = null;
278                 endpoints = null;
279             }
280             return new XdsSnapshotDto(resolvedAppId, readableGroups, served, resolvedGroup,
281                                       listeners, routes, clusters, endpoints);
282         }, controlPlaneExecutor);
283     }

285     /**
286      * Lists the application identities that have connected to the discovery API on this instance, together with
287      * the groups each currently has READ access to. The exact served snapshot for each is available through
288      * {@link #snapshot(String, String)} with its {@code appId}.
289      */
290     List<XdsAppDto> apps() {
291         final ImmutableList.Builder<XdsAppDto> result = ImmutableList.builder();
292         scopedClients.forEach((key, scopedClient) -> {
293             final Set<String> groups = scopedClient.readableGroups;
294             result.add(new XdsAppDto(
295                     scopedClient.appIdentity.login(),
296                     groups != null ? ImmutableList.copyOf(groups) : ImmutableList.of()));
297         });
298         return result.build();
299     }

305     private static XdsResourceTypeDto toResourceTypeDto(String version,
306                                                          Map<String, ? extends Message> resources) {
307         final ImmutableMap.Builder<String, JsonNode> builder = ImmutableMap.builder();
308         resources.forEach((name, message) -> {
309             try {
310                 builder.put(name, Jackson.readTree(JSON_MESSAGE_MARSHALLER.writeValueAsString(message)));
311             } catch (IOException e) {
312                 // Should never reach here.
313                 throw new UncheckedIOException(e);
314             }
315         });
316         return new XdsResourceTypeDto(version, builder.build());
317     }

319     @Nullable
320     private static Message marshallerPrototype(Marshaller<?> marshaller) {
321         if (marshaller instanceof MethodDescriptor.PrototypeMarshaller) {
322             final Object prototype = ((PrototypeMarshaller<?>) marshaller).getMessagePrototype();
323             if (prototype instanceof Message) {
324                 return (Message) prototype;
325             }
326         }
327         return null;
328     }

340     @Override
341     protected void handleXdsResource(String path, JsonNode content, String groupName)
342             throws IOException {
343         if (path.startsWith(CLUSTERS_DIRECTORY)) {
344             final Cluster.Builder builder = Cluster.newBuilder();
345             JSON_MESSAGE_MARSHALLER.mergeValue(content.traverse(), builder);
346             centralDogmaXdsResources.setCluster(groupName, builder.build());
347         } else if (path.startsWith(ENDPOINTS_DIRECTORY) || path.startsWith(K8S_ENDPOINTS_DIRECTORY)) {
348             final ClusterLoadAssignment.Builder builder = ClusterLoadAssignment.newBuilder();
349             JSON_MESSAGE_MARSHALLER.mergeValue(content.traverse(), builder);
350             centralDogmaXdsResources.setEndpoint(groupName, builder.build());
351         } else if (path.startsWith(LISTENERS_DIRECTORY)) {
352             final Listener.Builder builder = Listener.newBuilder();
353             JSON_MESSAGE_MARSHALLER.mergeValue(content.traverse(), builder);
354             centralDogmaXdsResources.setListener(groupName, builder.build());
355         } else if (path.startsWith(ROUTES_DIRECTORY)) {
356             final RouteConfiguration.Builder builder = RouteConfiguration.newBuilder();
357             JSON_MESSAGE_MARSHALLER.mergeValue(content.traverse(), builder);
358             centralDogmaXdsResources.setRoute(groupName, builder.build());
359         } else {
360             // ignore
361         }
362     }

370     @Override
371     protected void onFileRemoved(String groupName, String path) {
372         if (path.startsWith(CLUSTERS_DIRECTORY)) {
373             centralDogmaXdsResources.removeCluster(groupName, path);
374         } else if (path.startsWith(ENDPOINTS_DIRECTORY) ||
375                    path.startsWith(K8S_ENDPOINTS_DIRECTORY)) {
376             centralDogmaXdsResources.removeEndpoint(groupName, path);
377         } else if (path.startsWith(LISTENERS_DIRECTORY)) {
378             centralDogmaXdsResources.removeListener(groupName, path);
379         } else if (path.startsWith(ROUTES_DIRECTORY)) {
380             centralDogmaXdsResources.removeRoute(groupName, path);
381         }
382     }

389     @Override
390     protected void onMetadataChanged() {
391         if (scopedClients.isEmpty()) {
392             return;
393         }
394         // The cached Project.metadata() can lag behind this listener, so read the latest metadata.
395         fetchXdsMetadata().handleAsync((metadata, cause) -> {
396             if (cause != null) {
397                 logger.warn("Failed to read the xDS project metadata; scoped snapshots were not refreshed.",
398                             cause);
399                 return null;
400             }
401             // Permissions may have changed, so recompute each client's readable groups and rebuild its
402             // snapshot.
403             scopedClients.forEach((key, scopedClient) -> refreshScopedClient(key, scopedClient, metadata));
404             return null;
405         }, controlPlaneExecutor);
406     }

408     private CompletableFuture<ProjectMetadata> fetchXdsMetadata() {
409         return xdsProject().repos().get(Project.REPO_DOGMA)
410                            .get(Revision.HEAD, Query.ofJson(MetadataService.METADATA_JSON))
411                            .thenApply(entry -> {
412                                try {
413                                    return Jackson.treeToValue(entry.content(), ProjectMetadata.class);
414                                } catch (JsonProcessingException e) {
415                                    throw new CompletionException(e);
416                                }
417                            });
418     }

420     /**
421      * Returns a decorator that authenticates a client with either its mTLS client certificate or an application
422      * access token, but does NOT reject clients that present neither.
423      */
424     private static Function<? super HttpService, ? extends HttpService> optionalAppIdentityAuth(
425             MetadataService mds) {
426         final Authorizer<HttpRequest> authorizer =
427                 new ApplicationCertificateAuthorizer(mds::findCertificateById)
428                         .orElse(new ApplicationTokenAuthorizer(mds::findTokenBySecret));
429         return delegate -> new SimpleDecoratingHttpService(delegate) {
430             @Override
431             public HttpResponse serve(ServiceRequestContext ctx, HttpRequest req) {
432                 // The authorizer sets the authenticated user on the context when the certificate or token is
433                 // valid. The boolean result is ignored on purpose.
434                 return HttpResponse.of(authorizer.authorize(ctx, req).handle((authorized, cause) -> {
435                     try {
436                         return unwrap().serve(ctx, req);
437                     } catch (Exception e) {
438                         return Exceptions.throwUnsafely(e);
439                     }
440                 }));
441             }
442         };
443     }

445     /**
446      * Rebuilds the snapshots after a change. {@code changedGroup} is the group whose resources changed, or
447      * {@code null} to refresh every scoped client (e.g. on start). Only the clients affected by the change are
448      * rebuilt: a scoped client is refreshed when it can read {@code changedGroup}, or unconditionally when the
449      * change touched endpoints (which are shared unfiltered across all clients) or {@code changedGroup} is
450      * {@code null}.
451      */
452     private void updateAllSnapshots(@Nullable String changedGroup) {
453         // Read before snapshot() resets the update flags.
454         final boolean endpointsChanged = centralDogmaXdsResources.isEndpointUpdated();
455         final CentralDogmaSnapshot snapshot = centralDogmaXdsResources.snapshot();
456         cache.setSnapshot(DEFAULT_GROUP, snapshot);
457         metrics.onSnapshotUpdate(snapshot);
458         final boolean refreshAll = endpointsChanged || changedGroup == null;
459         scopedClients.forEach((key, scopedClient) -> {
460             final Set<String> groups = scopedClient.readableGroups;
461             if (groups != null && (refreshAll || groups.contains(changedGroup))) {
462                 cache.setSnapshot(key, centralDogmaXdsResources.snapshot(groups));
463             }
464         });
465     }

467     private String cacheKey() {
468         final User user = AuthUtil.currentUserOrNull();
469         if (!(user instanceof UserWithAppIdentity) || user.isSystemAdmin()) {
470             return DEFAULT_GROUP;
471         }
472         final UserWithAppIdentity appIdentity = (UserWithAppIdentity) user;
473         // login is the app id for UserWithAppIdentity.
474         final String key = appCacheKey(appIdentity.login());
475         if (!scopedClients.containsKey(key)) {
476             // First time we see this app identity; compute its readable groups and build its scoped snapshot
477             // synchronously. The snapshot is set in the cache BEFORE the client is registered in scopedClients,
478             // so a concurrent request that resolves to this key never observes it without a snapshot. A racing
479             // first request may redundantly recompute it, which is harmless.
480             final ScopedClient scopedClient = new ScopedClient(appIdentity);
481             final ProjectMetadata metadata = xdsProject().metadata();
482             assert metadata != null;
483             refreshScopedClient(key, scopedClient, metadata);
484             scopedClients.putIfAbsent(key, scopedClient);
485         }
486         return key;
487     }
488 
489     private void refreshScopedClient(String key, ScopedClient scopedClient, ProjectMetadata metadata) {
490         final Set<String> groups = groupsWithReadAccess(metadata, scopedClient.appIdentity);
491         scopedClient.readableGroups = groups;
492         cache.setSnapshot(key, centralDogmaXdsResources.snapshot(groups));
493     }
494 
495     private static Set<String> groupsWithReadAccess(ProjectMetadata metadata, User user) {
496         final ImmutableSet.Builder<String> groups = ImmutableSet.builder();
497         for (String groupName : metadata.repos().keySet()) {
498             // Any repository role implies at least READ access.
499             if (MetadataService.findRepositoryRole(metadata, groupName, user) != null) {
500                 groups.add(groupName);
501             }
502         }
503         return groups.build();
504     }

506     @Override

511     void stop() {
512         stop = true;
513         metrics.onStopped();
514         final XdsEndpointService xdsEndpointService = this.xdsEndpointService;
515         if (xdsEndpointService != null) {
516             if (xdsEndpointService.batchUpdateTaskSize() > 0) {
517                 logger.info("Waiting for {} xDS endpoint batch update tasks to finish up to 5 seconds...",
518                             xdsEndpointService.batchUpdateTaskSize());
519                 for (int i = 0; i < 5; i++) {
520                     try {
521                         if (xdsEndpointService.batchUpdateTaskSize() == 0) {
522                             break;
523                         }
524                         Thread.sleep(1000);
525                     } catch (InterruptedException e) {
526                         Thread.currentThread().interrupt();
527                         break;
528                     }
529                 }
530             }
531         }
532 
533         final boolean interrupted = terminate(controlPlaneExecutor);
534         if (interrupted) {
535             Thread.currentThread().interrupt();
536         }
537     }

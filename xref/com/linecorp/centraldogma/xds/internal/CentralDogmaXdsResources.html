16  package com.linecorp.centraldogma.xds.internal;
17  
18  import java.util.HashMap;
19  import java.util.Map;
20  import java.util.regex.Pattern;
21  
22  import com.google.common.collect.ImmutableList;
23  
24  import io.envoyproxy.controlplane.cache.Resources.ResourceType;
25  import io.envoyproxy.controlplane.cache.SnapshotResources;
26  import io.envoyproxy.controlplane.cache.VersionedResource;
27  import io.envoyproxy.envoy.config.cluster.v3.Cluster;
28  import io.envoyproxy.envoy.config.endpoint.v3.ClusterLoadAssignment;
29  import io.envoyproxy.envoy.config.listener.v3.Listener;
30  import io.envoyproxy.envoy.config.route.v3.RouteConfiguration;
31  import io.envoyproxy.envoy.extensions.transport_sockets.tls.v3.Secret;
32  
33  final class CentralDogmaXdsResources {
34  
35      private static final Pattern ENDPOINTS_PATTERN = Pattern.compile("/endpoints/");
36  
37      private final Map<String, Map<String, VersionedResource<Cluster>>> clusterResources = new HashMap<>();
38      private final Map<String, Map<String, VersionedResource<ClusterLoadAssignment>>> endpointResources =
39              new HashMap<>();
40      private final Map<String, Map<String, VersionedResource<Listener>>> listenerResources = new HashMap<>();
41      private final Map<String, Map<String, VersionedResource<RouteConfiguration>>> routeResources =
42              new HashMap<>();
43      private boolean clusterUpdated;
44      private boolean endpointUpdated;
45      private boolean listenerUpdated;
46      private boolean routeUpdated;
47  
48      private CentralDogmaSnapshot currentSnapshot;
49  
50      CentralDogmaXdsResources() {
51          final SnapshotResources<?> emptyResources = SnapshotResources.create(ImmutableList.of(),
52                                                                               "empty_resources");
53          currentSnapshot = new CentralDogmaSnapshot((SnapshotResources<Cluster>) emptyResources,
54                                                     (SnapshotResources<ClusterLoadAssignment>) emptyResources,
55                                                     (SnapshotResources<Listener>) emptyResources,
56                                                     (SnapshotResources<RouteConfiguration>) emptyResources,
57                                                     (SnapshotResources<Secret>) emptyResources);
58      }
59  
60      void setCluster(String groupName, Cluster cluster) {
61          final Map<String, VersionedResource<Cluster>> groupClusters =
62                  clusterResources.computeIfAbsent(groupName, k -> new HashMap<>());
63          groupClusters.put(cluster.getName(), VersionedResource.create(cluster));
64          clusterUpdated = true;
65      }
66  
67      void setEndpoint(String groupName, ClusterLoadAssignment endpoint) {
68          final Map<String, VersionedResource<ClusterLoadAssignment>> groupEndpoints =
69                  endpointResources.computeIfAbsent(groupName, k -> new HashMap<>());
70          groupEndpoints.put(endpoint.getClusterName(), VersionedResource.create(endpoint));
71          endpointUpdated = true;
72      }
73  
74      void setListener(String groupName, Listener listener) {
75          final Map<String, VersionedResource<Listener>> groupListeners =
76                  listenerResources.computeIfAbsent(groupName, k -> new HashMap<>());
77          groupListeners.put(listener.getName(), VersionedResource.create(listener));
78          listenerUpdated = true;
79      }
80  
81      void setRoute(String groupName, RouteConfiguration route) {
82          final Map<String, VersionedResource<RouteConfiguration>> groupRoutes =
83                  routeResources.computeIfAbsent(groupName, k -> new HashMap<>());
84          groupRoutes.put(route.getName(), VersionedResource.create(route));
85          routeUpdated = true;
86      }
87  
88      void removeCluster(String groupName, String path) {
89          final Map<String, VersionedResource<Cluster>> groupClusters = clusterResources.get(groupName);
90          if (groupClusters == null) {
91              return;
92          }
93          clusterUpdated |= groupClusters.remove(getResourceName(groupName, path)) != null;
94      }
95  
96      private static String getResourceName(String groupName, String path) {
97          return "groups/" + groupName + path.substring(0, path.length() - 5); // Remove .json
98      }
99  
100     void removeEndpoint(String groupName, String path) {
101         final Map<String, VersionedResource<ClusterLoadAssignment>> groupEndpoints =
102                 endpointResources.get(groupName);
103         if (groupEndpoints == null) {
104             return;
105         }
106         // e.g. /endpoints/foo-cluster.json file with group foo -> groups/foo/clusters/foo-cluster
107         // e.g. /k8s/endpoints/foo-cluster.json file with group foo -> groups/foo/k8s/clusters/foo-cluster
108         final String clusterName =
109                 "groups/" + groupName +
110                 ENDPOINTS_PATTERN.matcher(path.substring(0, path.length() - 5) /* remove .json */)
111                                  .replaceFirst("/clusters/");
112         endpointUpdated |= groupEndpoints.remove(clusterName) != null;
113     }
114 
115     void removeListener(String groupName, String path) {
116         final Map<String, VersionedResource<Listener>> groupListeners = listenerResources.get(groupName);
117         if (groupListeners == null) {
118             return;
119         }
120         listenerUpdated |= groupListeners.remove(getResourceName(groupName, path)) != null;
121     }
122 
123     void removeRoute(String groupName, String path) {
124         final Map<String, VersionedResource<RouteConfiguration>> groupRoutes =
125                 routeResources.get(groupName);
126         if (groupRoutes == null) {
127             return;
128         }
129         routeUpdated |= groupRoutes.remove(getResourceName(groupName, path)) != null;
130     }
131 
132     CentralDogmaSnapshot snapshot() {
133         final SnapshotResources<Cluster> clusters;
134         if (clusterUpdated) {
135             clusters = CentralDogmaSnapshotResources.create(clusterResources, ResourceType.CLUSTER);
136             clusterUpdated = false;
137         } else {
138             clusters = currentSnapshot.clusters();
139         }
140 
141         final SnapshotResources<ClusterLoadAssignment> endpoints;
142         if (endpointUpdated) {
143             endpoints = CentralDogmaSnapshotResources.create(endpointResources, ResourceType.ENDPOINT);
144             endpointUpdated = false;
145         } else {
146             endpoints = currentSnapshot.endpoints();
147         }
148 
149         final SnapshotResources<Listener> listeners;
150         if (listenerUpdated) {
151             listeners = CentralDogmaSnapshotResources.create(listenerResources, ResourceType.LISTENER);
152             listenerUpdated = false;
153         } else {
154             listeners = currentSnapshot.listeners();
155         }
156 
157         final SnapshotResources<RouteConfiguration> routes;
158         if (routeUpdated) {
159             routes = CentralDogmaSnapshotResources.create(routeResources, ResourceType.ROUTE);
160             routeUpdated = false;
161         } else {
162             routes = currentSnapshot.routes();
163         }
164 
165         return currentSnapshot =
166                 new CentralDogmaSnapshot(clusters, endpoints, listeners, routes, currentSnapshot.secrets());
167     }
168 
169     void removeGroup(String groupName) {
170         clusterUpdated |= clusterResources.remove(groupName) != null;
171         endpointUpdated |= endpointResources.remove(groupName) != null;
172         listenerUpdated |= listenerResources.remove(groupName) != null;
173         routeUpdated |= routeResources.remove(groupName) != null;
174     }
175 }

17  package com.linecorp.centraldogma.server.mirror;
18  
19  import static java.util.Objects.requireNonNull;
20  
21  import java.time.Instant;
22  import java.util.Objects;
23  
24  import javax.annotation.Nullable;
25  
26  import com.fasterxml.jackson.annotation.JsonCreator;
27  import com.fasterxml.jackson.annotation.JsonIgnoreProperties;
28  import com.fasterxml.jackson.annotation.JsonProperty;
29  import com.google.common.base.MoreObjects;
30  
31  /**
32   * The result of a mirroring operation.
33   */
34  @JsonIgnoreProperties(ignoreUnknown = true)
35  public final class MirrorResult {
36  
37      private final String mirrorId;
38      private final String projectName;
39      private final String repoName;
40      private final MirrorStatus mirrorStatus;
41      @Nullable
42      private final String description;
43      private final Instant triggeredTime;
44      private final Instant completedTime;
45  
46      /**
47       * Creates a new instance.
48       */
49      @JsonCreator
50      public MirrorResult(@JsonProperty("mirrorId") String mirrorId,
51                          @JsonProperty("projectName") String projectName,
52                          @JsonProperty("repoName") String repoName,
53                          @JsonProperty("mirrorStatus") MirrorStatus mirrorStatus,
54                          @JsonProperty("description") @Nullable String description,
55                          @JsonProperty("triggeredTime") Instant triggeredTime,
56                          @JsonProperty("completedTime") Instant completedTime) {
57          this.mirrorId = requireNonNull(mirrorId, "mirrorId");
58          this.projectName = requireNonNull(projectName, "projectName");
59          this.repoName = requireNonNull(repoName, "repoName");
60          this.mirrorStatus = requireNonNull(mirrorStatus, "mirrorStatus");
61          this.description = description;
62          this.triggeredTime = requireNonNull(triggeredTime, "triggeredTime");
63          this.completedTime = requireNonNull(completedTime, "completedTime");
64      }
65  
66      /**
67       * Returns the ID of the mirror.
68       */
69      @JsonProperty("mirrorId")
70      public String mirrorId() {
71          return mirrorId;
72      }
73  
74      /**
75       * Returns the project name which {@link #mirrorId()} belongs to.
76       */
77      @JsonProperty("projectName")
78      public String projectName() {
79          return projectName;
80      }
81  
82      /**
83       * Returns the repository name where the mirroring operation is performed.
84       */
85      @JsonProperty("repoName")
86      public String repoName() {
87          return repoName;
88      }
89  
90      /**
91       * Returns the status of the mirroring operation.
92       */
93      @JsonProperty("mirrorStatus")
94      public MirrorStatus mirrorStatus() {
95          return mirrorStatus;
96      }
97  
98      /**
99       * Returns the description of the mirroring operation.
100      */
101     @Nullable
102     @JsonProperty("description")
103     public String description() {
104         return description;
105     }
106 
107     /**
108      * Returns the time when the mirroring operation was triggered.
109      */
110     @JsonProperty("triggeredTime")
111     public Instant triggeredTime() {
112         return triggeredTime;
113     }
114 
115     /**
116      * Returns the time when the mirroring operation was completed.
117      */
118     @JsonProperty("completedTime")
119     public Instant completedTime() {
120         return completedTime;
121     }
122 
123     @Override
124     public boolean equals(Object o) {
125         if (this == o) {
126             return true;
127         }
128         if (!(o instanceof MirrorResult)) {
129             return false;
130         }
131         final MirrorResult that = (MirrorResult) o;
132         return mirrorId.equals(that.mirrorId) &&
133                projectName.equals(that.projectName) &&
134                repoName.equals(that.repoName) &&
135                mirrorStatus == that.mirrorStatus &&
136                Objects.equals(description, that.description) &&
137                triggeredTime.equals(that.triggeredTime) &&
138                completedTime.equals(that.completedTime);
139     }
140 
141     @Override
142     public int hashCode() {
143         return Objects.hash(mirrorId, projectName, repoName, mirrorStatus, description,
144                             triggeredTime, completedTime);
145     }
146 
147     @Override
148     public String toString() {
149         return MoreObjects.toStringHelper(this)
150                           .omitNullValues()
151                           .add("mirrorId", mirrorId)
152                           .add("projectName", projectName)
153                           .add("repoName", repoName)
154                           .add("mirrorStatus", mirrorStatus)
155                           .add("description", description)
156                           .add("triggeredTime", triggeredTime)
157                           .add("completedTime", completedTime)
158                           .toString();
159     }
160 }

17  package com.linecorp.centraldogma.server.mirror;
18  
19  import java.util.ServiceLoader;
20  
21  import javax.annotation.Nullable;
22  
23  /**
24   * A listener which is notified when a {@link Mirror} operation is started, completed or failed.
25   *
26   * <p>Implement this interface and register it using {@link ServiceLoader} to override the default behavior:
27   * <ul>
28   *     <li>{@link #onStart(MirrorTask)}: Logs an info message for the start of a scheduled {@link Mirror}.</li>
29   *     <li>{@link #onComplete(MirrorTask, MirrorResult)}: Does nothing.</li>
30   *     <li>{@link #onError(MirrorTask, Throwable)}: Logs a warning message for the error of a scheduled
31   *         {@link Mirror}.</li>
32   * </ul>
33   */
34  @Nullable
35  public interface MirrorListener {
36  
37      /**
38       * Invoked when the {@link Mirror} operation is started.
39       */
40      void onStart(MirrorTask mirror);
41  
42      /**
43       * Invoked when the {@link Mirror} operation is completed.
44       */
45      void onComplete(MirrorTask mirror, MirrorResult result);
46  
47      /**
48       * Invoked when the {@link Mirror} operation is failed.
49       */
50      void onError(MirrorTask mirror, Throwable cause);
51  }

19  import static com.linecorp.centraldogma.server.mirror.MirrorSchemes.SCHEME_GIT;
20  import static com.linecorp.centraldogma.server.mirror.MirrorSchemes.SCHEME_GIT_FILE;
21  import static com.linecorp.centraldogma.server.mirror.MirrorSchemes.SCHEME_GIT_HTTP;
22  import static com.linecorp.centraldogma.server.mirror.MirrorSchemes.SCHEME_GIT_HTTPS;
23  import static com.linecorp.centraldogma.server.mirror.MirrorSchemes.SCHEME_GIT_SSH;
24  import static com.linecorp.centraldogma.server.mirror.MirrorUtil.DOGMA_PATH_PATTERN;
25  import static com.linecorp.centraldogma.server.mirror.MirrorUtil.split;
26  import static java.util.Objects.requireNonNull;
27  
28  import java.io.File;
29  import java.net.URI;
30  import java.time.ZonedDateTime;
31  import java.util.regex.Matcher;
32  
33  import javax.annotation.Nullable;
34  
35  import com.cronutils.model.Cron;
36  
37  import com.linecorp.centraldogma.server.MirrorException;
38  import com.linecorp.centraldogma.server.command.CommandExecutor;

41  import com.linecorp.centraldogma.server.storage.repository.Repository;

43  /**
44   * Contains the properties for a mirroring task and performs the task.
45   */
46  public interface Mirror {
47  
48      /**
49       * Creates a new instance.
50       *
51       * @param schedule a cron expression that describes when the mirroring task is supposed to be triggered.
52       *                 If unspecified, {@code 0 * * * * ?} (every minute) is used.
53       * @param direction the direction of mirror
54       * @param credential the authentication credentials which are required when accessing the Git repositories
55       * @param localRepo the Central Dogma repository name
56       * @param localPath the directory path in the {@code localRepo}
57       * @param remoteUri the URI of the Git repository which will be mirrored from
58       * @param gitignore the file pattern for the files in {@code remoteUri} which will not be mirrored.
59       *                  It follows the same format to <a href="https://git-scm.com/docs/gitignore">gitignore</a>
60       */
61      static Mirror of(Cron schedule, MirrorDirection direction, MirrorCredential credential,
62                       Repository localRepo, String localPath, URI remoteUri,
63                       @Nullable String gitignore) {
64          requireNonNull(schedule, "schedule");
65          requireNonNull(direction, "direction");
66          requireNonNull(credential, "credential");
67          requireNonNull(localRepo, "localRepo");
68          requireNonNull(localPath, "localPath");
69          requireNonNull(remoteUri, "remoteUri");
70  
71          final String scheme = remoteUri.getScheme();
72          if (scheme == null) {
73              throw new IllegalArgumentException("no scheme in remoteUri: " + remoteUri);
74          }
75  
76          switch (scheme) {
77              case SCHEME_DOGMA: {
78                  final String[] components = split(remoteUri, "dogma");
79                  final URI remoteRepoUri = URI.create(components[0]);
80                  final Matcher matcher = DOGMA_PATH_PATTERN.matcher(remoteRepoUri.getPath());
81                  if (!matcher.find()) {
82                      throw new IllegalArgumentException(
83                              "cannot determine project name and repository name: " + remoteUri +
84                              " (expected: dogma://<host>[:<port>]/<project>/<repository>.dogma[<remotePath>])");
85                  }
86  
87                  final String remoteProject = matcher.group(1);
88                  final String remoteRepo = matcher.group(2);
89  
90                  return new CentralDogmaMirror(schedule, direction, credential, localRepo, localPath,
91                                                remoteRepoUri, remoteProject, remoteRepo, components[1],
92                                                gitignore);
93              }
94              case SCHEME_GIT:
95              case SCHEME_GIT_SSH:
96              case SCHEME_GIT_HTTP:
97              case SCHEME_GIT_HTTPS:
98              case SCHEME_GIT_FILE: {
99                  final String[] components = split(remoteUri, "git");
100                 return new GitMirror(schedule, direction, credential, localRepo, localPath,
101                                      URI.create(components[0]), components[1], components[2],
102                                      gitignore);
103             }
104         }
105 
106         throw new IllegalArgumentException("unsupported scheme in remoteUri: " + remoteUri);
107     }
108 
109     /**
110      * Returns the schedule for the mirroring task.
111      */
112     Cron schedule();
113 
114     /**
115      * Returns the next execution time of the mirroring task.
116      *
117      * @param lastExecutionTime the last execution time of the mirroring task
118      */
119     ZonedDateTime nextExecutionTime(ZonedDateTime lastExecutionTime);
120 
121     /**
122      * Returns the direction of the mirroring task.
123      */
124     MirrorDirection direction();
125 
126     /**
127      * Returns the authentication credentials which are required when accessing the Git repositories.
128      */
129     MirrorCredential credential();
130 
131     /**
132      * Returns the Central Dogma repository where is supposed to keep the mirrored files.
133      */
134     Repository localRepo();
135 
136     /**
137      * Returns the path in the Central Dogma repository where is supposed to keep the mirrored files.
138      */
139     String localPath();
140 
141     /**
142      * Returns the URI of the Git repository which will be mirrored from.
143      */
144     URI remoteRepoUri();
145 
146     /**
147      * Returns the path of the Git repository where is supposed to be mirrored.
148      */
149     String remotePath();
150 
151     /**
152      * Returns the name of the branch in the Git repository where is supposed to be mirrored.
153      */
154     @Nullable
155     String remoteBranch();
156 
157     /**
158      * Returns a <a href="https://git-scm.com/docs/gitignore">gitignore</a> pattern for the files
159      * which won't be mirrored.
160      */
161     @Nullable
162     String gitignore();
163 
164     /**
165      * Performs the mirroring task.
166      *
167      * @param workDir the local directory where keeps the mirrored files
168      * @param executor the {@link CommandExecutor} which is used to perform operation to the Central Dogma
169      *                 storage
170      * @param maxNumFiles the maximum number of files allowed to the mirroring task. A {@link MirrorException}
171      *                    would be raised if the number of files to be mirrored exceeds it.
172      * @param maxNumBytes the maximum bytes allowed to the mirroring task. A {@link MirrorException} would be
173      *                    raised if the total size of the files to be mirrored exceeds it.
174      */
175     void mirror(File workDir, CommandExecutor executor, int maxNumFiles, long maxNumBytes);
176 }

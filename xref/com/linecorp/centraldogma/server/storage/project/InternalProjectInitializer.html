62  /**
63   * Initializes the internal project and repositories.
64   */
65  public final class InternalProjectInitializer {
66  
67      private static final Logger logger = LoggerFactory.getLogger(InternalProjectInitializer.class);
68  
69      public static final String INTERNAL_PROJECT_DOGMA = "dogma";
70  
71      private final CommandExecutor executor;
72      private final ProjectManager projectManager;
73      private final EncryptionStorageManager encryptionStorageManager;
74      private final CompletableFuture<Void> initialFuture = new CompletableFuture<>();
75  
76      @Nullable
77      private volatile Revision lastTokensRevision;
78      @Nullable
79      private volatile Tokens tokens;
80  
81      /**
82       * Creates a new instance.
83       */
84      public InternalProjectInitializer(CommandExecutor executor, ProjectManager projectManager,
85                                        EncryptionStorageManager encryptionStorageManager) {
86          this.executor = requireNonNull(executor, "executor");
87          this.projectManager = requireNonNull(projectManager, "projectManager");
88          this.encryptionStorageManager = requireNonNull(encryptionStorageManager, "encryptionStorageManager");
89      }

91      /**
92       * Initializes the internal project and repositories in read-only mode.
93       */
94      public void initializeInReadOnlyMode() {
95          try {
96              if (!projectManager.exists(INTERNAL_PROJECT_DOGMA)) {
97                  throw new IllegalStateException(
98                          INTERNAL_PROJECT_DOGMA + " project does not exist. " +
99                          "Cannot initialize in read-only mode.");
100             }
101             final Repository dogmaRepo = projectManager.get(INTERNAL_PROJECT_DOGMA)
102                                                        .repos()
103                                                        .get(Project.REPO_DOGMA);
104             final Entry<JsonNode> entry = dogmaRepo.getOrNull(Revision.HEAD, Query.ofJson(TOKEN_JSON)).join();
105             if (entry == null || !entry.hasContent()) {
106                 throw new IllegalStateException(
107                         TOKEN_JSON + " file does not exist in " + INTERNAL_PROJECT_DOGMA +
108                         '/' + Project.REPO_DOGMA + ". Cannot initialize in read-only mode.");
109             }
110             setTokens(entry, dogmaRepo);
111             if (encryptionStorageManager.encryptSessionCookie()) {
112                 if (encryptionStorageManager.getCurrentSessionKey().join() == null) {
113                     throw new IllegalStateException(
114                             "Session key does not exist. Cannot initialize in read-only mode.");
115                 }
116             }
117 
118             initialFuture.complete(null);
119         } catch (Throwable t) {
120             initialFuture.completeExceptionally(t);
121             Exceptions.throwUnsafely(t);
122         }
123     }
124 
125     /**
126      * Creates an internal project and repositories and a token storage to {@code dogma/dogma/tokens.json}.
127      */
128     public void initialize() {
129         try {
130             initialize0(INTERNAL_PROJECT_DOGMA);
131             initializeTokens();
132             initializeSessionKey();
133             initialFuture.complete(null);
134         } catch (Exception cause) {
135             initialFuture.completeExceptionally(cause);
136         }
137     }
138 
139     /**
140      * Creates the specified internal project and its internal repositories.
141      */
142     public void initialize(String projectName) {
143         requireNonNull(projectName, "projectName");
144         checkArgument(!INTERNAL_PROJECT_DOGMA.equals(projectName),
145                       "Use initialize() to create %s", projectName);
146         initialize0(projectName);
147     }
148 
149     /**
150      * Creates an internal project and repositories such as a token storage.
151      */
152     private void initialize0(String projectName) {
153         final long creationTimeMillis = System.currentTimeMillis();
154         if (!projectManager.exists(projectName)) {
155             try {
156                 executor.execute(Command.forcePush(
157                                 createProject(creationTimeMillis, Author.SYSTEM, projectName)))
158                         .get();
159             } catch (Throwable cause) {
160                 final Throwable peeled = Exceptions.peel(cause);
161                 if (!(peeled instanceof ProjectExistsException)) {
162                     throw new Error("failed to initialize an internal project: " + projectName, peeled);
163                 }
164             }
165         }
166 
167         // These repositories might be created when creating an internal project, but we try to create them
168         // again here in order to make sure them exist because sometimes their names are changed.
169         initializeInternalRepos(projectName, Project.internalRepos(), creationTimeMillis);
170     }
171 
172     private void initializeTokens() {
173         final Repository dogmaRepo = projectManager.get(INTERNAL_PROJECT_DOGMA).repos().get(Project.REPO_DOGMA);
174         final Entry<JsonNode> entry = dogmaRepo.getOrNull(Revision.HEAD, Query.ofJson(TOKEN_JSON)).join();
175         if (entry != null && entry.hasContent()) {
176             setTokens(entry, dogmaRepo);
177             return;
178         }
179         try {
180             final Change<?> change = Change.ofJsonPatch(TOKEN_JSON,
181                                                         null, Jackson.valueToTree(new Tokens()));
182             final String commitSummary = "Initialize the token list file: /" + INTERNAL_PROJECT_DOGMA + '/' +
183                                          Project.REPO_DOGMA + TOKEN_JSON;
184             executor.execute(Command.forcePush(push(Author.SYSTEM, INTERNAL_PROJECT_DOGMA, Project.REPO_DOGMA,
185                                                     Revision.HEAD, commitSummary, "", Markup.PLAINTEXT,
186                                                     ImmutableList.of(change))))
187                     .get();
188             final Entry<JsonNode> entry1 = dogmaRepo.getOrNull(Revision.HEAD, Query.ofJson(TOKEN_JSON)).join();
189             assert entry1 != null;
190             setTokens(entry1, dogmaRepo);
191         } catch (Throwable cause) {
192             final Throwable peeled = Exceptions.peel(cause);
193             if (peeled instanceof ChangeConflictException) {
194                 return;
195             }
196             throw new Error("failed to initialize the token list file", peeled);
197         }
198     }
199 
200     private void setTokens(Entry<JsonNode> entry, Repository dogmaRepo) {
201         try {
202             final Tokens tokens = Jackson.treeToValue(entry.content(), Tokens.class);
203             lastTokensRevision = entry.revision();
204             this.tokens = tokens;
205             attachTokensListener(dogmaRepo);
206         } catch (JsonParseException | JsonMappingException e) {
207             throw new RuntimeException(String.format("failed to parse %s/%s/%s", INTERNAL_PROJECT_DOGMA,
208                                                      Project.REPO_DOGMA, TOKEN_JSON), e);
209         }
210     }
211 
212     private void attachTokensListener(Repository dogmaRepo) {
213         dogmaRepo.addListener(RepositoryListener.of(Query.ofJson(TOKEN_JSON), entry -> {
214             if (entry == null) {
215                 logger.warn("{} file is missing in {}/{}", TOKEN_JSON, INTERNAL_PROJECT_DOGMA,
216                             Project.REPO_DOGMA);
217                 return;
218             }
219 
220             final Revision lastRevision = entry.revision();
221             final Revision lastTokensRevision = this.lastTokensRevision;
222             if (lastTokensRevision != null && lastRevision.compareTo(lastTokensRevision) <= 0) {
223                 // An old data.
224                 return;
225             }
226 
227             try {
228                 final Tokens tokens = Jackson.treeToValue(entry.content(), Tokens.class);
229                 this.lastTokensRevision = lastRevision;
230                 this.tokens = tokens;
231             } catch (JsonParseException | JsonMappingException e) {
232                 logger.warn("Invalid {} file in {}/{}", TOKEN_JSON, INTERNAL_PROJECT_DOGMA,
233                             Project.REPO_DOGMA, e);
234             }
235         }));
236     }
237 
238     private void initializeSessionKey() {
239         if (!encryptionStorageManager.encryptSessionCookie()) {
240             return;
241         }
242         final SessionKey sessionKey = encryptionStorageManager.getCurrentSessionKey().join();
243         if (sessionKey != null) {
244             return;
245         }
246 
247         final SessionMasterKey sessionMasterKey = encryptionStorageManager.generateSessionMasterKey().join();
248         try {
249             executor.execute(Command.createSessionMasterKey(sessionMasterKey))
250                     .get();
251         } catch (Throwable cause) {
252             final Throwable peeled = Exceptions.peel(cause);
253             if (peeled instanceof EncryptionEntryExistsException) {
254                 return;
255             }
256             throw new Error("failed to initialize a session master key", peeled);
257         }
258     }
259 
260     /**
261      * Returns a {@link CompletableFuture} which is completed when the internal project and repositories are
262      * ready.
263      */
264     public CompletableFuture<Void> whenInitialized() {
265         return initialFuture;
266     }
267 
268     /**
269      * Returns the {@link Tokens}.
270      */
271     public Tokens tokens() {
272         final Tokens tokens = this.tokens;
273         checkState(tokens != null, "tokens have not been loaded yet");
274         return tokens;
275     }
276 
277     /**
278      * Creates the specified internal repositories in the internal project.
279      */
280     private void initializeInternalRepos(String projectName, List<String> internalRepos,
281                                          long creationTimeMillis) {
282         requireNonNull(internalRepos, "internalRepos");
283         final Project project = projectManager.get(projectName);
284         assert project != null;
285         for (final String repo : internalRepos) {
286             if (project.repos().exists(repo)) {
287                 continue;
288             }
289             try {
290                 executor.execute(Command.forcePush(
291                                 createRepository(creationTimeMillis, Author.SYSTEM, projectName, repo)))
292                         .get();
293             } catch (Throwable cause) {
294                 final Throwable peeled = Exceptions.peel(cause);
295                 if (peeled instanceof ReadOnlyException) {
296                     // The executor has stopped right after starting up.
297                     return;
298                 }
299                 if (!(peeled instanceof RepositoryExistsException)) {
300                     throw new Error("failed to initialize an internal repository: " + projectName +
301                                     '/' + repo, peeled);
302                 }
303             }
304         }
305     }

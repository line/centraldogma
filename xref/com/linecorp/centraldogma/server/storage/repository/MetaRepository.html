17  package com.linecorp.centraldogma.server.storage.repository;
18  
19  import java.util.List;
20  import java.util.concurrent.CompletableFuture;
21  
22  import com.linecorp.centraldogma.common.Author;
23  import com.linecorp.centraldogma.internal.api.v1.MirrorDto;
24  import com.linecorp.centraldogma.server.command.Command;
25  import com.linecorp.centraldogma.server.command.CommitResult;
26  import com.linecorp.centraldogma.server.credential.Credential;
27  import com.linecorp.centraldogma.server.mirror.Mirror;
28  
29  /**
30   * A Revision-controlled filesystem-like repository which is named {@code "meta"}.
31   */
32  public interface MetaRepository extends Repository {
33  
34      /**
35       * Returns active mirroring tasks.
36       */
37      default CompletableFuture<List<Mirror>> mirrors() {
38          return mirrors(false);
39      }
40  
41      /**
42       * Returns a set of mirroring tasks. If {@code includeDisabled} is {@code true}, disabled mirroring tasks
43       * are also included in the returned {@link Mirror}s.
44       */
45      CompletableFuture<List<Mirror>> mirrors(boolean includeDisabled);
46  
47      /**
48       * Returns a mirroring task of the specified {@code id}.
49       */
50      CompletableFuture<Mirror> mirror(String id);
51  
52      /**
53       * Returns a list of mirroring credentials.
54       */
55      CompletableFuture<List<Credential>> credentials();
56  
57      /**
58       * Returns a mirroring credential of the specified {@code id}.
59       */
60      CompletableFuture<Credential> credential(String id);
61  
62      /**
63       * Create a push {@link Command} for the {@link MirrorDto}.
64       */
65      CompletableFuture<Command<CommitResult>> createPushCommand(MirrorDto mirrorDto, Author author,
66                                                                 boolean update);
67  
68      /**
69       * Create a push {@link Command} for the {@link Credential}.
70       */
71      CompletableFuture<Command<CommitResult>> createPushCommand(Credential credential, Author author,
72                                                                 boolean update);
73  }

63      private static final CancellationException CANCELLATION_EXCEPTION =
64              Exceptions.clearTrace(new CancellationException("parent complete"));
65  
66      static CompletableFuture<MergedEntry<?>> mergeEntries(
67              List<CompletableFuture<Entry<?>>> entryFutures, Revision revision,
68              MergeQuery<?> query) {
69          requireNonNull(entryFutures, "entryFutures");
70          requireNonNull(revision, "revision");
71          requireNonNull(query, "query");
72  
73          final CompletableFuture<MergedEntry<JsonNode>> future = new CompletableFuture<>();
74          CompletableFutures.allAsList(entryFutures).handle((entries, cause) -> {
75              if (cause != null) {
76                  future.completeExceptionally(Exceptions.peel(cause));
77                  return null;
78              }
79  
80              final Builder<JsonNode> jsonNodesBuilder = ImmutableList.builder();
81              final Builder<String> pathsBuilder = ImmutableList.builder();
82              for (Entry<?> entry : entries) {
83                  if (entry == null) {
84                      continue;
85                  }
86                  try {
87                      jsonNodesBuilder.add(entry.contentAsJson());
88                      pathsBuilder.add(entry.path());
89                  } catch (JsonParseException e) {
90                      future.completeExceptionally(e);
91                      return null;
92                  }
93              }
94  
95              JsonNode result;
96              try {
97                  final List<JsonNode> jsonNodes = jsonNodesBuilder.build();
98                  if (jsonNodes.isEmpty()) {
99                      throw new EntryNotFoundException(revision, concatenatePaths(query.mergeSources()));
100                 }
101 
102                 result = Jackson.mergeTree(jsonNodes);
103                 final List<String> expressions = query.expressions();
104                 if (!Iterables.isEmpty(expressions)) {
105                     result = Jackson.extractTree(result, expressions);
106                 }
107             } catch (Exception e) {
108                 future.completeExceptionally(e);
109                 return null;
110             }
111 
112             future.complete(MergedEntry.of(revision, EntryType.JSON, result, pathsBuilder.build()));
113             return null;
114         });
115         return unsafeCast(future);
116     }
117 
118     private static String concatenatePaths(Iterable<MergeSource> mergeSources) {
119         return Streams.stream(mergeSources).map(MergeSource::path).collect(Collectors.joining(","));
120     }

122     /**
123      * Applies the specified {@link Query} to the {@link Entry#content()} of the specified {@link Entry} and
124      * returns the query result.
125      *
126      * @throws IllegalStateException if the specified {@link Entry} is a directory
127      * @throws QuerySyntaxException if the syntax of specified {@link Query} is invalid
128      * @throws QueryExecutionException if an {@link Exception} is raised while applying the specified
129      *                                 {@link Query} to the {@link Entry#content()}
130      */
131     static <T> Entry<T> applyQuery(Entry<T> entry, Query<T> query) {
132         requireNonNull(query, "query");
133         entry.content(); // Ensure that content is not null.
134         final EntryType entryType = entry.type();
135 
136         final QueryType queryType = query.type();
137         if (!queryType.supportedEntryTypes().contains(entryType)) {
138             throw new QueryExecutionException("Unsupported entry type: " + entryType +
139                                               " (query: " + query + ')');
140         }
141 
142         if (queryType == IDENTITY || queryType == IDENTITY_TEXT || queryType == IDENTITY_JSON ||
143             queryType == IDENTITY_YAML) {
144             return entry;
145         } else if (queryType == JSON_PATH) {
146             return Entry.of(entry.revision(), query.path(), entryType, query.apply(entry.content()));
147         } else {
148             throw new QueryExecutionException("Unsupported entry type: " + entryType +
149                                               " (query: " + query + ')');
150         }
151     }

153     static <T> CompletableFuture<Entry<T>> watch(Repository repo, Revision lastKnownRev, Query<T> query,
154                                                  boolean errorOnEntryNotFound) {
155         requireNonNull(repo, "repo");
156         requireNonNull(lastKnownRev, "lastKnownRev");
157         requireNonNull(query, "query");
158 
159         final Query<Object> castQuery = unsafeCast(query);
160         final CompletableFuture<Entry<Object>> parentFuture = new CompletableFuture<>();
161         repo.getOrNull(lastKnownRev, castQuery)
162             .thenAccept(oldResult -> watch(repo, castQuery, lastKnownRev, oldResult,
163                                            parentFuture, errorOnEntryNotFound))
164             .exceptionally(voidFunction(parentFuture::completeExceptionally));
165 
166         return unsafeCast(parentFuture);
167     }
168 
169     private static void watch(Repository repo, Query<Object> query,
170                               Revision lastKnownRev, @Nullable Entry<Object> oldResult,
171                               CompletableFuture<Entry<Object>> parentFuture, boolean errorOnEntryNotFound) {
172 
173         final CompletableFuture<Revision> future = repo.watch(lastKnownRev, query.path(), errorOnEntryNotFound);
174         parentFuture.whenComplete((res, cause) -> future.completeExceptionally(CANCELLATION_EXCEPTION));
175 
176         future.thenCompose(newRev -> repo.getOrNull(newRev, query).thenAccept(newResult -> {
177             if (errorOnEntryNotFound && newResult == null) {
178                 // The entry is removed.
179                 parentFuture.completeExceptionally(new EntryNotFoundException(newRev, query.path()));
180                 return;
181             }
182 
183             if (newResult == null ||
184                 oldResult != null && Objects.equals(oldResult.content(), newResult.content())) {
185                 // Entry does not exist or did not change; watch again for more changes.
186                 if (!parentFuture.isDone()) {
187                     // ... only when the parent future has not been cancelled.
188                     watch(repo, query, newRev, oldResult, parentFuture, errorOnEntryNotFound);
189                 }
190             } else {
191                 parentFuture.complete(newResult);
192             }
193         })).exceptionally(voidFunction(parentFuture::completeExceptionally));
194     }

16  package com.linecorp.centraldogma.server.auth;
17  
18  import java.util.concurrent.CompletableFuture;
19  
20  /**
21   * An interface for session management.
22   */
23  public interface SessionManager extends AutoCloseable {
24      /**
25       * Returns a new session ID.
26       */
27      String generateSessionId();
28  
29      /**
30       * Returns whether the session which has the specified {@code sessionId} exists.
31       *
32       * @param sessionId the session ID
33       */
34      CompletableFuture<Boolean> exists(String sessionId);
35  
36      /**
37       * Returns the session which has the specified {@code sessionId}.
38       *
39       * @param sessionId the session ID
40       */
41      CompletableFuture<Session> get(String sessionId);
42  
43      /**
44       * Creates the specified {@code session} in the session storage.
45       *
46       * @param session the session to be stored
47       */
48      CompletableFuture<Void> create(Session session);
49  
50      /**
51       * Updates the specified {@code session} if it exists.
52       *
53       * @param session the session to be updated
54       */
55      CompletableFuture<Void> update(Session session);
56  
57      /**
58       * Deletes the specified session from the session storage.
59       *
60       * @param sessionId the session ID to be removed
61       */
62      CompletableFuture<Void> delete(String sessionId);
63  }

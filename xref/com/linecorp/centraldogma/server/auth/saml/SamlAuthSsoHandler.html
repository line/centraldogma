58  /**
59   * A {@link SamlSingleSignOnHandler} implementation for the Central Dogma server.
60   */
61  final class SamlAuthSsoHandler implements SamlSingleSignOnHandler {
62  
63      private final Supplier<String> sessionIdGenerator;
64      private final Function<Session, CompletableFuture<Void>> loginSessionPropagator;
65      private final Duration sessionValidDuration;
66      private final Function<String, String> loginNameNormalizer;
67      @Nullable
68      private final String subjectLoginNameIdFormat;
69      @Nullable
70      private final String attributeLoginName;
71  
72      SamlAuthSsoHandler(
73              Supplier<String> sessionIdGenerator,
74              Function<Session, CompletableFuture<Void>> loginSessionPropagator,
75              Duration sessionValidDuration, Function<String, String> loginNameNormalizer,
76              @Nullable String subjectLoginNameIdFormat, @Nullable String attributeLoginName) {
77          this.sessionIdGenerator = requireNonNull(sessionIdGenerator, "sessionIdGenerator");
78          this.loginSessionPropagator = requireNonNull(loginSessionPropagator, "loginSessionPropagator");
79          this.sessionValidDuration = requireNonNull(sessionValidDuration, "sessionValidDuration");
80          this.loginNameNormalizer = requireNonNull(loginNameNormalizer, "loginNameNormalizer");
81          checkArgument(!Strings.isNullOrEmpty(subjectLoginNameIdFormat) ||
82                        !Strings.isNullOrEmpty(attributeLoginName),
83                        "a name ID format of a subject or an attribute name should be specified " +
84                        "for finding a login name");
85          this.subjectLoginNameIdFormat = subjectLoginNameIdFormat;
86          this.attributeLoginName = attributeLoginName;
87      }
88  
89      @Override
90      public CompletionStage<Void> beforeInitiatingSso(ServiceRequestContext ctx, HttpRequest req,
91                                                       MessageContext<AuthnRequest> message,
92                                                       SamlIdentityProviderConfig idpConfig) {
93          final QueryStringDecoder decoder = new QueryStringDecoder(req.path(), true);
94          final List<String> ref = decoder.parameters().get("ref");
95          if (ref == null || ref.isEmpty()) {
96              return CompletableFuture.completedFuture(null);
97          }
98  
99          final String relayState = ref.get(0);
100         if (idpConfig.ssoEndpoint().bindingProtocol() == SamlBindingProtocol.HTTP_REDIRECT &&
101             relayState.length() > 80) {
102             return CompletableFuture.completedFuture(null);
103         }
104 
105         final SAMLBindingContext sub = message.getSubcontext(SAMLBindingContext.class, true);
106         assert sub != null : SAMLBindingContext.class.getName();
107         sub.setRelayState(relayState);
108         return CompletableFuture.completedFuture(null);
109     }
110 
111     @Override
112     public HttpResponse loginSucceeded(ServiceRequestContext ctx, AggregatedHttpRequest req,
113                                        MessageContext<Response> message, @Nullable String sessionIndex,
114                                        @Nullable String relayState) {
115         final Response response = requireNonNull(message, "message").getMessage();
116         final String username = Optional.ofNullable(findLoginNameFromSubjects(response))
117                                         .orElseGet(() -> findLoginNameFromAttributes(response));
118         if (Strings.isNullOrEmpty(username)) {
119             return loginFailed(ctx, req, message,
120                                new IllegalStateException("Cannot get a username from the response"));
121         }
122 
123         final String sessionId = sessionIdGenerator.get();
124         final Session session =
125                 new Session(sessionId, loginNameNormalizer.apply(username), sessionValidDuration);
126 
127         final String redirectionScript;
128         if (!Strings.isNullOrEmpty(relayState)) {
129             redirectionScript = "window.location.href='/#" +
130                                 HtmlEscapers.htmlEscaper().escape(relayState) + '\'';
131         } else {
132             redirectionScript = "window.location.href='/'";
133         }
134         return HttpResponse.of(loginSessionPropagator.apply(session).thenApply(
135                 unused -> HttpResponse.of(HttpStatus.OK, MediaType.HTML_UTF_8, getHtmlWithOnload(
136                         "localStorage.setItem('sessionId','" + sessionId + "')",
137                         redirectionScript))));
138     }
139 
140     @Nullable
141     private String findLoginNameFromSubjects(Response response) {
142         if (Strings.isNullOrEmpty(subjectLoginNameIdFormat)) {
143             return null;
144         }
145         return response.getAssertions()
146                        .stream()
147                        .map(s -> s.getSubject().getNameID())
148                        .filter(nameId -> nameId.getFormat().equals(subjectLoginNameIdFormat))
149                        .map(NameIDType::getValue)
150                        .findFirst()
151                        .orElse(null);
152     }
153 
154     @Nullable
155     private String findLoginNameFromAttributes(Response response) {
156         if (Strings.isNullOrEmpty(attributeLoginName)) {
157             return null;
158         }
159         return response.getAssertions()
160                        .stream()
161                        .flatMap(s -> s.getAttributeStatements().stream())
162                        .flatMap(s -> s.getAttributes().stream())
163                        .filter(attr -> attr.getName().equals(attributeLoginName))
164                        .findFirst()
165                        .map(attr -> {
166                            final XMLObject v = attr.getAttributeValues().get(0);
167                            if (v instanceof XSString) {
168                                return ((XSString) v).getValue();
169                            } else {
170                                return null;
171                            }
172                        })
173                        .orElse(null);
174     }
175 
176     @Override
177     public HttpResponse loginFailed(ServiceRequestContext ctx, AggregatedHttpRequest req,
178                                     @Nullable MessageContext<Response> message, Throwable cause) {
179         final HttpStatus status =
180                 cause instanceof InvalidSamlRequestException ? HttpStatus.BAD_REQUEST
181                                                              : HttpStatus.INTERNAL_SERVER_ERROR;
182         return HttpApiUtil.newResponse(ctx, status, cause);
183     }

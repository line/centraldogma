16  package com.linecorp.centraldogma.server.internal.storage;
17  
18  import java.security.SecureRandom;
19  import java.security.Security;
20  
21  import javax.crypto.Cipher;
22  import javax.crypto.SecretKey;
23  import javax.crypto.spec.GCMParameterSpec;
24  import javax.crypto.spec.SecretKeySpec;
25  
26  import org.bouncycastle.jce.provider.BouncyCastleProvider;
27  import org.conscrypt.Conscrypt;
28  import org.slf4j.Logger;
29  import org.slf4j.LoggerFactory;
30  
31  import com.linecorp.armeria.common.util.SystemInfo;
32  
33  public final class AesGcmSivCipher {
34  
35      private static final Logger logger = LoggerFactory.getLogger(AesGcmSivCipher.class);
36  
37      private static final String ALGORITHM = "AES/GCM-SIV/NoPadding";
38      // https://datatracker.ietf.org/doc/html/rfc8452#section-4
39      public static final int KEY_SIZE_BYTES = 32;
40      public static final int NONCE_SIZE_BYTES = 12;
41      private static final int TAG_SIZE_BITS = 128;
42  
43      private static final SecureRandom SECURE_RANDOM = new SecureRandom();
44  
45      private static final String PROVIDER;
46  
47      static {
48          if (SystemInfo.isLinux()) {
49              Security.addProvider(Conscrypt.newProvider());
50              PROVIDER = "Conscrypt";
51          } else {
52              Security.addProvider(new BouncyCastleProvider());
53              PROVIDER = "BC";
54          }
55          logger.info("Using cryptographic provider: {}", PROVIDER);
56      }
57  
58      public static byte[] generateAes256Key() {
59          final byte[] keyBytes = new byte[KEY_SIZE_BYTES]; // 256bit
60          // Consider sharding the SecureRandom instance if this method is called frequently.
61          SECURE_RANDOM.nextBytes(keyBytes);
62          return keyBytes;
63      }
64  
65      public static SecretKeySpec aesSecretKey(byte[] key) {
66          return new SecretKeySpec(key, "AES");
67      }
68  
69      public static byte[] generateNonce() {
70          final byte[] nonce = new byte[NONCE_SIZE_BYTES];
71          SECURE_RANDOM.nextBytes(nonce);
72          return nonce;
73      }
74  
75      public static byte[] encrypt(SecretKey key, byte[] nonce, byte[] data) throws Exception {
76          return encrypt(key, nonce, data, 0, data.length);
77      }
78  
79      public static byte[] encrypt(SecretKey key, byte[] nonce, byte[] data, int off, int len) throws Exception {
80          final Cipher cipher = Cipher.getInstance(ALGORITHM, PROVIDER);
81          final GCMParameterSpec parameterSpec = new GCMParameterSpec(TAG_SIZE_BITS, nonce);
82          cipher.init(Cipher.ENCRYPT_MODE, key, parameterSpec);
83          return cipher.doFinal(data, off, len);
84      }
85  
86      public static byte[] decrypt(SecretKey key, byte[] nonce, byte[] ciphertext) throws Exception {
87          final Cipher cipher = Cipher.getInstance(ALGORITHM, PROVIDER);
88          final GCMParameterSpec parameterSpec = new GCMParameterSpec(TAG_SIZE_BITS, nonce);
89          cipher.init(Cipher.DECRYPT_MODE, key, parameterSpec);
90          return cipher.doFinal(ciphertext);
91      }
92  
93      private AesGcmSivCipher() {}
94  }

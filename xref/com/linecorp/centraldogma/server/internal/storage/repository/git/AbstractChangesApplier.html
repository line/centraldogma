36  import com.fasterxml.jackson.databind.JsonNode;

38  import com.linecorp.centraldogma.common.CentralDogmaException;

42  
43  abstract class AbstractChangesApplier {
44  
45      private static final byte[] EMPTY_BYTE = new byte[0];
46  
47      int apply(Repository jGitRepository, Revision headRevision,
48                @Nullable ObjectId baseTreeId, DirCache dirCache) {
49          try (ObjectInserter inserter = jGitRepository.newObjectInserter();
50               ObjectReader reader = jGitRepository.newObjectReader()) {
51  
52              if (baseTreeId != null) {
53                  // the DirCacheBuilder is to used for doing update operations on the given DirCache object
54                  final DirCacheBuilder builder = dirCache.builder();
55  
56                  // Add the tree object indicated by the prevRevision to the temporary DirCache object.
57                  builder.addTree(EMPTY_BYTE, 0, reader, baseTreeId);
58                  builder.finish();
59              }
60  
61              return doApply(headRevision, dirCache, reader, inserter);
62          } catch (CentralDogmaException e) {
63              throw e;
64          } catch (Exception e) {
65              throw new StorageException("failed to apply changes on revision: " + headRevision.major(), e);
66          }
67      }
68  
69      abstract int doApply(Revision headRevision, DirCache dirCache,
70                           ObjectReader reader, ObjectInserter inserter) throws IOException;
71  
72      static void applyPathEdit(DirCache dirCache, PathEdit edit) {
73          final DirCacheEditor e = dirCache.editor();
74          e.add(edit);
75          e.finish();
76      }
77  
78      // PathEdit implementations which is used when applying changes.
79  
80      static final class InsertJson extends PathEdit {
81          private final ObjectInserter inserter;
82          private final JsonNode jsonNode;
83  
84          InsertJson(String entryPath, ObjectInserter inserter, JsonNode jsonNode) {
85              super(entryPath);
86              this.inserter = inserter;
87              this.jsonNode = jsonNode;
88          }
89  
90          @Override
91          public void apply(DirCacheEntry ent) {
92              try {
93                  ent.setObjectId(inserter.insert(Constants.OBJ_BLOB, Jackson.writeValueAsBytes(jsonNode)));
94                  ent.setFileMode(FileMode.REGULAR_FILE);
95              } catch (IOException e) {
96                  throw new StorageException("failed to create a new JSON blob", e);
97              }
98          }
99      }
100 
101     static final class InsertText extends PathEdit {
102         private final ObjectInserter inserter;
103         private final String text;
104 
105         InsertText(String entryPath, ObjectInserter inserter, String text) {
106             super(entryPath);
107             this.inserter = inserter;
108             this.text = text;
109         }
110 
111         @Override
112         public void apply(DirCacheEntry ent) {
113             try {
114                 ent.setObjectId(inserter.insert(Constants.OBJ_BLOB, text.getBytes(UTF_8)));
115                 ent.setFileMode(FileMode.REGULAR_FILE);
116             } catch (IOException e) {
117                 throw new StorageException("failed to create a new text blob", e);
118             }
119         }
120     }
121 
122     static final class CopyOldEntry extends PathEdit {
123         private final DirCacheEntry oldEntry;
124 
125         CopyOldEntry(String entryPath, DirCacheEntry oldEntry) {
126             super(entryPath);
127             this.oldEntry = oldEntry;
128         }
129 
130         @Override
131         public void apply(DirCacheEntry ent) {
132             ent.setFileMode(oldEntry.getFileMode());
133             ent.setObjectId(oldEntry.getObjectId());
134         }
135     }
136 }

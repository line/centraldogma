17  package com.linecorp.centraldogma.server.internal.storage.repository;
18  
19  import static java.util.Objects.requireNonNull;
20  
21  import com.linecorp.centraldogma.common.Revision;
22  
23  /**
24   * An interface that provides a {@link Revision} with an object.
25   */
26  public interface HasRevision<T> {
27  
28      /**
29       * Creates a new instance with the specified object and revision.
30       */
31      static <T> HasRevision<T> of(T object, Revision revision) {
32          requireNonNull(object, "object");
33          requireNonNull(revision, "revision");
34          return new DefaultHasRevision<>(object, revision);
35      }
36  
37      /**
38       * Returns the {@link Revision}.
39       */
40      Revision revision();
41  
42      /**
43       * Returns the object.
44       */
45      T object();
46  }

53  final class DefaultChangesApplier extends AbstractChangesApplier {
54  
55      private final Iterable<Change<?>> changes;
56  
57      DefaultChangesApplier(Iterable<Change<?>> changes) {
58          this.changes = changes;
59      }
60  
61      @Override
62      int doApply(Revision unused, DirCache dirCache,
63                  ObjectReader reader, ObjectInserter inserter) throws IOException {
64          int numEdits = 0;
65          // loop over the specified changes.
66          for (Change<?> change : changes) {
67              final String changePath = change.path().substring(1); // Strip the leading '/'.
68              final DirCacheEntry oldEntry = dirCache.getEntry(changePath);
69              final byte[] oldContent = oldEntry != null ? reader.open(oldEntry.getObjectId()).getBytes()
70                                                         : null;
71  
72              switch (change.type()) {
73                  case UPSERT_JSON: {
74                      final JsonNode oldJsonNode = oldContent != null ? Jackson.readTree(oldContent) : null;
75                      final JsonNode newJsonNode = firstNonNull((JsonNode) change.content(),
76                                                                JsonNodeFactory.instance.nullNode());
77  
78                      // Upsert only when the contents are really different.
79                      if (!Objects.equals(newJsonNode, oldJsonNode)) {
80                          applyPathEdit(dirCache, new InsertJson(changePath, inserter, newJsonNode));
81                          numEdits++;
82                      }
83                      break;
84                  }
85                  case UPSERT_TEXT: {
86                      final String sanitizedOldText;
87                      if (oldContent != null) {
88                          sanitizedOldText = sanitizeText(new String(oldContent, UTF_8));
89                      } else {
90                          sanitizedOldText = null;
91                      }
92  
93                      final String sanitizedNewText = sanitizeText(change.contentAsText());
94  
95                      // Upsert only when the contents are really different.
96                      if (!sanitizedNewText.equals(sanitizedOldText)) {
97                          applyPathEdit(dirCache, new InsertText(changePath, inserter, sanitizedNewText));
98                          numEdits++;
99                      }
100                     break;
101                 }
102                 case REMOVE:
103                     if (oldEntry != null) {
104                         applyPathEdit(dirCache, new DeletePath(changePath));
105                         numEdits++;
106                         break;
107                     }
108 
109                     // The path might be a directory.
110                     if (applyDirectoryEdits(dirCache, changePath, null, change)) {
111                         numEdits++;
112                     } else {
113                         // Was not a directory either; conflict.
114                         reportNonExistentEntry(change);
115                         break;
116                     }
117                     break;
118                 case RENAME: {
119                     final String newPath =
120                             ((String) change.content()).substring(1); // Strip the leading '/'.
121 
122                     if (dirCache.getEntry(newPath) != null) {
123                         throw new ChangeConflictException("a file exists at the target path: " + change);
124                     }
125 
126                     if (oldEntry != null) {
127                         if (changePath.equals(newPath)) {
128                             // Redundant rename request - old path and new path are same.
129                             break;
130                         }
131 
132                         final DirCacheEditor editor = dirCache.editor();
133                         editor.add(new DeletePath(changePath));
134                         editor.add(new CopyOldEntry(newPath, oldEntry));
135                         editor.finish();
136                         numEdits++;
137                         break;
138                     }
139 
140                     // The path might be a directory.
141                     if (applyDirectoryEdits(dirCache, changePath, newPath, change)) {
142                         numEdits++;
143                     } else {
144                         // Was not a directory either; conflict.
145                         reportNonExistentEntry(change);
146                     }
147                     break;
148                 }
149                 case APPLY_JSON_PATCH: {
150                     final JsonNode oldJsonNode;
151                     if (oldContent != null) {
152                         oldJsonNode = Jackson.readTree(oldContent);
153                     } else {
154                         oldJsonNode = Jackson.nullNode;
155                     }
156 
157                     final JsonNode newJsonNode;
158                     try {
159                         newJsonNode = JsonPatch.fromJson((JsonNode) change.content()).apply(oldJsonNode);
160                     } catch (Exception e) {
161                         throw new ChangeConflictException("failed to apply JSON patch: " + change, e);
162                     }
163 
164                     // Apply only when the contents are really different.
165                     if (!newJsonNode.equals(oldJsonNode)) {
166                         applyPathEdit(dirCache, new InsertJson(changePath, inserter, newJsonNode));
167                         numEdits++;
168                     }
169                     break;
170                 }
171                 case APPLY_TEXT_PATCH:
172                     final Patch<String> patch = DiffUtils.parseUnifiedDiff(
173                             Util.stringToLines(sanitizeText((String) change.content())));
174 
175                     final String sanitizedOldText;
176                     final List<String> sanitizedOldTextLines;
177                     if (oldContent != null) {
178                         sanitizedOldText = sanitizeText(new String(oldContent, UTF_8));
179                         sanitizedOldTextLines = Util.stringToLines(sanitizedOldText);
180                     } else {
181                         sanitizedOldText = null;
182                         sanitizedOldTextLines = Collections.emptyList();
183                     }
184 
185                     final String newText;
186                     try {
187                         final List<String> newTextLines = DiffUtils.patch(sanitizedOldTextLines, patch);
188                         if (newTextLines.isEmpty()) {
189                             newText = "";
190                         } else {
191                             final StringJoiner joiner = new StringJoiner("\n", "", "\n");
192                             for (String line : newTextLines) {
193                                 joiner.add(line);
194                             }
195                             newText = joiner.toString();
196                         }
197                     } catch (Exception e) {
198                         throw new ChangeConflictException("failed to apply text patch: " + change, e);
199                     }
200 
201                     // Apply only when the contents are really different.
202                     if (!newText.equals(sanitizedOldText)) {
203                         applyPathEdit(dirCache, new InsertText(changePath, inserter, newText));
204                         numEdits++;
205                     }
206                     break;
207             }
208         }
209         return numEdits;
210     }
211 
212     @Override
213     public String toString() {
214         return MoreObjects.toStringHelper(this)
215                           .add("changes", changes)
216                           .toString();
217     }

219     /**
220      * Applies recursive directory edits.
221      *
222      * @param oldDir the path to the directory to make a recursive change
223      * @param newDir the path to the renamed directory, or {@code null} to remove the directory.
224      *
225      * @return {@code true} if any edits were made to {@code dirCache}, {@code false} otherwise
226      */
227     private static boolean applyDirectoryEdits(DirCache dirCache,
228                                                String oldDir, @Nullable String newDir, Change<?> change) {
229 
230         if (!oldDir.endsWith("/")) {
231             oldDir += '/';
232         }
233         if (newDir != null && !newDir.endsWith("/")) {
234             newDir += '/';
235         }
236 
237         final byte[] rawOldDir = Constants.encode(oldDir);
238         final byte[] rawNewDir = newDir != null ? Constants.encode(newDir) : null;
239         final int numEntries = dirCache.getEntryCount();
240         DirCacheEditor editor = null;
241 
242         loop:
243         for (int i = 0; i < numEntries; i++) {
244             final DirCacheEntry e = dirCache.getEntry(i);
245             final byte[] rawPath = e.getRawPath();
246 
247             // Ensure that there are no entries under the newDir; we have a conflict otherwise.
248             if (rawNewDir != null) {
249                 boolean conflict = true;
250                 if (rawPath.length > rawNewDir.length) {
251                     // Check if there is a file whose path starts with 'newDir'.
252                     for (int j = 0; j < rawNewDir.length; j++) {
253                         if (rawNewDir[j] != rawPath[j]) {
254                             conflict = false;
255                             break;
256                         }
257                     }
258                 } else if (rawPath.length == rawNewDir.length - 1) {
259                     // Check if there is a file whose path is exactly same with newDir without trailing '/'.
260                     for (int j = 0; j < rawNewDir.length - 1; j++) {
261                         if (rawNewDir[j] != rawPath[j]) {
262                             conflict = false;
263                             break;
264                         }
265                     }
266                 } else {
267                     conflict = false;
268                 }
269 
270                 if (conflict) {
271                     throw new ChangeConflictException("target directory exists already: " + change);
272                 }
273             }
274 
275             // Skip the entries that do not belong to the oldDir.
276             if (rawPath.length <= rawOldDir.length) {
277                 continue;
278             }
279             for (int j = 0; j < rawOldDir.length; j++) {
280                 if (rawOldDir[j] != rawPath[j]) {
281                     continue loop;
282                 }
283             }
284 
285             // Do not create an editor until we find an entry to rename/remove.
286             // We can tell if there was any matching entries or not from the nullness of editor later.
287             if (editor == null) {
288                 editor = dirCache.editor();
289                 editor.add(new DeleteTree(oldDir));
290                 if (newDir == null) {
291                     // Recursive removal
292                     break;
293                 }
294             }
295 
296             assert newDir != null; // We should get here only when it's a recursive rename.
297 
298             final String oldPath = e.getPathString();
299             final String newPath = newDir + oldPath.substring(oldDir.length());
300             editor.add(new CopyOldEntry(newPath, e));
301         }
302 
303         if (editor != null) {
304             editor.finish();
305             return true;
306         } else {
307             return false;
308         }
309     }
310 
311     private static void reportNonExistentEntry(Change<?> change) {
312         throw new ChangeConflictException("non-existent file/directory: " + change);
313     }

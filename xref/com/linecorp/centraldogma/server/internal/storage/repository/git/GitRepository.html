195     /**
196      * Creates a new Git-backed repository.
197      *
198      * @param repoDir the location of this repository
199      * @param repositoryWorker the {@link Executor} which will perform the blocking repository operations
200      * @param creationTimeMillis the creation time
201      * @param author the user who initiated the creation of this repository
202      *
203      * @throws StorageException if failed to create a new repository
204      */
205     GitRepository(Project parent, File repoDir, Executor repositoryWorker,
206                   long creationTimeMillis, Author author, @Nullable RepositoryCache cache) {
207 
208         this.parent = requireNonNull(parent, "parent");
209         name = requireNonNull(repoDir, "repoDir").getName();
210         this.repositoryWorker = requireNonNull(repositoryWorker, "repositoryWorker");
211         this.creationTimeMillis = creationTimeMillis;
212         this.author = requireNonNull(author, "author");
213         this.cache = cache;
214 
215         final RepositoryBuilder repositoryBuilder = new RepositoryBuilder().setGitDir(repoDir).setBare();
216         boolean success = false;
217         try {
218             // Create an empty repository with format version 0 first.
219             try (org.eclipse.jgit.lib.Repository initRepo = repositoryBuilder.build()) {
220                 if (exist(repoDir)) {
221                     throw new StorageException(
222                             "failed to create a repository at: " + repoDir + " (exists already)");
223                 }
224                 initRepo.create(true);
225 
226                 // Save the initial default settings.
227                 JGitUtil.applyDefaultsAndSave(initRepo.getConfig());
228             }
229 
230             // Re-open the repository with the updated settings.
231             jGitRepository = new RepositoryBuilder().setGitDir(repoDir).build();
232 
233             // Initialize the master branch.
234             final RefUpdate head = jGitRepository.updateRef(Constants.HEAD);
235             head.disableRefLog();
236             head.link(Constants.R_HEADS + Constants.MASTER);
237 
238             // Initialize the commit ID database.
239             commitIdDatabase = new CommitIdDatabase(jGitRepository);
240 
241             // Insert the initial commit into the master branch.
242             commit0(null, Revision.INIT, creationTimeMillis, author,
243                     "Create a new repository", "", Markup.PLAINTEXT,
244                     Collections.emptyList(), true);
245 
246             headRevision = Revision.INIT;
247             success = true;
248         } catch (IOException e) {
249             throw new StorageException("failed to create a repository at: " + repoDir, e);
250         } finally {
251             if (!success) {
252                 internalClose();
253                 // Failed to create a repository. Remove any cruft so that it is not loaded on the next run.
254                 deleteCruft(repoDir);
255             }
256         }
257     }

259     /**
260      * Opens an existing Git-backed repository.
261      *
262      * @param repoDir the location of this repository
263      * @param repositoryWorker the {@link Executor} which will perform the blocking repository operations
264      *
265      * @throws StorageException if failed to open the repository at the specified location
266      */
267     GitRepository(Project parent, File repoDir, Executor repositoryWorker, @Nullable RepositoryCache cache) {
268         this.parent = requireNonNull(parent, "parent");
269         name = requireNonNull(repoDir, "repoDir").getName();
270         this.repositoryWorker = requireNonNull(repositoryWorker, "repositoryWorker");
271         this.cache = cache;
272 
273         final RepositoryBuilder repositoryBuilder = new RepositoryBuilder().setGitDir(repoDir).setBare();
274         try {
275             jGitRepository = repositoryBuilder.build();
276             if (!exist(repoDir)) {
277                 throw new RepositoryNotFoundException(repoDir.toString());
278             }
279 
280             // Retrieve the repository format.
281             final int formatVersion = jGitRepository.getConfig().getInt(
282                     CONFIG_CORE_SECTION, null, CONFIG_KEY_REPO_FORMAT_VERSION, 0);
283             if (formatVersion != JGitUtil.REPO_FORMAT_VERSION) {
284                 throw new StorageException("unsupported repository format version: " + formatVersion +
285                                            " (expected: " + JGitUtil.REPO_FORMAT_VERSION + ')');
286             }
287 
288             // Update the default settings if necessary.
289             JGitUtil.applyDefaultsAndSave(jGitRepository.getConfig());
290         } catch (IOException e) {
291             throw new StorageException("failed to open a repository at: " + repoDir, e);
292         }
293 
294         boolean success = false;
295         try {
296             headRevision = uncachedHeadRevision();
297             commitIdDatabase = new CommitIdDatabase(jGitRepository);
298             if (!headRevision.equals(commitIdDatabase.headRevision())) {
299                 commitIdDatabase.rebuild(jGitRepository);
300                 assert headRevision.equals(commitIdDatabase.headRevision());
301             }
302             final Commit initialCommit = blockingHistory(Revision.INIT, Revision.INIT, ALL_PATH, 1).get(0);
303             creationTimeMillis = initialCommit.when();
304             author = initialCommit.author();
305             success = true;
306         } finally {
307             if (!success) {
308                 internalClose();
309             }
310         }
311     }

326     /**
327      * Waits until all pending operations are complete and closes this repository.
328      *
329      * @param failureCauseSupplier the {@link Supplier} that creates a new {@link CentralDogmaException}
330      *                             which will be used to fail the operations issued after this method is called
331      */
332     void close(Supplier<CentralDogmaException> failureCauseSupplier) {
333         requireNonNull(failureCauseSupplier, "failureCauseSupplier");
334         if (closePending.compareAndSet(null, failureCauseSupplier)) {
335             repositoryWorker.execute(() -> {
336                 // MUST acquire gcLock first to prevent a dead lock
337                 rwLock.writeLock().lock();
338                 try {
339                     if (commitIdDatabase != null) {
340                         try {
341                             commitIdDatabase.close();
342                         } catch (Exception e) {
343                             logger.warn("Failed to close a commitId database:", e);
344                         }
345                     }
346 
347                     if (jGitRepository != null) {
348                         try {
349                             jGitRepository.close();
350                         } catch (Exception e) {
351                             logger.warn("Failed to close a Git repository: {}",
352                                         jGitRepository.getDirectory(), e);
353                         }
354                     }
355                 } finally {
356                     try {
357                         rwLock.writeLock().unlock();
358                     } finally {
359                         commitWatchers.close(failureCauseSupplier);
360                         closeFuture.complete(null);
361                     }
362                 }
363             });
364         }
365 
366         closeFuture.join();
367     }

398     private static Revision normalizeNow(Revision revision, int baseMajor) {
399         requireNonNull(revision, "revision");
400 
401         int major = revision.major();
402 
403         if (major >= 0) {
404             if (major > baseMajor) {
405                 throw new RevisionNotFoundException(revision);
406             }
407         } else {
408             major = baseMajor + major + 1;
409             if (major <= 0) {
410                 throw new RevisionNotFoundException(revision);
411             }
412         }
413 
414         // Create a new instance only when necessary.
415         if (revision.major() == major) {
416             return revision;
417         } else {
418             return new Revision(major);
419         }
420     }

438     private Map<String, Entry<?>> blockingFind(
439             Revision revision, String pathPattern, Map<FindOption<?>, ?> options) {
440 
441         requireNonNull(pathPattern, "pathPattern");
442         requireNonNull(revision, "revision");
443         requireNonNull(options, "options");
444 
445         final Revision normRevision = normalizeNow(revision);
446         final boolean fetchContent = FindOption.FETCH_CONTENT.get(options);
447         final int maxEntries = FindOption.MAX_ENTRIES.get(options);
448 
449         readLock();
450         try (ObjectReader reader = jGitRepository.newObjectReader();
451              TreeWalk treeWalk = new TreeWalk(reader);
452              RevWalk revWalk = newRevWalk(reader)) {
453 
454             // Query on a non-exist revision will return empty result.
455             final Revision headRevision = cachedHeadRevision();
456             if (normRevision.compareTo(headRevision) > 0) {
457                 return Collections.emptyMap();
458             }
459 
460             if ("/".equals(pathPattern)) {
461                 return Collections.singletonMap(pathPattern, Entry.ofDirectory(normRevision, "/"));
462             }
463 
464             final Map<String, Entry<?>> result = new LinkedHashMap<>();
465             final ObjectId commitId = commitIdDatabase.get(normRevision);
466             final RevCommit revCommit = revWalk.parseCommit(commitId);
467             final PathPatternFilter filter = PathPatternFilter.of(pathPattern);
468 
469             final RevTree revTree = revCommit.getTree();
470             treeWalk.addTree(revTree.getId());
471             while (treeWalk.next() && result.size() < maxEntries) {
472                 final boolean matches = filter.matches(treeWalk);
473                 final String path = '/' + treeWalk.getPathString();
474 
475                 // Recurse into a directory if necessary.
476                 if (treeWalk.isSubtree()) {
477                     if (matches) {
478                         // Add the directory itself to the result set if its path matches the pattern.
479                         result.put(path, Entry.ofDirectory(normRevision, path));
480                     }
481 
482                     treeWalk.enterSubtree();
483                     continue;
484                 }
485 
486                 if (!matches) {
487                     continue;
488                 }
489 
490                 // Build an entry as requested.
491                 final Entry<?> entry;
492                 final EntryType entryType = EntryType.guessFromPath(path);
493                 if (fetchContent) {
494                     final byte[] content = reader.open(treeWalk.getObjectId(0)).getBytes();
495                     switch (entryType) {
496                         case JSON:
497                             final JsonNode jsonNode = Jackson.readTree(content);
498                             entry = Entry.ofJson(normRevision, path, jsonNode);
499                             break;
500                         case TEXT:
501                             final String strVal = sanitizeText(new String(content, UTF_8));
502                             entry = Entry.ofText(normRevision, path, strVal);
503                             break;
504                         default:
505                             throw new Error("unexpected entry type: " + entryType);
506                     }
507                 } else {
508                     switch (entryType) {
509                         case JSON:
510                             entry = Entry.ofJson(normRevision, path, Jackson.nullNode);
511                             break;
512                         case TEXT:
513                             entry = Entry.ofText(normRevision, path, "");
514                             break;
515                         default:
516                             throw new Error("unexpected entry type: " + entryType);
517                     }
518                 }
519 
520                 result.put(path, entry);
521             }
522 
523             return Util.unsafeCast(result);
524         } catch (CentralDogmaException | IllegalArgumentException e) {
525             throw e;
526         } catch (Exception e) {
527             throw new StorageException(
528                     "failed to get data from '" + parent.name() + '/' + name + "' at " + pathPattern +
529                     " for " + revision, e);
530         } finally {
531             readUnlock();
532         }
533     }

546     private List<Commit> blockingHistory(Revision from, Revision to, String pathPattern, int maxCommits) {
547         requireNonNull(pathPattern, "pathPattern");
548         requireNonNull(from, "from");
549         requireNonNull(to, "to");
550         if (maxCommits <= 0) {
551             throw new IllegalArgumentException("maxCommits: " + maxCommits + " (expected: > 0)");
552         }
553 
554         maxCommits = Math.min(maxCommits, MAX_MAX_COMMITS);
555 
556         final RevisionRange range = normalizeNow(from, to);
557         final RevisionRange descendingRange = range.toDescending();
558 
559         // At this point, we are sure: from.major >= to.major
560         readLock();
561         try (RevWalk revWalk = newRevWalk()) {
562             final ObjectIdOwnerMap<?> revWalkInternalMap =
563                     (ObjectIdOwnerMap<?>) revWalkObjectsField.get(revWalk);
564 
565             final ObjectId fromCommitId = commitIdDatabase.get(descendingRange.from());
566             final ObjectId toCommitId = commitIdDatabase.get(descendingRange.to());
567 
568             revWalk.markStart(revWalk.parseCommit(fromCommitId));
569             revWalk.setRetainBody(false);
570 
571             // Instead of relying on RevWalk to filter the commits,
572             // we let RevWalk yield all commits so we can:
573             // - Have more control on when iteration should be stopped.
574             //   (A single Iterator.next() doesn't take long.)
575             // - Clean up the internal map as early as possible.
576             final RevFilter filter = new TreeRevFilter(revWalk, AndTreeFilter.create(
577                     TreeFilter.ANY_DIFF, PathPatternFilter.of(pathPattern)));
578 
579             // Search up to 1000 commits when maxCommits <= 100.
580             // Search up to (maxCommits * 10) commits when 100 < maxCommits <= 1000.
581             final int maxNumProcessedCommits = Math.max(maxCommits * 10, MAX_MAX_COMMITS);
582 
583             final List<Commit> commitList = new ArrayList<>();
584             int numProcessedCommits = 0;
585             for (RevCommit revCommit : revWalk) {
586                 numProcessedCommits++;
587 
588                 if (filter.include(revWalk, revCommit)) {
589                     revWalk.parseBody(revCommit);
590                     commitList.add(toCommit(revCommit));
591                     revCommit.disposeBody();
592                 }
593 
594                 if (revCommit.getId().equals(toCommitId) ||
595                     commitList.size() >= maxCommits ||
596                     // Prevent from iterating for too long.
597                     numProcessedCommits >= maxNumProcessedCommits) {
598                     break;
599                 }
600 
601                 // Clear the internal lookup table of RevWalk to reduce the memory usage.
602                 // This is safe because we have linear history and traverse in one direction.
603                 if (numProcessedCommits % 16 == 0) {
604                     revWalkInternalMap.clear();
605                 }
606             }
607 
608             // Include the initial empty commit only when the caller specified
609             // the initial revision (1) in the range and the pathPattern contains '/**'.
610             if (commitList.size() < maxCommits &&
611                 descendingRange.to().major() == 1 &&
612                 pathPattern.contains(ALL_PATH)) {
613                 try (RevWalk tmpRevWalk = newRevWalk()) {
614                     final RevCommit lastRevCommit = tmpRevWalk.parseCommit(toCommitId);
615                     commitList.add(toCommit(lastRevCommit));
616                 }
617             }
618 
619             if (!descendingRange.equals(range)) { // from and to is swapped so reverse the list.
620                 Collections.reverse(commitList);
621             }
622 
623             return commitList;
624         } catch (CentralDogmaException e) {
625             throw e;
626         } catch (Exception e) {
627             throw new StorageException(
628                     "failed to retrieve the history: " + parent.name() + '/' + name +
629                     " (" + pathPattern + ", " + from + ".." + to + ')', e);
630         } finally {
631             readUnlock();
632         }
633     }

635     private static Commit toCommit(RevCommit revCommit) {
636         final Author author;
637         final PersonIdent committerIdent = revCommit.getCommitterIdent();
638         final long when;
639         if (committerIdent == null) {
640             author = Author.UNKNOWN;
641             when = 0;
642         } else {
643             author = new Author(committerIdent.getName(), committerIdent.getEmailAddress());
644             when = committerIdent.getWhen().getTime();
645         }
646 
647         try {
648             return CommitUtil.newCommit(author, when, revCommit.getFullMessage());
649         } catch (Exception e) {
650             throw new StorageException("failed to create a Commit", e);
651         }

668             requireNonNull(to, "to");

741     private Map<String, Change<?>> toChangeMap(List<DiffEntry> diffEntryList) {
742         try (ObjectReader reader = jGitRepository.newObjectReader()) {
743             final Map<String, Change<?>> changeMap = new LinkedHashMap<>();
744 
745             for (DiffEntry diffEntry : diffEntryList) {
746                 final String oldPath = '/' + diffEntry.getOldPath();
747                 final String newPath = '/' + diffEntry.getNewPath();
748 
749                 switch (diffEntry.getChangeType()) {
750                     case MODIFY:
751                         final EntryType oldEntryType = EntryType.guessFromPath(oldPath);
752                         switch (oldEntryType) {
753                             case JSON:
754                                 if (!oldPath.equals(newPath)) {
755                                     putChange(changeMap, oldPath, Change.ofRename(oldPath, newPath));
756                                 }
757 
758                                 final JsonNode oldJsonNode =
759                                         Jackson.readTree(
760                                                 reader.open(diffEntry.getOldId().toObjectId()).getBytes());
761                                 final JsonNode newJsonNode =
762                                         Jackson.readTree(
763                                                 reader.open(diffEntry.getNewId().toObjectId()).getBytes());
764                                 final JsonPatch patch =
765                                         JsonPatch.generate(oldJsonNode, newJsonNode, ReplaceMode.SAFE);
766 
767                                 if (!patch.isEmpty()) {
768                                     putChange(changeMap, newPath,
769                                               Change.ofJsonPatch(newPath, Jackson.valueToTree(patch)));
770                                 }
771                                 break;
772                             case TEXT:
773                                 final String oldText = sanitizeText(new String(
774                                         reader.open(diffEntry.getOldId().toObjectId()).getBytes(), UTF_8));
775 
776                                 final String newText = sanitizeText(new String(
777                                         reader.open(diffEntry.getNewId().toObjectId()).getBytes(), UTF_8));
778 
779                                 if (!oldPath.equals(newPath)) {
780                                     putChange(changeMap, oldPath, Change.ofRename(oldPath, newPath));
781                                 }
782 
783                                 if (!oldText.equals(newText)) {
784                                     putChange(changeMap, newPath,
785                                               Change.ofTextPatch(newPath, oldText, newText));
786                                 }
787                                 break;
788                             default:
789                                 throw new Error("unexpected old entry type: " + oldEntryType);
790                         }
791                         break;
792                     case ADD:
793                         final EntryType newEntryType = EntryType.guessFromPath(newPath);
794                         switch (newEntryType) {
795                             case JSON: {
796                                 final JsonNode jsonNode = Jackson.readTree(
797                                         reader.open(diffEntry.getNewId().toObjectId()).getBytes());
798 
799                                 putChange(changeMap, newPath, Change.ofJsonUpsert(newPath, jsonNode));
800                                 break;
801                             }
802                             case TEXT: {
803                                 final String text = sanitizeText(new String(
804                                         reader.open(diffEntry.getNewId().toObjectId()).getBytes(), UTF_8));
805 
806                                 putChange(changeMap, newPath, Change.ofTextUpsert(newPath, text));
807                                 break;
808                             }
809                             default:
810                                 throw new Error("unexpected new entry type: " + newEntryType);
811                         }
812                         break;
813                     case DELETE:
814                         putChange(changeMap, oldPath, Change.ofRemoval(oldPath));
815                         break;
816                     default:
817                         throw new Error();
818                 }
819             }
820             return changeMap;
821         } catch (Exception e) {
822             throw new StorageException("failed to convert list of DiffEntry to Changes map", e);
823         }
824     }

850     private CommitResult blockingCommit(
851             Revision baseRevision, long commitTimeMillis, Author author, String summary,
852             String detail, Markup markup, Iterable<Change<?>> changes, boolean allowEmptyCommit,
853             boolean directExecution) {
854 
855         requireNonNull(baseRevision, "baseRevision");
856 
857         final RevisionAndEntries res;
858         final Iterable<Change<?>> applyingChanges;
859         writeLock();
860         try {
861             final Revision normBaseRevision = normalizeNow(baseRevision);
862             final Revision headRevision = cachedHeadRevision();
863             if (headRevision.major() != normBaseRevision.major()) {
864                 throw new ChangeConflictException(
865                         "invalid baseRevision: " + baseRevision + " (expected: " + headRevision +
866                         " or equivalent)");
867             }
868 
869             if (directExecution) {
870                 applyingChanges = blockingPreviewDiff(normBaseRevision, changes).values();
871             } else {
872                 applyingChanges = changes;
873             }
874             res = commit0(headRevision, headRevision.forward(1), commitTimeMillis,
875                           author, summary, detail, markup, applyingChanges, allowEmptyCommit);
876 
877             this.headRevision = res.revision;
878         } finally {
879             writeUnLock();
880         }
881 
882         // Note that the notification is made while no lock is held to avoid the risk of a dead lock.
883         notifyWatchers(res.revision, res.diffEntries);
884         return CommitResult.of(res.revision, applyingChanges);
885     }

887     private RevisionAndEntries commit0(@Nullable Revision prevRevision, Revision nextRevision,
888                                        long commitTimeMillis, Author author, String summary,
889                                        String detail, Markup markup,
890                                        Iterable<Change<?>> changes, boolean allowEmpty) {
891 
892         requireNonNull(author, "author");
893         requireNonNull(summary, "summary");
894         requireNonNull(changes, "changes");
895         requireNonNull(detail, "detail");
896         requireNonNull(markup, "markup");
897 
898         assert prevRevision == null || prevRevision.major() > 0;
899         assert nextRevision.major() > 0;
900 
901         try (ObjectInserter inserter = jGitRepository.newObjectInserter();
902              ObjectReader reader = jGitRepository.newObjectReader();
903              RevWalk revWalk = newRevWalk(reader)) {
904 
905             final ObjectId prevTreeId = prevRevision != null ? toTree(revWalk, prevRevision) : null;
906 
907             // The staging area that keeps the entries of the new tree.
908             // It starts with the entries of the tree at the prevRevision (or with no entries if the
909             // prevRevision is the initial commit), and then this method will apply the requested changes
910             // to build the new tree.
911             final DirCache dirCache = DirCache.newInCore();
912 
913             // Apply the changes and retrieve the list of the affected files.
914             final int numEdits = applyChanges(prevRevision, prevTreeId, dirCache, changes);
915 
916             // Reject empty commit if necessary.
917             final List<DiffEntry> diffEntries;
918             boolean isEmpty = numEdits == 0;
919             if (!isEmpty) {
920                 // Even if there are edits, the resulting tree might be identical with the previous tree.
921                 final CanonicalTreeParser p = new CanonicalTreeParser();
922                 p.reset(reader, prevTreeId);
923                 final DiffFormatter diffFormatter = new DiffFormatter(null);
924                 diffFormatter.setRepository(jGitRepository);
925                 diffEntries = diffFormatter.scan(p, new DirCacheIterator(dirCache));
926                 isEmpty = diffEntries.isEmpty();
927             } else {
928                 diffEntries = ImmutableList.of();
929             }
930 
931             if (!allowEmpty && isEmpty) {
932                 throw new RedundantChangeException(
933                         "changes did not change anything in " + parent().name() + '/' + name() +
934                         " at revision " + (prevRevision != null ? prevRevision.major() : 0) +
935                         ": " + changes);
936             }
937 
938             // flush the current index to repository and get the result tree object id.
939             final ObjectId nextTreeId = dirCache.writeTree(inserter);
940 
941             // build a commit object
942             final PersonIdent personIdent = new PersonIdent(author.name(), author.email(),
943                                                             commitTimeMillis / 1000L * 1000L, 0);
944 
945             final CommitBuilder commitBuilder = new CommitBuilder();
946 
947             commitBuilder.setAuthor(personIdent);
948             commitBuilder.setCommitter(personIdent);
949             commitBuilder.setTreeId(nextTreeId);
950             commitBuilder.setEncoding(UTF_8);
951 
952             // Write summary, detail and revision to commit's message as JSON format.
953             commitBuilder.setMessage(CommitUtil.toJsonString(summary, detail, markup, nextRevision));
954 
955             // if the head commit exists, use it as the parent commit.
956             if (prevRevision != null) {
957                 commitBuilder.setParentId(commitIdDatabase.get(prevRevision));
958             }
959 
960             final ObjectId nextCommitId = inserter.insert(commitBuilder);
961             inserter.flush();
962 
963             // tagging the revision object, for history lookup purpose.
964             commitIdDatabase.put(nextRevision, nextCommitId);
965             doRefUpdate(revWalk, R_HEADS_MASTER, nextCommitId);
966 
967             return new RevisionAndEntries(nextRevision, diffEntries);
968         } catch (CentralDogmaException | IllegalArgumentException e) {
969             throw e;
970         } catch (Exception e) {
971             throw new StorageException("failed to push at '" + parent.name() + '/' + name + '\'', e);
972         }
973     }

975     private int applyChanges(@Nullable Revision baseRevision, @Nullable ObjectId baseTreeId, DirCache dirCache,
976                              Iterable<Change<?>> changes) {
977 
978         int numEdits = 0;
979 
980         try (ObjectInserter inserter = jGitRepository.newObjectInserter();
981              ObjectReader reader = jGitRepository.newObjectReader()) {
982 
983             if (baseTreeId != null) {
984                 // the DirCacheBuilder is to used for doing update operations on the given DirCache object
985                 final DirCacheBuilder builder = dirCache.builder();
986 
987                 // Add the tree object indicated by the prevRevision to the temporary DirCache object.
988                 builder.addTree(EMPTY_BYTE, 0, reader, baseTreeId);
989                 builder.finish();
990             }
991 
992             // loop over the specified changes.
993             for (Change<?> change : changes) {
994                 final String changePath = change.path().substring(1); // Strip the leading '/'.
995                 final DirCacheEntry oldEntry = dirCache.getEntry(changePath);
996                 final byte[] oldContent = oldEntry != null ? reader.open(oldEntry.getObjectId()).getBytes()
997                                                            : null;
998 
999                 switch (change.type()) {
1000                     case UPSERT_JSON: {
1001                         final JsonNode oldJsonNode = oldContent != null ? Jackson.readTree(oldContent) : null;
1002                         final JsonNode newJsonNode = firstNonNull((JsonNode) change.content(),
1003                                                                   JsonNodeFactory.instance.nullNode());
1004 
1005                         // Upsert only when the contents are really different.
1006                         if (!Objects.equals(newJsonNode, oldJsonNode)) {
1007                             applyPathEdit(dirCache, new InsertJson(changePath, inserter, newJsonNode));
1008                             numEdits++;
1009                         }
1010                         break;
1011                     }
1012                     case UPSERT_TEXT: {
1013                         final String sanitizedOldText;
1014                         if (oldContent != null) {
1015                             sanitizedOldText = sanitizeText(new String(oldContent, UTF_8));
1016                         } else {
1017                             sanitizedOldText = null;
1018                         }
1019 
1020                         final String sanitizedNewText = sanitizeText(change.contentAsText());
1021 
1022                         // Upsert only when the contents are really different.
1023                         if (!sanitizedNewText.equals(sanitizedOldText)) {
1024                             applyPathEdit(dirCache, new InsertText(changePath, inserter, sanitizedNewText));
1025                             numEdits++;
1026                         }
1027                         break;
1028                     }
1029                     case REMOVE:
1030                         if (oldEntry != null) {
1031                             applyPathEdit(dirCache, new DeletePath(changePath));
1032                             numEdits++;
1033                             break;
1034                         }
1035 
1036                         // The path might be a directory.
1037                         if (applyDirectoryEdits(dirCache, changePath, null, change)) {
1038                             numEdits++;
1039                         } else {
1040                             // Was not a directory either; conflict.
1041                             reportNonExistentEntry(change);
1042                             break;
1043                         }
1044                         break;
1045                     case RENAME: {
1046                         final String newPath =
1047                                 ((String) change.content()).substring(1); // Strip the leading '/'.
1048 
1049                         if (dirCache.getEntry(newPath) != null) {
1050                             throw new ChangeConflictException("a file exists at the target path: " + change);
1051                         }
1052 
1053                         if (oldEntry != null) {
1054                             if (changePath.equals(newPath)) {
1055                                 // Redundant rename request - old path and new path are same.
1056                                 break;
1057                             }
1058 
1059                             final DirCacheEditor editor = dirCache.editor();
1060                             editor.add(new DeletePath(changePath));
1061                             editor.add(new CopyOldEntry(newPath, oldEntry));
1062                             editor.finish();
1063                             numEdits++;
1064                             break;
1065                         }
1066 
1067                         // The path might be a directory.
1068                         if (applyDirectoryEdits(dirCache, changePath, newPath, change)) {
1069                             numEdits++;
1070                         } else {
1071                             // Was not a directory either; conflict.
1072                             reportNonExistentEntry(change);
1073                         }
1074                         break;
1075                     }
1076                     case APPLY_JSON_PATCH: {
1077                         final JsonNode oldJsonNode;
1078                         if (oldContent != null) {
1079                             oldJsonNode = Jackson.readTree(oldContent);
1080                         } else {
1081                             oldJsonNode = Jackson.nullNode;
1082                         }
1083 
1084                         final JsonNode newJsonNode;
1085                         try {
1086                             newJsonNode = JsonPatch.fromJson((JsonNode) change.content()).apply(oldJsonNode);
1087                         } catch (Exception e) {
1088                             throw new ChangeConflictException("failed to apply JSON patch: " + change, e);
1089                         }
1090 
1091                         // Apply only when the contents are really different.
1092                         if (!newJsonNode.equals(oldJsonNode)) {
1093                             applyPathEdit(dirCache, new InsertJson(changePath, inserter, newJsonNode));
1094                             numEdits++;
1095                         }
1096                         break;
1097                     }
1098                     case APPLY_TEXT_PATCH:
1099                         final Patch<String> patch = DiffUtils.parseUnifiedDiff(
1100                                 Util.stringToLines(sanitizeText((String) change.content())));
1101 
1102                         final String sanitizedOldText;
1103                         final List<String> sanitizedOldTextLines;
1104                         if (oldContent != null) {
1105                             sanitizedOldText = sanitizeText(new String(oldContent, UTF_8));
1106                             sanitizedOldTextLines = Util.stringToLines(sanitizedOldText);
1107                         } else {
1108                             sanitizedOldText = null;
1109                             sanitizedOldTextLines = Collections.emptyList();
1110                         }
1111 
1112                         final String newText;
1113                         try {
1114                             final List<String> newTextLines = DiffUtils.patch(sanitizedOldTextLines, patch);
1115                             if (newTextLines.isEmpty()) {
1116                                 newText = "";
1117                             } else {
1118                                 final StringJoiner joiner = new StringJoiner("\n", "", "\n");
1119                                 for (String line : newTextLines) {
1120                                     joiner.add(line);
1121                                 }
1122                                 newText = joiner.toString();
1123                             }
1124                         } catch (Exception e) {
1125                             throw new ChangeConflictException("failed to apply text patch: " + change, e);
1126                         }
1127 
1128                         // Apply only when the contents are really different.
1129                         if (!newText.equals(sanitizedOldText)) {
1130                             applyPathEdit(dirCache, new InsertText(changePath, inserter, newText));
1131                             numEdits++;
1132                         }
1133                         break;
1134                 }
1135             }
1136         } catch (CentralDogmaException | IllegalArgumentException e) {
1137             throw e;
1138         } catch (Exception e) {
1139             throw new StorageException("failed to apply changes on revision " + baseRevision, e);
1140         }
1141         return numEdits;
1142     }

1167     /**
1168      * Applies recursive directory edits.
1169      *
1170      * @param oldDir the path to the directory to make a recursive change
1171      * @param newDir the path to the renamed directory, or {@code null} to remove the directory.
1172      *
1173      * @return {@code true} if any edits were made to {@code dirCache}, {@code false} otherwise
1174      */
1175     private static boolean applyDirectoryEdits(DirCache dirCache,
1176                                                String oldDir, @Nullable String newDir, Change<?> change) {
1177 
1178         if (!oldDir.endsWith("/")) {
1179             oldDir += '/';
1180         }
1181         if (newDir != null && !newDir.endsWith("/")) {
1182             newDir += '/';
1183         }
1184 
1185         final byte[] rawOldDir = Constants.encode(oldDir);
1186         final byte[] rawNewDir = newDir != null ? Constants.encode(newDir) : null;
1187         final int numEntries = dirCache.getEntryCount();
1188         DirCacheEditor editor = null;
1189 
1190         loop:
1191         for (int i = 0; i < numEntries; i++) {
1192             final DirCacheEntry e = dirCache.getEntry(i);
1193             final byte[] rawPath = e.getRawPath();
1194 
1195             // Ensure that there are no entries under the newDir; we have a conflict otherwise.
1196             if (rawNewDir != null) {
1197                 boolean conflict = true;
1198                 if (rawPath.length > rawNewDir.length) {
1199                     // Check if there is a file whose path starts with 'newDir'.
1200                     for (int j = 0; j < rawNewDir.length; j++) {
1201                         if (rawNewDir[j] != rawPath[j]) {
1202                             conflict = false;
1203                             break;
1204                         }
1205                     }
1206                 } else if (rawPath.length == rawNewDir.length - 1) {
1207                     // Check if there is a file whose path is exactly same with newDir without trailing '/'.
1208                     for (int j = 0; j < rawNewDir.length - 1; j++) {
1209                         if (rawNewDir[j] != rawPath[j]) {
1210                             conflict = false;
1211                             break;
1212                         }
1213                     }
1214                 } else {
1215                     conflict = false;
1216                 }
1217 
1218                 if (conflict) {
1219                     throw new ChangeConflictException("target directory exists already: " + change);
1220                 }
1221             }
1222 
1223             // Skip the entries that do not belong to the oldDir.
1224             if (rawPath.length <= rawOldDir.length) {
1225                 continue;
1226             }
1227             for (int j = 0; j < rawOldDir.length; j++) {
1228                 if (rawOldDir[j] != rawPath[j]) {
1229                     continue loop;
1230                 }
1231             }
1232 
1233             // Do not create an editor until we find an entry to rename/remove.
1234             // We can tell if there was any matching entries or not from the nullness of editor later.
1235             if (editor == null) {
1236                 editor = dirCache.editor();
1237                 editor.add(new DeleteTree(oldDir));
1238                 if (newDir == null) {
1239                     // Recursive removal
1240                     break;
1241                 }
1242             }
1243 
1244             assert newDir != null; // We should get here only when it's a recursive rename.
1245 
1246             final String oldPath = e.getPathString();
1247             final String newPath = newDir + oldPath.substring(oldDir.length());
1248             editor.add(new CopyOldEntry(newPath, e));
1249         }
1250 
1251         if (editor != null) {
1252             editor.finish();
1253             return true;
1254         } else {
1255             return false;
1256         }
1257     }

1296             failFastIfTimedOut(this, logger, ctx, "findLatestRevision", lastKnownRevision, pathPattern);
1297             return blockingFindLatestRevision(lastKnownRevision, pathPattern, errorOnEntryNotFound);
1298         }, repositoryWorker);
1299     }

1301     @Nullable
1302     private Revision blockingFindLatestRevision(Revision lastKnownRevision, String pathPattern,
1303                                                 boolean errorOnEntryNotFound) {
1304         final RevisionRange range = normalizeNow(lastKnownRevision, Revision.HEAD);
1305         if (range.from().equals(range.to())) {
1306             // Empty range.
1307             if (!errorOnEntryNotFound) {
1308                 return null;
1309             }
1310             // We have to check if we have the entry.
1311             final Map<String, Entry<?>> entries =
1312                     blockingFind(range.to(), pathPattern, FindOptions.FIND_ONE_WITHOUT_CONTENT);
1313             if (!entries.isEmpty()) {
1314                 // We have the entry so just return null because there's no change.
1315                 return null;
1316             }
1317             throw new EntryNotFoundException(lastKnownRevision, pathPattern);
1318         }
1319 
1320         if (range.from().major() == 1) {
1321             // Fast path: no need to compare because we are sure there is nothing at revision 1.
1322             final Map<String, Entry<?>> entries =
1323                     blockingFind(range.to(), pathPattern, FindOptions.FIND_ONE_WITHOUT_CONTENT);
1324             if (entries.isEmpty()) {
1325                 if (!errorOnEntryNotFound) {
1326                     return null;
1327                 }
1328                 throw new EntryNotFoundException(lastKnownRevision, pathPattern);
1329             } else {
1330                 return range.to();
1331             }
1332         }
1333 
1334         // Slow path: compare the two trees.
1335         final PathPatternFilter filter = PathPatternFilter.of(pathPattern);
1336         // Convert the revisions to Git trees.
1337         final List<DiffEntry> diffEntries;
1338         readLock();
1339         try (RevWalk revWalk = newRevWalk()) {
1340             final RevTree treeA = toTree(revWalk, range.from());
1341             final RevTree treeB = toTree(revWalk, range.to());
1342             diffEntries = blockingCompareTrees(treeA, treeB);
1343         } finally {
1344             readUnlock();
1345         }
1346 
1347         // Return the latest revision if the changes between the two trees contain the file.
1348         for (DiffEntry e : diffEntries) {
1349             final String path;
1350             switch (e.getChangeType()) {
1351                 case ADD:
1352                     path = e.getNewPath();
1353                     break;
1354                 case MODIFY:
1355                 case DELETE:
1356                     path = e.getOldPath();
1357                     break;
1358                 default:
1359                     throw new Error();
1360             }
1361 
1362             if (filter.matches(path)) {
1363                 return range.to();
1364             }
1365         }
1366 
1367         if (!errorOnEntryNotFound) {
1368             return null;
1369         }
1370         if (!blockingFind(range.to(), pathPattern, FindOptions.FIND_ONE_WITHOUT_CONTENT).isEmpty()) {
1371             // We have to make sure that the entry does not exist because the size of diffEntries can be 0
1372             // when the contents of range.from() and range.to() are identical. (e.g. add, remove and add again)
1373             return null;
1374         }
1375         throw new EntryNotFoundException(lastKnownRevision, pathPattern);
1376     }

1378     /**
1379      * Compares the two Git trees (with caching).
1380      */
1381     private List<DiffEntry> blockingCompareTrees(RevTree treeA, RevTree treeB) {
1382         if (cache == null) {
1383             return blockingCompareTreesUncached(treeA, treeB, TreeFilter.ALL);
1384         }
1385 
1386         final CacheableCompareTreesCall key = new CacheableCompareTreesCall(this, treeA, treeB);
1387         CompletableFuture<List<DiffEntry>> existingFuture = cache.getIfPresent(key);
1388         if (existingFuture != null) {
1389             final List<DiffEntry> existingDiffEntries = existingFuture.getNow(null);
1390             if (existingDiffEntries != null) {
1391                 // Cached already.
1392                 return existingDiffEntries;
1393             }
1394         }
1395 
1396         // Not cached yet. Acquire a lock so that we do not compare the same tree pairs simultaneously.
1397         final List<DiffEntry> newDiffEntries;
1398         final Lock lock = key.coarseGrainedLock();
1399         lock.lock();
1400         try {
1401             existingFuture = cache.getIfPresent(key);
1402             if (existingFuture != null) {
1403                 final List<DiffEntry> existingDiffEntries = existingFuture.getNow(null);
1404                 if (existingDiffEntries != null) {
1405                     // Other thread already put the entries to the cache before we acquire the lock.
1406                     return existingDiffEntries;
1407                 }
1408             }
1409 
1410             newDiffEntries = blockingCompareTreesUncached(treeA, treeB, TreeFilter.ALL);
1411             cache.put(key, newDiffEntries);
1412         } finally {
1413             lock.unlock();
1414         }
1415 
1416         logger.debug("Cache miss: {}", key);
1417         return newDiffEntries;
1418     }

1420     private List<DiffEntry> blockingCompareTreesUncached(@Nullable RevTree treeA,
1421                                                          @Nullable RevTree treeB,
1422                                                          TreeFilter filter) {
1423         readLock();
1424         try (DiffFormatter diffFormatter = new DiffFormatter(null)) {
1425             diffFormatter.setRepository(jGitRepository);
1426             diffFormatter.setPathFilter(filter);
1427             return ImmutableList.copyOf(diffFormatter.scan(treeA, treeB));
1428         } catch (IOException e) {
1429             throw new StorageException("failed to compare two trees: " + treeA + " vs. " + treeB, e);
1430         } finally {
1431             readUnlock();
1432         }
1433     }

1435     @Override
1436     public CompletableFuture<Revision> watch(Revision lastKnownRevision, String pathPattern,
1437                                              boolean errorOnEntryNotFound) {
1438         requireNonNull(lastKnownRevision, "lastKnownRevision");
1439         requireNonNull(pathPattern, "pathPattern");
1440 
1441         final ServiceRequestContext ctx = context();
1442         final Revision normLastKnownRevision = normalizeNow(lastKnownRevision);
1443         final CompletableFuture<Revision> future = new CompletableFuture<>();
1444         CompletableFuture.runAsync(() -> {
1445             failFastIfTimedOut(this, logger, ctx, "watch", lastKnownRevision, pathPattern);
1446             readLock();
1447             try {
1448                 // If lastKnownRevision is outdated already and the recent changes match,
1449                 // there's no need to watch.
1450                 final Revision latestRevision = blockingFindLatestRevision(normLastKnownRevision, pathPattern,
1451                                                                            errorOnEntryNotFound);
1452                 if (latestRevision != null) {
1453                     future.complete(latestRevision);
1454                 } else {
1455                     commitWatchers.add(normLastKnownRevision, pathPattern, future);
1456                 }
1457             } finally {
1458                 readUnlock();
1459             }
1460         }, repositoryWorker).exceptionally(cause -> {
1461             future.completeExceptionally(cause);
1462             return null;
1463         });
1464 
1465         return future;
1466     }

1564     /**
1565      * Clones this repository into a new one.
1566      */
1567     public void cloneTo(File newRepoDir, BiConsumer<Integer, Integer> progressListener) {
1568         requireNonNull(newRepoDir, "newRepoDir");
1569         requireNonNull(progressListener, "progressListener");
1570 
1571         final Revision endRevision = normalizeNow(Revision.HEAD);
1572         final GitRepository newRepo = new GitRepository(parent, newRepoDir, repositoryWorker,
1573                                                         creationTimeMillis(), author(), cache);
1574 
1575         progressListener.accept(1, endRevision.major());
1576         boolean success = false;
1577         try {
1578             // Replay all commits.
1579             Revision previousNonEmptyRevision = null;
1580             for (int i = 2; i <= endRevision.major();) {
1581                 // Fetch up to 16 commits at once.
1582                 final int batch = 16;
1583                 final List<Commit> commits = blockingHistory(
1584                         new Revision(i), new Revision(Math.min(endRevision.major(), i + batch - 1)),
1585                         Repository.ALL_PATH, batch);
1586                 checkState(!commits.isEmpty(), "empty commits");
1587 
1588                 if (previousNonEmptyRevision == null) {
1589                     previousNonEmptyRevision = commits.get(0).revision().backward(1);
1590                 }
1591                 for (Commit c : commits) {
1592                     final Revision revision = c.revision();
1593                     checkState(revision.major() == i,
1594                                "mismatching revision: %s (expected: %s)", revision.major(), i);
1595 
1596                     final Revision baseRevision = revision.backward(1);
1597                     final Collection<Change<?>> changes =
1598                             diff(previousNonEmptyRevision, revision, Repository.ALL_PATH).join().values();
1599 
1600                     try {
1601                         newRepo.blockingCommit(
1602                                 baseRevision, c.when(), c.author(), c.summary(), c.detail(), c.markup(),
1603                                 changes, /* allowEmptyCommit */ false, false);
1604                         previousNonEmptyRevision = revision;
1605                     } catch (RedundantChangeException e) {
1606                         // NB: We allow an empty commit here because an old version of Central Dogma had a bug
1607                         //     which allowed the creation of an empty commit.
1608                         newRepo.blockingCommit(
1609                                 baseRevision, c.when(), c.author(), c.summary(), c.detail(), c.markup(),
1610                                 changes, /* allowEmptyCommit */ true, false);
1611                     }
1612 
1613                     progressListener.accept(i, endRevision.major());
1614                     i++;
1615                 }
1616             }
1617 
1618             success = true;
1619         } finally {
1620             newRepo.internalClose();
1621             if (!success) {
1622                 deleteCruft(newRepoDir);
1623             }
1624         }
1625     }

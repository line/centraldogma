76      public EncryptionGitStorage(String projectName, String repoName,
77                                  EncryptionStorageManager encryptionStorageManager) {
78          this.projectName = projectName;
79          this.repoName = repoName;
80          final String projectRepoPrefix = projectName + '/' + repoName + '/';
81          objectKeyPrefix = (projectRepoPrefix + OBJS).getBytes(StandardCharsets.UTF_8);
82          // Suffixes start with refs/ so do not add REFS.
83          refsKeyPrefix = projectRepoPrefix.getBytes(StandardCharsets.UTF_8);
84          rev2ShaPrefix = (projectRepoPrefix + REV2SHA).getBytes(StandardCharsets.UTF_8);
85          this.encryptionStorageManager = encryptionStorageManager;
86          currentDek = encryptionStorageManager.getCurrentDek(projectName, repoName);
87  
88          encryptionStorageManager.addCurrentDekListener(projectName, repoName, newDek -> currentDek = newDek);
89      }

107     ObjectId insertObject(ObjectId objectId, int type, byte[] data, int off, int len) throws IOException {
108         if (off < 0 || len < 0 || off + len > data.length) {
109             throw new IllegalArgumentException("Invalid offset or length: " + off + ", " + len);
110         }
111         final byte[] metadataKey = objectMetadataKey(objectId);
112         if (encryptionStorageManager.containsMetadata(metadataKey)) {
113             return objectId;
114         }
115 
116         final SecretKeyWithVersion currentDek = this.currentDek;
117 
118         final byte[] nonce = AesGcmSivCipher.generateNonce();
119         // Generate a new DEK for the data so that we don't decrypt and encrypt the data again when the
120         // repository key is rotated.
121         final byte[] objectDek = AesGcmSivCipher.generateAes256Key();
122         final byte[] objectWdek = encrypt(currentDek.secretKey(), nonce, objectDek, 0, KEY_SIZE_BYTES);
123 
124         assert objectWdek.length == KEY_SIZE_BYTES + 16; // 16 bytes for the tag
125 
126         final GitObjectMetadata gitObjectMetadata = GitObjectMetadata.of(currentDek.version(), nonce, type,
127                                                                          objectWdek);
128 
129         final SecretKeySpec keySpec = aesSecretKey(objectDek);
130 
131         final byte[] encryptedId = encryptObjectId(keySpec, nonce, objectId);
132         final byte[] encryptedValue = encrypt(keySpec, nonce, data, off, len);
133         encryptionStorageManager.putObject(metadataKey, gitObjectMetadata.toBytes(),
134                                            encryptedId, encryptedValue);
135         return objectId;
136     }

138     @VisibleForTesting
139     public byte[] objectMetadataKey(ObjectId objectId) {
140         final ByteBuffer byteBuffer = ByteBuffer.allocate(objectKeyPrefix.length + 20);
141         byteBuffer.put(objectKeyPrefix);
142         objectId.copyRawTo(byteBuffer);
143         return byteBuffer.array();
144     }

146     private byte[] encryptObjectId(SecretKey dek, byte[] nonce, ObjectId objectId) {
147         final byte[] idBytes = new byte[20];
148         objectId.copyRawTo(idBytes, 0);
149         return encrypt(dek, nonce, idBytes, 0, 20);
150     }
151 
152     private byte[] encrypt(SecretKey dek, byte[] nonce, byte[] data, int offset, int length) {
153         try {
154             return AesGcmSivCipher.encrypt(dek, nonce, data, offset, length);
155         } catch (Exception e) {
156             throw new EncryptionStorageException(
157                     "Failed to encrypt data in " + projectName + '/' + repoName, e);
158         }
159     }
160 
161     private byte[] decrypt(SecretKey key, byte[] nonce, byte[] ciphertext) {
162         try {
163             return AesGcmSivCipher.decrypt(key, nonce, ciphertext);
164         } catch (Exception e) {
165             throw new EncryptionStorageException(
166                     "Failed to decrypt data in " + projectName + '/' + repoName, e);
167         }
168     }

170     @Nullable
171     public ObjectLoader getObject(ObjectId objectId, int typeHint) throws IncorrectObjectTypeException {
172         final byte[] metadataKey = objectMetadataKey(objectId);
173         final byte[] metadata = encryptionStorageManager.getMetadata(metadataKey);
174         if (metadata == null) {
175             return null;
176         }
177 
178         final int actualType = getInt(metadata, 4 + NONCE_SIZE_BYTES);
179         if (typeHint != OBJ_ANY && actualType != typeHint) {
180             throw new IncorrectObjectTypeException(objectId.copy(), typeHint);
181         }
182 
183         final int keyVersion = getInt(metadata, 0);
184         final SecretKey dek = getDek(keyVersion);
185 
186         final GitObjectMetadata gitObjectMetadata = GitObjectMetadata.fromBytes(metadata);
187         final SecretKeySpec objectDek;
188         try {
189             objectDek = gitObjectMetadata.objectDek(dek);
190         } catch (Exception e) {
191             throw new EncryptionStorageException(
192                     "Failed to get object dek in " + projectName + '/' + repoName + " for " + objectId, e);
193         }
194 
195         final byte[] encryptedKey = encryptObjectId(objectDek,
196                                                     gitObjectMetadata.nonce(), objectId);
197         final byte[] value = encryptionStorageManager.getObject(encryptedKey, metadataKey);
198         if (value == null) {
199             return null;
200         }
201 
202         final byte[] decrypted = decrypt(objectDek, gitObjectMetadata.nonce(), value);
203         return new DecryptedObjectLoader(decrypted, actualType);
204     }

206     @Nullable
207     @VisibleForTesting
208     public Ref readRef(String refName) {
209         final byte[] refNameBytes = refName.getBytes(StandardCharsets.UTF_8);
210         final byte[] metadataKey = refMetadataKey(refNameBytes);
211         final byte[] metadata = encryptionStorageManager.getMetadata(metadataKey);
212         if (metadata == null) {
213             return null;
214         }
215         final byte[] nonce = new byte[NONCE_SIZE_BYTES];
216         System.arraycopy(metadata, 4, nonce, 0, NONCE_SIZE_BYTES);
217         final int keyVersion = getInt(metadata, 0);
218         final SecretKey dek = getDek(keyVersion);
219 
220         final byte[] encryptedRefName = encrypt(dek, nonce, refNameBytes, 0, refNameBytes.length);
221         final byte[] encryptedRefValue = encryptionStorageManager.getObjectId(encryptedRefName, metadataKey);
222         if (encryptedRefValue == null) {
223             return null;
224         }
225 
226         final byte[] refValue = decrypt(dek, nonce, encryptedRefValue);
227         if (!isSymRef(refValue, refValue.length)) {
228             // We don't use annotated tag.
229             return new ObjectIdRef.PeeledNonTag(
230                     // Just use LOOSE because we don't distinguish between LOOSE and PACKED.
231                     Storage.LOOSE, refName, ObjectId.fromRaw(refValue));
232         }
233         final byte[] refValueWithoutSymRef = new byte[refValue.length - 5];
234         System.arraycopy(refValue, 5, refValueWithoutSymRef, 0, refValueWithoutSymRef.length);
235         final String targetRefName = new String(refValueWithoutSymRef, StandardCharsets.UTF_8);
236         final Ref targetRef = readRef(targetRefName);
237         if (targetRef != null) {
238             return new SymbolicRef(refName, targetRef);
239         }
240         return new SymbolicRef(refName, new ObjectIdRef.Unpeeled(NEW, targetRefName, null));
241     }

243     private static boolean isSymRef(byte[] buf, int n) {
244         if (n < 6) {
245             return false;
246         }
247         return buf[0] == 'r' && buf[1] == 'e' && buf[2] == 'f' && buf[3] == ':' && buf[4] == ' ';
248     }

250     Result updateRef(String refName, ObjectId objectId, Result desiredResult) {
251         final byte[] refNameBytes = refName.getBytes(StandardCharsets.UTF_8);
252 
253         final SecretKeyWithVersion currentDek = this.currentDek;
254         final byte[] metadataKey = refMetadataKey(refNameBytes);
255         final byte[] metadata = new byte[4 + NONCE_SIZE_BYTES];
256         putInt(metadata, 0, currentDek.version());
257         final byte[] nonce = AesGcmSivCipher.generateNonce();
258         System.arraycopy(nonce, 0, metadata, 4, NONCE_SIZE_BYTES);
259 
260         final byte[] encryptedRefName =
261                 encrypt(currentDek.secretKey(), nonce, refNameBytes, 0, refNameBytes.length);
262         final byte[] encryptedId = encryptObjectId(currentDek.secretKey(), nonce, objectId);
263 
264         // We should remove the previous ref name if it exists.
265 
266         final byte[] previousEncryptedRefName;
267         final byte[] previousMetadata = encryptionStorageManager.getMetadata(metadataKey);
268         if (previousMetadata == null) {
269             previousEncryptedRefName = null;
270         } else {
271             final byte[] previousNonce = new byte[NONCE_SIZE_BYTES];
272             System.arraycopy(previousMetadata, 4, previousNonce, 0, NONCE_SIZE_BYTES);
273             final SecretKey previousDek = getDek(getInt(previousMetadata, 0));
274             previousEncryptedRefName =
275                     encrypt(previousDek, previousNonce, refNameBytes, 0, refNameBytes.length);
276         }
277         encryptionStorageManager.putObjectId(metadataKey, metadata, encryptedRefName,
278                                              encryptedId, previousEncryptedRefName);
279         return desiredResult;
280     }

290     void deleteRef(String refName) {
291         final byte[] refNameBytes = refName.getBytes(StandardCharsets.UTF_8);
292         final byte[] metadataKey = refMetadataKey(refNameBytes);
293         final byte[] metadata = encryptionStorageManager.getMetadata(metadataKey);
294         if (metadata == null) {
295             return;
296         }
297 
298         final byte[] nonce = new byte[NONCE_SIZE_BYTES];
299         System.arraycopy(metadata, 4, nonce, 0, NONCE_SIZE_BYTES);
300         final int keyVersion = getInt(metadata, 0);
301         final SecretKey dek = getDek(keyVersion);
302 
303         final byte[] encryptedRefName = encrypt(dek, nonce, refNameBytes, 0, refNameBytes.length);
304         encryptionStorageManager.deleteObjectId(metadataKey, encryptedRefName);
305     }

307     void linkRef(String refName, String target) {
308         final byte[] refNameBytes = refName.getBytes(StandardCharsets.UTF_8);
309         final byte[] metadataKey = refMetadataKey(refNameBytes);
310         final byte[] nonce = AesGcmSivCipher.generateNonce();
311         final byte[] metadata = new byte[4 + NONCE_SIZE_BYTES];
312         final SecretKeyWithVersion currentDek = this.currentDek;
313         putInt(metadata, 0, currentDek.version());
314         System.arraycopy(nonce, 0, metadata, 4, NONCE_SIZE_BYTES);
315 
316         final byte[] encryptedRefName =
317                 encrypt(currentDek.secretKey(), nonce, refNameBytes, 0, refNameBytes.length);
318         final byte[] encoded = encode(RefDirectory.SYMREF + target);
319         final byte[] encryptedTarget = encrypt(currentDek.secretKey(), nonce, encoded, 0, encoded.length);
320         encryptionStorageManager.putObjectId(metadataKey, metadata, encryptedRefName, encryptedTarget, null);
321     }

323     @VisibleForTesting
324     public ObjectId getRevisionObjectId(Revision revision) {
325         final byte[] metadataKey = rev2ShaMetadataKey(revision);
326         final byte[] metadata = encryptionStorageManager.getMetadata(metadataKey);
327         if (metadata == null) {
328             throw new RevisionNotFoundException(revision);
329         }
330         final int keyVersion = getInt(metadata, 0);
331         final SecretKey dek = getDek(keyVersion);
332 
333         final byte[] nonce = new byte[NONCE_SIZE_BYTES];
334         System.arraycopy(metadata, 4, nonce, 0, NONCE_SIZE_BYTES);
335 
336         final byte[] encryptedKey = encrypt(dek, nonce, Ints.toByteArray(revision.major()), 0, 4);
337         final byte[] value = encryptionStorageManager.getObjectId(encryptedKey, metadataKey);
338         if (value == null) {
339             throw new RevisionNotFoundException(revision);
340         }
341         final byte[] raw = decrypt(dek, nonce, value);
342         if (raw.length != 20) {
343             throw new EncryptionStorageException(
344                     "Corrupted commit ID for " + revision + " in " + projectName + '/' + repoName + ": " +
345                     " expected 20 bytes but got " + raw.length);
346         }
347         return ObjectId.fromRaw(raw);
348     }

358     void putRevisionObjectId(Revision revision, ObjectId objectId) {
359         final byte[] metadataKey = rev2ShaMetadataKey(revision);
360         if (encryptionStorageManager.containsMetadata(metadataKey)) {
361             throw new EncryptionStorageException(
362                     "Revision already exists: " + revision + " in " + projectName + '/' + repoName);
363         }
364         final SecretKeyWithVersion currentDek = this.currentDek;
365         final byte[] metadata = new byte[4 + NONCE_SIZE_BYTES];
366         final byte[] nonce = AesGcmSivCipher.generateNonce();
367         putInt(metadata, 0, currentDek.version());
368         System.arraycopy(nonce, 0, metadata, 4, NONCE_SIZE_BYTES);
369 
370         final byte[] encryptedRevision =
371                 encrypt(currentDek.secretKey(), nonce, Ints.toByteArray(revision.major()), 0, 4);
372         final byte[] encryptedId = encryptObjectId(currentDek.secretKey(), nonce, objectId);
373         encryptionStorageManager.putObjectId(metadataKey, metadata, encryptedRevision, encryptedId, null);
374     }

381     private static final class DecryptedObjectLoader extends ObjectLoader {
382         private final byte[] decrypted;
383         private final int type;
384 
385         DecryptedObjectLoader(byte[] decrypted, int type) {
386             this.decrypted = decrypted;
387             this.type = type;
388         }
389 
390         @Override
391         public int getType() {
392             return type;
393         }
394 
395         @Override
396         public long getSize() {
397             return decrypted.length;
398         }
399 
400         @Override
401         public byte[] getCachedBytes() {
402             return decrypted;
403         }
404 
405         @Override
406         public ObjectStream openStream() {
407             return new ObjectStream.SmallStream(this);
408         }
409 
410         @Override
411         public String toString() {
412             return MoreObjects.toStringHelper(this)
413                               .add("type", type)
414                               .add("size", decrypted.length)
415                               .toString();
416         }
417     }

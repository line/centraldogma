74      public EncryptionGitStorage(String projectName, String repoName,
75                                  EncryptionStorageManager encryptionStorageManager) {
76          this.projectName = projectName;
77          this.repoName = repoName;
78          final String projectRepoPrefix = projectName + '/' + repoName + '/';
79          objectKeyPrefix = (projectRepoPrefix + OBJS).getBytes(StandardCharsets.UTF_8);
80          // Suffixes start with refs/ so do not add REFS.
81          refsKeyPrefix = projectRepoPrefix.getBytes(StandardCharsets.UTF_8);
82          rev2ShaPrefix = (projectRepoPrefix + REV2SHA).getBytes(StandardCharsets.UTF_8);
83          this.encryptionStorageManager = encryptionStorageManager;
84          currentDek = encryptionStorageManager.getCurrentDek(projectName, repoName);
85          deks.put(currentDek.version(), currentDek.secretKey());
86      }

96      private SecretKey dek(int version) {
97          return deks.computeIfAbsent(version, key -> encryptionStorageManager.getDek(
98                  projectName, repoName, version));
99      }

101     ObjectId insertObject(ObjectId objectId, int type, byte[] data, int off, int len) throws IOException {
102         if (off < 0 || len < 0 || off + len > data.length) {
103             throw new IllegalArgumentException("Invalid offset or length: " + off + ", " + len);
104         }
105         final byte[] metadataKey = objectMetadataKey(objectId);
106         if (encryptionStorageManager.containsMetadata(metadataKey)) {
107             return objectId;
108         }
109 
110         final SecretKeyWithVersion currentDek = this.currentDek;
111 
112         final byte[] nonce = AesGcmSivCipher.generateNonce();
113         // Generate a new DEK for the data so that we don't decrypt and encrypt the data again when the
114         // repository key is rotated.
115         final byte[] objectDek = AesGcmSivCipher.generateAes256Key();
116         final byte[] objectWdek = encrypt(currentDek.secretKey(), nonce, objectDek, 0, KEY_SIZE_BYTES);
117 
118         assert objectWdek.length == KEY_SIZE_BYTES + 16; // 16 bytes for the tag
119 
120         final GitObjectMetadata gitObjectMetadata = GitObjectMetadata.of(currentDek.version(), nonce, type,
121                                                                          objectWdek);
122 
123         final SecretKeySpec keySpec = aesSecretKey(objectDek);
124 
125         final byte[] encryptedId = encryptObjectId(keySpec, nonce, objectId);
126         final byte[] encryptedValue = encrypt(keySpec, nonce, data, off, len);
127         encryptionStorageManager.putObject(metadataKey, gitObjectMetadata.toBytes(),
128                                            encryptedId, encryptedValue);
129         return objectId;
130     }

132     @VisibleForTesting
133     public byte[] objectMetadataKey(ObjectId objectId) {
134         final ByteBuffer byteBuffer = ByteBuffer.allocate(objectKeyPrefix.length + 20);
135         byteBuffer.put(objectKeyPrefix);
136         objectId.copyRawTo(byteBuffer);
137         return byteBuffer.array();
138     }

140     private byte[] encryptObjectId(SecretKey dek, byte[] nonce, ObjectId objectId) {
141         final byte[] idBytes = new byte[20];
142         objectId.copyRawTo(idBytes, 0);
143         return encrypt(dek, nonce, idBytes, 0, 20);
144     }
145 
146     private byte[] encrypt(SecretKey dek, byte[] nonce, byte[] data, int offset, int length) {
147         try {
148             return AesGcmSivCipher.encrypt(dek, nonce, data, offset, length);
149         } catch (Exception e) {
150             throw new EncryptionStorageException(
151                     "Failed to encrypt data in " + projectName + '/' + repoName, e);
152         }
153     }
154 
155     private byte[] decrypt(SecretKey key, byte[] nonce, byte[] ciphertext) {
156         try {
157             return AesGcmSivCipher.decrypt(key, nonce, ciphertext);
158         } catch (Exception e) {
159             throw new EncryptionStorageException(
160                     "Failed to decrypt data in " + projectName + '/' + repoName, e);
161         }
162     }

164     @Nullable
165     public ObjectLoader getObject(ObjectId objectId, int typeHint) throws IncorrectObjectTypeException {
166         final byte[] metadataKey = objectMetadataKey(objectId);
167         final byte[] metadata = encryptionStorageManager.getMetadata(metadataKey);
168         if (metadata == null) {
169             return null;
170         }
171 
172         final int actualType = getInt(metadata, 4 + NONCE_SIZE_BYTES);
173         if (typeHint != OBJ_ANY && actualType != typeHint) {
174             throw new IncorrectObjectTypeException(objectId.copy(), typeHint);
175         }
176 
177         final int keyVersion = getInt(metadata, 0);
178         final SecretKey dek = dek(keyVersion);
179 
180         final GitObjectMetadata gitObjectMetadata = GitObjectMetadata.fromBytes(metadata);
181         final SecretKeySpec objectDek;
182         try {
183             objectDek = gitObjectMetadata.objectDek(dek);
184         } catch (Exception e) {
185             throw new EncryptionStorageException(
186                     "Failed to get object dek in " + projectName + '/' + repoName + " for " + objectId, e);
187         }
188 
189         final byte[] encryptedKey = encryptObjectId(objectDek,
190                                                     gitObjectMetadata.nonce(), objectId);
191         final byte[] value = encryptionStorageManager.getObject(encryptedKey, metadataKey);
192         if (value == null) {
193             return null;
194         }
195 
196         final byte[] decrypted = decrypt(objectDek, gitObjectMetadata.nonce(), value);
197         return new DecryptedObjectLoader(decrypted, actualType);
198     }

200     @Nullable
201     @VisibleForTesting
202     public Ref readRef(String refName) {
203         final byte[] refNameBytes = refName.getBytes(StandardCharsets.UTF_8);
204         final byte[] metadataKey = refMetadataKey(refNameBytes);
205         final byte[] metadata = encryptionStorageManager.getMetadata(metadataKey);
206         if (metadata == null) {
207             return null;
208         }
209         final byte[] nonce = new byte[NONCE_SIZE_BYTES];
210         System.arraycopy(metadata, 4, nonce, 0, NONCE_SIZE_BYTES);
211         final int keyVersion = getInt(metadata, 0);
212         final SecretKey dek = dek(keyVersion);
213 
214         final byte[] encryptedRefName = encrypt(dek, nonce, refNameBytes, 0, refNameBytes.length);
215         final byte[] encryptedRefValue = encryptionStorageManager.getObjectId(encryptedRefName, metadataKey);
216         if (encryptedRefValue == null) {
217             return null;
218         }
219 
220         final byte[] refValue = decrypt(dek, nonce, encryptedRefValue);
221         if (!isSymRef(refValue, refValue.length)) {
222             // We don't use annotated tag.
223             return new ObjectIdRef.PeeledNonTag(
224                     // Just use LOOSE because we don't distinguish between LOOSE and PACKED.
225                     Storage.LOOSE, refName, ObjectId.fromRaw(refValue));
226         }
227         final byte[] refValueWithoutSymRef = new byte[refValue.length - 5];
228         System.arraycopy(refValue, 5, refValueWithoutSymRef, 0, refValueWithoutSymRef.length);
229         final String targetRefName = new String(refValueWithoutSymRef, StandardCharsets.UTF_8);
230         final Ref targetRef = readRef(targetRefName);
231         if (targetRef != null) {
232             return new SymbolicRef(refName, targetRef);
233         }
234         return new SymbolicRef(refName, new ObjectIdRef.Unpeeled(NEW, targetRefName, null));
235     }

237     private static boolean isSymRef(byte[] buf, int n) {
238         if (n < 6) {
239             return false;
240         }
241         return buf[0] == 'r' && buf[1] == 'e' && buf[2] == 'f' && buf[3] == ':' && buf[4] == ' ';
242     }

244     Result updateRef(String refName, ObjectId objectId, Result desiredResult) {
245         final byte[] refNameBytes = refName.getBytes(StandardCharsets.UTF_8);
246 
247         final SecretKeyWithVersion currentDek = this.currentDek;
248         final byte[] metadataKey = refMetadataKey(refNameBytes);
249         final byte[] metadata = new byte[4 + NONCE_SIZE_BYTES];
250         putInt(metadata, 0, currentDek.version());
251         final byte[] nonce = AesGcmSivCipher.generateNonce();
252         System.arraycopy(nonce, 0, metadata, 4, NONCE_SIZE_BYTES);
253 
254         final byte[] encryptedRefName =
255                 encrypt(currentDek.secretKey(), nonce, refNameBytes, 0, refNameBytes.length);
256         final byte[] encryptedId = encryptObjectId(currentDek.secretKey(), nonce, objectId);
257 
258         // We should remove the previous ref name if it exists.
259 
260         final byte[] previousEncryptedRefName;
261         final byte[] previousMetadata = encryptionStorageManager.getMetadata(metadataKey);
262         if (previousMetadata == null) {
263             previousEncryptedRefName = null;
264         } else {
265             final byte[] previousNonce = new byte[NONCE_SIZE_BYTES];
266             System.arraycopy(previousMetadata, 4, previousNonce, 0, NONCE_SIZE_BYTES);
267             final SecretKey previousDek = dek(getInt(previousMetadata, 0));
268             previousEncryptedRefName =
269                     encrypt(previousDek, previousNonce, refNameBytes, 0, refNameBytes.length);
270         }
271         encryptionStorageManager.putObjectId(metadataKey, metadata, encryptedRefName,
272                                              encryptedId, previousEncryptedRefName);
273         return desiredResult;
274     }

284     void deleteRef(String refName) {
285         final byte[] refNameBytes = refName.getBytes(StandardCharsets.UTF_8);
286         final byte[] metadataKey = refMetadataKey(refNameBytes);
287         final byte[] metadata = encryptionStorageManager.getMetadata(metadataKey);
288         if (metadata == null) {
289             return;
290         }
291 
292         final byte[] nonce = new byte[NONCE_SIZE_BYTES];
293         System.arraycopy(metadata, 4, nonce, 0, NONCE_SIZE_BYTES);
294         final int keyVersion = getInt(metadata, 0);
295         final SecretKey dek = dek(keyVersion);
296 
297         final byte[] encryptedRefName = encrypt(dek, nonce, refNameBytes, 0, refNameBytes.length);
298         encryptionStorageManager.deleteObjectId(metadataKey, encryptedRefName);
299     }

301     void linkRef(String refName, String target) {
302         final byte[] refNameBytes = refName.getBytes(StandardCharsets.UTF_8);
303         final byte[] metadataKey = refMetadataKey(refNameBytes);
304         final byte[] nonce = AesGcmSivCipher.generateNonce();
305         final byte[] metadata = new byte[4 + NONCE_SIZE_BYTES];
306         final SecretKeyWithVersion currentDek = this.currentDek;
307         putInt(metadata, 0, currentDek.version());
308         System.arraycopy(nonce, 0, metadata, 4, NONCE_SIZE_BYTES);
309 
310         final byte[] encryptedRefName =
311                 encrypt(currentDek.secretKey(), nonce, refNameBytes, 0, refNameBytes.length);
312         final byte[] encoded = encode(RefDirectory.SYMREF + target);
313         final byte[] encryptedTarget = encrypt(currentDek.secretKey(), nonce, encoded, 0, encoded.length);
314         encryptionStorageManager.putObjectId(metadataKey, metadata, encryptedRefName, encryptedTarget, null);
315     }

317     @VisibleForTesting
318     public ObjectId getRevisionObjectId(Revision revision) {
319         final byte[] metadataKey = rev2ShaMetadataKey(revision);
320         final byte[] metadata = encryptionStorageManager.getMetadata(metadataKey);
321         if (metadata == null) {
322             throw new RevisionNotFoundException(revision);
323         }
324         final int keyVersion = getInt(metadata, 0);
325         final SecretKey dek = dek(keyVersion);
326 
327         final byte[] nonce = new byte[NONCE_SIZE_BYTES];
328         System.arraycopy(metadata, 4, nonce, 0, NONCE_SIZE_BYTES);
329 
330         final byte[] encryptedKey = encrypt(dek, nonce, Ints.toByteArray(revision.major()), 0, 4);
331         final byte[] value = encryptionStorageManager.getObjectId(encryptedKey, metadataKey);
332         if (value == null) {
333             throw new RevisionNotFoundException(revision);
334         }
335         final byte[] raw = decrypt(dek, nonce, value);
336         if (raw.length != 20) {
337             throw new EncryptionStorageException(
338                     "Corrupted commit ID for " + revision + " in " + projectName + '/' + repoName + ": " +
339                     " expected 20 bytes but got " + raw.length);
340         }
341         return ObjectId.fromRaw(raw);
342     }

352     void putRevisionObjectId(Revision revision, ObjectId objectId) {
353         final byte[] metadataKey = rev2ShaMetadataKey(revision);
354         if (encryptionStorageManager.containsMetadata(metadataKey)) {
355             throw new EncryptionStorageException(
356                     "Revision already exists: " + revision + " in " + projectName + '/' + repoName);
357         }
358         final SecretKeyWithVersion currentDek = this.currentDek;
359         final byte[] metadata = new byte[4 + NONCE_SIZE_BYTES];
360         final byte[] nonce = AesGcmSivCipher.generateNonce();
361         putInt(metadata, 0, currentDek.version());
362         System.arraycopy(nonce, 0, metadata, 4, NONCE_SIZE_BYTES);
363 
364         final byte[] encryptedRevision =
365                 encrypt(currentDek.secretKey(), nonce, Ints.toByteArray(revision.major()), 0, 4);
366         final byte[] encryptedId = encryptObjectId(currentDek.secretKey(), nonce, objectId);
367         encryptionStorageManager.putObjectId(metadataKey, metadata, encryptedRevision, encryptedId, null);
368     }

370     private static final class DecryptedObjectLoader extends ObjectLoader {
371         private final byte[] decrypted;
372         private final int type;
373 
374         DecryptedObjectLoader(byte[] decrypted, int type) {
375             this.decrypted = decrypted;
376             this.type = type;
377         }
378 
379         @Override
380         public int getType() {
381             return type;
382         }
383 
384         @Override
385         public long getSize() {
386             return decrypted.length;
387         }
388 
389         @Override
390         public byte[] getCachedBytes() {
391             return decrypted;
392         }
393 
394         @Override
395         public ObjectStream openStream() {
396             return new ObjectStream.SmallStream(this);
397         }
398 
399         @Override
400         public String toString() {
401             return MoreObjects.toStringHelper(this)
402                               .add("type", type)
403                               .add("size", decrypted.length)
404                               .toString();
405         }
406     }

65      public static boolean isMetaFile(String path) {
66          return "/mirrors.json".equals(path) || "/credentials.json".equals(path) ||
67                 (path.endsWith(".json") && (path.startsWith(PATH_CREDENTIALS) || path.startsWith(PATH_MIRRORS)));
68      }
69  
70      public static boolean isMirrorFile(String path) {
71          return path.endsWith(".json") && (path.startsWith(PATH_CREDENTIALS) || path.startsWith(PATH_MIRRORS));
72      }

76      }

79          return PATH_MIRRORS + mirrorId + ".json";

91      @Override
92      public CompletableFuture<List<Mirror>> mirrors(boolean includeDisabled) {
93          if (includeDisabled) {
94              return allMirrors();
95          }
96          return allMirrors().thenApply(mirrors -> {
97              return mirrors.stream().filter(Mirror::enabled).collect(toImmutableList());
98          });
99      }

101     @Override
102     public CompletableFuture<Mirror> mirror(String id) {
103         final String mirrorFile = mirrorFile(id);
104         return find(mirrorFile).thenCompose(entries -> {
105             @SuppressWarnings("unchecked")
106             final Entry<JsonNode> entry = (Entry<JsonNode>) entries.get(mirrorFile);
107             if (entry == null) {
108                 throw new EntryNotFoundException("failed to find mirror '" + mirrorFile + "' in " +
109                                                  parent().name() + '/' + name());
110             }
111 
112             final JsonNode mirrorJson = entry.content();
113             if (!mirrorJson.isObject()) {
114                 throw newInvalidJsonTypeException(mirrorFile, mirrorJson);
115             }
116             final MirrorConfig c;
117             try {
118                 c = Jackson.treeToValue(mirrorJson, MirrorConfig.class);
119             } catch (JsonProcessingException e) {
120                 throw new RepositoryMetadataException("failed to load the mirror configuration", e);
121             }
122 
123             final CompletableFuture<List<Credential>> credentials;
124             if (Strings.isNullOrEmpty(c.credentialId())) {
125                 credentials = credentials();
126             } else {
127                 credentials = credential(c.credentialId()).thenApply(ImmutableList::of);
128             }
129             return credentials.thenApply(credentials0 -> {
130                 final Mirror mirror = c.toMirror(parent(), credentials0);
131                 if (mirror == null) {
132                     throw new EntryNotFoundException("failed to find a mirror config for '" + mirrorFile +
133                                                      "' in " + parent().name() + '/' + name());
134                 }
135                 return mirror;
136             });
137         });
138     }

140     private CompletableFuture<List<Mirror>> allMirrors() {
141         return find(PATH_MIRRORS + "*.json").thenCompose(entries -> {
142             if (entries.isEmpty()) {
143                 return UnmodifiableFuture.completedFuture(ImmutableList.of());
144             }
145 
146             return credentials().thenApply(credentials -> {
147                 try {
148                     return parseMirrors(entries, credentials);
149                 } catch (JsonProcessingException e) {
150                     return Exceptions.throwUnsafely(e);
151                 }
152             });
153         });
154     }
155 
156     private List<Mirror> parseMirrors(Map<String, Entry<?>> entries, List<Credential> credentials)
157             throws JsonProcessingException {
158 
159         return entries.entrySet().stream().map(entry -> {
160                           final JsonNode mirrorJson = (JsonNode) entry.getValue().content();
161                           if (!mirrorJson.isObject()) {
162                               throw newInvalidJsonTypeException(entry.getKey(), mirrorJson);
163                           }
164                           final MirrorConfig c;
165                           try {
166                               c = Jackson.treeToValue(mirrorJson, MirrorConfig.class);
167                           } catch (JsonProcessingException e) {
168                               return Exceptions.throwUnsafely(e);
169                           }
170                           return c.toMirror(parent(), credentials);
171                       })
172                       .filter(Objects::nonNull)
173                       .collect(toImmutableList());
174     }

176     @Override
177     public CompletableFuture<List<Credential>> credentials() {
178         return find(PATH_CREDENTIALS + "*.json").thenApply(entries -> {
179             if (entries.isEmpty()) {
180                 return ImmutableList.of();
181             }
182             try {
183                 return parseCredentials(entries);
184             } catch (Exception e) {
185                 throw new RepositoryMetadataException("failed to load the credential configuration", e);
186             }
187         });
188     }

190     @Override
191     public CompletableFuture<Credential> credential(String credentialId) {
192         final String credentialFile = credentialFile(credentialId);
193         return find(credentialFile).thenApply(entries -> {
194             @SuppressWarnings("unchecked")
195             final Entry<JsonNode> entry = (Entry<JsonNode>) entries.get(credentialFile);
196             if (entry == null) {
197                 throw new EntryNotFoundException("failed to find credential '" + credentialId + "' in " +
198                                                  parent().name() + '/' + name());
199             }
200 
201             try {
202                 return parseCredential(credentialFile, entry);
203             } catch (Exception e) {
204                 throw new RepositoryMetadataException("failed to load the credential configuration", e);
205             }
206         });
207     }
208 
209     private List<Credential> parseCredentials(Map<String, Entry<?>> entries)
210             throws JsonProcessingException {
211         return entries.entrySet().stream()
212                       .map(entry -> {
213                           try {
214                               //noinspection unchecked
215                               return parseCredential(entry.getKey(), (Entry<JsonNode>) entry.getValue());
216                           } catch (JsonProcessingException e) {
217                               return Exceptions.throwUnsafely(e);
218                           }
219                       })
220                       .collect(toImmutableList());
221     }
222 
223     private Credential parseCredential(String credentialFile, Entry<JsonNode> entry)
224             throws JsonProcessingException {
225         final JsonNode credentialJson = entry.content();
226         if (!credentialJson.isObject()) {
227             throw newInvalidJsonTypeException(credentialFile, credentialJson);
228         }
229         return Jackson.treeToValue(credentialJson, Credential.class);
230     }
231 
232     private RepositoryMetadataException newInvalidJsonTypeException(String fileName, JsonNode credentialJson) {
233         return new RepositoryMetadataException(parent().name() + '/' + name() + fileName +
234                                                " must be an object: " + credentialJson.getNodeType());
235     }

237     private CompletableFuture<Map<String, Entry<?>>> find(String filePattern) {
238         return find(Revision.HEAD, filePattern, ImmutableMap.of());
239     }

241     @Override
242     public CompletableFuture<Command<CommitResult>> createPushCommand(MirrorDto mirrorDto, Author author,
243                                                                       @Nullable ZoneConfig zoneConfig,
244                                                                       boolean update) {
245         validateMirror(mirrorDto, zoneConfig);
246         if (update) {
247             final String summary = "Update the mirror '" + mirrorDto.id() + '\'';
248             return mirror(mirrorDto.id()).thenApply(mirror -> {
249                 // Perform the update operation only if the mirror exists.
250                 return newCommand(mirrorDto, author, summary);
251             });
252         } else {
253             String summary = "Create a new mirror from " + mirrorDto.remoteUrl() +
254                              mirrorDto.remotePath() + '#' + mirrorDto.remoteBranch() + " into " +
255                              mirrorDto.localRepo() + mirrorDto.localPath();
256             if (MirrorDirection.valueOf(mirrorDto.direction()) == MirrorDirection.REMOTE_TO_LOCAL) {
257                 summary = "[Remote-to-local] " + summary;
258             } else {
259                 summary = "[Local-to-remote] " + summary;
260             }
261             return UnmodifiableFuture.completedFuture(newCommand(mirrorDto, author, summary));
262         }
263     }
264 
265     @Override
266     public CompletableFuture<Command<CommitResult>> createPushCommand(Credential credential,
267                                                                       Author author, boolean update) {
268         checkArgument(!credential.id().isEmpty(), "Credential ID should not be empty");
269 
270         if (update) {
271             return credential(credential.id()).thenApply(c -> {
272                 assert c.id().equals(credential.id());
273                 final String summary = "Update the mirror credential '" + credential.id() + '\'';
274                 return newCommand(credential, author, summary);
275             });
276         } else {
277             final String summary = "Create a new mirror credential for " + credential.id();
278             return UnmodifiableFuture.completedFuture(newCommand(credential, author, summary));
279         }
280     }
281 
282     private Command<CommitResult> newCommand(MirrorDto mirrorDto, Author author, String summary) {
283         final MirrorConfig mirrorConfig = converterToMirrorConfig(mirrorDto);
284         final JsonNode jsonNode = Jackson.valueToTree(mirrorConfig);
285         final Change<JsonNode> change = Change.ofJsonUpsert(mirrorFile(mirrorConfig.id()), jsonNode);
286         return Command.push(author, parent().name(), name(), Revision.HEAD, summary, "", Markup.PLAINTEXT,
287                             change);
288     }
289 
290     private Command<CommitResult> newCommand(Credential credential, Author author, String summary) {
291         final JsonNode jsonNode = Jackson.valueToTree(credential);
292         final Change<JsonNode> change = Change.ofJsonUpsert(credentialFile(credential.id()), jsonNode);
293         return Command.push(author, parent().name(), name(), Revision.HEAD, summary, "", Markup.PLAINTEXT,
294                             change);
295     }

297     private static void validateMirror(MirrorDto mirror, @Nullable ZoneConfig zoneConfig) {
298         checkArgument(!Strings.isNullOrEmpty(mirror.id()), "Mirror ID is empty");
299         final String scheduleString = mirror.schedule();
300         if (scheduleString != null) {
301             final Cron schedule = MirrorConfig.CRON_PARSER.parse(scheduleString);
302             final CronField secondField = schedule.retrieve(CronFieldName.SECOND);
303             checkArgument(!secondField.getExpression().asString().contains("*"),
304                           "The second field of the schedule must be specified. (seconds: *, expected: 0-59)");
305         }
306 
307         final String zone = mirror.zone();
308         if (zone != null) {
309             checkArgument(zoneConfig != null, "Zone configuration is missing");
310             checkArgument(zoneConfig.allZones().contains(zone),
311                           "The zone '%s' is not in the zone configuration: %s", zone, zoneConfig);
312         }
313     }

315     private static MirrorConfig converterToMirrorConfig(MirrorDto mirrorDto) {
316         final String remoteUri =
317                 mirrorDto.remoteScheme() + "://" + mirrorDto.remoteUrl() +
318                 MirrorUtil.normalizePath(mirrorDto.remotePath()) + '#' + mirrorDto.remoteBranch();
319 
320         return new MirrorConfig(
321                 mirrorDto.id(),
322                 mirrorDto.enabled(),
323                 mirrorDto.schedule(),
324                 MirrorDirection.valueOf(mirrorDto.direction()),
325                 mirrorDto.localRepo(),
326                 mirrorDto.localPath(),
327                 URI.create(remoteUri),
328                 mirrorDto.gitignore(),
329                 mirrorDto.credentialId(),
330                 mirrorDto.zone());
331     }

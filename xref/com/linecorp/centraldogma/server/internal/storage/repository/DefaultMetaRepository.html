107     private static CompletableFuture<List<Mirror>> maybeFilter(CompletableFuture<List<Mirror>> future,
108                                                                boolean includeDisabled) {
109         if (includeDisabled) {
110             return future;
111         }
112         return future.thenApply(mirrors -> mirrors.stream().filter(Mirror::enabled).collect(toImmutableList()));
113     }

115     @Override
116     public CompletableFuture<Mirror> mirror(String repoName, String id, Revision revision) {
117         final String mirrorFile = mirrorFile(repoName, id);
118         return find(revision, mirrorFile).thenCompose(entries -> {
119             @SuppressWarnings("unchecked")
120             final Entry<JsonNode> entry = (Entry<JsonNode>) entries.get(mirrorFile);
121             if (entry == null) {
122                 throw new EntryNotFoundException(
123                         "failed to find mirror '" + mirrorFile + "' in " + parent().name() + '/' + name() +
124                         " (revision: " + revision + ')');
125             }
126 
127             final JsonNode mirrorJson = entry.content();
128             if (!mirrorJson.isObject()) {
129                 throw newInvalidJsonTypeException(mirrorFile, mirrorJson);
130             }
131             final MirrorConfig c;
132             try {
133                 c = Jackson.treeToValue(mirrorJson, MirrorConfig.class);
134             } catch (JsonProcessingException e) {
135                 throw new RepositoryMetadataException("failed to load the mirror configuration", e);
136             }
137 
138             if (c.credentialName().isEmpty()) {
139                 if (!parent().repos().exists(repoName)) {
140                     throw mirrorNotFound(revision, mirrorFile);
141                 }
142                 return CompletableFuture.completedFuture(convertToMirror(c, parent(), Credential.NONE));
143             }
144 
145             final CompletableFuture<Credential> future = credential(c.credentialName());
146             return future.thenApply(credential -> convertToMirror(c, parent(), credential));
147         });
148     }
149 
150     private EntryNotFoundException mirrorNotFound(Revision revision, String mirrorFile) {
151         return new EntryNotFoundException(
152                 "failed to find a mirror config for '" + mirrorFile + "' in " +
153                 parent().name() + '/' + name() + " (revision: " + revision + ')');
154     }

156     private CompletableFuture<List<Mirror>> allMirrors() {
157         return find(ALL_MIRRORS).thenCompose(this::handleAllMirrors);
158     }

160     private CompletableFuture<List<Mirror>> allMirrors(String repoName) {
161         return find("/repos/" + repoName + "/mirrors/*.json").thenCompose(this::handleAllMirrors);
162     }
163 
164     private CompletableFuture<List<Mirror>> handleAllMirrors(Map<String, Entry<?>> entries) {
165         if (entries.isEmpty()) {
166             return UnmodifiableFuture.completedFuture(ImmutableList.of());
167         }
168 
169         final CompletableFuture<List<Credential>> future = allCredentials();
170         return future.thenApply(credentials -> {
171             final List<MirrorConfig> mirrorConfigs = toMirrorConfigs(entries);
172             return mirrorConfigs.stream()
173                                 .map(mirrorConfig -> convertToMirror(
174                                         mirrorConfig, parent(), credentials))
175                                 .filter(Objects::nonNull)
176                                 .collect(toImmutableList());
177         });
178     }
179 
180     private CompletableFuture<List<Credential>> allCredentials() {
181         // TODO(minwoox): Optimize to read only the necessary files.
182         return find("/credentials/*.json,/repos/*/credentials/*.json").thenApply(
183                 entries -> credentials(entries, null));
184     }
185 
186     private List<MirrorConfig> toMirrorConfigs(Map<String, Entry<?>> entries) {
187         return entries.entrySet().stream().map(entry -> {
188                           final JsonNode mirrorJson = (JsonNode) entry.getValue().content();
189                           if (!mirrorJson.isObject()) {
190                               throw newInvalidJsonTypeException(entry.getKey(), mirrorJson);
191                           }
192                           try {
193                               return Jackson.treeToValue(mirrorJson, MirrorConfig.class);
194                           } catch (JsonProcessingException e) {
195                               return Exceptions.throwUnsafely(e);
196                           }
197                       })
198                       .collect(toImmutableList());
199     }

213     private List<Credential> credentials(Map<String, Entry<?>> entries, @Nullable String repoName) {
214         if (entries.isEmpty()) {
215             return ImmutableList.of();
216         }
217         try {
218             return parseCredentials(entries);
219         } catch (Exception e) {
220             String message = "failed to load the credential configuration";
221             if (repoName != null) {
222                 message += " for " + repoName;
223             }
224             throw new RepositoryMetadataException(message, e);
225         }
226     }

228     @Override
229     public CompletableFuture<Credential> credential(String credentialName) {
230         final String credentialFile = credentialFile(credentialName);
231         return credential0(credentialFile);
232     }
233 
234     private CompletableFuture<Credential> credential0(String credentialFile) {
235         return find(credentialFile).thenApply(entries -> {
236             @SuppressWarnings("unchecked")
237             final Entry<JsonNode> entry = (Entry<JsonNode>) entries.get(credentialFile);
238             if (entry == null) {
239                 throw new EntryNotFoundException("failed to find credential file '" + credentialFile + "' in " +
240                                                  parent().name() + '/' + name());
241             }
242 
243             try {
244                 return parseCredential(credentialFile, entry);
245             } catch (Exception e) {
246                 throw new RepositoryMetadataException(
247                         "failed to load the credential configuration. credential file: " + credentialFile, e);
248             }
249         });
250     }
251 
252     private List<Credential> parseCredentials(Map<String, Entry<?>> entries)
253             throws JsonProcessingException {
254         return entries.entrySet().stream()
255                       .map(entry -> {
256                           try {
257                               //noinspection unchecked
258                               return parseCredential(entry.getKey(),
259                                                      (Entry<JsonNode>) entry.getValue());
260                           } catch (JsonProcessingException e) {
261                               return Exceptions.throwUnsafely(e);
262                           }
263                       })
264                       .collect(toImmutableList());
265     }
266 
267     private Credential parseCredential(String credentialFile, Entry<JsonNode> entry)
268             throws JsonProcessingException {
269         final JsonNode credentialJson = entry.content();
270         if (!credentialJson.isObject()) {
271             throw newInvalidJsonTypeException(credentialFile, credentialJson);
272         }
273         return Jackson.treeToValue(credentialJson, Credential.class);
274     }

276     private RepositoryMetadataException newInvalidJsonTypeException(
277             String fileName, JsonNode credentialJson) {
278         return new RepositoryMetadataException(parent().name() + '/' + name() + fileName +
279                                                " must be an object: " + credentialJson.getNodeType());
280     }

286     @Override
287     public CompletableFuture<Command<Revision>> createMirrorPushCommand(
288             String repoName, MirrorRequest mirrorRequest, Author author,
289             @Nullable ZoneConfig zoneConfig, boolean update) {
290         validateMirror(mirrorRequest, zoneConfig);
291         if (update) {
292             final String summary = "Update the mirror '" + mirrorRequest.id() + "' in " + repoName;
293             return mirror(repoName, mirrorRequest.id()).thenApply(mirror -> {
294                 return newMirrorCommand(repoName, mirrorRequest, author, summary);
295             });
296         } else {
297             String summary = "Create a new mirror from " + mirrorRequest.remoteUrl() +
298                              mirrorRequest.remotePath() + '#' + mirrorRequest.remoteBranch() + " into " +
299                              repoName + mirrorRequest.localPath();
300             if (MirrorDirection.valueOf(mirrorRequest.direction()) == MirrorDirection.REMOTE_TO_LOCAL) {
301                 summary = "[Remote-to-local] " + summary;
302             } else {
303                 summary = "[Local-to-remote] " + summary;
304             }
305             return UnmodifiableFuture.completedFuture(
306                     newMirrorCommand(repoName, mirrorRequest, author, summary));
307         }
308     }

310     @Override
311     public CompletableFuture<Command<Revision>> createCredentialPushCommand(Credential credential,
312                                                                             Author author, boolean update) {
313         final String credentialName = credential.name();
314         if (update) {
315             return credential(credentialName).thenApply(c -> {
316                 final String summary = "Update the mirror credential '" + credentialName + '\'';
317                 return newCredentialCommand(credentialFile(credentialName), credential, author, summary);
318             });
319         }
320         final String summary = "Create a new mirror credential for " + credential.name();
321         return UnmodifiableFuture.completedFuture(newCredentialCommand(
322                 credentialFile(credentialName), credential, author, summary));
323     }

326     public CompletableFuture<Command<Revision>> createCredentialPushCommand(String repoName,
327                                                                             Credential credential,
328                                                                             Author author, boolean update) {

337         }

343     private Command<Revision> newCredentialCommand(String credentialFile, Credential credential,
344                                                    Author author, String summary) {
345         final JsonNode jsonNode = Jackson.valueToTree(credential);
346         final Change<JsonNode> change = Change.ofJsonUpsert(credentialFile, jsonNode);
347         return Command.push(author, parent().name(), name(), Revision.HEAD, summary, "", Markup.PLAINTEXT,
348                             change);
349     }
350 
351     private Command<Revision> newMirrorCommand(String repoName, MirrorRequest mirrorRequest,
352                                                Author author, String summary) {
353         final MirrorConfig mirrorConfig = converterToMirrorConfig(mirrorRequest);
354         final JsonNode jsonNode = Jackson.valueToTree(mirrorConfig);
355         final Change<JsonNode> change =
356                 Change.ofJsonUpsert(mirrorFile(repoName, mirrorConfig.id()), jsonNode);
357         return Command.push(author, parent().name(), name(), Revision.HEAD, summary, "", Markup.PLAINTEXT,
358                             change);
359     }

361     private static void validateMirror(MirrorRequest mirror, @Nullable ZoneConfig zoneConfig) {
362         checkArgument(!Strings.isNullOrEmpty(mirror.id()), "Mirror ID is empty");
363         final String scheduleString = mirror.schedule();
364         if (scheduleString != null) {
365             final Cron schedule = MirrorConfig.CRON_PARSER.parse(scheduleString);
366             final CronField secondField = schedule.retrieve(CronFieldName.SECOND);
367             checkArgument(!secondField.getExpression().asString().contains("*"),
368                           "The second field of the schedule must be specified. (seconds: *, expected: 0-59)");
369         }
370 
371         final String zone = mirror.zone();
372         if (zone != null) {
373             checkArgument(zoneConfig != null, "Zone configuration is missing");
374             checkArgument(zoneConfig.allZones().contains(zone),
375                           "The zone '%s' is not in the zone configuration: %s", zone, zoneConfig);
376         }
377     }

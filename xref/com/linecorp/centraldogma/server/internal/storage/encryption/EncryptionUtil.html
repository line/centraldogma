16  package com.linecorp.centraldogma.server.internal.storage.encryption;
17  
18  public final class EncryptionUtil {
19  
20      public static void putInt(byte[] metadata, int offset, int num) {
21          metadata[offset] = (byte) (num >> 24);
22          metadata[offset + 1] = (byte) (num >> 16);
23          metadata[offset + 2] = (byte) (num >> 8);
24          metadata[offset + 3] = (byte) num;
25      }
26  
27      public static int getInt(byte[] metadata, int offset) {
28          return ((metadata[offset] & 0xFF) << 24) |
29                 ((metadata[offset + 1] & 0xFF) << 16) |
30                 ((metadata[offset + 2] & 0xFF) << 8) |
31                 (metadata[offset + 3] & 0xFF);
32      }
33  
34      private EncryptionUtil() {}
35  }

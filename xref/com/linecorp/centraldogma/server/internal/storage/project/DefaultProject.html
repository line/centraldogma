73      private final long creationTimeMillis;
74      private final Author author;

83      /**
84       * Opens an existing project.
85       */
86      DefaultProject(File rootDir, Executor repositoryWorker, Executor purgeWorker,
87                     @Nullable RepositoryCache cache) {
88          requireNonNull(rootDir, "rootDir");
89          requireNonNull(repositoryWorker, "repositoryWorker");
90  
91          if (!rootDir.exists()) {
92              throw new ProjectNotFoundException(rootDir.toString());
93          }
94  
95          name = rootDir.getName();
96          repos = newRepoManager(rootDir, repositoryWorker, purgeWorker, cache);
97  
98          boolean success = false;
99          try {
100             createReservedRepos(System.currentTimeMillis());
101             final ProjectMetadata projectedMetadata = initialMetadata();
102             if (projectedMetadata != null) {
103                 final UserAndTimestamp creation = projectedMetadata.creation();
104                 creationTimeMillis = creation.timestampMillis();
105                 author = Author.ofEmail(creation.user());
106                 attachMetadataListener();
107             } else {
108                 creationTimeMillis = repos.get(REPO_DOGMA).creationTimeMillis();
109                 author = repos.get(REPO_DOGMA).author();
110             }
111             success = true;
112         } catch (Exception e) {
113             throw new CentralDogmaException("failed to initialize internal repositories of " + name, e);
114         } finally {
115             if (!success) {
116                 repos.close(() -> new CentralDogmaException(
117                         "failed to initialize internal repositories of " + name));
118             }
119         }
120     }
121 
122     /**
123      * Creates a new project.
124      */
125     DefaultProject(File rootDir, Executor repositoryWorker, Executor purgeWorker,
126                    long creationTimeMillis, Author author, @Nullable RepositoryCache cache) {
127         requireNonNull(rootDir, "rootDir");
128         requireNonNull(repositoryWorker, "repositoryWorker");
129 
130         if (rootDir.exists()) {
131             throw new ProjectExistsException(rootDir.getName());
132         }
133 
134         name = rootDir.getName();
135         repos = newRepoManager(rootDir, repositoryWorker, purgeWorker, cache);
136 
137         boolean success = false;
138         try {
139             createReservedRepos(creationTimeMillis);
140             initializeMetadata(creationTimeMillis, author);
141             this.creationTimeMillis = creationTimeMillis;
142             this.author = author;
143             attachMetadataListener();
144             success = true;
145         } finally {
146             if (!success) {
147                 repos.close(() -> new CentralDogmaException(
148                         "failed to initialize internal repositories of " + name));
149             }
150         }
151     }
152 
153     private RepositoryManager newRepoManager(File rootDir, Executor repositoryWorker, Executor purgeWorker,
154                                              @Nullable RepositoryCache cache) {
155         // Enable caching if 'cache' is not null.
156         final GitRepositoryManager gitRepos =
157                 new GitRepositoryManager(this, rootDir, repositoryWorker, purgeWorker, cache);
158         return cache == null ? gitRepos : new CachingRepositoryManager(gitRepos, cache);
159     }
160 
161     private void createReservedRepos(long creationTimeMillis) {
162         if (!repos.exists(REPO_DOGMA)) {
163             try {
164                 repos.create(REPO_DOGMA, creationTimeMillis, Author.SYSTEM);
165             } catch (RepositoryExistsException ignored) {
166                 // Just in case there's a race.
167             }
168         }
169         if (!repos.exists(REPO_META)) {
170             try {
171                 repos.create(REPO_META, creationTimeMillis, Author.SYSTEM);
172             } catch (RepositoryExistsException ignored) {
173                 // Just in case there's a race.
174             }
175         }
176     }

178     @Nullable
179     @Override
180     public ProjectMetadata metadata() {
181         return projectMetadata;
182     }
183 
184     private void initializeMetadata(long creationTimeMillis, Author author) {
185         // Do not generate a metadata file for internal projects.
186         if (name.equals(INTERNAL_PROJECT_DOGMA)) {
187             return;
188         }
189 
190         final Repository dogmaRepo = repos.get(REPO_DOGMA);
191         final Revision headRev = dogmaRepo.normalizeNow(Revision.HEAD);
192         if (!dogmaRepo.exists(headRev, METADATA_JSON).join()) {
193             logger.info("Initializing metadata: {}", name);
194 
195             final UserAndTimestamp userAndTimestamp = UserAndTimestamp.of(author);
196             final RepositoryMetadata repo = new RepositoryMetadata(REPO_META, userAndTimestamp,
197                                                                    PerRolePermissions.ofInternal());
198             final Member member = new Member(author, ProjectRole.OWNER, userAndTimestamp);
199             final ProjectMetadata metadata = new ProjectMetadata(name,
200                                                                  ImmutableMap.of(repo.id(), repo),
201                                                                  ImmutableMap.of(member.id(), member),
202                                                                  ImmutableMap.of(),
203                                                                  userAndTimestamp, null);
204             final CommitResult result =
205                     dogmaRepo.commit(headRev, creationTimeMillis, Author.SYSTEM,
206                                      "Initialize metadata", "",
207                                      Markup.PLAINTEXT,
208                                      Change.ofJsonUpsert(METADATA_JSON, Jackson.valueToTree(metadata)))
209                              .join();
210             lastMetadataRevision = result.revision();
211             projectMetadata = metadata;
212         }
213     }
214 
215     @Nullable
216     private ProjectMetadata initialMetadata()
217             throws ExecutionException, InterruptedException, JsonProcessingException {
218         if (name.equals(INTERNAL_PROJECT_DOGMA)) {
219             return null;
220         }
221         final Entry<JsonNode> metadata = repos.get(REPO_DOGMA).get(Revision.HEAD, Query.ofJson(METADATA_JSON))
222                                               .get();
223         final ProjectMetadata projectMetadata = Jackson.treeToValue(metadata.content(),
224                                                                     ProjectMetadata.class);
225         lastMetadataRevision = metadata.revision();
226         this.projectMetadata = projectMetadata;
227         return projectMetadata;
228     }
229 
230     /**
231      * Listens to new changes for "metadata.json" and updates the information to {@link #lastMetadataRevision}
232      * and {@link #projectMetadata}.
233      */
234     private void attachMetadataListener() {
235         if (name.equals(INTERNAL_PROJECT_DOGMA)) {
236             return;
237         }
238 
239         final Repository dogmaRepo = repos.get(REPO_DOGMA);
240         dogmaRepo.addListener(RepositoryListener.of(Query.ofJson(METADATA_JSON), entry -> {
241             if (entry == null) {
242                 logger.warn("{} file is missing in {}/{}", METADATA_JSON, name, REPO_DOGMA);
243                 return;
244             }
245 
246             final Revision lastRevision = entry.revision();
247             final Revision lastMetadataRevision = this.lastMetadataRevision;
248             assert lastMetadataRevision != null;
249             if (lastRevision.compareTo(lastMetadataRevision) <= 0) {
250                 // An old data.
251                 return;
252             }
253 
254             try {
255                 final ProjectMetadata projectMetadata = Jackson.treeToValue(entry.content(),
256                                                                             ProjectMetadata.class);
257                 this.lastMetadataRevision = lastRevision;
258                 this.projectMetadata = projectMetadata;
259             } catch (JsonParseException | JsonMappingException e) {
260                 logger.warn("Invalid {} file in {}/{}", METADATA_JSON, name, REPO_DOGMA, e);
261             }
262         }));
263     }

280     @Override
281     public MetaRepository metaRepo() {
282         MetaRepository metaRepo = this.metaRepo.get();
283         if (metaRepo != null) {
284             return metaRepo;
285         }
286 
287         metaRepo = new DefaultMetaRepository(repos.get(REPO_META));
288         if (this.metaRepo.compareAndSet(null, metaRepo)) {
289             return metaRepo;
290         } else {
291             return this.metaRepo.get();
292         }
293     }

303                           .add("name", name)
304                           .add("author", author)
305                           .add("repos", repos)
306                           .toString();
307     }

94      private static void handle(CompletableFuture<?> future, AsyncMethodCallback resultHandler) {
95          future.handle((res, cause) -> {
96              if (cause != null) {
97                  resultHandler.onError(convert(cause));
98              } else {
99                  resultHandler.onComplete(res);
100             }
101             return null;
102         });
103     }
104 
105     private static void handle(Callable<?> task, AsyncMethodCallback resultHandler) {
106         try {
107             resultHandler.onComplete(task.call());
108         } catch (Throwable cause) {
109             resultHandler.onError(convert(cause));
110         }
111     }
112 
113     private static void handleAsVoidResult(CompletableFuture<?> future, AsyncMethodCallback resultHandler) {
114         future.handle((res, cause) -> {
115             if (cause != null) {
116                 resultHandler.onError(convert(cause));
117             } else {
118                 resultHandler.onComplete(null);
119             }
120             return null;
121         });
122     }

164     @Override
165     public void createRepository(String projectName, String repositoryName,
166                                  AsyncMethodCallback resultHandler) {
167         validateRepositoryName(repositoryName, "repositoryName");
168         // HTTP v1 API will return '403 forbidden' in this case, but we deal it as '400 bad request' here.
169         if (isInternalRepo(repositoryName)) {
170             resultHandler.onError(convert(RESERVED_REPOSITORY_EXCEPTION));
171             return;
172         }
173         handleAsVoidResult(executor.execute(Command.createRepository(SYSTEM, projectName, repositoryName))
174                                    .thenCompose(unused -> mds.addRepo(SYSTEM, projectName, repositoryName)),
175                            resultHandler);
176     }

230     private com.linecorp.centraldogma.common.Revision normalizeRevision(
231             String projectName, String repositoryName, Revision revision) {
232         final Repository repository = projectApiManager.getProject(projectName, null).repos().get(
233                 repositoryName);
234         final com.linecorp.centraldogma.common.Revision normalized =
235                 repository.normalizeNow(convert(revision));
236         final com.linecorp.centraldogma.common.Revision head = repository.normalizeNow(HEAD);
237         increaseCounterIfOldRevisionUsed(RequestContext.current(), repository, normalized, head);
238         return normalized;
239     }

241     @Override
242     public void listFiles(String projectName, String repositoryName, Revision revision, String pathPattern,
243                           AsyncMethodCallback resultHandler) {
244         // Call normalizeRevision() first to check if the specified revision needs to be recorded.
245         normalizeRevision(projectName, repositoryName, revision);
246         handle(projectApiManager.getProject(projectName, null).repos().get(repositoryName)
247                                 .find(convert(revision), pathPattern, FIND_ALL_WITHOUT_CONTENT)
248                                 .thenApply(entries -> {
249                                     final List<Entry> ret = new ArrayList<>(entries.size());
250                                     entries.forEach((path, entry) -> ret.add(
251                                             new Entry(path, convert(entry.type()))));
252                                     return ret;
253                                 }),
254                resultHandler);
255     }

311     @Override
312     public void push(String projectName, String repositoryName, Revision baseRevision, Author author,
313                      String summary, Comment detail, List<Change> changes, AsyncMethodCallback resultHandler) {
314         final List<com.linecorp.centraldogma.common.Change<?>> convertedChanges =
315                 convert(changes, Converter::convert);
316         try {
317             checkMetaRepoPush(repositoryName, convertedChanges);
318         } catch (Exception e) {
319             resultHandler.onError(e);
320             return;
321         }
322         // TODO(trustin): Change Repository.commit() to return a Commit.
323         handle(executor.execute(Command.push(convert(author), projectName, repositoryName,
324                                              convert(baseRevision), summary, detail.getContent(),
325                                              convert(detail.getMarkup()), convertedChanges))
326                        .thenCompose(newRev -> {
327                            return projectApiManager.getProject(projectName, null).repos().get(repositoryName)
328                                                    .history(newRev, newRev, "/**");
329                        })
330                        .thenApply(commits -> convert(commits.get(0))),
331                resultHandler);
332     }

373     @Override
374     public void watchRepository(
375             String projectName, String repositoryName, Revision lastKnownRevision,
376             String pathPattern, long timeoutMillis, AsyncMethodCallback resultHandler) {
377         // Call normalizeRevision() first to check if the specified revision needs to be recorded.
378         normalizeRevision(projectName, repositoryName, lastKnownRevision);
379         if (timeoutMillis <= 0) {
380             rejectInvalidWatchTimeout("watchRepository", resultHandler);
381             return;
382         }
383 
384         final Repository repo = projectApiManager.getProject(projectName, null).repos().get(repositoryName);
385         final CompletableFuture<com.linecorp.centraldogma.common.Revision> future =
386                 watchService.watchRepository(repo, convert(lastKnownRevision), pathPattern, timeoutMillis,
387                                              false);
388         handleWatchRepositoryResult(future, resultHandler);
389     }

391     private static void handleWatchRepositoryResult(
392             CompletableFuture<com.linecorp.centraldogma.common.Revision> future,
393             AsyncMethodCallback resultHandler) {
394         future.handle((res, cause) -> {
395             if (cause == null) {
396                 final WatchRepositoryResult wrr = new WatchRepositoryResult();
397                 wrr.setRevision(convert(res));
398                 resultHandler.onComplete(wrr);
399             } else if (cause instanceof CancellationException) {
400                 resultHandler.onComplete(CentralDogmaConstants.EMPTY_WATCH_REPOSITORY_RESULT);
401             } else if (cause instanceof RequestAlreadyTimedOutException) {
402                 logger.warn("Ignoring the exception raised when a request has already timed out.");
403             } else {
404                 logAndInvokeOnError("watchRepository", resultHandler, cause);
405             }
406             return null;
407         });
408     }

411     public void watchFile(
412             String projectName, String repositoryName, Revision lastKnownRevision,
413             Query query, long timeoutMillis, AsyncMethodCallback resultHandler) {
414         // Call normalizeRevision() first to check if the specified revision needs to be recorded.
415         normalizeRevision(projectName, repositoryName, lastKnownRevision);
416         if (timeoutMillis <= 0) {
417             rejectInvalidWatchTimeout("watchFile", resultHandler);
418             return;

420

423                 watchService.watchFile(repo, convert(lastKnownRevision), convert(query), timeoutMillis,
424                                        false);
425 
426         handleWatchFileResult(future, resultHandler);
427     }

429     private static void rejectInvalidWatchTimeout(String operationName, AsyncMethodCallback resultHandler) {
430         final CentralDogmaException cde = new CentralDogmaException(ErrorCode.BAD_REQUEST);
431         CentralDogmaExceptions.log(operationName, cde);
432         resultHandler.onError(cde);
433     }
434 
435     private static void handleWatchFileResult(
436             CompletableFuture<com.linecorp.centraldogma.common.Entry<Object>> future,
437             AsyncMethodCallback resultHandler) {
438         future.handle((res, cause) -> {
439             if (cause == null) {
440                 final WatchFileResult wfr = new WatchFileResult();
441                 wfr.setRevision(convert(res.revision()));
442                 wfr.setType(convert(res.type()));
443                 wfr.setContent(res.contentAsText());
444                 resultHandler.onComplete(wfr);
445             } else if (cause instanceof CancellationException) {
446                 resultHandler.onComplete(CentralDogmaConstants.EMPTY_WATCH_FILE_RESULT);
447             } else if (cause instanceof RequestAlreadyTimedOutException) {
448                 logger.warn("Ignoring the exception raised when a request has already timed out.");
449             } else {
450                 logAndInvokeOnError("watchFile", resultHandler, cause);
451             }
452             return null;
453         });
454     }
455 
456     private static void logAndInvokeOnError(
457             String operationName, AsyncMethodCallback resultHandler, Throwable cause) {
458         final CentralDogmaException cde = convert(cause);
459         CentralDogmaExceptions.log(operationName, cde);
460         resultHandler.onError(cde);
461     }

89      private static void handle(CompletableFuture<?> future, AsyncMethodCallback resultHandler) {
90          future.handle((res, cause) -> {
91              if (cause != null) {
92                  resultHandler.onError(convert(cause));
93              } else {
94                  resultHandler.onComplete(res);
95              }
96              return null;
97          });
98      }
99  
100     private static void handle(Callable<?> task, AsyncMethodCallback resultHandler) {
101         try {
102             resultHandler.onComplete(task.call());
103         } catch (Throwable cause) {
104             resultHandler.onError(convert(cause));
105         }
106     }
107 
108     private static void handleAsVoidResult(CompletableFuture<?> future, AsyncMethodCallback resultHandler) {
109         future.handle((res, cause) -> {
110             if (cause != null) {
111                 resultHandler.onError(convert(cause));
112             } else {
113                 resultHandler.onComplete(null);
114             }
115             return null;
116         });
117     }

125     @Override
126     public void removeProject(String name, AsyncMethodCallback resultHandler) {
127         // Metadata must be updated first because it cannot be updated if the project is removed.
128         handle(mds.removeProject(SYSTEM, name)

162     @Override
163     public void createRepository(String projectName, String repositoryName,
164                                  AsyncMethodCallback resultHandler) {
165         // HTTP v1 API will return '403 forbidden' in this case, but we deal it as '400 bad request' here.
166         if (isReservedRepoName(repositoryName)) {
167             resultHandler.onError(convert(RESERVED_REPOSITORY_EXCEPTION));
168             return;
169         }
170         handleAsVoidResult(executor.execute(Command.createRepository(SYSTEM, projectName, repositoryName))
171                                    .thenCompose(unused -> mds.addRepo(SYSTEM, projectName, repositoryName)),
172                            resultHandler);
173     }

226     @Override
227     public void listFiles(String projectName, String repositoryName, Revision revision, String pathPattern,
228                           AsyncMethodCallback resultHandler) {
229 
230         handle(projectManager.get(projectName).repos().get(repositoryName)
231                              .find(convert(revision), pathPattern, FIND_ALL_WITHOUT_CONTENT)
232                              .thenApply(entries -> {
233                                  final List<Entry> ret = new ArrayList<>(entries.size());
234                                  entries.forEach((path, entry) -> ret.add(
235                                          new Entry(path, convert(entry.type()))));
236                                  return ret;
237                              }),
238                resultHandler);
239     }

289     @Override
290     public void push(String projectName, String repositoryName, Revision baseRevision, Author author,
291                      String summary, Comment detail, List<Change> changes, AsyncMethodCallback resultHandler) {
292         final List<com.linecorp.centraldogma.common.Change<?>> convertedChanges =
293                 convert(changes, Converter::convert);
294         try {
295             checkPush(repositoryName, convertedChanges);
296         } catch (Exception e) {
297             resultHandler.onError(e);
298             return;
299         }
300         // TODO(trustin): Change Repository.commit() to return a Commit.
301         handle(executor.execute(Command.push(convert(author), projectName, repositoryName,
302                                              convert(baseRevision), summary, detail.getContent(),
303                                              convert(detail.getMarkup()), convertedChanges))
304                        .thenCompose(commitResult -> {
305                            final com.linecorp.centraldogma.common.Revision newRev = commitResult.revision();
306                            return projectManager.get(projectName).repos().get(repositoryName)
307                                                 .history(newRev, newRev, "/**");
308                        })
309                        .thenApply(commits -> convert(commits.get(0))),
310                resultHandler);
311     }

347     @Override
348     public void watchRepository(
349             String projectName, String repositoryName, Revision lastKnownRevision,
350             String pathPattern, long timeoutMillis, AsyncMethodCallback resultHandler) {
351 
352         if (timeoutMillis <= 0) {
353             rejectInvalidWatchTimeout("watchRepository", resultHandler);
354             return;
355         }
356 
357         final Repository repo = projectManager.get(projectName).repos().get(repositoryName);
358         final CompletableFuture<com.linecorp.centraldogma.common.Revision> future =
359                 watchService.watchRepository(repo, convert(lastKnownRevision), pathPattern, timeoutMillis,
360                                              false);
361         handleWatchRepositoryResult(future, resultHandler);
362     }
363 
364     private static void handleWatchRepositoryResult(
365             CompletableFuture<com.linecorp.centraldogma.common.Revision> future,
366             AsyncMethodCallback resultHandler) {
367         future.handle((res, cause) -> {
368             if (cause == null) {
369                 final WatchRepositoryResult wrr = new WatchRepositoryResult();
370                 wrr.setRevision(convert(res));
371                 resultHandler.onComplete(wrr);
372             } else if (cause instanceof CancellationException) {
373                 resultHandler.onComplete(CentralDogmaConstants.EMPTY_WATCH_REPOSITORY_RESULT);
374             } else if (cause instanceof RequestAlreadyTimedOutException) {
375                 logger.warn("Ignoring the exception raised when a request has already timed out.");
376             } else {
377                 logAndInvokeOnError("watchRepository", resultHandler, cause);
378             }
379             return null;
380         });
381     }

383     @Override
384     public void watchFile(
385             String projectName, String repositoryName, Revision lastKnownRevision,
386             Query query, long timeoutMillis, AsyncMethodCallback resultHandler) {
387 
388         if (timeoutMillis <= 0) {
389             rejectInvalidWatchTimeout("watchFile", resultHandler);
390             return;
391         }
392 
393         final Repository repo = projectManager.get(projectName).repos().get(repositoryName);
394         final CompletableFuture<com.linecorp.centraldogma.common.Entry<Object>> future =
395                 watchService.watchFile(repo, convert(lastKnownRevision), convert(query), timeoutMillis,
396                                        false);
397 
398         handleWatchFileResult(future, resultHandler);
399     }
400 
401     private static void rejectInvalidWatchTimeout(String operationName, AsyncMethodCallback resultHandler) {
402         final CentralDogmaException cde = new CentralDogmaException(ErrorCode.BAD_REQUEST);
403         CentralDogmaExceptions.log(operationName, cde);
404         resultHandler.onError(cde);
405     }
406 
407     private static void handleWatchFileResult(
408             CompletableFuture<com.linecorp.centraldogma.common.Entry<Object>> future,
409             AsyncMethodCallback resultHandler) {
410         future.handle((res, cause) -> {
411             if (cause == null) {
412                 final WatchFileResult wfr = new WatchFileResult();
413                 wfr.setRevision(convert(res.revision()));
414                 wfr.setType(convert(res.type()));
415                 wfr.setContent(res.contentAsText());
416                 resultHandler.onComplete(wfr);
417             } else if (cause instanceof CancellationException) {
418                 resultHandler.onComplete(CentralDogmaConstants.EMPTY_WATCH_FILE_RESULT);
419             } else if (cause instanceof RequestAlreadyTimedOutException) {
420                 logger.warn("Ignoring the exception raised when a request has already timed out.");
421             } else {
422                 logAndInvokeOnError("watchFile", resultHandler, cause);
423             }
424             return null;
425         });
426     }
427 
428     private static void logAndInvokeOnError(
429             String operationName, AsyncMethodCallback resultHandler, Throwable cause) {
430         final CentralDogmaException cde = convert(cause);
431         CentralDogmaExceptions.log(operationName, cde);
432         resultHandler.onError(cde);
433     }

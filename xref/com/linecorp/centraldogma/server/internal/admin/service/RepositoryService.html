82      /**
83       * GET /projects/{projectName}/repositories/{repoName}/revision/{revision}
84       * Normalizes the revision into an absolute revision.
85       */
86      @Get("/projects/{projectName}/repositories/{repoName}/revision/{revision}")
87      public RevisionDto normalizeRevision(@Param String projectName,
88                                           @Param String repoName,
89                                           @Param String revision) {
90          return DtoConverter.convert(projectManager().get(projectName).repos().get(repoName)
91                                                      .normalizeNow(new Revision(revision)));
92      }

94      /**
95       * GET /projects/{projectName}/repositories/{repoName}/files/revisions/{revision}{path}
96       * Returns the blob in the path.
97       */
98      @Get("regex:/projects/(?<projectName>[^/]+)/repositories/(?<repoName>[^/]+)" +
99           "/files/revisions/(?<revision>[^/]+)(?<path>/.*$)")
100     public CompletionStage<EntryDto> getFile(@Param String projectName,
101                                              @Param String repoName,
102                                              @Param String revision,
103                                              @Param String path,
104                                              @Param @Default("IDENTITY") QueryType queryType,
105                                              @Param @Default("") String expression) {
106 
107         final Query<?> query = Query.of(queryType,path, expression);
108         final Repository repo = projectManager().get(projectName).repos().get(repoName);
109         return repo.get(repo.normalizeNow(new Revision(revision)), query)
110                    .thenApply(DtoConverter::convert);
111     }

113     /**
114      * POST|PUT /projects/{projectName}/repositories/{repoName}/files/revisions/{revision}
115      * Adds a new file or edits the existing file.
116      */
117     @Post
118     @Put
119     @Path("/projects/{projectName}/repositories/{repoName}/files/revisions/{revision}")
120     @RequiresWritePermission
121     public CompletionStage<Object> addOrEditFile(@Param String projectName,
122                                                  @Param String repoName,
123                                                  @Param String revision,
124                                                  AggregatedHttpRequest request,
125                                                  ServiceRequestContext ctx) {
126         final Entry<CommitMessageDto, Change<?>> p = commitMessageAndChange(request);
127         final CommitMessageDto commitMessage = p.getKey();
128         final Change<?> change = p.getValue();
129         return push(projectName, repoName, new Revision(revision), AuthUtil.currentAuthor(ctx),
130                     commitMessage.getSummary(), commitMessage.getDetail().getContent(),
131                     Markup.valueOf(commitMessage.getDetail().getMarkup()), change)
132                 // This is so weird but there is no way to find a converter for 'null' with the current
133                 // Armeria's converter implementation. We will figure out a better way to improve it.
134                 .thenApply(unused -> VOID);
135     }

137     /**
138      * POST /projects/{projectName}/repositories/{repoName}/delete/revisions/{revision}{path}
139      * Deletes a file.
140      */
141     @Post("regex:/projects/(?<projectName>[^/]+)/repositories/(?<repoName>[^/]+)" +
142           "/delete/revisions/(?<revision>[^/]+)(?<path>/.*$)")
143     @RequiresWritePermission
144     public HttpResponse deleteFile(@Param String projectName,
145                                    @Param String repoName,
146                                    @Param String revision,
147                                    @Param String path,
148                                    AggregatedHttpRequest request,
149                                    ServiceRequestContext ctx) {
150         final CommitMessageDto commitMessage;
151         try {
152             final JsonNode node = Jackson.readTree(request.contentUtf8());
153             commitMessage = Jackson.convertValue(node.get("commitMessage"), CommitMessageDto.class);
154         } catch (IOException e) {
155             throw new IllegalArgumentException("invalid data to be parsed", e);
156         }
157 
158         final CompletableFuture<?> future =
159                 push(projectName, repoName, new Revision(revision), AuthUtil.currentAuthor(ctx),
160                      commitMessage.getSummary(), commitMessage.getDetail().getContent(),
161                      Markup.valueOf(commitMessage.getDetail().getMarkup()), Change.ofRemoval(path));
162 
163         return HttpResponse.from(future.thenApply(unused -> HttpResponse.of(HttpStatus.OK)));
164     }

166     /**
167      * GET /projects/{projectName}/repositories/{repoName}/history{path}?from=x.x&amp;to=x.x
168      * Returns a history between the specified revisions.
169      */
170     @Get("regex:/projects/(?<projectName>[^/]+)/repositories/(?<repoName>[^/]+)" +
171          "/history(?<path>/.*$)")
172     public CompletionStage<List<CommitDto>> getHistory(@Param String projectName,
173                                                        @Param String repoName,
174                                                        @Param String path,
175                                                        @Param @Default("-1") String from,
176                                                        @Param @Default("1") String to) {
177         return projectManager().get(projectName).repos().get(repoName)
178                                .history(new Revision(from),
179                                         new Revision(to),
180                                         path + "**")
181                                .thenApply(commits -> commits.stream()
182                                                             .map(DtoConverter::convert)
183                                                             .collect(toList()));
184     }

186     /**
187      * GET /projects/{projectName}/repositories/{repoName}/search/revisions/{revision}?term={term}
188      * Finds the files matched by {@code term}.
189      */
190     @Get("/projects/{projectName}/repositories/{repoName}/search/revisions/{revision}")
191     public CompletionStage<List<EntryDto>> search(@Param String projectName,
192                                                   @Param String repoName,
193                                                   @Param String revision,
194                                                   @Param String term) {
195         return projectManager().get(projectName).repos().get(repoName)
196                                .find(new Revision(revision), normalizeSearchTerm(term), FIND_ALL_WITH_CONTENT)
197                                .thenApply(entries -> entries.values().stream()
198                                                             .map(DtoConverter::convert)
199                                                             .collect(toList()));
200     }

213         return projectManager().get(projectName).repos().get(repoName)

220     private CompletableFuture<?> push(String projectName, String repoName,
221                                       Revision revision, Author author,
222                                       String commitSummary, String commitDetail, Markup commitMarkup,
223                                       Change<?> change) {
224         final Repository repo = projectManager().get(projectName).repos().get(repoName);
225         return push0(projectName, repoName, repo.normalizeNow(revision), author,
226                      commitSummary, commitDetail, commitMarkup, change);
227     }
228 
229     private CompletableFuture<?> push0(String projectName, String repoName,
230                                        Revision normalizedRev, Author author,
231                                        String commitSummary, String commitDetail, Markup commitMarkup,
232                                        Change<?> change) {
233         return execute(Command.push(author, projectName, repoName, normalizedRev,
234                                     commitSummary, commitDetail, commitMarkup, ImmutableList.of(change)));
235     }

237     private static Entry<CommitMessageDto, Change<?>> commitMessageAndChange(AggregatedHttpRequest request) {
238         try {
239             final JsonNode node = Jackson.readTree(request.contentUtf8());
240             final CommitMessageDto commitMessage =
241                     Jackson.convertValue(node.get("commitMessage"), CommitMessageDto.class);
242             final EntryDto file = Jackson.convertValue(node.get("file"), EntryDto.class);
243             final Change<?> change;
244             switch (file.getType()) {
245                 case "JSON":
246                     change = Change.ofJsonUpsert(file.getPath(), file.getContent());
247                     break;
248                 case "TEXT":
249                     change = Change.ofTextUpsert(file.getPath(), file.getContent());
250                     break;
251                 default:
252                     throw new IllegalArgumentException("unsupported file type: " + file.getType());
253             }
254 
255             return Maps.immutableEntry(commitMessage, change);
256         } catch (IOException e) {
257             throw new IllegalArgumentException("invalid data to be parsed", e);
258         }
259     }
260 
261     private static String normalizeSearchTerm(final String term) {
262         if (Strings.isNullOrEmpty(term)) {
263             throw new IllegalArgumentException("term should not be empty");
264         }
265 
266         final StringBuilder sb = new StringBuilder();
267 
268         for (final String term0 : termSplitter.split(term)) {
269             if (sb.length() > 0) {
270                 sb.append(',');
271             }
272 
273             if (term0.matches(".*[/*]+.*")) {
274                 sb.append(term0);
275             } else {
276                 sb.append('*').append(term0).append('*');
277             }
278         }
279         return sb.toString();
280     }

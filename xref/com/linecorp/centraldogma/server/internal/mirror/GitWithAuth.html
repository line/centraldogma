91      /**
92       * One of the Locks in this array is locked while a Git repository is accessed so that other GitMirrors
93       * that access the same repository cannot access it at the same time. The lock is chosen based on the
94       * hash code of the Git repository path. See {@link #getLock(File)} for more information.
95       *
96       * <p>The number of available locks is hard-coded, but it should be large enough for most use cases.
97       */
98      private static final Lock[] locks = new Lock[1024];
99  
100     static {
101         IsolatedSystemReader.install();
102 
103         for (int i = 0; i < locks.length; i++) {
104             locks[i] = new ReentrantLock();
105         }
106     }
107 
108     private static Lock getLock(File repoDir) {
109         final int h = repoDir.getPath().hashCode();
110         return locks[Math.abs((h ^ h >>> 16) % locks.length)];
111     }
112 
113     private final GitMirror mirror;
114     private final Lock lock;
115     private final Map<String, ProgressMonitor> progressMonitors = new HashMap<>();
116 
117     GitWithAuth(GitMirror mirror, File repoDir) throws IOException {
118         super(repo(repoDir));
119         this.mirror = mirror;
120         lock = getLock(repoDir);
121     }
122 
123     private static Repository repo(File repoDir) throws IOException {
124         final Lock lock = getLock(repoDir);
125         boolean success = false;
126         lock.lock();
127         try {
128             repoDir.getParentFile().mkdirs();
129             final Repository repo = new RepositoryBuilder().setGitDir(repoDir).setBare().build();
130             if (!repo.getObjectDatabase().exists()) {
131                 repo.create(true);
132             }
133 
134             JGitUtil.applyDefaultsAndSave(repo.getConfig());
135             success = true;
136             return repo;
137         } finally {
138             if (!success) {
139                 lock.unlock();
140             }
141         }
142     }
143 
144     @Override
145     public void close() {
146         try {
147             super.close();
148         } finally {
149             try {
150                 getRepository().close();
151             } finally {
152                 lock.unlock();
153             }
154         }
155     }

181     private <T extends TransportCommand<?, ?>> T configure(T command) {
182         final MirrorCredential c = mirror.credential();
183         switch (mirror.remoteRepoUri().getScheme()) {
184             case SCHEME_GIT_HTTP:
185             case SCHEME_GIT_HTTPS:
186                 if (c instanceof PasswordMirrorCredential) {
187                     configureHttp(command, (PasswordMirrorCredential) c);
188                 } else if (c instanceof AccessTokenMirrorCredential) {
189                     configureHttp(command, (AccessTokenMirrorCredential) c);
190                 }
191                 break;
192             case SCHEME_GIT_SSH:
193                 if (c instanceof PasswordMirrorCredential) {
194                     configureSsh(command, (PasswordMirrorCredential) c);
195                 } else if (c instanceof PublicKeyMirrorCredential) {
196                     configureSsh(command, (PublicKeyMirrorCredential) c);
197                 }
198                 break;
199         }
200 
201         return command;
202     }
203 
204     private static <T extends TransportCommand<?, ?>> void configureSsh(T cmd, PublicKeyMirrorCredential cred) {
205         final Collection<KeyPair> keyPairs;
206         try {
207             keyPairs = keyPairResourceParser.loadKeyPairs(null, NamedResource.ofName(cred.username()),
208                                                           passwordProvider(cred.passphrase()),
209                                                           cred.privateKey());
210         } catch (IOException | GeneralSecurityException e) {
211             throw new MirrorException("Unexpected exception while loading private key. username: " +
212                                       cred.username() + ", publicKey: " +
213                                       publicKeyPreview(cred.publicKey()), e);
214         }
215         cmd.setTransportConfigCallback(transport -> {
216             final GitSshdSessionFactory factory = new DefaultGitSshdSessionFactory() {
217                 @Override
218                 void onClientCreated(SshClient client) {
219                     client.setKeyIdentityProvider(KeyIdentityProvider.wrapKeyPairs(keyPairs));
220                 }
221             };
222             final SshTransport sshTransport = (SshTransport) transport;
223             sshTransport.setSshSessionFactory(factory);
224         });
225     }
226 
227     private static <T extends TransportCommand<?, ?>> void configureSsh(T cmd, PasswordMirrorCredential cred) {
228         cmd.setTransportConfigCallback(transport -> {
229             final GitSshdSessionFactory factory = new DefaultGitSshdSessionFactory() {
230                 @Override
231                 void onClientCreated(SshClient client) {
232                     client.setFilePasswordProvider(passwordProvider(cred.password()));
233                 }
234             };
235             final SshTransport sshTransport = (SshTransport) transport;
236             sshTransport.setSshSessionFactory(factory);
237         });
238     }

240     private static FilePasswordProvider passwordProvider(@Nullable String passphrase) {
241         if (passphrase == null) {
242             return FilePasswordProvider.EMPTY;
243         }
244 
245         return FilePasswordProvider.of(passphrase);
246     }

248     private static <T extends TransportCommand<?, ?>> void configureHttp(T cmd, PasswordMirrorCredential cred) {
249         cmd.setCredentialsProvider(new UsernamePasswordCredentialsProvider(cred.username(), cred.password()));
250     }
251 
252     private static <T extends TransportCommand<?, ?>> void configureHttp(
253             T cmd, AccessTokenMirrorCredential cred) {
254         cmd.setCredentialsProvider(new UsernamePasswordCredentialsProvider("token", cred.accessToken()));
255     }

257     private final class MirrorProgressMonitor extends EmptyProgressMonitor {
258 
259         private final String operationName;
260 
261         MirrorProgressMonitor(String operationName) {
262             this.operationName = requireNonNull(operationName, "operationName");
263         }
264 
265         @Override
266         public void beginTask(String title, int totalWork) {
267             if (totalWork > 0 && logger.isInfoEnabled()) {
268                 logger.info("[{}] {} ({}, total: {})", operationName, mirror.remoteRepoUri(), title, totalWork);
269             }
270         }
271     }

273     private static class DefaultGitSshdSessionFactory extends GitSshdSessionFactory {
274         @Override
275         public RemoteSession getSession(URIish uri, CredentialsProvider credentialsProvider,
276                                         FS fs, int tms) throws TransportException {
277             try {
278                 return new GitSshdSession(uri, NoopCredentialsProvider.INSTANCE, fs, tms) {
279                     @Override
280                     protected SshClient createClient() {
281                         // Not an Armeria but an SSHD client.
282                         final ClientBuilder builder = ClientBuilder.builder();
283                         // Do not use local file system.
284                         builder.hostConfigEntryResolver(HostConfigEntryResolver.EMPTY);
285                         builder.fileSystemFactory(NoneFileSystemFactory.INSTANCE);
286                         // Do not verify the server key.
287                         builder.serverKeyVerifier((clientSession, remoteAddress, serverKey) -> true);
288                         final SshClient client = builder.build();
289                         onClientCreated(client);
290                         return client;
291                     }
292 
293                     @Override
294                     protected ClientSession createClientSession(
295                             SshClient clientInstance, String host, String username, int port,
296                             String... passwords) throws IOException, InterruptedException {
297                         if (port <= 0) {
298                             port = 22; // Use the SSH default port it unspecified.
299                         }
300                         return super.createClientSession(clientInstance, host, username,
301                                                          port, passwords);
302                     }
303                 };
304             } catch (Exception e) {
305                 throw new TransportException("Unable to connect to: " + uri +
306                                              " CredentialsProvider: " + credentialsProvider, e);
307             }
308         }
309 
310         void onClientCreated(SshClient client) {}
311     }

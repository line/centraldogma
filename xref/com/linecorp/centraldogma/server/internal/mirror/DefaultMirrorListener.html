17  package com.linecorp.centraldogma.server.internal.mirror;
18  
19  import org.slf4j.Logger;
20  import org.slf4j.LoggerFactory;
21  
22  import com.linecorp.centraldogma.server.metadata.User;
23  import com.linecorp.centraldogma.server.mirror.Mirror;
24  import com.linecorp.centraldogma.server.mirror.MirrorAccessController;
25  import com.linecorp.centraldogma.server.mirror.MirrorListener;
26  import com.linecorp.centraldogma.server.mirror.MirrorResult;
27  import com.linecorp.centraldogma.server.mirror.MirrorTask;
28  
29  enum DefaultMirrorListener implements MirrorListener {
30  
31      INSTANCE;
32  
33      private static final Logger logger = LoggerFactory.getLogger(DefaultMirrorListener.class);
34  
35      @Override
36      public void onCreate(Mirror mirror, User creator, MirrorAccessController accessController) {
37          logger.debug("A new mirroring from {} is created. mirror: {}", mirror.remoteRepoUri(), mirror);
38      }
39  
40      @Override
41      public void onUpdate(Mirror mirror, User updater, MirrorAccessController accessController) {
42          logger.debug("The mirroring ID {} is updated. mirror: {}", mirror.id(), mirror);
43      }
44  
45      @Override
46      public void onDisallowed(Mirror mirror) {
47          logger.debug("The mirroring from {} is not allowed. mirror: {}",
48                       mirror.remoteRepoUri(), mirror);
49      }
50  
51      @Override
52      public void onStart(MirrorTask mirrorTask) {
53          if (mirrorTask.scheduled()) {
54              logger.info("Mirroring: {}", mirrorTask);
55          }
56      }
57  
58      @Override
59      public void onComplete(MirrorTask mirrorTask, MirrorResult result) {
60          // Do nothing
61      }
62  
63      @Override
64      public void onError(MirrorTask mirrorTask, Throwable cause) {
65          if (mirrorTask.scheduled()) {
66              logger.warn("Unexpected exception while mirroring: {}", mirrorTask, cause);
67          }
68      }
69  }

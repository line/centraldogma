68  public final class MirrorSchedulingService implements MirroringService {
69  
70      private static final Logger logger = LoggerFactory.getLogger(MirrorSchedulingService.class);
71  
72      static final MirrorListener mirrorListener;
73  
74      static {
75          final List<MirrorListener> listeners =
76                  ImmutableList.copyOf(ServiceLoader.load(MirrorListener.class,
77                                                          DefaultMirroringServicePlugin.class.getClassLoader()));
78          if (listeners.isEmpty()) {
79              mirrorListener = DefaultMirrorListener.INSTANCE;
80          } else {
81              mirrorListener = new CompositeMirrorListener(listeners);
82          }
83      }
84  
85      /**
86       * How often to check the mirroring schedules. i.e. every second.
87       */
88      private static final Duration TICK = Duration.ofSeconds(1);
89  
90      private final File workDir;
91      private final ProjectManager projectManager;
92      private final int numThreads;
93      private final int maxNumFilesPerMirror;
94      private final long maxNumBytesPerMirror;
95      @Nullable
96      private final ZoneConfig zoneConfig;
97      @Nullable
98      private final String currentZone;
99  
100     private volatile CommandExecutor commandExecutor;
101     private volatile ListeningScheduledExecutorService scheduler;
102     private volatile ListeningExecutorService worker;
103 
104     private ZonedDateTime lastExecutionTime;
105     private final MeterRegistry meterRegistry;
106 
107     @VisibleForTesting
108     public MirrorSchedulingService(File workDir, ProjectManager projectManager, MeterRegistry meterRegistry,
109                                    int numThreads, int maxNumFilesPerMirror, long maxNumBytesPerMirror,
110                                    @Nullable ZoneConfig zoneConfig) {
111 
112         this.workDir = requireNonNull(workDir, "workDir");
113         this.projectManager = requireNonNull(projectManager, "projectManager");
114         this.meterRegistry = requireNonNull(meterRegistry, "meterRegistry");
115 
116         checkArgument(numThreads > 0, "numThreads: %s (expected: > 0)", numThreads);
117         checkArgument(maxNumFilesPerMirror > 0,
118                       "maxNumFilesPerMirror: %s (expected: > 0)", maxNumFilesPerMirror);
119         checkArgument(maxNumBytesPerMirror > 0,
120                       "maxNumBytesPerMirror: %s (expected: > 0)", maxNumBytesPerMirror);
121         this.numThreads = numThreads;
122         this.maxNumFilesPerMirror = maxNumFilesPerMirror;
123         this.maxNumBytesPerMirror = maxNumBytesPerMirror;
124         this.zoneConfig = zoneConfig;
125         if (zoneConfig != null) {
126             currentZone = zoneConfig.currentZone();
127         } else {
128             currentZone = null;
129         }
130     }
131 
132     public boolean isStarted() {
133         return scheduler != null;
134     }
135 
136     public synchronized void start(CommandExecutor commandExecutor) {
137         if (isStarted()) {
138             return;
139         }
140 
141         this.commandExecutor = requireNonNull(commandExecutor, "commandExecutor");
142 
143         ScheduledExecutorService executorService = Executors.newSingleThreadScheduledExecutor(
144                 new DefaultThreadFactory("mirroring-scheduler", true));
145         executorService = ExecutorServiceMetrics.monitor(meterRegistry, executorService,
146                                                          "mirroringScheduler");
147         scheduler = MoreExecutors.listeningDecorator(executorService);
148 
149         // Use SynchronousQueue to prevent the work queue from growing infinitely
150         // when the workers cannot handle the mirroring tasks fast enough.
151         final SynchronousQueue<Runnable> workQueue = new SynchronousQueue<>();
152         worker = MoreExecutors.listeningDecorator(new ThreadPoolExecutor(
153                 0, numThreads, 90, TimeUnit.SECONDS, workQueue,
154                 new DefaultThreadFactory("mirroring-worker", true),
155                 (rejectedTask, executor) -> {
156                     // We do not want the mirroring tasks to be rejected.
157                     // Just wait until a worker thread takes it.
158                     try {
159                         workQueue.put(rejectedTask);
160                     } catch (InterruptedException e) {
161                         // Propagate the interrupt to the scheduler.
162                         Thread.currentThread().interrupt();
163                     }
164                 }));
165 
166         final ListenableScheduledFuture<?> future = scheduler.scheduleWithFixedDelay(
167                 this::scheduleMirrors,
168                 TICK.getSeconds(), TICK.getSeconds(), TimeUnit.SECONDS);
169 
170         Futures.addCallback(future, new FutureCallback<Object>() {
171             @Override
172             public void onSuccess(@Nullable Object result) {}
173 
174             @Override
175             public void onFailure(Throwable cause) {
176                 logger.error("Git mirroring scheduler stopped due to an unexpected exception:", cause);
177             }
178         }, MoreExecutors.directExecutor());
179     }
180 
181     public synchronized void stop() {
182         final ExecutorService scheduler = this.scheduler;
183         final ExecutorService worker = this.worker;
184 
185         try {
186             final boolean interrupted = terminate(scheduler) || terminate(worker);
187             if (interrupted) {
188                 Thread.currentThread().interrupt();
189             }
190         } finally {
191             this.scheduler = null;
192             this.worker = null;
193         }
194     }
195 
196     private void scheduleMirrors() {
197         final ZonedDateTime now = ZonedDateTime.now();
198         if (lastExecutionTime == null) {
199             lastExecutionTime = now.minus(TICK);
200         }
201 
202         final ZonedDateTime currentLastExecutionTime = lastExecutionTime;
203         lastExecutionTime = now;
204 
205         projectManager.list()
206                       .values()
207                       .forEach(project -> {
208                           final List<Mirror> mirrors;
209                           try {
210                               mirrors = project.metaRepo().mirrors()
211                                                .get(5, TimeUnit.SECONDS);
212                           } catch (TimeoutException e) {
213                               logger.warn("Failed to load the mirror list within 5 seconds. project: {}",
214                                           project.name(), e);
215                               return;
216                           } catch (Exception e) {
217                               logger.warn("Failed to load the mirror list from: {}", project.name(), e);
218                               return;
219                           }
220                           for (Mirror m : mirrors) {
221                               if (m.schedule() == null) {
222                                   continue;
223                               }
224                               if (zoneConfig != null) {
225                                   String pinnedZone = m.zone();
226                                   if (pinnedZone == null) {
227                                       // Use the first zone if the mirror does not specify a zone.
228                                       pinnedZone = zoneConfig.allZones().get(0);
229                                   }
230                                   if (!pinnedZone.equals(currentZone)) {
231                                       // Skip the mirror if it is pinned to a different zone.
232                                       if (!zoneConfig.allZones().contains(pinnedZone)) {
233                                           // The mirror is pinned to an invalid zone.
234                                           final MirrorTask invalidMirror =
235                                                   new MirrorTask(m, User.SYSTEM, Instant.now(),
236                                                                  pinnedZone, true);
237                                           mirrorListener.onStart(invalidMirror);
238                                           mirrorListener.onError(invalidMirror, new MirrorException(
239                                                   "The mirror is pinned to an unknown zone: " + pinnedZone +
240                                                   " (valid zones: " + zoneConfig.allZones() + ')'));
241                                       }
242                                       continue;
243                                   }
244                               }
245                               try {
246                                   if (m.nextExecutionTime(currentLastExecutionTime).compareTo(now) < 0) {
247                                       runAsync(new MirrorTask(m, User.SYSTEM, Instant.now(),
248                                                               currentZone, true));
249                                   }
250                               } catch (Exception e) {
251                                   logger.warn("Unexpected exception while mirroring: {}", m, e);
252                               }
253                           }
254                       });
255     }
256 
257     @Override
258     public CompletableFuture<Void> mirror() {
259         // Note: This method is not used in the production code.
260         if (commandExecutor == null) {
261             return CompletableFuture.completedFuture(null);
262         }
263 
264         return CompletableFuture.runAsync(
265                 () -> projectManager.list().values().forEach(p -> {
266                     try {
267                         p.metaRepo().mirrors().get(5, TimeUnit.SECONDS)
268                          .forEach(m -> run(new MirrorTask(m, User.SYSTEM, Instant.now(), currentZone, false)));
269                     } catch (InterruptedException | TimeoutException | ExecutionException e) {
270                         throw new IllegalStateException(
271                                 "Failed to load mirror list with in 5 seconds. project: " + p.name(), e);
272                     }
273                 }),
274                 worker);
275     }
276 
277     private void runAsync(MirrorTask m) {
278         worker.submit(() -> run(m));
279     }
280 
281     private void run(MirrorTask mirrorTask) {
282         try {
283             mirrorListener.onStart(mirrorTask);
284             final MirrorResult result =
285                     new InstrumentedMirroringJob(mirrorTask, meterRegistry)
286                             .run(workDir, commandExecutor, maxNumFilesPerMirror, maxNumBytesPerMirror);
287             mirrorListener.onComplete(mirrorTask, result);
288         } catch (Exception e) {
289             mirrorListener.onError(mirrorTask, e);
290             if (!mirrorTask.scheduled()) {
291                 if (e instanceof MirrorException) {
292                     throw (MirrorException) e;
293                 } else {
294                     throw new MirrorException(e);
295                 }
296             }
297         }
298     }

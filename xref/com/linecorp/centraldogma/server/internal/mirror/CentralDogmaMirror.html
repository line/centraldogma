96      private CentralDogma createRemoteClient(long maxNumBytes) throws UnknownHostException {
97          final URI uri = remoteUri().uri();
98          final ArmeriaCentralDogmaBuilder builder = new ArmeriaCentralDogmaBuilder();
99          final int port = uri.getPort();
100         if (port > 0) {
101             builder.host(uri.getHost(), port);
102         } else {
103             builder.host(uri.getHost());
104         }
105         if (SCHEME_DOGMA_HTTPS.equals(uri.getScheme())) {
106             builder.useTls(true);
107         }
108         final Credential cred = credential();
109         if (cred instanceof AccessTokenCredential) {
110             builder.accessToken(((AccessTokenCredential) cred).accessToken());
111         }
112         // TODO(minwoox) Support mTLS authentication as well.
113 
114         builder.clientConfigurator(cb -> cb.maxResponseLength(maxNumBytes));
115         return builder.build();
116     }

118     @Override
119     protected MirrorResult mirrorLocalToRemote(File workDir, int maxNumFiles, long maxNumBytes,
120                                                Instant triggeredTime) throws Exception {
121         final Revision localHead = localRepo().normalizeNow(Revision.HEAD);
122 
123         try (CentralDogma remote = createRemoteClient(maxNumBytes)) {
124             final CentralDogmaRepository repo = remote.forRepo(remoteProject, remoteRepo);
125             // Get remote HEAD revision.
126             final Revision remoteHead = repo.normalize(Revision.HEAD).join();
127 
128             // Load old MirrorState from remote.
129             final String mirrorStatePath = remotePath() + LOCAL_TO_REMOTE_MIRROR_STATE_FILE_NAME;
130             final MirrorState oldMirrorState = loadRemoteMirrorState(repo, mirrorStatePath, remoteHead);
131 
132             // Determine whether to run.
133             final MirrorDecision decision = shouldRunLocalToRemote(
134                     oldMirrorState, localHead, remoteHead);
135             if (decision == MirrorDecision.SKIP) {
136                 final String description = String.format(
137                         "The remote repository '%s' already at %s. Local repository: '%s/%s'",
138                         remoteUri(), localHead, localRepo().parent().name(), localRepo().name());
139                 logger.debug(description);
140                 return newMirrorResult(MirrorStatus.UP_TO_DATE, description, triggeredTime);
141             }
142 
143             // Read local files under localPath and apply gitignore filter.
144             final Map<String, Entry<?>> localEntries = filterByGitignore(
145                     localRepo().find(localHead, localPath() + "**", FIND_ALL_WITH_CONTENT).join(),
146                     localPath());
147 
148             // Fetch remote files for comparison/removal detection.
149             final Map<String, Entry<?>> remoteEntries =
150                     repo.file(PathPattern.of(remotePath() + "**")).viewRaw(true).get(remoteHead).join();
151 
152             // Build changes.
153             final List<Change<?>> changes = new ArrayList<>();
154             long numFiles = 0;
155             long numBytes = 0;
156 
157             for (Map.Entry<String, Entry<?>> entry : localEntries.entrySet()) {
158                 final String localFilePath = entry.getKey();
159                 final Entry<?> localEntry = entry.getValue();
160 
161                 if (localEntry.type() == EntryType.DIRECTORY) {
162                     continue;
163                 }
164 
165                 // Skip the mirror state file.
166                 if (localFilePath.endsWith(MIRROR_STATE_FILE_NAME)) {
167                     continue;
168                 }
169 
170                 // Map local path to remote path.
171                 final String remoteFilePath = remotePath() +
172                                               localFilePath.substring(localPath().length());
173 
174                 if (!Util.isValidFilePath(remoteFilePath)) {
175                     continue;
176                 }
177 
178                 if (++numFiles > maxNumFiles) {
179                     throw newMirrorException(maxNumFiles, "files");
180                 }
181 
182                 final String content = localEntry.contentAsText();
183                 if (content != null) {
184                     numBytes += content.length();
185                     if (numBytes > maxNumBytes) {
186                         throw newMirrorException(maxNumBytes, "bytes");
187                     }
188                 }
189 
190                 changes.add(newChange(remoteFilePath, localEntry));
191             }
192 
193             // Detect removals: files on remote that don't have a corresponding local file.
194             for (Map.Entry<String, Entry<?>> entry : remoteEntries.entrySet()) {
195                 final String remoteFilePath = entry.getKey();
196                 final Entry<?> remoteEntry = entry.getValue();
197 
198                 if (remoteEntry.type() == EntryType.DIRECTORY) {
199                     continue;
200                 }
201 
202                 // Skip the mirror state file.
203                 if (remoteFilePath.endsWith(LOCAL_TO_REMOTE_MIRROR_STATE_FILE_NAME)) {
204                     continue;
205                 }
206 
207                 // Map remote path to local path.
208                 final String localFilePath = localPath() +
209                                              remoteFilePath.substring(remotePath().length());
210 
211                 if (!localEntries.containsKey(localFilePath)) {
212                     changes.add(Change.ofRemoval(remoteFilePath));
213                 }
214             }
215 
216             // Add the mirror state file.
217             final MirrorState newMirrorState = new MirrorState(
218                     localHead.text(), remoteHead.text(), localHead.text(),
219                     MirrorDirection.LOCAL_TO_REMOTE, hashString());
220             changes.add(Change.ofJsonUpsert(mirrorStatePath,
221                                             Jackson.valueToTree(newMirrorState)));
222 
223             if (decision == MirrorDecision.COMPARE_AND_RUN &&
224                 !hasLocalToRemoteChanges(changes, remoteEntries)) {
225                 final String description = String.format(
226                         "The remote repository '%s' already at %s. Local repository: '%s/%s'",
227                         remoteUri(), localHead, localRepo().parent().name(), localRepo().name());
228                 logger.debug(description);
229                 return newMirrorResult(MirrorStatus.UP_TO_DATE, description, triggeredTime);
230             }
231 
232             final String summary = "Mirror '" + localRepo().name() + "' at " + localHead +
233                                    " to the repository '" + remoteUri() + '\'';
234 
235             try {
236                 repo.commit(summary, changes).push(Revision.HEAD).join();
237                 logger.info(summary);
238                 return newMirrorResult(MirrorStatus.SUCCESS, summary, triggeredTime);
239             } catch (Throwable t) {
240                 final Throwable peeled = Exceptions.peel(t);
241                 if (peeled instanceof RedundantChangeException) {
242                     return newMirrorResult(MirrorStatus.UP_TO_DATE, summary, triggeredTime);
243                 }
244                 return Exceptions.throwUnsafely(peeled);
245             }
246         } catch (Throwable t) {
247             final Throwable peeled = Exceptions.peel(t);
248             if (peeled instanceof MirrorException) {
249                 return Exceptions.throwUnsafely(peeled);
250             }
251             throw new MirrorException("unexpected exception while mirror local to remote repository. " +
252                                       "local: " + localRepo().parent().name() + '/' + localRepo().name() +
253                                       ", remote: " + remoteProject + '/' + remoteRepo, t);
254         }
255     }

257     @Override
258     protected MirrorResult mirrorRemoteToLocal(File workDir, CommandExecutor executor,
259                                                int maxNumFiles, long maxNumBytes, Instant triggeredTime)
260             throws Exception {
261         final Revision localRev = localRepo().normalizeNow(Revision.HEAD);
262         final String mirrorStatePath = localPath() + MIRROR_STATE_FILE_NAME;
263 
264         try (CentralDogma remote = createRemoteClient(maxNumBytes)) {
265             final CentralDogmaRepository repo = remote.forRepo(remoteProject, remoteRepo);
266             // Get remote HEAD revision.
267             final Revision remoteHead = repo.normalize(Revision.HEAD).join();
268 
269             // Load old MirrorState from local repo.
270             final MirrorState oldMirrorState = loadLocalMirrorState(mirrorStatePath, localRev);
271 
272             // Determine whether to run.
273             final MirrorDecision decision = shouldRunRemoteToLocal(oldMirrorState, localRev, remoteHead);
274             if (decision == MirrorDecision.SKIP) {
275                 final String description = String.format(
276                         "Repository '%s/%s' already at %s, %s",
277                         localRepo().parent().name(), localRepo().name(),
278                         remoteHead, remoteUri());
279                 logger.debug(description);
280                 return newMirrorResult(MirrorStatus.UP_TO_DATE, description, triggeredTime);
281             }
282 
283             // Fetch all remote files under remotePath and apply gitignore filter.
284             final Map<String, Entry<?>> remoteEntries = filterByGitignore(
285                     repo.file(PathPattern.of(remotePath() + "**")).viewRaw(true).get(remoteHead).join(),
286                     remotePath());
287 
288             // Build Change objects.
289             final Map<String, Change<?>> changes = new HashMap<>();
290 
291             // Add mirror_state.json.
292             final MirrorState newMirrorState = new MirrorState(
293                     remoteHead.text(), remoteHead.text(), localRev.text(),
294                     MirrorDirection.REMOTE_TO_LOCAL, hashString());
295             changes.put(mirrorStatePath,
296                         Change.ofJsonUpsert(mirrorStatePath, Jackson.valueToTree(newMirrorState)));
297 
298             long numFiles = 0;
299             long numBytes = 0;
300 
301             for (Map.Entry<String, Entry<?>> entry : remoteEntries.entrySet()) {
302                 final String remoteFilePath = entry.getKey();
303                 final Entry<?> remoteEntry = entry.getValue();
304 
305                 if (remoteEntry.type() == EntryType.DIRECTORY) {
306                     continue;
307                 }
308 
309                 // Skip files not under remotePath.
310                 if (!remoteFilePath.startsWith(remotePath())) {
311                     continue;
312                 }
313 
314                 // Map remote path to local path.
315                 final String localFilePath = localPath() +
316                                              remoteFilePath.substring(remotePath().length());
317 
318                 if (!Util.isValidFilePath(localFilePath)) {
319                     continue;
320                 }
321 
322                 if (++numFiles > maxNumFiles) {
323                     throw newMirrorException(maxNumFiles, "files");
324                 }
325 
326                 final String content = remoteEntry.contentAsText();
327                 if (content != null) {
328                     numBytes += content.length();
329                     if (numBytes > maxNumBytes) {
330                         throw newMirrorException(maxNumBytes, "bytes");
331                     }
332                 }
333 
334                 changes.put(localFilePath, newChange(localFilePath, remoteEntry));
335             }
336 
337             // Detect removals.
338             final Map<FindOption<?>, ?> findOptions =
339                     decision == MirrorDecision.COMPARE_AND_RUN ? FIND_ALL_WITH_CONTENT
340                                                                : FIND_ALL_WITHOUT_CONTENT;
341             final Map<String, Entry<?>> oldEntries =
342                     localRepo().find(localRev, localPath() + "**", findOptions).join();
343 
344             if (decision == MirrorDecision.COMPARE_AND_RUN) {
345                 if (!hasRemoteToLocalChanges(changes, oldEntries)) {
346                     final String description = String.format(
347                             "Repository '%s/%s' already at %s, %s",
348                             localRepo().parent().name(), localRepo().name(),
349                             remoteHead, remoteUri());
350                     logger.debug(description);
351                     return newMirrorResult(MirrorStatus.UP_TO_DATE, description, triggeredTime);
352                 }
353             }
354 
355             oldEntries.keySet().removeAll(changes.keySet());
356             oldEntries.forEach((path, entry) -> {
357                 if (entry.type() != EntryType.DIRECTORY && !changes.containsKey(path)) {
358                     changes.put(path, Change.ofRemoval(path));
359                 }
360             });
361 
362             final String summary = "Mirror " + remoteHead + ", '" + remoteUri() +
363                                    "' to the repository '" + localRepo().name() + '\'';
364             try {
365                 final Revision revision = executor.execute(Command.push(
366                         MIRROR_AUTHOR, localRepo().parent().name(), localRepo().name(),
367                         Revision.HEAD, summary, "", Markup.PLAINTEXT,
368                         changes.values())).join();
369                 final String description = summary + ", revision: " + revision.text();
370                 return newMirrorResult(MirrorStatus.SUCCESS, description, triggeredTime);
371             } catch (Throwable t) {
372                 final Throwable peeled = Exceptions.peel(t);
373                 if (peeled instanceof RedundantChangeException) {
374                     return newMirrorResult(MirrorStatus.UP_TO_DATE, summary, triggeredTime);
375                 }
376                 return Exceptions.throwUnsafely(peeled);
377             }
378         } catch (Throwable t) {
379             final Throwable peeled = Exceptions.peel(t);
380             if (peeled instanceof MirrorException) {
381                 return Exceptions.throwUnsafely(peeled);
382             }
383             throw new MirrorException("unexpected exception while mirror remote to local repository. " +
384                                       "local: " + localRepo().parent().name() + '/' + localRepo().name() +
385                                       ", remote: " + remoteProject + '/' + remoteRepo, t);
386         }
387     }

389     private MirrorDecision shouldRunLocalToRemote(@Nullable MirrorState oldMirrorState,
390                                                   Revision localHead,
391                                                   Revision remoteHead) {
392         if (oldMirrorState == null) {
393             return MirrorDecision.RUN;
394         }
395         if (!hashString().equals(oldMirrorState.configHash())) {
396             return MirrorDecision.RUN;
397         }
398         if (oldMirrorState.remoteRevision() == null || oldMirrorState.localRevision() == null) {
399             return MirrorDecision.RUN;
400         }
401         if (!localHead.text().equals(oldMirrorState.localRevision())) {
402             return MirrorDecision.RUN;
403         }
404 
405         // Use backward(1) because the previous mirroring itself created a commit on the remote.
406         if (!remoteHead.backward(1).text().equals(oldMirrorState.remoteRevision())) {
407             // Something changed in the mirrored remote repository since the last mirroring.
408             final String remotePath = remotePath();
409             if ("/".equals(remotePath)) {
410                 return MirrorDecision.RUN;
411             }
412             // If the remote path is not the root, check whether there are changes under the remote path only.
413             return MirrorDecision.COMPARE_AND_RUN;
414         }
415 
416         return MirrorDecision.SKIP;
417     }

419     private MirrorDecision shouldRunRemoteToLocal(@Nullable MirrorState oldMirrorState,
420                                                   Revision localHead,
421                                                   Revision remoteHead) {
422         if (oldMirrorState == null) {
423             return MirrorDecision.RUN;
424         }
425         if (!hashString().equals(oldMirrorState.configHash())) {
426             return MirrorDecision.RUN;
427         }
428         if (oldMirrorState.remoteRevision() == null || oldMirrorState.localRevision() == null) {
429             return MirrorDecision.RUN;
430         }
431         if (!remoteHead.text().equals(oldMirrorState.remoteRevision())) {
432             return MirrorDecision.RUN;
433         }
434         // Use backward(1) because the previous mirroring itself created a commit on the local.
435         if (!localHead.backward(1).text().equals(oldMirrorState.localRevision())) {
436             // Something changed in the mirrored local repository since the last mirroring.
437             final String localPath = localPath();
438             if ("/".equals(localPath)) {
439                 return MirrorDecision.RUN;
440             }
441             // If the local path is not the root, check whether there are changes under the local path only.
442             return MirrorDecision.COMPARE_AND_RUN;
443         }
444         return MirrorDecision.SKIP;
445     }

447     @Nullable
448     private static MirrorState loadRemoteMirrorState(CentralDogmaRepository repo, String mirrorStatePath,
449                                                      Revision remoteHead) {
450         final Entry<?> entry;
451         try {
452             entry = repo.file(mirrorStatePath).get(remoteHead).join();
453             if (entry.type() != EntryType.JSON) {
454                 return null;
455             }
456         } catch (Exception e) {
457             final Throwable peeled = Exceptions.peel(e);
458             if (peeled instanceof EntryNotFoundException) {
459                 return null;
460             }
461             return Exceptions.throwUnsafely(peeled);
462         }
463         try {
464             return Jackson.treeToValue((TreeNode) entry.content(), MirrorState.class);
465         } catch (Exception e) {
466             logger.warn("Failed to load mirror state from remote: {}", mirrorStatePath, e);
467             return null;
468         }
469     }
470 
471     @Nullable
472     private MirrorState loadLocalMirrorState(String mirrorStatePath, Revision localRev) {
473         final Entry<?> entry = localRepo().getOrNull(localRev, mirrorStatePath).join();
474         if (entry == null || entry.type() != EntryType.JSON) {
475             return null;
476         }
477         try {
478             return Jackson.treeToValue((TreeNode) entry.content(), MirrorState.class);
479         } catch (Exception e) {
480             logger.warn("Failed to load mirror state from local: {}", mirrorStatePath, e);
481             return null;
482         }
483     }

485     private static Change<?> newChange(String path, Entry<?> entry) {
486         switch (EntryType.guessFromPath(path)) {
487             case JSON:
488                 return Change.ofJsonUpsert(path, entry.contentAsText());
489             case YAML:
490                 return Change.ofYamlUpsert(path, entry.contentAsText());
491             case TEXT:
492             default:
493                 return Change.ofTextUpsert(path, entry.contentAsText());
494         }
495     }

497     private MirrorException newMirrorException(long limit, String filesOrBytes) {
498         return new MirrorException(
499                 "mirror (id: " + id() + ", localRepo: " + localRepo().parent().name() +
500                 '/' + localRepo().name() + ", remoteUri: " + remoteUri() +
501                 ") contains more than " + limit + ' ' + filesOrBytes);
502     }

504     private static boolean hasLocalToRemoteChanges(List<Change<?>> changes,
505                                                    Map<String, Entry<?>> remoteEntries) {
506         for (Change<?> change : changes) {
507             final String path = change.path();
508             if (path.endsWith(LOCAL_TO_REMOTE_MIRROR_STATE_FILE_NAME)) {
509                 continue;
510             }
511             // A removal is always a real change.
512             if (change.type() == ChangeType.REMOVE) {
513                 return true;
514             }
515             final Entry<?> remoteEntry = remoteEntries.get(path);
516             if (remoteEntry == null) {
517                 // New file added.
518                 return true;
519             }
520             final String newContent = change.contentAsText();
521             final String oldContent = remoteEntry.contentAsText();
522             if (newContent != null && !newContent.equals(oldContent)) {
523                 return true;
524             }
525         }
526         return false;
527     }
528 
529     private static boolean hasRemoteToLocalChanges(Map<String, Change<?>> newChanges,
530                                                    Map<String, Entry<?>> oldEntries) {
531         for (Change<?> change : newChanges.values()) {
532             final String path = change.path();
533             if (path.endsWith(MIRROR_STATE_FILE_NAME)) {
534                 continue;
535             }
536             if (!oldEntries.containsKey(path)) {
537                 return true;
538             }
539         }
540         for (Entry<?> entry : oldEntries.values()) {
541             if (entry.type() == EntryType.DIRECTORY) {
542                 continue;
543             }
544             final String path = entry.path();
545             if (path.endsWith(MIRROR_STATE_FILE_NAME)) {
546                 continue;
547             }
548             if (!newChanges.containsKey(path)) {
549                 return true;
550             }
551         }
552         for (Change<?> change : newChanges.values()) {
553             final String path = change.path();
554             if (path.endsWith(MIRROR_STATE_FILE_NAME)) {
555                 continue;
556             }
557             final Entry<?> oldEntry = oldEntries.get(path);
558             if (oldEntry == null) {
559                 continue;
560             }
561             final String newContent = change.contentAsText();
562             final String oldContent = oldEntry.contentAsText();
563             if (newContent != null && !newContent.equals(oldContent)) {
564                 return true;
565             }
566         }
567         return false;
568     }

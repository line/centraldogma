54  public final class MirrorRunner implements SafeCloseable {
55  
56      private final ProjectApiManager projectApiManager;
57      private final CommandExecutor commandExecutor;
58      private final File workDir;
59      private final MirroringServicePluginConfig mirrorConfig;
60      private final ExecutorService worker;
61  
62      private final Map<MirrorKey, CompletableFuture<MirrorResult>> inflightRequests = new ConcurrentHashMap<>();
63      @Nullable
64      private final String currentZone;
65      private final MirrorAccessController mirrorAccessController;
66  
67      public MirrorRunner(ProjectApiManager projectApiManager, CommandExecutor commandExecutor,
68                          CentralDogmaConfig cfg, MeterRegistry meterRegistry,
69                          MirrorAccessController mirrorAccessController) {
70          this.projectApiManager = projectApiManager;
71          this.commandExecutor = commandExecutor;
72          // TODO(ikhoon): Periodically clean up stale repositories.
73          workDir = new File(cfg.dataDir(), "_mirrors_manual");
74          MirroringServicePluginConfig mirrorConfig =
75                  (MirroringServicePluginConfig) cfg.pluginConfigMap()
76                                                    .get(MirroringServicePluginConfig.class);
77          if (mirrorConfig == null) {
78              mirrorConfig = MirroringServicePluginConfig.INSTANCE;
79          }
80          this.mirrorConfig = mirrorConfig;
81          if (cfg.zone() != null) {
82              currentZone = cfg.zone().currentZone();
83          } else {
84              currentZone = null;
85          }
86          this.mirrorAccessController = mirrorAccessController;
87  
88          final ThreadPoolExecutor threadPoolExecutor = new ThreadPoolExecutor(
89                  0, mirrorConfig.numMirroringThreads(),
90                  // TODO(minwoox): Use LinkedTransferQueue when we upgrade to JDK 21.
91                  60, TimeUnit.SECONDS, new LinkedBlockingQueue<>(),
92                  new DefaultThreadFactory("mirror-api-worker", true));
93          threadPoolExecutor.allowCoreThreadTimeOut(true);
94          worker = ExecutorServiceMetrics.monitor(meterRegistry, threadPoolExecutor,
95                                                  "mirrorApiWorker");
96      }
97  
98      public CompletableFuture<MirrorResult> run(String projectName, String repoName,
99                                                 String mirrorId, User user) {
100         // If there is an inflight request, return it to avoid running the same mirror task multiple times.
101         return inflightRequests.computeIfAbsent(new MirrorKey(projectName, repoName, mirrorId),
102                                                 key -> run(key, user));
103     }
104 
105     private CompletableFuture<MirrorResult> run(MirrorKey mirrorKey, User user) {
106         try {
107             final CompletableFuture<MirrorResult> future =
108                     metaRepo(mirrorKey.projectName).mirror(mirrorKey.repoName, mirrorKey.mirrorId).thenCompose(
109                             mirror -> {
110                                 if (!mirror.enabled()) {
111                                     throw new MirrorException("The mirror is disabled: " +
112                                                               mirrorKey.repoName + '/' + mirrorKey.mirrorId);
113                                 }
114 
115                                 return mirrorAccessController.isAllowed(mirror).thenApplyAsync(allowed -> {
116                                     if (!allowed) {
117                                         throw new MirrorAccessException(
118                                                 "The mirroring from " + mirror.remoteRepoUri() +
119                                                 " is not allowed: " +
120                                                 mirrorKey.repoName + '/' + mirrorKey.mirrorId);
121                                     }
122 
123                                     final String zone = mirror.zone();
124                                     if (zone != null && !zone.equals(currentZone)) {
125                                         throw new MirrorException(
126                                                 "The mirror is not in the current zone: " + currentZone);
127                                     }
128                                     final MirrorTask mirrorTask = new MirrorTask(mirror, user, Instant.now(),
129                                                                                  currentZone, false);
130                                     final MirrorListener listener = MirrorSchedulingService.mirrorListener();
131                                     listener.onStart(mirrorTask);
132                                     try {
133                                         final MirrorResult mirrorResult =
134                                                 mirror.mirror(workDir, commandExecutor,
135                                                               mirrorConfig.maxNumFilesPerMirror(),
136                                                               mirrorConfig.maxNumBytesPerMirror(),
137                                                               mirrorTask.triggeredTime());
138                                         listener.onComplete(mirrorTask, mirrorResult);
139                                         return mirrorResult;
140                                     } catch (Exception e) {
141                                         listener.onError(mirrorTask, e);
142                                         throw e;
143                                     }
144                                 }, worker);
145                             });
146             // Remove the inflight request when the mirror task is done.
147             future.handleAsync((unused0, unused1) -> inflightRequests.remove(mirrorKey), worker);
148             return future;
149         } catch (Throwable e) {
150             inflightRequests.remove(mirrorKey);
151             throw e;
152         }
153     }
154 
155     private MetaRepository metaRepo(String projectName) {
156         // Assume that internal projects does not have mirrors.
157         return projectApiManager.getProject(projectName, null).metaRepo();
158     }
159 
160     @Override
161     public void close() {
162         final boolean interrupted = terminate(worker);
163         if (interrupted) {
164             Thread.currentThread().interrupt();
165         }
166         inflightRequests.clear();
167     }
168 
169     private static final class MirrorKey {
170         private final String projectName;
171         private final String repoName;
172         private final String mirrorId;
173 
174         private MirrorKey(String projectName, String repoName, String mirrorId) {
175             this.projectName = projectName;
176             this.repoName = repoName;
177             this.mirrorId = mirrorId;
178         }
179 
180         @Override
181         public boolean equals(Object o) {
182             if (this == o) {
183                 return true;
184             }
185             if (!(o instanceof MirrorKey)) {
186                 return false;
187             }
188             final MirrorKey mirrorKey = (MirrorKey) o;
189             return projectName.equals(mirrorKey.projectName) &&
190                    repoName.equals(mirrorKey.repoName) &&
191                    mirrorId.equals(mirrorKey.mirrorId);
192         }
193 
194         @Override
195         public int hashCode() {
196             return Objects.hash(projectName, repoName, mirrorId);
197         }
198 
199         @Override
200         public String toString() {
201             return MoreObjects.toStringHelper(this)
202                               .add("projectName", projectName)
203                               .add("repoName", repoName)
204                               .add("mirrorId", mirrorId)
205                               .toString();
206         }
207     }

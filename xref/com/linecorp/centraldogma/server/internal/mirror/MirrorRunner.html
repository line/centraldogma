50  public final class MirrorRunner implements SafeCloseable {
51  
52      private final ProjectApiManager projectApiManager;
53      private final CommandExecutor commandExecutor;
54      private final File workDir;
55      private final MirroringServicePluginConfig mirrorConfig;
56      private final ExecutorService worker;
57  
58      private final Map<MirrorKey, CompletableFuture<MirrorResult>> inflightRequests = new ConcurrentHashMap<>();
59  
60      public MirrorRunner(ProjectApiManager projectApiManager, CommandExecutor commandExecutor,
61                          CentralDogmaConfig cfg, MeterRegistry meterRegistry) {
62          this.projectApiManager = projectApiManager;
63          this.commandExecutor = commandExecutor;
64          // TODO(ikhoon): Periodically clean up stale repositories.
65          workDir = new File(cfg.dataDir(), "_mirrors_manual");
66          MirroringServicePluginConfig mirrorConfig =
67                  (MirroringServicePluginConfig) cfg.pluginConfigMap()
68                                                    .get(MirroringServicePluginConfig.class);
69          if (mirrorConfig == null) {
70              mirrorConfig = MirroringServicePluginConfig.INSTANCE;
71          }
72          this.mirrorConfig = mirrorConfig;
73  
74          final ThreadPoolExecutor threadPoolExecutor = new ThreadPoolExecutor(
75                  0, mirrorConfig.numMirroringThreads(),
76                  // TODO(minwoox): Use LinkedTransferQueue when we upgrade to JDK 21.
77                  60, TimeUnit.SECONDS, new LinkedBlockingQueue<>(),
78                  new DefaultThreadFactory("mirror-api-worker", true));
79          threadPoolExecutor.allowCoreThreadTimeOut(true);
80          worker = ExecutorServiceMetrics.monitor(meterRegistry, threadPoolExecutor,
81                                                  "mirrorApiWorker");
82      }
83  
84      public CompletableFuture<MirrorResult> run(String projectName, String mirrorId, User user) {
85          // If there is an inflight request, return it to avoid running the same mirror task multiple times.
86          return inflightRequests.computeIfAbsent(new MirrorKey(projectName, mirrorId), key -> run(key, user));
87      }
88  
89      private CompletableFuture<MirrorResult> run(MirrorKey mirrorKey, User user) {
90          try {
91              final CompletableFuture<MirrorResult> future =
92                      metaRepo(mirrorKey.projectName).mirror(mirrorKey.mirrorId).thenApplyAsync(mirror -> {
93                          if (!mirror.enabled()) {
94                              throw new MirrorException("The mirror is disabled: " + mirrorKey);
95                          }
96  
97                          final MirrorTask mirrorTask = new MirrorTask(mirror, user, Instant.now(), true);
98                          final MirrorListener listener = MirrorSchedulingService.mirrorListener;
99                          listener.onStart(mirrorTask);
100                         try {
101                             final MirrorResult mirrorResult = mirror.mirror(workDir, commandExecutor,
102                                                                             mirrorConfig.maxNumFilesPerMirror(),
103                                                                             mirrorConfig.maxNumBytesPerMirror(),
104                                                                             mirrorTask.triggeredTime());
105                             listener.onComplete(mirrorTask, mirrorResult);
106                             return mirrorResult;
107                         } catch (Exception e) {
108                             listener.onError(mirrorTask, e);
109                             throw e;
110                         }
111                     }, worker);
112             // Remove the inflight request when the mirror task is done.
113             future.handleAsync((unused0, unused1) -> inflightRequests.remove(mirrorKey), worker);
114             return future;
115         } catch (Throwable e) {
116             inflightRequests.remove(mirrorKey);
117             throw e;
118         }
119     }
120 
121     private MetaRepository metaRepo(String projectName) {
122         // Assume that internal projects does not have mirrors.
123         return projectApiManager.getProject(projectName, null).metaRepo();
124     }
125 
126     @Override
127     public void close() {
128         final boolean interrupted = terminate(worker);
129         if (interrupted) {
130             Thread.currentThread().interrupt();
131         }
132         inflightRequests.clear();
133     }
134 
135     private static final class MirrorKey {
136         private final String projectName;
137         private final String mirrorId;
138 
139         private MirrorKey(String projectName, String mirrorId) {
140             this.projectName = projectName;
141             this.mirrorId = mirrorId;
142         }
143 
144         @Override
145         public boolean equals(Object o) {
146             if (this == o) {
147                 return true;
148             }
149             if (!(o instanceof MirrorKey)) {
150                 return false;
151             }
152             final MirrorKey mirrorKey = (MirrorKey) o;
153             return projectName.equals(mirrorKey.projectName) &&
154                    mirrorId.equals(mirrorKey.mirrorId);
155         }
156 
157         @Override
158         public int hashCode() {
159             return Objects.hash(projectName, mirrorId);
160         }
161 
162         @Override
163         public String toString() {
164             return MoreObjects.toStringHelper(this)
165                               .add("projectName", projectName)
166                               .add("mirrorId", mirrorId)
167                               .toString();
168         }
169     }
170 }

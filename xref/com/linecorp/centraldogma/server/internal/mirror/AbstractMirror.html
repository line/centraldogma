114     protected AbstractMirror(String id, boolean enabled, @Nullable Cron schedule, MirrorDirection direction,
115                              Credential credential, Repository localRepo, String localPath,
116                              RepositoryUri remoteUri,
117                              @Nullable String gitignore, @Nullable String zone) {
118         this.id = requireNonNull(id, "id");
119         this.enabled = enabled;
120         this.direction = requireNonNull(direction, "direction");
121         this.credential = requireNonNull(credential, "credential");
122         this.localRepo = requireNonNull(localRepo, "localRepo");
123         this.localPath = normalizePath(requireNonNull(localPath, "localPath"));
124         this.remoteUri = remoteUri;
125         this.gitignore = gitignore;
126         this.zone = zone;
127 
128         if (gitignore != null) {
129             ignoreNode = new IgnoreNode();
130             try {
131                 ignoreNode.parse(new ByteArrayInputStream(gitignore.getBytes(UTF_8)));
132             } catch (IOException e) {
133                 throw new IllegalArgumentException("Failed to read gitignore: " + gitignore, e);
134             }
135         } else {
136             ignoreNode = null;
137         }
138 
139         if (schedule != null) {
140             this.schedule = requireNonNull(schedule, "schedule");
141             executionTime = ExecutionTime.forCron(this.schedule);
142 
143             // Pre-calculate a constant jitter value up to 1 minute for a mirror.
144             // Use the properties' hash code so that the same properties result in the same jitter.
145             jitterMillis = Math.abs(Objects.hash(this.schedule.asString(), this.direction,
146                                                  this.localRepo.parent().name(), this.localRepo.name(),
147                                                  this.remoteUri) /
148                                     (Integer.MAX_VALUE / 60000));
149         } else {
150             this.schedule = null;
151             executionTime = null;
152             jitterMillis = -1;
153         }
154     }

171     @VisibleForTesting
172     ZonedDateTime nextExecutionTime(ZonedDateTime lastExecutionTime, long jitterMillis) {
173         requireNonNull(lastExecutionTime, "lastExecutionTime");
174         final Optional<ZonedDateTime> next = executionTime.nextExecution(
175                 lastExecutionTime.minus(jitterMillis, ChronoUnit.MILLIS));
176         if (next.isPresent()) {
177             return next.get().plus(jitterMillis, ChronoUnit.MILLIS);
178         }
179         throw new IllegalArgumentException(
180                 "no next execution time for " + CRON_DESCRIPTOR.describe(schedule) + ", lastExecutionTime: " +
181                 lastExecutionTime);
182     }

226     @Override
227     public final MirrorResult mirror(File workDir, CommandExecutor executor, int maxNumFiles,
228                                      long maxNumBytes, Instant triggeredTime) {
229         try {
230             switch (direction()) {
231                 case LOCAL_TO_REMOTE:
232                     return mirrorLocalToRemote(workDir, maxNumFiles, maxNumBytes, triggeredTime);
233                 case REMOTE_TO_LOCAL:
234                     return mirrorRemoteToLocal(workDir, executor, maxNumFiles, maxNumBytes, triggeredTime);
235                 default:
236                     throw new Error("Should never reach here");
237             }
238         } catch (InterruptedException e) {
239             // Propagate the interruption.
240             Thread.currentThread().interrupt();
241             throw new MirrorException(e);
242         } catch (MirrorException e) {
243             throw e;
244         } catch (Exception e) {
245             final String message = e.getMessage();
246             if (message != null) {
247                 throw new MirrorException(message, e);
248             } else {
249                 throw new MirrorException(e);
250             }
251         }
252     }

272     /**
273      * Validates the given changes using all registered {@link MirrorFileValidator}s before they are
274      * committed. Mirror state files are excluded from validation.
275      *
276      * @throws MirrorException if any change fails validation
277      */
278     protected final void validateChanges(Map<String, Change<?>> changes) {
279         if (FILE_VALIDATORS.isEmpty()) {
280             return;
281         }
282         final String projectName = localRepo().parent().name();
283         final String repoName = localRepo().name();
284         final List<String> errors = new ArrayList<>();
285         for (Change<?> change : changes.values()) {
286             // Skip mirror state files — they are internal bookkeeping, not user content.
287             if (change.path().endsWith(MIRROR_STATE_FILE_NAME)) {
288                 continue;
289             }
290             for (MirrorFileValidator validator : FILE_VALIDATORS) {
291                 try {
292                     validator.validate(projectName, repoName, change);
293                 } catch (MirrorException e) {
294                     errors.add(e.getMessage());
295                 }
296             }
297         }
298         if (!errors.isEmpty()) {
299             throw new MirrorException(
300                     "Mirror validation failed for '" + projectName + '/' + repoName + "':\n" +
301                     String.join("\n", errors));
302         }
303     }

305     /**
306      * Filters the entries using gitignore patterns. Returns the entries as-is if no gitignore is configured.
307      * The entries should be sorted by path so that directory entries come before their children.
308      */
309     protected final Map<String, Entry<?>> filterByGitignore(Map<String, Entry<?>> entries, String basePath) {
310         if (ignoreNode == null) {
311             // Return a mutable copy because callers may modify the returned map.
312             return new HashMap<>(entries);
313         }
314         final Map<String, Entry<?>> result = new HashMap<>(entries.size());
315         String lastIgnoredDirectory = null;
316         for (Map.Entry<String, Entry<?>> entry : entries.entrySet()) {
317             final String path = entry.getKey();
318             final boolean isDirectory = entry.getValue().type() == EntryType.DIRECTORY;
319             final String relativePath = path.substring(basePath.length());
320             final MatchResult ignoreResult = ignoreNode.isIgnored(relativePath, isDirectory);
321             if (ignoreResult == MatchResult.IGNORED) {
322                 if (isDirectory) {
323                     lastIgnoredDirectory = path.endsWith("/") ? path : path + '/';
324                 }
325                 continue;
326             }
327             if (ignoreResult == MatchResult.CHECK_PARENT) {
328                 if (lastIgnoredDirectory != null && path.startsWith(lastIgnoredDirectory)) {
329                     continue;
330                 }
331             }
332             result.put(path, entry.getValue());
333         }
334         return result;
335     }

337     String hashString() {
338         if (hashString != null) {
339             return hashString;
340         }
341         return hashString = Hashing.sha256().hashString(toString(), UTF_8).toString();
342     }

344     @Override
345     public String toString() {
346         final ToStringHelper helper = MoreObjects.toStringHelper("")
347                                                  .omitNullValues()
348                                                  .add("direction", direction)
349                                                  .add("localProj", localRepo.parent().name())
350                                                  .add("localRepo", localRepo.name())
351                                                  .add("localPath", localPath)
352                                                  .add("remoteUri", remoteUri)
353                                                  .add("gitignore", gitignore)
354                                                  .add("credential", credential);
355         if (schedule != null) {
356             helper.add("schedule", CronDescriptor.instance().describe(schedule));
357         }
358         return helper.toString();
359     }

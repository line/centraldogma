60  final class MigratingMirrorToRepositoryService {
61  
62      private static final Logger logger = LoggerFactory.getLogger(MigratingMirrorToRepositoryService.class);
63  
64      @VisibleForTesting
65      static final String MIRROR_TO_REPOSITORY_MIGRATION_JOB_LOG = "/mirror-to-repository-migration-job.json";
66  
67      private final ProjectManager projectManager;
68      private final CommandExecutor commandExecutor;
69  
70      MigratingMirrorToRepositoryService(ProjectManager projectManager, CommandExecutor commandExecutor) {
71          this.projectManager = projectManager;
72          this.commandExecutor = commandExecutor;
73      }
74  
75      void migrate() throws Exception {
76          if (hasMigrationLog()) {
77              logger.debug("Mirrors have already been migrated to each repository. Skipping auto migration...");
78              return;
79          }
80  
81          final Stopwatch stopwatch = Stopwatch.createStarted();
82          int numCredentialMigratedProjects = 0;
83          int numMirrorMigratedProjects = 0;
84          String projectName = "";
85          try {
86              for (Project project : projectManager.list().values()) {
87                  projectName = project.name();
88                  logger.info("Migrating credentials in the project: {} ...", projectName);
89                  final MetaRepository repository = project.metaRepo();
90                  if (migrateCredentials(projectName, repository)) {
91                      numCredentialMigratedProjects++;
92                      logger.info("Credentials in the project '{}' have been migrated.", projectName);
93                  } else {
94                      logger.info("No legacy configurations of credentials found in the project: {}.",
95                                  projectName);
96                  }
97  
98                  logger.info("Migrating mirrors in the project: {} ...", projectName);
99                  if (migrateMirrors(repository)) {
100                     numMirrorMigratedProjects++;
101                     logger.info("Mirrors in the project '{}' have been migrated.", projectName);
102                 } else {
103                     logger.info("No legacy configurations of mirrors found in the project: {}.",
104                                 projectName);
105                 }
106             }
107             logMigrationJob(numCredentialMigratedProjects, numMirrorMigratedProjects);
108         } catch (Exception ex) {
109             logger.warn("Failed to migrate credentials and mirrors. project: {}", projectName, ex);
110             return;
111         }
112         logger.info("Mirrors to each repository migration has been completed. (took: {} ms.) " +
113                     "migrated credentials: {}, migrated mirrors: {}",
114                     stopwatch.elapsed().toMillis(), numCredentialMigratedProjects, numMirrorMigratedProjects);
115     }
116 
117     private void logMigrationJob(int numCredentialMigratedProjects,
118                                  int numMirrorMigratedProjects) throws Exception {
119         final Map<String, Object> data =
120                 ImmutableMap.of("timestamp", Instant.now(),
121                                 "credentials", numCredentialMigratedProjects,
122                                 "mirrors", numMirrorMigratedProjects);
123         final Change<JsonNode> change = Change.ofJsonUpsert(MIRROR_TO_REPOSITORY_MIGRATION_JOB_LOG,
124                                                             Jackson.writeValueAsString(data));
125         final Command<CommitResult> command =
126                 Command.push(Author.SYSTEM, InternalProjectInitializer.INTERNAL_PROJECT_DOGMA,
127                              Project.REPO_DOGMA, Revision.HEAD,
128                              "Migration of mirrors to each repository has been done.", "",
129                              Markup.PLAINTEXT, change);
130         executeCommand(command);
131     }
132 
133     private boolean hasMigrationLog() throws Exception {
134         final Project internalProj = projectManager.get(InternalProjectInitializer.INTERNAL_PROJECT_DOGMA);
135         final Repository repository = internalProj.repos().get(Project.REPO_DOGMA);
136         final Map<String, Entry<?>> entries = repository.find(Revision.HEAD,
137                                                               MIRROR_TO_REPOSITORY_MIGRATION_JOB_LOG).get();
138         final Entry<?> entry = entries.get(MIRROR_TO_REPOSITORY_MIGRATION_JOB_LOG);
139         return entry != null;
140     }
141 
142     private boolean migrateCredentials(String projectName, MetaRepository repository) throws Exception {
143         final Map<String, Entry<?>> entries = repository.find(Revision.HEAD, CREDENTIALS + "*.json")
144                                                         .join();
145         if (entries.isEmpty()) {
146             return false;
147         }
148 
149         final List<Change<?>> changes = new ArrayList<>();
150         for (Map.Entry<String, Entry<?>> entry : entries.entrySet()) {
151             final JsonNode content = (JsonNode) entry.getValue().content();
152             if (!content.isObject()) {
153                 warnInvalidConfig(entry, "credential", content);
154                 continue;
155             }
156             try {
157                 final LegacyCredential legacyCredential = Jackson.treeToValue(content, LegacyCredential.class);
158                 // Migrate the credential to a project level one.
159                 final String name = credentialName(projectName, legacyCredential.id());
160                 final Credential newCredential = legacyCredential.toNewCredential(name);
161                 final String file = credentialFile(name);
162                 assert entry.getKey().equals(file);
163                 changes.add(Change.ofJsonUpsert(file, Jackson.valueToTree(newCredential)));
164             } catch (JsonProcessingException e) {
165                 warnInvalidConfig(entry, "credential", content);
166             }
167         }
168         if (changes.isEmpty()) {
169             return false;
170         }
171 
172         final Command<CommitResult> command =
173                 Command.push(Author.SYSTEM, repository.parent().name(), REPO_META, Revision.HEAD,
174                              "Migrate the credentials", "", Markup.PLAINTEXT, changes);
175         executeCommand(command);
176         return true;
177     }
178 
179     private boolean migrateMirrors(MetaRepository repository) throws Exception {
180         final Map<String, Entry<?>> entries = repository.find(Revision.HEAD, LEGACY_MIRRORS_PATH + "*.json")
181                                                         .join();
182         if (entries.isEmpty()) {
183             return false;
184         }
185         final List<Change<?>> changes = new ArrayList<>();
186         for (Map.Entry<String, Entry<?>> entry : entries.entrySet()) {
187             final JsonNode content = (JsonNode) entry.getValue().content();
188             if (!content.isObject()) {
189                 warnInvalidConfig(entry, "mirror", content);
190                 continue;
191             }
192             try {
193                 final MirrorConfig mirrorConfig = Jackson.treeToValue(content, MirrorConfig.class);
194                 final String repoName = mirrorConfig.localRepo();
195                 if (repoName == null) {
196                     warnInvalidConfig(entry, "mirror", content);
197                     continue;
198                 }
199                 final String credentialName = mirrorConfig.credentialName();
200                 assert !credentialName.startsWith("projects/")
201                         : "Legacy mirror configuration has invalid credential name: " + credentialName;
202 
203                 // Migrate the credentialName to a project level one.
204                 final MirrorConfig newMirrorConfig = mirrorConfig.withCredentialName(
205                         credentialName(repository.parent().name(), credentialName));
206                 changes.add(Change.ofJsonUpsert(mirrorFile(repoName, mirrorConfig.id()),
207                                                 Jackson.valueToTree(newMirrorConfig)));
208             } catch (JsonProcessingException e) {
209                 warnInvalidConfig(entry, "mirror", content);
210             }
211         }
212         if (changes.isEmpty()) {
213             return false;
214         }
215 
216         final Command<CommitResult> command =
217                 Command.push(Author.SYSTEM, repository.parent().name(), REPO_META, Revision.HEAD,
218                              "Migrate the mirrors to each repository", "", Markup.PLAINTEXT, changes);
219         executeCommand(command);
220         return true;
221     }
222 
223     private static void warnInvalidConfig(Map.Entry<String, Entry<?>> entry, String type, JsonNode content) {
224         logger.warn("Ignoring an invalid {} configuration: {}, {}", type, entry.getKey(), content);
225     }
226 
227     private void executeCommand(Command<CommitResult> command)
228             throws InterruptedException, ExecutionException, TimeoutException {
229         commandExecutor.execute(Command.forcePush(command)).get(1, TimeUnit.MINUTES);
230     }

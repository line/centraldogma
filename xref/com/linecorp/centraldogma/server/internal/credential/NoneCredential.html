17  package com.linecorp.centraldogma.server.internal.credential;
18  
19  import static java.util.Objects.requireNonNull;
20  
21  import javax.annotation.Nullable;
22  
23  import com.fasterxml.jackson.annotation.JsonCreator;
24  import com.fasterxml.jackson.annotation.JsonProperty;
25  import com.google.common.base.MoreObjects.ToStringHelper;
26  
27  import com.linecorp.centraldogma.server.credential.Credential;
28  import com.linecorp.centraldogma.server.credential.CredentialType;
29  
30  public final class NoneCredential extends AbstractCredential {
31  
32      @JsonCreator
33      public NoneCredential(@JsonProperty("name") @Nullable String name) {
34          super(name, CredentialType.NONE);
35      }
36  
37      @Override
38      void addProperties(ToStringHelper helper) {
39          // No properties to add
40      }
41  
42      @Override
43      public Credential withoutSecret() {
44          return this;
45      }
46  
47      @Override
48      public Credential withName(String credentialName) {
49          requireNonNull(credentialName, "credentialName");
50          return new NoneCredential(credentialName);
51      }
52  }

63      // The first found variable file will be used from the list.
64      private static final List<String> VARIABLE_FILES =
65              ImmutableList.of("/.variables.json", "/.variables.json5", "/.variables.yaml", "/.variables.yml");
66      private static final String VARIABLE_FILES_PATTERN = Joiner.on(",").join(VARIABLE_FILES);
67  
68      private static final UnmodifiableFuture<Map<String, Object>> EMPTY_MAP_FUTURE =
69              UnmodifiableFuture.completedFuture(ImmutableMap.of());
70  
71      private final CrudOperation<Variable> crudRepo;
72      private final LoadingCache<Entry<?>, Template> cache;
73  
74      public Templater(CommandExecutor executor, ProjectManager pm) {
75          crudRepo = new DefaultCrudOperation<>(Variable.class, executor, pm);
76          final Configuration cfg = new Configuration(Configuration.VERSION_2_3_32);
77          cfg.setDefaultEncoding("UTF-8");
78          cfg.setLogTemplateExceptions(false);
79          cfg.setWrapUncheckedExceptions(true);
80          cfg.setFallbackOnNullLoopVariable(false);
81          cfg.setClassicCompatible(false);
82          cfg.setNumberFormat("computer");
83          cfg.setBooleanFormat("c");
84          cfg.setAPIBuiltinEnabled(false);
85          cfg.setNewBuiltinClassResolver(TemplateClassResolver.ALLOWS_NOTHING_RESOLVER);
86  
87          cache = Caffeine.newBuilder()
88                          .expireAfterAccess(Duration.ofHours(1))
89                          .maximumSize(4096)
90                          .build(key -> {
91                              final String content;
92                              if (key.rawContent() != null) {
93                                  content = key.rawContent();
94                              } else {
95                                  // The entry processed by JSON-Path does not have a raw content.
96                                  content = key.contentAsPrettyText();
97                              }
98                              return new Template(key.path(), content, cfg);
99                          });
100     }

102     public <T> CompletableFuture<Entry<T>> render(Repository repo, Entry<T> entry,
103                                                   @Nullable String variableFile,
104                                                   @Nullable Revision templateRevision) {
105         if (!entry.hasContent()) {
106             return UnmodifiableFuture.completedFuture(entry);
107         }
108         final Project project = repo.parent();
109         final Repository dogmaRepo = project.repos().get(Project.REPO_DOGMA);
110         final Revision normTemplateRevision;
111         if (templateRevision == null) {
112             normTemplateRevision = dogmaRepo.normalizeNow(Revision.HEAD);
113         } else {
114             normTemplateRevision = dogmaRepo.normalizeNow(templateRevision);
115         }
116 
117         final String projectName = project.name();
118         // TODO(ikhoon): Optimize by caching the rendering result for the same set of variables and template.
119         return mergeVariables(crudRepo.findAll(crudContext(projectName, normTemplateRevision)),
120                               crudRepo.findAll(crudContext(projectName, repo.name(), normTemplateRevision)),
121                               findRepoVariableFile(repo, entry),
122                               findEntryPathVariableFile(repo, entry),
123                               findClientVariableFile(repo, entry, variableFile))
124                 .thenApply(variables -> process(entry, variables, normTemplateRevision))
125                 .toCompletableFuture();
126     }
127 
128     private static CompletableFuture<Map<String, Object>> findRepoVariableFile(Repository repo,
129                                                                                Entry<?> entry) {
130         return findVariableFile(repo, entry.revision(), VARIABLE_FILES_PATTERN);
131     }
132 
133     private static CompletableFuture<Map<String, Object>> findEntryPathVariableFile(Repository repo,
134                                                                                     Entry<?> entry) {
135         final String entryPath = entry.path();
136         final Revision revision = entry.revision();
137         final int lastSlash = entryPath.lastIndexOf('/');
138         final String directory = entryPath.substring(0, lastSlash);
139         final String filePattern = VARIABLE_FILES.stream()
140                                                  .map(file -> directory + file)
141                                                  .collect(Collectors.joining(","));
142         // Try to find variable files in the same directory as the template.
143         return findVariableFile(repo, revision, filePattern);
144     }
145 
146     private static CompletableFuture<Map<String, Object>> findVariableFile(
147             Repository repo, Revision revision, String pathPattern) {
148         return repo.find(revision, pathPattern).thenApply(entries -> {
149             final Entry<?> chosen = chooseVariableFile(entries);
150             if (chosen == null) {
151                 return ImmutableMap.of();
152             }
153             return parseVariableFile(chosen);
154         });
155     }
156 
157     private static CompletableFuture<Map<String, Object>> findClientVariableFile(
158             Repository repo, Entry<?> entry, @Nullable String variableFile) {
159         if (Strings.isNullOrEmpty(variableFile)) {
160             return EMPTY_MAP_FUTURE;
161         }
162         return repo.get(entry.revision(), variableFile).thenApply(entry0 -> {
163             if (entry0.type().type() != JsonNode.class) {
164                 throw new TemplateProcessingException(
165                         "The variable file must be a JSON or YAML type: " + variableFile);
166             }
167             return parseVariableFile(entry0);
168         });
169     }
170 
171     @Nullable
172     private static Entry<?> chooseVariableFile(Map<String, Entry<?>> entries) {
173         if (entries.isEmpty()) {
174             return null;
175         }
176         if (entries.size() == 1) {
177             return entries.values().iterator().next();
178         }
179 
180         final Collection<Entry<?>> values = entries.values();
181         for (String variableFile : VARIABLE_FILES) {
182             for (Entry<?> entry : values) {
183                 if (entry.path().endsWith(variableFile)) {
184                     return entry;
185                 }
186             }
187         }
188         return null;
189     }
190 
191     private static Map<String, Object> parseVariableFile(Entry<?> entry) {
192         try {
193             final JsonNode jsonNode = entry.contentAsJson();
194             if (!jsonNode.isObject()) {
195                 throw new TemplateProcessingException(
196                         "The variable file must contain a JSON object: " + entry.path());
197             }
198             final ObjectNode node = (ObjectNode) jsonNode;
199             //noinspection unchecked
200             return Jackson.treeToValue(node, Map.class);
201         } catch (JsonProcessingException e) {
202             throw new TemplateProcessingException(
203                     "Failed to parse the variable file: " + entry.path() + ".\n" + e.getMessage(), e);
204         }
205     }
206 
207     private <T> Entry<T> process(Entry<T> entry, Map<String, Object> variables, Revision templateRevision) {
208         final StringWriter out = new StringWriter();
209         final Template template = cache.get(entry);
210         try {
211             template.process(variables, out);
212             //noinspection unchecked
213             final Entry<T> newEntry = (Entry<T>) newEntryWithContent(entry, out.toString());
214             return newEntry.withTemplateRevision(templateRevision);
215         } catch (Exception e) {
216             throw new TemplateProcessingException(
217                     "Failed to process the template for " + entry.path() + ".\n" + e.getMessage(), e);
218         }
219     }
220 
221     private static <T> Entry<?> newEntryWithContent(Entry<?> entry, String content) {
222         try {
223             switch (entry.type()) {
224                 case TEXT:
225                     return Entry.ofText(entry.revision(), entry.path(), content);
226                 case JSON:
227                     return Entry.ofJson(entry.revision(), entry.path(), content);
228                 case YAML:
229                     return Entry.ofYaml(entry.revision(), entry.path(), content);
230                 case DIRECTORY:
231                 default:
232                     // Should not reach here.
233                     throw new Error();
234             }
235         } catch (JsonProcessingException e) {
236             throw new IllegalStateException(e);
237         }
238     }

240     private static CompletionStage<Map<String, Object>> mergeVariables(
241             CompletableFuture<List<HasRevision<Variable>>> projFuture,
242             CompletableFuture<List<HasRevision<Variable>>> repoFuture,
243             CompletableFuture<Map<String, Object>> repoFileFuture,
244             CompletableFuture<Map<String, Object>> entryPathFuture,
245             CompletableFuture<Map<String, Object>> clientFileFuture) {
246         return CompletableFutures.combine(
247                 projFuture, repoFuture, repoFileFuture, entryPathFuture, clientFileFuture,
248                 (projVars, repoVars, repoFileVars, entryPathVars, clientFileVars) -> {
249                     final ImmutableMap.Builder<String, Object> builder =
250                             ImmutableMap.builderWithExpectedSize(
251                                     projVars.size() + repoVars.size() + repoFileVars.size() +
252                                     entryPathVars.size() + clientFileVars.size());
253 
254                     for (HasRevision<Variable> it : projVars) {
255                         final Variable variable = it.object();
256                         builder.put(variable.id(), parseValue(variable));
257                     }
258                     for (HasRevision<Variable> it : repoVars) {
259                         final Variable variable = it.object();
260                         // Repo-level variables override project-level ones.
261                         builder.put(variable.id(), parseValue(variable));
262                     }
263                     // The higher precedence variables override the lower precedence ones.
264                     builder.putAll(repoFileVars);
265                     builder.putAll(entryPathVars);
266                     builder.putAll(clientFileVars);
267 
268                     final Map<String, Object> variables = builder.buildKeepingLast();
269                     // Prefix variables map with "vars" key.
270                     // This allows using "vars.varName" in the template.
271                     // TODO(ikhoon): Support secret variables that will be prefixed with "secrets" key.
272                     final Map<String, Object> vars = new HashMap<>();
273                     vars.put("vars", variables);
274                     return vars;
275                 });
276     }
277 
278     private static Object parseValue(Variable variable) {
279         switch (variable.type()) {
280             case JSON:
281                 try {
282                     return Jackson.readValue(variable.value(), Object.class);
283                 } catch (JsonProcessingException e) {
284                     // Should not reach here as the value has been validated before storing.
285                     throw new IllegalStateException(e);
286                 }
287             case STRING:
288                 return variable.value();
289             default:
290                 // Should not reach here.
291                 throw new Error();
292         }
293     }

17  package com.linecorp.centraldogma.server.internal.api.sysadmin;
18  
19  import java.util.concurrent.CompletableFuture;
20  
21  import com.linecorp.armeria.common.HttpStatus;
22  import com.linecorp.armeria.server.HttpStatusException;
23  import com.linecorp.armeria.server.annotation.Consumes;
24  import com.linecorp.armeria.server.annotation.Get;
25  import com.linecorp.armeria.server.annotation.ProducesJson;
26  import com.linecorp.armeria.server.annotation.Put;
27  import com.linecorp.centraldogma.server.command.Command;
28  import com.linecorp.centraldogma.server.command.CommandExecutor;
29  import com.linecorp.centraldogma.server.internal.api.AbstractService;
30  import com.linecorp.centraldogma.server.internal.api.auth.RequiresSystemAdministrator;
31  import com.linecorp.centraldogma.server.internal.api.sysadmin.UpdateServerStatusRequest.Scope;
32  import com.linecorp.centraldogma.server.management.ServerStatus;
33  import com.linecorp.centraldogma.server.management.ServerStatusManager;
34  
35  @ProducesJson
36  public final class ServerStatusService extends AbstractService {
37  
38      private final ServerStatusManager serverStatusManager;
39  
40      public ServerStatusService(CommandExecutor executor, ServerStatusManager serverStatusManager) {
41          super(executor);
42          this.serverStatusManager = serverStatusManager;
43      }
44  
45      /**
46       * GET /status
47       *
48       * <p>Returns the server status.
49       */
50      @Get("/status")
51      public ServerStatus status() {
52          return ServerStatus.of(executor().isWritable(), executor().isStarted());
53      }
54  
55      /**
56       * PUT /status
57       *
58       * <p>Patches the server status with a JSON patch. Currently used only for entering read-only.
59       *
60       * <p>If {@link UpdateServerStatusRequest#scope()} is omitted, defaults to {@link Scope#ALL}.
61       * If the scope is {@link Scope#ALL}, the new status is propagated to all cluster servers.
62       * If the scope is {@link Scope#LOCAL}, the new status is only applied to the current server.
63       */
64      @Put("/status")
65      @Consumes("application/json")
66      @RequiresSystemAdministrator
67      public CompletableFuture<ServerStatus> updateStatus(UpdateServerStatusRequest statusRequest)
68              throws Exception {
69          // TODO(trustin): Consider extracting this into common utility or Armeria.
70          final ServerStatus oldStatus = status();
71  
72          final ServerStatus newStatus = statusRequest.serverStatus();
73          if (statusRequest.scope() == Scope.LOCAL) {
74              // Validate the new status for the local scope. Other servers may have different status.
75              if (oldStatus == newStatus) {
76                  throw HttpStatusException.of(HttpStatus.NOT_MODIFIED);
77              }
78  
79              return CompletableFuture.supplyAsync(() -> {
80                  executor().statusManager().updateStatus(newStatus);
81                  serverStatusManager.updateStatus(newStatus);
82                  return status();
83              }, serverStatusManager.sequentialExecutor());
84          } else {
85              return execute(Command.updateServerStatus(newStatus))
86                      .thenApply(unused -> status());
87          }
88      }
89  }

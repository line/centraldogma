70              ImmutableList.of(
71                      ImmutableMap.of("op", "replace",
72                                      "path", "/status",
73                                      "value", "active")));

78                                      "value", "inactive")));

89      /**
90       * GET /appIdentities
91       *
92       * <p>Returns the list of the app identities.
93       */
94      @Get("/appIdentities")
95      public Collection<AppIdentity> listAppIdentities(User loginUser) {
96          if (loginUser.isSystemAdmin()) {
97              return mds.getAppIdentityRegistry().appIds().values();
98          } else {
99              return mds.getAppIdentityRegistry()
100                       .withoutSecret()
101                       .appIds()
102                       .values().stream().filter(appIdentity -> !appIdentity.isSystemAdmin())
103                       .collect(toImmutableList());
104         }
105     }

107     /**
108      * POST /appIdentities
109      *
110      * <p>Returns a newly-generated app identity belonging to the current login user.
111      */
112     @Post("/appIdentities")
113     @StatusCode(201)
114     @ResponseConverter(CreateApiResponseConverter.class)
115     public CompletableFuture<ResponseEntity<AppIdentity>> createAppIdentity(
116             @Param String appId,
117             @Param @Default("false") boolean isSystemAdmin,
118             @Param AppIdentityType type,
119             @Param @Nullable String secret,
120             @Param @Nullable String certificateId,
121             Author author, User loginUser) {
122         if (!mtlsEnabled && type == AppIdentityType.CERTIFICATE) {
123             throw new IllegalArgumentException(
124                     "Cannot create a CERTIFICATE type app identity when mTLS is disabled.");
125         }
126 
127         if (!loginUser.isSystemAdmin()) {
128             checkArgument(
129                     !isSystemAdmin,
130                     "Only system administrators are allowed to create a system admin-level app identity.");
131             checkArgument(
132                     secret == null,
133                     "Only system administrators are allowed to create a new application token from " +
134                     "the given secret string");
135         }
136         final CompletableFuture<Revision> future;
137         if (type == AppIdentityType.TOKEN) {
138             checkArgument(certificateId == null,
139                           "TOKEN type cannot have a certificateId: %s", certificateId);
140             if (secret != null) {
141                 future = mds.createToken(author, appId, secret, isSystemAdmin);
142             } else {
143                 future = mds.createToken(author, appId, isSystemAdmin);
144             }
145         } else {
146             checkArgument(certificateId != null, "CERTIFICATE type must have a certificateId.");
147             checkArgument(secret == null,
148                           "CERTIFICATE type cannot have a secret: %s", secret);
149             future = mds.createCertificate(author, appId, certificateId, isSystemAdmin);
150         }
151         return future.thenCompose(unused -> fetchAppIdentity(appId))
152                      .thenApply(appIdentity -> {
153                          final ResponseHeaders headers = ResponseHeaders.of(HttpStatus.CREATED,
154                                                                             HttpHeaderNames.LOCATION,
155                                                                             "/appIdentities/" + appId);
156                          return ResponseEntity.of(headers, appIdentity);
157                      });
158     }

160     /**
161      * DELETE /appIdentities/{appId}
162      *
163      * <p>Deletes an app identity of the specified ID then returns it.
164      */
165     @Delete("/appIdentities/{appId}")
166     public CompletableFuture<AppIdentity> deleteAppIdentity(ServiceRequestContext ctx,
167                                                             @Param String appId,
168                                                             Author author, User loginUser) {
169         return getAppIdentityOrRespondForbidden(ctx, appId, loginUser).thenCompose(
170                 appIdentities -> {
171                     if (appIdentities.type() == AppIdentityType.TOKEN) {
172                         return mds.destroyToken(author, appId)
173                                   .thenApply(unused -> ((Token) appIdentities).withoutSecret());
174                     }
175                     return mds.destroyCertificate(author, appId).thenApply(unused -> appIdentities);
176                 });
177     }

179     /**
180      * DELETE /appIdentities/{appId}/removed
181      *
182      * <p>Purges an app identity of the specified ID that was deleted before.
183      */
184     @Delete("/appIdentities/{appId}/removed")
185     @RequiresSystemAdministrator
186     public CompletableFuture<AppIdentity> purgeAppIdentity(ServiceRequestContext ctx,
187                                                            @Param String appId,
188                                                            Author author, User loginUser) {
189         return getAppIdentityOrRespondForbidden(ctx, appId, loginUser).thenApplyAsync(
190                 appIdentity -> {
191                     mds.purgeAppIdentity(author, appId);
192                     if (appIdentity.type() == AppIdentityType.TOKEN) {
193                         return ((Token) appIdentity).withoutSecret();
194                     }
195                     return appIdentity;
196                 }, ctx.blockingTaskExecutor());
197     }

199     /**
200      * PATCH /appIdentities/{appId}
201      *
202      * <p>Activates or deactivates the app identity of the specified {@code appId}.
203      */
204     @Patch("/appIdentities/{appId}")
205     @ConsumesJson
206     public CompletableFuture<AppIdentity> updateAppIdentity(ServiceRequestContext ctx,
207                                                             @Param String appId,
208                                                             JsonNode node, Author author, User loginUser) {
209         // {"status":"active"} or {"status":"inactive"}
210         final JsonNode status = node.get("status");
211         if (status == null || status.isNull() || !status.isTextual()) {
212             throw new IllegalArgumentException(
213                     "The request must contain a 'status' field with a string value.");
214         }
215         final String text = status.asText();
216         if (!"active".equals(text) && !"inactive".equals(text)) {
217             throw new IllegalArgumentException(
218                     "The 'status' field must be either 'active' or 'inactive': " + text);
219         }
220 
221         return getAppIdentityOrRespondForbidden(ctx, appId, loginUser).thenCompose(
222                 appIdentity -> {
223                     if (appIdentity.isDeleted()) {
224                         throw new IllegalArgumentException(
225                                 "You can't update the status of the app identity scheduled for deletion.");
226                     }
227                     if ("active".equals(text)) {
228                         if (appIdentity.type() == AppIdentityType.TOKEN) {
229                             return mds.activateToken(author, appId);
230                         } else {
231                             return mds.activateCertificate(author, appId);
232                         }
233                     }
234                     if (appIdentity.type() == AppIdentityType.TOKEN) {
235                         return mds.deactivateToken(author, appId);
236                     } else {
237                         return mds.deactivateCertificate(author, appId);
238                     }
239                 }
240         ).thenCompose(unused -> {
241             // Fetch the updated app identity.
242             return fetchAppIdentity(appId).thenApply(updated -> {
243                 if (updated.type() == AppIdentityType.TOKEN) {
244                     return ((Token) updated).withoutSecret();
245                 }
246                 return updated;
247             });
248         });
249     }

251     /**
252      * PATCH /appIdentities/{appId}/level
253      *
254      * <p>Updates a level of the app identity of the specified ID.
255      */
256     @Patch("/appIdentities/{appId}/level")
257     @ConsumesJson
258     @RequiresSystemAdministrator
259     public CompletableFuture<AppIdentity> updateAppIdentityLevel(
260             ServiceRequestContext ctx,
261             @Param String appId,
262             AppIdentityLevelRequest appIdentityLevelRequest,
263             Author author, User loginUser) {
264         final String newAppIdentityLevel = appIdentityLevelRequest.level().toLowerCase();
265         checkArgument("user".equals(newAppIdentityLevel) || "systemadmin".equals(newAppIdentityLevel),
266                       "app identity level: %s (expected: user or systemadmin)",
267                       appIdentityLevelRequest.level());
268 
269         return getAppIdentityOrRespondForbidden(ctx, appId, loginUser).thenCompose(
270                 appIdentity -> {
271                     boolean toBeSystemAdmin = false;
272 
273                     switch (newAppIdentityLevel) {
274                         case "user":
275                             if (!appIdentity.isSystemAdmin()) {
276                                 throw HttpStatusException.of(HttpStatus.NOT_MODIFIED);
277                             }
278                             break;
279                         case "systemadmin":
280                             if (appIdentity.isSystemAdmin()) {
281                                 throw HttpStatusException.of(HttpStatus.NOT_MODIFIED);
282                             }
283                             toBeSystemAdmin = true;
284                             break;
285                     }
286                     return mds.updateAppIdentityLevel(author, appId, toBeSystemAdmin).thenCompose(
287                             unused -> fetchAppIdentity(appId).thenApply(updated -> {
288                                 if (updated.type() == AppIdentityType.TOKEN) {
289                                     return ((Token) updated).withoutSecret();
290                                 }
291                                 return updated;
292                             }));
293                 });
294     }

389     /**
390      * PATCH /tokens/{appId}
391      *
392      * <p>Activates or deactivates the token of the specified {@code appId}.
393      *
394      * @deprecated Use {@link #updateAppIdentity(ServiceRequestContext, String, JsonNode, Author, User)}.
395      */
396     @Patch("/tokens/{appId}")
397     @Consumes("application/json-patch+json")
398     @Deprecated
399     public CompletableFuture<Token> updateToken(ServiceRequestContext ctx,
400                                                 @Param String appId,
401                                                 JsonNode node, Author author, User loginUser) {
402         return getTokenOrRespondForbidden(ctx, appId, loginUser).thenCompose(
403                 token -> {
404                     if (token.isDeleted()) {
405                         throw new IllegalArgumentException(
406                                 "You can't update the status of the token scheduled for deletion.");
407                     }
408                     if (node.equals(LEGACY_ACTIVATION)) {
409                         return mds.activateToken(author, appId)
410                                   .thenApply(unused -> token.withoutSecret());
411                     }
412                     if (node.equals(LEGACY_DEACTIVATION)) {
413                         return mds.deactivateToken(author, appId)
414                                   .thenApply(unused -> token.withoutSecret());
415                     }
416                     throw new IllegalArgumentException("Unsupported JSON patch: " + node +
417                                                        " (expected: " + LEGACY_ACTIVATION + " or " +
418                                                        LEGACY_DEACTIVATION + ')');
419                 }
420         );
421     }

445     private CompletableFuture<AppIdentity> fetchAppIdentity(String appId) {
446         return mds.fetchAppIdentityRegistry().thenApply(registry -> registry.get(appId));
447     }
448 
449     private CompletableFuture<AppIdentity> getAppIdentityOrRespondForbidden(ServiceRequestContext ctx,
450                                                                             String appId, User loginUser) {
451         return fetchAppIdentity(appId).thenApply(appIdentity -> {
452             // Give permission to the system administrators.
453             if (!loginUser.isSystemAdmin() &&
454                 !appIdentity.creation().user().equals(loginUser.id())) {
455                 return HttpApiUtil.throwResponse(ctx, HttpStatus.FORBIDDEN,
456                                                  "Do not have permission for app identity: %s",
457                                                  appIdentity.appId());
458             }
459             return appIdentity;
460         });
461     }
462 
463     private CompletableFuture<Token> getTokenOrRespondForbidden(ServiceRequestContext ctx,
464                                                                 String appId, User loginUser) {
465         return getAppIdentityOrRespondForbidden(ctx, appId, loginUser).thenApply(appIdentity -> {
466             if (appIdentity.type() != AppIdentityType.TOKEN) {
467                 return HttpApiUtil.throwResponse(
468                         ctx, HttpStatus.NOT_FOUND, "%s is not a token but a %s",
469                         appIdentity.appId(), appIdentity.type());
470             }
471 
472             return (Token) appIdentity;
473         });
474     }

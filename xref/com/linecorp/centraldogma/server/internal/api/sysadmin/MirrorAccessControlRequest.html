17  package com.linecorp.centraldogma.server.internal.api.sysadmin;
18  
19  import static com.linecorp.centraldogma.server.internal.storage.repository.git.GitCrudRepository.validateId;
20  import static java.util.Objects.requireNonNull;
21  
22  import java.util.Objects;
23  import java.util.regex.Pattern;
24  import java.util.regex.PatternSyntaxException;
25  
26  import com.fasterxml.jackson.annotation.JsonCreator;
27  import com.fasterxml.jackson.annotation.JsonProperty;
28  import com.google.common.base.MoreObjects;
29  
30  public final class MirrorAccessControlRequest {
31      private final String id;
32      private final String targetPattern;
33      private final boolean allow;
34      private final String description;
35      private final int order;
36  
37      @JsonCreator
38      public MirrorAccessControlRequest(@JsonProperty("id") String id,
39                                        @JsonProperty("targetPattern") String targetPattern,
40                                        @JsonProperty("allow") boolean allow,
41                                        @JsonProperty("description") String description,
42                                        @JsonProperty("order") int order) {
43          this.id = validateId(id);
44          // Validate the target pattern.
45          try {
46              Pattern.compile(targetPattern);
47          } catch (PatternSyntaxException e) {
48              throw new IllegalArgumentException("invalid targetPattern: " + targetPattern, e);
49          }
50          this.targetPattern = requireNonNull(targetPattern, "targetPattern");
51          this.allow = allow;
52          this.description = requireNonNull(description, "description");
53          this.order = order;
54      }
55  
56      /**
57       * Returns the ID of the mirror access control.
58       */
59      @JsonProperty
60      public String id() {
61          return id;
62      }
63  
64      /**
65       * Returns the target pattern of the mirror.
66       */
67      @JsonProperty
68      public String targetPattern() {
69          return targetPattern;
70      }
71  
72      /**
73       * Returns whether the mirror ACL allows or denies the target pattern.
74       */
75      @JsonProperty
76      public boolean allow() {
77          return allow;
78      }
79  
80      /**
81       * Returns the description of the mirror access control.
82       */
83      @JsonProperty
84      public String description() {
85          return description;
86      }
87  
88      /**
89       * Returns the order of the mirror access control.
90       */
91      @JsonProperty
92      public int order() {
93          return order;
94      }
95  
96      @Override
97      public boolean equals(Object o) {
98          if (!(o instanceof MirrorAccessControlRequest)) {
99              return false;
100         }
101         final MirrorAccessControlRequest that = (MirrorAccessControlRequest) o;
102         return allow == that.allow &&
103                order == that.order &&
104                id.equals(that.id) &&
105                targetPattern.equals(that.targetPattern) &&
106                description.equals(that.description);
107     }
108 
109     @Override
110     public int hashCode() {
111         return Objects.hash(id, targetPattern, allow, description, order);
112     }
113 
114     @Override
115     public String toString() {
116         return MoreObjects.toStringHelper(this)
117                           .add("id", id)
118                           .add("targetPattern", targetPattern)
119                           .add("allow", allow)
120                           .add("description", description)
121                           .add("order", order)
122                           .toString();
123     }
124 }

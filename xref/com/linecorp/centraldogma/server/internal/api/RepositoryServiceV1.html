78      /**
79       * GET /projects/{projectName}/repos?status={status}
80       *
81       * <p>Returns the list of the repositories or removed repositories.
82       */
83      @Get("/projects/{projectName}/repos")
84      public CompletableFuture<List<RepositoryDto>> listRepositories(ServiceRequestContext ctx, Project project,
85                                                                     @Param @Nullable String status, User user) {
86          if (status != null) {
87              HttpApiUtil.checkStatusArgument(status);
88          }
89  
90          return mds.findProjectRole(project.name(), user).handle((role, throwable) -> {
91              final boolean hasOwnerRole = role == ProjectRole.OWNER;
92              if (status != null) {
93                  if (hasOwnerRole) {
94                      return project.repos().listRemoved().keySet().stream().map(RepositoryDto::new)
95                                    .collect(toImmutableList());
96                  }
97                  return HttpApiUtil.throwResponse(
98                          ctx, HttpStatus.FORBIDDEN,
99                          "You must be an owner of project '%s' to retrieve removed repositories.",
100                         project.name());
101             }
102 
103             return project.repos().list().values().stream()
104                           .filter(r -> user.isSystemAdmin() || !Project.REPO_DOGMA.equals(r.name()))
105                           .filter(r -> hasOwnerRole || !Project.REPO_META.equals(r.name()))
106                           .map(DtoConverter::convert)
107                           .collect(toImmutableList());
108         });
109     }

111     /**
112      * POST /projects/{projectName}/repos
113      *
114      * <p>Creates a new repository.
115      */
116     @Post("/projects/{projectName}/repos")
117     @StatusCode(201)
118     @ResponseConverter(CreateApiResponseConverter.class)
119     @RequiresProjectRole(ProjectRole.OWNER)
120     public CompletableFuture<RepositoryDto> createRepository(ServiceRequestContext ctx, Project project,
121                                                              CreateRepositoryRequest request,
122                                                              Author author) {
123         final String repoName = request.name();
124         if (Project.isReservedRepoName(repoName)) {
125             return HttpApiUtil.throwResponse(ctx, HttpStatus.FORBIDDEN,
126                                              "A reserved repository cannot be created.");
127         }
128         final CommandExecutor commandExecutor = executor();
129         final CompletableFuture<Revision> future =
130                 RepositoryServiceUtil.createRepository(commandExecutor, mds, author, project.name(), repoName);
131         return future.handle(returnOrThrow(() -> DtoConverter.convert(project.repos().get(repoName))));
132     }

134     /**
135      * DELETE /projects/{projectName}/repos/{repoName}
136      *
137      * <p>Removes a repository.
138      */
139     @Delete("/projects/{projectName}/repos/{repoName}")
140     @RequiresProjectRole(ProjectRole.OWNER)
141     public CompletableFuture<Void> removeRepository(ServiceRequestContext ctx,
142                                                     @Param String repoName,
143                                                     Repository repository,
144                                                     Author author) {
145         if (Project.isReservedRepoName(repoName)) {
146             return HttpApiUtil.throwResponse(ctx, HttpStatus.FORBIDDEN,
147                                              "A reserved repository cannot be removed.");
148         }
149         return RepositoryServiceUtil.removeRepository(executor(), mds, author,
150                                                       repository.parent().name(), repoName)
151                                     .handle(HttpApiUtil::throwUnsafelyIfNonNull);
152     }
153 
154     /**
155      * DELETE /projects/{projectName}/repos/{repoName}/removed
156      *
157      * <p>Purges a repository that was removed before.
158      */
159     @Delete("/projects/{projectName}/repos/{repoName}/removed")
160     @RequiresProjectRole(ProjectRole.OWNER)
161     public CompletableFuture<Void> purgeRepository(@Param String repoName,
162                                                    Project project, Author author) {
163         return execute(Command.purgeRepository(author, project.name(), repoName))
164                 .thenCompose(unused -> mds.purgeRepo(author, project.name(), repoName)
165                                           .handle(HttpApiUtil::throwUnsafelyIfNonNull));
166     }

168     /**
169      * PATCH /projects/{projectName}/repos/{repoName}
170      *
171      * <p>Patches a repository with the JSON_PATCH. Currently, only unremove repository operation is supported.
172      */
173     @Consumes("application/json-patch+json")
174     @Patch("/projects/{projectName}/repos/{repoName}")
175     @RequiresProjectRole(ProjectRole.OWNER)
176     public CompletableFuture<RepositoryDto> patchRepository(@Param String repoName,
177                                                             Project project,
178                                                             JsonNode node,
179                                                             Author author) {
180         checkUnremoveArgument(node);
181         return execute(Command.unremoveRepository(author, project.name(), repoName))
182                 .thenCompose(unused -> mds.restoreRepo(author, project.name(), repoName))
183                 .handle(returnOrThrow(() -> DtoConverter.convert(project.repos().get(repoName))));
184     }

186     /**
187      * GET /projects/{projectName}/repos/{repoName}/revision/{revision}
188      *
189      * <p>Normalizes the revision into an absolute revision.
190      */
191     @Get("/projects/{projectName}/repos/{repoName}/revision/{revision}")
192     @RequiresRepositoryRole(RepositoryRole.READ)
193     public Map<String, Integer> normalizeRevision(ServiceRequestContext ctx,
194                                                   Repository repository, @Param String revision) {
195         final Revision normalizedRevision = repository.normalizeNow(new Revision(revision));
196         final Revision head = repository.normalizeNow(Revision.HEAD);
197         increaseCounterIfOldRevisionUsed(ctx, repository, normalizedRevision, head);
198         return ImmutableMap.of("revision", normalizedRevision.major());
199     }
200 
201     static void increaseCounterIfOldRevisionUsed(ServiceRequestContext ctx, Repository repository,
202                                                  Revision revision) {
203         final Revision normalized = repository.normalizeNow(revision);
204         final Revision head = repository.normalizeNow(Revision.HEAD);
205         increaseCounterIfOldRevisionUsed(ctx, repository, normalized, head);
206     }
207 
208     public static void increaseCounterIfOldRevisionUsed(
209             ServiceRequestContext ctx, Repository repository, Revision normalized, Revision head) {
210         final String projectName = repository.parent().name();
211         final String repoName = repository.name();
212         if (normalized.major() == 1) {
213             ctx.log().whenRequestComplete().thenAccept(
214                     log -> ctx.meterRegistry()
215                               .counter("revisions.init", generateTags(projectName, repoName, log).build())
216                               .increment());
217         }
218         if (head.major() - normalized.major() >= 5000) {
219             ctx.log().whenRequestComplete().thenAccept(
220                     log -> ctx.meterRegistry()
221                               .summary("revisions.old",
222                                        generateTags(projectName, repoName, log)
223                                                .add(Tag.of("init", Boolean.toString(normalized.major() == 1)))
224                                                .build())
225                               .record(head.major() - normalized.major()));
226         }
227     }
228 
229     private static ImmutableList.Builder<Tag> generateTags(
230             String projectName, String repoName, RequestOnlyLog log) {
231         final ImmutableList.Builder<Tag> builder = ImmutableList.builder();
232         return builder.add(Tag.of("project", projectName),
233                            Tag.of("repo", repoName),
234                            Tag.of("service", firstNonNull(log.serviceName(), "none")),
235                            Tag.of("method", log.name()));
236     }

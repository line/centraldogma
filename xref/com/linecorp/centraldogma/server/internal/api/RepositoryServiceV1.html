96      /**
97       * GET /projects/{projectName}/repos?status={status}
98       *
99       * <p>Returns the list of the repositories or removed repositories.
100      */
101     @Get("/projects/{projectName}/repos")
102     public CompletableFuture<List<RepositoryDto>> listRepositories(ServiceRequestContext ctx, Project project,
103                                                                    @Param @Nullable String status, User user) {
104         if (status != null) {
105             HttpApiUtil.checkStatusArgument(status);
106         }
107 
108         if (InternalProjectInitializer.INTERNAL_PROJECT_DOGMA.equals(project.name())) {
109             if (user.isSystemAdmin()) {
110                 if (status != null) {
111                     return CompletableFuture.completedFuture(removedRepositories(project));
112                 }
113                 return CompletableFuture.completedFuture(
114                         project.repos().list().values().stream()
115                                .map(repository -> DtoConverter.convert(repository, RepositoryStatus.ACTIVE))
116                                .collect(toImmutableList()));
117             }
118             return HttpApiUtil.throwResponse(
119                     ctx, HttpStatus.FORBIDDEN,
120                     "You must be a system administrator to retrieve repositories of the dogma project.");
121         }
122 
123         if (status == null) {
124             final Map<String, RepositoryMetadata> repos = projectMetadata(project).repos();
125             return CompletableFuture.completedFuture(
126                     project.repos().list().values().stream()
127                            .filter(r -> user.isSystemAdmin() || !Project.isInternalRepo(r.name()))
128                            .map(repository -> DtoConverter.convert(repository, repos))
129                            .collect(toImmutableList()));
130         }
131 
132         return mds.findProjectRole(project.name(), user).handle((role, throwable) -> {
133             final boolean hasOwnerRole = role == ProjectRole.OWNER;
134             if (hasOwnerRole) {
135                 return removedRepositories(project);
136             }
137             return HttpApiUtil.throwResponse(
138                     ctx, HttpStatus.FORBIDDEN,
139                     "You must be an owner of project '%s' to retrieve removed repositories.",
140                     project.name());
141         });
142     }

144     private static ProjectMetadata projectMetadata(Project project) {
145         assert !InternalProjectInitializer.INTERNAL_PROJECT_DOGMA.equals(project.name());
146         final ProjectMetadata metadata = project.metadata();
147         assert metadata != null; // not null because the project is not dogma project.
148         return metadata;
149     }
150 
151     private static RepositoryStatus repositoryStatus(Repository repository) {
152         final Map<String, RepositoryMetadata> repos = projectMetadata(repository.parent()).repos();
153         final String repoName = normalizeRepositoryName(repository);
154 
155         final RepositoryMetadata metadata = repos.get(repoName);
156         if (metadata == null) {
157             return RepositoryStatus.ACTIVE;
158         } else {
159             return metadata.status();
160         }
161     }
162 
163     private static String normalizeRepositoryName(Repository repository) {
164         final String repoName = repository.name();
165         if (!Project.REPO_META.equals(repoName)) {
166             return repoName;
167         }
168         // Use dogma repository for the meta repository because the meta repository will be removed.
169         return Project.REPO_DOGMA;
170     }
171 
172     private static ImmutableList<RepositoryDto> removedRepositories(Project project) {
173         return project.repos().listRemoved().keySet().stream().map(RepositoryDto::removed)
174                       .collect(toImmutableList());
175     }

177     /**
178      * POST /projects/{projectName}/repos
179      *
180      * <p>Creates a new repository.
181      */
182     @Post("/projects/{projectName}/repos")
183     @StatusCode(201)
184     @ResponseConverter(CreateApiResponseConverter.class)
185     @RequiresProjectRole(ProjectRole.MEMBER)
186     public CompletableFuture<RepositoryDto> createRepository(ServiceRequestContext ctx, Project project,
187                                                              CreateRepositoryRequest request,
188                                                              Author author) {
189         final String repoName = request.name();
190         if (Project.isInternalRepo(repoName)) {
191             return HttpApiUtil.throwResponse(ctx, HttpStatus.FORBIDDEN,
192                                              "An internal repository cannot be created.");
193         }
194         if (request.encrypt() && !encryptionStorageManager.enabled()) {
195             return HttpApiUtil.throwResponse(ctx, HttpStatus.BAD_REQUEST,
196                                              "Encryption is not enabled in the server.");
197         }
198 
199         final CommandExecutor commandExecutor = executor();
200         final CompletableFuture<Revision> future =
201                 RepositoryServiceUtil.createRepository(commandExecutor, mds, author, project.name(), repoName,
202                                                        request.encrypt(), encryptionStorageManager);
203         return future.handle(returnOrThrow(() -> {
204             final Repository repository = project.repos().get(repoName);
205             return DtoConverter.convert(repository, repositoryStatus(repository));
206         }));
207     }

209     /**
210      * DELETE /projects/{projectName}/repos/{repoName}
211      *
212      * <p>Removes a repository.
213      */
214     @Delete("/projects/{projectName}/repos/{repoName}")
215     @RequiresRepositoryRole(RepositoryRole.ADMIN)
216     public CompletableFuture<Void> removeRepository(ServiceRequestContext ctx,
217                                                     @Param String repoName,
218                                                     Repository repository,
219                                                     Author author) {
220         if (Project.isInternalRepo(repoName)) {
221             return HttpApiUtil.throwResponse(ctx, HttpStatus.FORBIDDEN,
222                                              "An internal repository cannot be removed.");
223         }
224         return RepositoryServiceUtil.removeRepository(executor(), mds, author,
225                                                       repository.parent().name(), repoName)
226                                     .handle(HttpApiUtil::throwUnsafelyIfNonNull);
227     }

245     /**
246      * PATCH /projects/{projectName}/repos/{repoName}
247      *
248      * <p>Patches a repository with the JSON_PATCH. Currently, only unremove repository operation is supported.
249      */
250     @Consumes("application/json-patch+json")
251     @Patch("/projects/{projectName}/repos/{repoName}")
252     @RequiresProjectRole(ProjectRole.OWNER)
253     public CompletableFuture<RepositoryDto> patchRepository(@Param String repoName,
254                                                             Project project,
255                                                             JsonNode node,
256                                                             Author author) {
257         checkUnremoveArgument(node);
258         return execute(Command.unremoveRepository(author, project.name(), repoName))
259                 .thenCompose(unused -> mds.restoreRepo(author, project.name(), repoName))
260                 .handle(returnOrThrow(() -> {
261                     final Repository repository = project.repos().get(repoName);
262                     return DtoConverter.convert(repository, repositoryStatus(repository));
263                 }));
264     }

267      * GET /projects/{projectName}/repos/{repoName}/revision/{revision}
268      *
269      * <p>Normalizes the revision into an absolute revision.
270      */
271     @Get("/projects/{projectName}/repos/{repoName}/revision/{revision}")
272     @RequiresRepositoryRole(RepositoryRole.READ)
273     public Map<String, Integer> normalizeRevision(ServiceRequestContext ctx,
274                                                   Repository repository, @Param String revision) {
275         final Revision normalizedRevision = repository.normalizeNow(new Revision(revision));
276         final Revision head = repository.normalizeNow(Revision.HEAD);
277         increaseCounterIfOldRevisionUsed(ctx, repository, normalizedRevision, head);
278         return ImmutableMap.of("revision", normalizedRevision.major());
279     }

293     /**
294      * PUT /projects/{projectName}/repos/{repoName}/status
295      *
296      * <p>Changes the repository status.
297      */
298     @Put("/projects/{projectName}/repos/{repoName}/status")
299     @Consumes("application/json")
300     @RequiresRepositoryRole(RepositoryRole.ADMIN)
301     public CompletableFuture<RepositoryDto> updateStatus(Project project,
302                                                          Repository repository,
303                                                          Author author,
304                                                          UpdateRepositoryStatusRequest statusRequest) {
305         rejectIfDogmaProject(project);
306         final RepositoryStatus oldStatus = repositoryStatus(repository);
307         final RepositoryStatus newStatus = statusRequest.status();
308         if (oldStatus == newStatus) {
309             // No need to update the status, just return the current status.
310             return CompletableFuture.completedFuture(DtoConverter.convert(repository, oldStatus));
311         }
312 
313         return mds.updateRepositoryStatus(author, project.name(),
314                                           normalizeRepositoryName(repository), newStatus)
315                   .thenApply(unused -> DtoConverter.convert(repository, newStatus));
316     }

318     /**
319      * POST /projects/{projectName}/repos/{repoName}/migrate/encrypted
320      *
321      * <p>Migrates the repository to an encrypted repository.
322      */
323     @Post("/projects/{projectName}/repos/{repoName}/migrate/encrypted")
324     @RequiresSystemAdministrator
325     public CompletableFuture<RepositoryDto> migrateToEncryptedRepository(ServiceRequestContext ctx,
326                                                                          Project project,
327                                                                          Repository repository,
328                                                                          Author author) {
329         validateMigrationPrerequisites(ctx, project, repository);
330         ctx.setRequestTimeoutMillis(Long.MAX_VALUE); // Disable the request timeout for migration.
331 
332         return encryptionStorageManager
333                 .generateWdek()
334                 .thenCompose(wdek -> setRepositoryStatus(author, project, repository,
335                                                          RepositoryStatus.READ_ONLY)
336                         .thenCompose(unused -> migrate(author, project, repository, wdek)));
337     }
338 
339     private void validateMigrationPrerequisites(ServiceRequestContext ctx, Project project,
340                                                 Repository repository) {
341         if (!encryptionStorageManager.enabled()) {
342             throw new IllegalArgumentException(
343                     "Encryption is not enabled in the server. Cannot migrate to an encrypted repository.");
344         }
345 
346         // TODO(minwoox): Dogma project and dogma repository will be migrated one day later by a plugin.
347         if (InternalProjectInitializer.INTERNAL_PROJECT_DOGMA.equals(project.name()) ||
348             project.name().startsWith("@") || Project.REPO_DOGMA.equals(repository.name())) {
349             throw new IllegalArgumentException(
350                     "Cannot migrate the internal project or repository to an encrypted repository. project: " +
351                     project.name() + ", repository: " + repository.name());
352         }
353 
354         final RepositoryStatus currentStatus = repositoryStatus(repository);
355         if (repository.isEncrypted()) {
356             throw new IllegalArgumentException(
357                     "The repository is already encrypted. Cannot migrate to an encrypted repository again." +
358                     " project: " + project.name() + ", repository: " + repository.name());
359         }
360 
361         final Revision normalizeNow = repository.normalizeNow(Revision.HEAD);
362         if (normalizeNow.major() > 1000) {
363             // Prohibit migration to an encrypted repository if the repository has more than 1000 revisions.
364             // After we implement the repository history rollover feature,
365             // the repository can be migrated to an encrypted repository.
366             throw new IllegalArgumentException(
367                     "Cannot migrate a repository with more than 1000 revisions to an encrypted repository.");
368         }
369 
370         if (currentStatus == RepositoryStatus.READ_ONLY) {
371             HttpApiUtil.throwResponse(
372                     ctx, HttpStatus.CONFLICT,
373                     "Cannot migrate a read-only repository to an encrypted repository. " +
374                     "Please change the status to ACTIVE first.");
375         }
376     }

378     private CompletableFuture<Void> setRepositoryStatus(Author author, Project project, Repository repository,
379                                                         RepositoryStatus status) {
380         final String projectName = project.name();
381         final String repoName = repository.name();
382         logger.info("Changing repository status: project={}, repository={}, status={}",
383                     projectName, repoName, status);
384         return mds.updateRepositoryStatus(author, projectName, repoName, status)
385                   .handle((unused, cause) -> {
386                       if (cause != null) {
387                           logger.warn("Failed to change the repository status: project={}, repository={}, " +
388                                       "status={}", projectName, repoName, status, cause);
389                           Exceptions.throwUnsafely(cause);
390                       } else {
391                           logger.info("Changed repository status: project={}, repository={}, status={}",
392                                       projectName, repoName, status);
393                       }
394                       return null;
395                   });
396     }

398     private CompletionStage<RepositoryDto> migrate(Author author, Project project,
399                                                    Repository repository, byte[] wdek) {
400         final String projectName = project.name();
401         final String repoName = repository.name();
402         logger.info("Starting repository encryption migration: project={}, repository={}",
403                     projectName, repoName);
404 
405         final Command<Void> command = Command.migrateToEncryptedRepository(
406                 null, author, projectName, repoName, wdek);
407 
408         return executor().execute(command)
409                          .handle((unused, cause) -> {
410                              if (cause != null) {
411                                  logger.warn("failed to migrate repository to an encrypted repository: " +
412                                              "project={}, repository={}", projectName, repoName, cause);
413                                  return setRepositoryStatus(author, project, repository,
414                                                             RepositoryStatus.ACTIVE)
415                                          .thenApply(unused1 -> (RepositoryDto) Exceptions.throwUnsafely(cause));
416                              }
417                              logger.info("Successfully migrated repository to an encrypted repository: " +
418                                          "project={}, repository={}", projectName, repoName);
419                              return setRepositoryStatus(author, project, repository, RepositoryStatus.ACTIVE)
420                                      .thenApply(unused1 -> {
421                                          final Repository updatedRepository =
422                                                  project.repos().get(repository.name());
423                                          return DtoConverter.convert(updatedRepository,
424                                                                      RepositoryStatus.ACTIVE);
425                                      });
426                          }).thenCompose(Function.identity());
427     }

429     static void increaseCounterIfOldRevisionUsed(ServiceRequestContext ctx, Repository repository,
430                                                  Revision revision) {
431         final Revision normalized = repository.normalizeNow(revision);
432         final Revision head = repository.normalizeNow(Revision.HEAD);
433         increaseCounterIfOldRevisionUsed(ctx, repository, normalized, head);
434     }
435 
436     public static void increaseCounterIfOldRevisionUsed(
437             ServiceRequestContext ctx, Repository repository, Revision normalized, Revision head) {
438         final String projectName = repository.parent().name();
439         final String repoName = repository.name();
440         if (normalized.major() == 1) {
441             ctx.log().whenRequestComplete().thenAccept(
442                     log -> ctx.meterRegistry()
443                               .counter("revisions.init", generateTags(projectName, repoName, log).build())
444                               .increment());
445         }
446         if (head.major() - normalized.major() >= 5000) {
447             ctx.log().whenRequestComplete().thenAccept(
448                     log -> ctx.meterRegistry()
449                               .summary("revisions.old",
450                                        generateTags(projectName, repoName, log)
451                                                .add(Tag.of("init", Boolean.toString(normalized.major() == 1)))
452                                                .build())
453                               .record(head.major() - normalized.major()));
454         }
455     }

457     private static ImmutableList.Builder<Tag> generateTags(
458             String projectName, String repoName, RequestOnlyLog log) {
459         final ImmutableList.Builder<Tag> builder = ImmutableList.builder();
460         return builder.add(Tag.of("project", projectName),
461                            Tag.of("repo", repoName),
462                            Tag.of("service", firstNonNull(log.serviceName(), "none")),
463                            Tag.of("method", log.name()));
464     }

470         }

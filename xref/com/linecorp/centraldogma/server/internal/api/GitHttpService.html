61   * A service that provides Git HTTP protocol.
62   */
63  @RequestConverter(HttpApiRequestConverter.class)
64  @RequiresRepositoryRole(RepositoryRole.READ)
65  public final class GitHttpService {
66  
67      private static final Logger logger = LoggerFactory.getLogger(GitHttpService.class);
68  
69      private static final char[] HEX_DIGITS = "0123456789abcdef".toCharArray();
70  
71      // TODO(minwoox): Add the headers in this class to Armeria.
72      private static final AggregatedHttpResponse CAPABILITY_ADVERTISEMENT_RESPONSE = AggregatedHttpResponse.of(
73              ResponseHeaders.builder(200)
74                             .contentType(MediaType.GIT_UPLOAD_PACK_ADVERTISEMENT)
75                             .add(HttpHeaderNames.CACHE_CONTROL, ServerCacheControl.REVALIDATED.asHeaderValue())
76                             .build(),
77              HttpData.ofUtf8(capabilityAdvertisement()));
78  
79      // https://git-scm.com/docs/protocol-capabilities/
80      private static String capabilityAdvertisement() {
81          final PacketLineFraming packetLineFraming = new PacketLineFraming();
82          packetLineFraming.putWithoutPktLine("001e# service=git-upload-pack\n");
83          packetLineFraming.flush();
84          packetLineFraming.put("version 2");
85          packetLineFraming.put(COMMAND_LS_REFS);
86          // Support limited options for now due to the unique characteristics of Git repositories in
87          // Central Dogma, such as having only a master branch and no tags, among other specifics.
88          packetLineFraming.put(COMMAND_FETCH + '=' + OPTION_WAIT_FOR_DONE + ' ' + OPTION_SHALLOW);
89          // TODO(minwoox): Migrate hash function https://git-scm.com/docs/hash-function-transition
90          packetLineFraming.put("object-format=sha1");
91          packetLineFraming.flush();
92          return packetLineFraming.toString();
93      }

95      private final ProjectApiManager projectApiManager;

97      public GitHttpService(ProjectApiManager projectApiManager) {
98          this.projectApiManager = requireNonNull(projectApiManager, "projectApiManager");
99      }

101     // https://www.git-scm.com/docs/gitprotocol-http#_smart_clients
102     @Get("/{projectName}/{repoName}/info/refs")
103     public HttpResponse advertiseCapability(@Header("git-protocol") @Nullable String gitProtocol,
104                                             @Param String service,
105                                             @Param String projectName, @Param String repoName, User user) {
106         repoName = maybeRemoveGitSuffix(repoName);
107         if (!"git-upload-pack".equals(service)) {
108             // Return 403 https://www.git-scm.com/docs/http-protocol#_smart_server_response
109             return HttpResponse.of(HttpStatus.FORBIDDEN, MediaType.PLAIN_TEXT_UTF_8,
110                                    "Unsupported service: " + service);
111         }
112 
113         if (gitProtocol == null || !gitProtocol.contains(VERSION_2_REQUEST)) {
114             return HttpResponse.of(HttpStatus.BAD_REQUEST, MediaType.PLAIN_TEXT_UTF_8,
115                                    "Unsupported git-protocol: " + gitProtocol);
116         }
117 
118         if (!projectApiManager.exists(projectName)) {
119             return HttpResponse.of(HttpStatus.NOT_FOUND, MediaType.PLAIN_TEXT_UTF_8,
120                                    "Project not found: " + projectName);
121         }
122         if (!projectApiManager.getProject(projectName, user).repos().exists(repoName)) {
123             return HttpResponse.of(HttpStatus.NOT_FOUND, MediaType.PLAIN_TEXT_UTF_8,
124                                    "Repository not found: " + repoName);
125         }
126         return CAPABILITY_ADVERTISEMENT_RESPONSE.toHttpResponse();
127     }
128 
129     private static String maybeRemoveGitSuffix(String repoName) {
130         if (repoName.length() >= 5 && repoName.endsWith(".git")) {
131             repoName = repoName.substring(0, repoName.length() - 4);
132         }
133         return repoName;
134     }
135 
136     // https://www.git-scm.com/docs/gitprotocol-http#_smart_service_git_upload_pack
137     @Post("/{projectName}/{repoName}/git-upload-pack")
138     public HttpResponse gitUploadPack(AggregatedHttpRequest req,
139                                       @Param String projectName, @Param String repoName, User user) {
140         repoName = maybeRemoveGitSuffix(repoName);
141         final String gitProtocol = req.headers().get(HttpHeaderNames.GIT_PROTOCOL);
142         if (gitProtocol == null || !gitProtocol.contains(VERSION_2_REQUEST)) {
143             return HttpResponse.of(HttpStatus.BAD_REQUEST, MediaType.PLAIN_TEXT_UTF_8,
144                                    "Unsupported git-protocol: " + gitProtocol);
145         }
146 
147         final MediaType contentType = req.headers().contentType();
148         if (MediaType.GIT_UPLOAD_PACK_REQUEST != contentType) {
149             return HttpResponse.of(HttpStatus.BAD_REQUEST, MediaType.PLAIN_TEXT_UTF_8,
150                                    "Unsupported content-type: " + contentType);
151         }
152 
153         if (!projectApiManager.exists(projectName)) {
154             return HttpResponse.of(HttpStatus.NOT_FOUND, MediaType.PLAIN_TEXT_UTF_8,
155                                    "Project not found: " + projectName);
156         }
157         final Project project = projectApiManager.getProject(projectName, user);
158         if (!project.repos().exists(repoName)) {
159             return HttpResponse.of(HttpStatus.NOT_FOUND, MediaType.PLAIN_TEXT_UTF_8,
160                                    "Repository not found: " + repoName);
161         }
162 
163         final Repository jGitRepository = project.repos().get(repoName).jGitRepository();
164 
165         final ByteStreamMessage body = StreamMessage.fromOutputStream(os -> {
166             // Don't need to close the input stream.
167             final ByteArrayInputStream inputStream = new ByteArrayInputStream(req.content().byteBuf().array());
168             // Don't need to close because we don't use the timer inside it.
169             final UploadPack uploadPack = new UploadPack(jGitRepository);
170             uploadPack.setTimeout(0); // Disable timeout because Armeria server will handle it.
171             // HTTP does not use bidirectional pipe.
172             uploadPack.setBiDirectionalPipe(false);
173             uploadPack.setExtraParameters(ImmutableList.of(VERSION_2_REQUEST));
174             try {
175                 uploadPack.upload(inputStream, os, null);
176             } catch (IOException e) {
177                 // Log until https://github.com/line/centraldogma/pull/719 is implemented.
178                 logger.debug("Failed to respond git-upload-pack-request: {}", req.contentUtf8(), e);
179                 throw new RuntimeException("failed to respond git-upload-pack-request: " +
180                                            req.contentUtf8(), e);
181             }
182             try {
183                 os.close();
184             } catch (IOException e) {
185                 // Should never reach here because StreamWriterOutputStream.close() never throws an exception.
186                 logger.warn("Failed to close the output stream. request: {}", req.contentUtf8(), e);
187             }
188         });
189         return HttpResponse.of(
190                 ResponseHeaders.builder(200)
191                                .contentType(MediaType.GIT_UPLOAD_PACK_RESULT)
192                                .add(HttpHeaderNames.CACHE_CONTROL,
193                                     ServerCacheControl.REVALIDATED.asHeaderValue())
194                                .build(), body);
195     }

197     static class PacketLineFraming {
198         private final StringBuilder sb = new StringBuilder();
199 
200         // https://git-scm.com/docs/protocol-common#_pkt_line_format
201         void put(String line) {
202             lineLength(sb, line.getBytes(StandardCharsets.UTF_8).length + 5);
203             sb.append(line).append('\n');
204         }
205 
206         private static void lineLength(StringBuilder sb, int length) {
207             for (int i = 3; i >= 0; i--) {
208                 sb.append(HEX_DIGITS[(length >>> (4 * i)) & 0xf]);
209             }
210         }
211 
212         void putWithoutPktLine(String line) {
213             sb.append(line);
214         }
215 
216         void delim() {
217             sb.append("0001");
218         }
219 
220         void flush() {
221             // https: //git-scm.com/docs/protocol-v2/2.31.0#_packet_line_framing
222             sb.append("0000");
223         }
224 
225         @Override
226         public String toString() {
227             return sb.toString();
228         }
229     }

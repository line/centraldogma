584         return new CentralDogma(buildConfig(), meterRegistry, ImmutableList.copyOf(plugins));

586 
587     private CentralDogmaConfig buildConfig() {
588         final List<ServerPort> ports = !this.ports.isEmpty() ? this.ports
589                                                              : Collections.singletonList(DEFAULT_PORT);
590         final Set<String> systemAdminSet = systemAdministrators.build();
591         final AuthConfig authCfg;
592         if (authProviderFactory != null) {
593             authCfg = new AuthConfig(
594                     authProviderFactory, systemAdminSet, caseSensitiveLoginNames,
595                     sessionCacheSpec, sessionTimeoutMillis, sessionValidationSchedule,
596                     authProviderProperties != null ? Jackson.valueToTree(authProviderProperties) : null);
597         } else {
598             authCfg = null;
599             logger.info("{} is not specified, so {} will not be configured.",
600                         AuthProviderFactory.class.getSimpleName(),
601                         AuthConfig.class.getSimpleName());
602         }
603 
604         final QuotaConfig quotaConfig = writeQuota > 0 ? new QuotaConfig(writeQuota, timeWindowSeconds) : null;
605 
606         return new CentralDogmaConfig(dataDir, ports, tls, trustedProxyAddresses, clientAddressSources,
607                                       numWorkers, maxNumConnections,
608                                       requestTimeoutMillis, idleTimeoutMillis, maxFrameLength,
609                                       numRepositoryWorkers, repositoryCacheSpec,
610                                       maxRemovedRepositoryAgeMillis, gracefulShutdownTimeout,
611                                       webAppEnabled, webAppTitle, replicationConfig,
612                                       null, accessLogFormat, authCfg, quotaConfig,
613                                       corsConfig, pluginConfigs, managementConfig, zoneConfig);
614     }

101     private static final Map<String, ConfigValueConverter> CONFIG_VALUE_CONVERTERS;
102 
103     static {
104         final SimpleModule module = new SimpleModule();
105         module.addDeserializer(PluginConfig.class, new PluginConfigDeserializer());
106         // Use different ObjectMapper to avoid infinite recursion.
107         objectMapper.registerModule(module);
108 
109         final ArrayList<ConfigValueConverter> configValueConverters = new ArrayList<>();
110         Streams.stream(ServiceLoader.load(ConfigValueConverter.class)).forEach(configValueConverters::add);
111         configValueConverters.add(DefaultConfigValueConverter.INSTANCE);
112         final ImmutableMap.Builder<String, ConfigValueConverter> builder = ImmutableMap.builder();
113         for (ConfigValueConverter configValueConverter : configValueConverters) {
114             if (configValueConverter.supportedPrefixes().isEmpty()) {
115                 continue;
116             }
117             boolean addConverter = true;
118             for (String supportedPrefix : configValueConverter.supportedPrefixes()) {
119                 if (!PREFIX_PATTERN.matcher(supportedPrefix).matches()) {
120                     logger.warn("{} isn't used because it has an invalid prefix: {}. (expected: {})",
121                                 configValueConverter, supportedPrefix, PREFIX_PATTERN.pattern());
122                     addConverter = false;
123                     break;
124                 }
125             }
126             if (addConverter) {
127                 configValueConverter.supportedPrefixes()
128                                     .forEach(prefix -> builder.put(prefix, configValueConverter));
129             }
130         }
131         CONFIG_VALUE_CONVERTERS = ImmutableMap.copyOf(builder.buildOrThrow());
132 
133         final StringBuilder sb = new StringBuilder();
134         sb.append('{');
135         CONFIG_VALUE_CONVERTERS.entrySet().stream().sorted(Entry.comparingByKey()).forEach(
136                 entry -> sb.append(entry.getKey())
137                            .append('=')
138                            .append(entry.getValue().getClass().getName()).append(", "));
139         sb.setLength(sb.length() - 2);
140         sb.append('}');
141 
142         logger.debug("Available {}s: {}", ConfigValueConverter.class.getName(), sb);
143     }

145     /**
146      * Converts the specified {@code value} using {@link ConfigValueConverter} if the specified {@code value}
147      * starts with a prefix followed by a colon {@code ':'}.
148      */
149     @Nullable
150     public static String convertValue(@Nullable String value, String propertyName) {
151         if (value == null) {
152             return null;
153         }
154 
155         final int index = value.indexOf(':');
156         if (index <= 0) {
157             // no prefix or starts with ':'.
158             return value;
159         }
160 
161         final String prefix = value.substring(0, index);
162         if (!PREFIX_PATTERN.matcher(prefix).matches()) {
163             // Not a prefix.
164             return value;
165         }
166 
167         final String rest = value.substring(index + 1);
168 
169         final ConfigValueConverter converter = CONFIG_VALUE_CONVERTERS.get(prefix);
170         if (converter != null) {
171             return converter.convert(prefix, rest);
172         }
173         logger.warn("No {} found for {}. prefix: {}",
174                     ConfigValueConverter.class.getSimpleName(), propertyName, prefix);
175         return value;
176     }

618     private static Predicate<InetAddress> toTrustedProxyAddressPredicate(List<String> trustedProxyAddresses) {
619         final String first = trustedProxyAddresses.get(0);
620         Predicate<InetAddress> predicate = first.indexOf('/') < 0 ? ofExact(first) : ofCidr(first);
621         for (int i = 1; i < trustedProxyAddresses.size(); i++) {
622             final String next = trustedProxyAddresses.get(i);
623             predicate = predicate.or(next.indexOf('/') < 0 ? ofExact(next) : ofCidr(next));
624         }
625         return predicate;
626     }

628     private static List<ClientAddressSource> toClientAddressSourceList(
629             @Nullable List<String> clientAddressSources,
630             boolean useDefaultSources, boolean specifiedProxyProtocol) {
631         if (clientAddressSources != null && !clientAddressSources.isEmpty()) {
632             return clientAddressSources.stream().map(
633                     name -> "PROXY_PROTOCOL".equals(name) ? ofProxyProtocol() : ofHeader(name))
634                                        .collect(toImmutableList());
635         }
636 
637         if (useDefaultSources) {
638             final Builder<ClientAddressSource> builder = new Builder<>();
639             builder.add(ofHeader(HttpHeaderNames.FORWARDED));
640             builder.add(ofHeader(HttpHeaderNames.X_FORWARDED_FOR));
641             if (specifiedProxyProtocol) {
642                 builder.add(ofProxyProtocol());
643             }
644             return builder.build();
645         }
646 
647         return ImmutableList.of();
648     }

650     static final class ServerPortSerializer extends JsonSerializer<ServerPort> {
651         @Override
652         public void serialize(ServerPort value,
653                               JsonGenerator gen, SerializerProvider serializers) throws IOException {
654 
655             final InetSocketAddress localAddr = value.localAddress();
656             final int port = localAddr.getPort();
657             final String host;
658 
659             if (localAddr.getAddress().isAnyLocalAddress()) {
660                 host = "*";
661             } else {
662                 final String hs = localAddr.getHostString();
663                 if (NetUtil.isValidIpV6Address(hs)) {
664                     // Try to get the platform-independent consistent IPv6 address string.
665                     host = NetUtil.toAddressString(localAddr.getAddress());
666                 } else {
667                     host = hs;
668                 }
669             }
670 
671             gen.writeStartObject();
672             gen.writeObjectFieldStart("localAddress");
673             gen.writeStringField("host", host);
674             gen.writeNumberField("port", port);
675             gen.writeEndObject();
676             gen.writeArrayFieldStart("protocols");
677             for (final SessionProtocol protocol : value.protocols()) {
678                 gen.writeString(protocol.uriText());
679             }
680             gen.writeEndArray();
681             gen.writeEndObject();
682         }
683     }

685     static final class ServerPortDeserializer extends JsonDeserializer<ServerPort> {
686         @Override
687         public ServerPort deserialize(JsonParser p, DeserializationContext ctx) throws IOException {
688 
689             final JsonNode root = p.getCodec().readTree(p);
690             final JsonNode localAddress = root.get("localAddress");
691             if (localAddress == null || localAddress.getNodeType() != JsonNodeType.OBJECT) {
692                 return fail(ctx, root);
693             }
694 
695             final JsonNode host = localAddress.get("host");
696             if (host == null || host.getNodeType() != JsonNodeType.STRING) {
697                 return fail(ctx, root);
698             }
699 
700             final JsonNode port = localAddress.get("port");
701             if (port == null || port.getNodeType() != JsonNodeType.NUMBER) {
702                 return fail(ctx, root);
703             }
704 
705             final ImmutableSet.Builder<SessionProtocol> protocolsBuilder = ImmutableSet.builder();
706             final JsonNode protocols = root.get("protocols");
707             if (protocols != null) {
708                 if (protocols.getNodeType() != JsonNodeType.ARRAY) {
709                     return fail(ctx, root);
710                 }
711                 protocols.elements().forEachRemaining(
712                         protocol -> protocolsBuilder.add(SessionProtocol.of(protocol.textValue())));
713             } else {
714                 final JsonNode protocol = root.get("protocol");
715                 if (protocol == null || protocol.getNodeType() != JsonNodeType.STRING) {
716                     return fail(ctx, root);
717                 }
718                 protocolsBuilder.add(SessionProtocol.of(protocol.textValue()));
719             }
720 
721             final String hostVal = host.textValue();
722             final int portVal = port.intValue();
723 
724             final InetSocketAddress localAddressVal;
725             if ("*".equals(hostVal)) {
726                 localAddressVal = new InetSocketAddress(portVal);
727             } else {
728                 localAddressVal = new InetSocketAddress(hostVal, portVal);
729             }
730 
731             return new ServerPort(localAddressVal, protocolsBuilder.build());
732         }
733 
734         private static ServerPort fail(DeserializationContext ctx, JsonNode root) throws JsonMappingException {
735             ctx.reportInputMismatch(ServerPort.class, "invalid server port information: %s", root);
736             throw new Error(); // Should never reach here.
737         }
738     }

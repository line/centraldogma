17  package com.linecorp.centraldogma.server.credential;
18  
19  import com.fasterxml.jackson.annotation.JsonIgnoreProperties;
20  import com.fasterxml.jackson.annotation.JsonInclude;
21  import com.fasterxml.jackson.annotation.JsonProperty;
22  import com.fasterxml.jackson.annotation.JsonSubTypes;
23  import com.fasterxml.jackson.annotation.JsonSubTypes.Type;
24  import com.fasterxml.jackson.annotation.JsonTypeInfo;
25  
26  import com.linecorp.centraldogma.server.internal.credential.AccessTokenLegacyCredential;
27  import com.linecorp.centraldogma.server.internal.credential.NoneLegacyCredential;
28  import com.linecorp.centraldogma.server.internal.credential.PasswordLegacyCredential;
29  import com.linecorp.centraldogma.server.internal.credential.PublicKeyLegacyCredential;
30  
31  /**
32   * A credential used to access external resources such as Git repositories or the Kubernetes control plane.
33   *
34   * @deprecated Use {@link Credential} instead.
35   */
36  @JsonTypeInfo(use = JsonTypeInfo.Id.NAME, include = JsonTypeInfo.As.EXISTING_PROPERTY, property = "type")
37  @JsonSubTypes({
38          @Type(value = NoneLegacyCredential.class, name = "none"),
39          @Type(value = PasswordLegacyCredential.class, name = "password"),
40          @Type(value = PublicKeyLegacyCredential.class, name = "public_key"),
41          @Type(value = AccessTokenLegacyCredential.class, name = "access_token")
42  })
43  @JsonInclude(JsonInclude.Include.NON_NULL)
44  @JsonIgnoreProperties(ignoreUnknown = true)
45  @Deprecated
46  public interface LegacyCredential {
47  
48      /**
49       * Returns the ID of the credential.
50       */
51      @JsonProperty("id")
52      String id();
53  
54      /**
55       * Returns whether this {@link LegacyCredential} is enabled.
56       */
57      @JsonProperty("enabled")
58      boolean enabled();
59  
60      /**
61       * Converts this {@link LegacyCredential} into a new {@link Credential}.
62       */
63      Credential toNewCredential(String name);
64  
65      /**
66       * Returns a new {@link LegacyCredential} that does not contain any sensitive information.
67       */
68      LegacyCredential withoutSecret();
69  }

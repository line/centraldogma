303     CentralDogma(CentralDogmaConfig cfg, MeterRegistry meterRegistry, List<Plugin> plugins) {
304         this.cfg = requireNonNull(cfg, "cfg");
305         pluginGroups = PluginGroup.loadPlugins(CentralDogma.class.getClassLoader(), cfg, plugins);
306         pluginsForAllReplicas = pluginGroups.get(PluginTarget.ALL_REPLICAS);
307         pluginsForLeaderOnly = pluginGroups.get(PluginTarget.LEADER_ONLY);
308         pluginsForZoneLeaderOnly = pluginGroups.get(PluginTarget.ZONE_LEADER_ONLY);
309         if (pluginsForZoneLeaderOnly != null) {
310             checkState(cfg.zone() != null,
311                        "zone must be specified when zone leader plugins are enabled.");
312         }
313         startStop = new CentralDogmaStartStop(pluginsForAllReplicas);
314         this.meterRegistry = meterRegistry;
315     }

415     /**
416      * Stops the server. This method does nothing if the server is stopped already.
417      */
418     public CompletableFuture<Void> stop() {
419         serverHealth.setHealthy(false);
420 
421         final Optional<GracefulShutdownTimeout> gracefulTimeoutOpt = cfg.gracefulShutdownTimeout();
422         if (gracefulTimeoutOpt.isPresent()) {
423             try {
424                 // Sleep 1 second so that clients have some time to redirect traffic according
425                 // to the health status
426                 Thread.sleep(1000);
427             } catch (InterruptedException e) {
428                 logger.debug("Interrupted while waiting for quiet period", e);
429                 Thread.currentThread().interrupt();
430             }
431         }
432 
433         numPendingStopRequests.incrementAndGet();
434         return startStop.stop().thenRun(numPendingStopRequests::decrementAndGet);
435     }

442     private boolean doStart() throws Exception {
443         boolean success = false;
444         ExecutorService repositoryWorker = null;
445         ScheduledExecutorService purgeWorker = null;
446         EncryptionStorageManager encryptionStorageManager = null;
447         ProjectManager pm = null;
448         CommandExecutor executor = null;
449         Server server = null;
450         SessionManager sessionManager = null;
451         try {
452             logger.info("Starting the Central Dogma ..");
453 
454             final ThreadPoolExecutor repositoryWorkerImpl = new ThreadPoolExecutor(
455                     cfg.numRepositoryWorkers(), cfg.numRepositoryWorkers(),
456                     // TODO(minwoox): Use LinkedTransferQueue when we upgrade to JDK 21.
457                     60, TimeUnit.SECONDS, new LinkedBlockingQueue<>(),
458                     new DefaultThreadFactory("repository-worker", true));
459             repositoryWorkerImpl.allowCoreThreadTimeOut(true);
460             repositoryWorker = ExecutorServiceMetrics.monitor(meterRegistry, repositoryWorkerImpl,
461                                                               "repositoryWorker");
462 
463             logger.info("Starting the project manager: {}", cfg.dataDir());
464 
465             purgeWorker = Executors.newSingleThreadScheduledExecutor(
466                     new DefaultThreadFactory("purge-worker", true));
467 
468             encryptionStorageManager = EncryptionStorageManager.of(cfg);
469 
470             pm = new DefaultProjectManager(cfg.dataDir(), repositoryWorker, purgeWorker,
471                                            meterRegistry, cfg.repositoryCacheSpec(), encryptionStorageManager);
472 
473             logger.info("Started the project manager: {}", pm);
474 
475             logger.info("Current settings:\n{}", cfg);
476 
477             sessionManager = initializeSessionManager();
478 
479             logger.info("Starting the command executor ..");
480             executor = startCommandExecutor(pm, repositoryWorker, purgeWorker,
481                                             meterRegistry, sessionManager, encryptionStorageManager);
482             // The projectInitializer is set in startCommandExecutor.
483             assert projectInitializer != null;
484             if (executor.isWritable()) {
485                 logger.info("Started the command executor.");
486             }
487 
488             logger.info("Starting the RPC server.");
489             server = startServer(pm, executor, purgeWorker, meterRegistry, sessionManager,
490                                  projectInitializer, encryptionStorageManager);
491             logger.info("Started the RPC server at: {}", server.activePorts());
492             logger.info("Started the Central Dogma successfully.");
493             success = true;
494         } finally {
495             if (success) {
496                 this.repositoryWorker = repositoryWorker;
497                 this.purgeWorker = purgeWorker;
498                 this.encryptionStorageManager = encryptionStorageManager;
499                 this.pm = pm;
500                 this.executor = executor;
501                 this.server = server;
502                 this.sessionManager = sessionManager;
503             } else {
504                 doStop(server, executor, pm, repositoryWorker, purgeWorker, sessionManager, mirrorRunner,
505                        encryptionStorageManager);
506             }
507         }
508         return success;
509     }

517     private CommandExecutor startCommandExecutor(
518             ProjectManager pm, Executor repositoryWorker,
519             ScheduledExecutorService purgeWorker, MeterRegistry meterRegistry,
520             @Nullable SessionManager sessionManager, EncryptionStorageManager encryptionStorageManager) {
521 
522         final Consumer<CommandExecutor> onTakeLeadership = exec -> {
523             if (pluginsForLeaderOnly != null) {
524                 logger.info("Starting plugins on the leader replica ..");
525                 pluginsForLeaderOnly
526                         .start(cfg, pm, exec, meterRegistry, purgeWorker, projectInitializer,
527                                mirrorAccessController)
528                         .handle((unused, cause) -> {
529                             if (cause == null) {
530                                 logger.info("Started plugins on the leader replica.");
531                             } else {
532                                 logger.error("Failed to start plugins on the leader replica..", cause);
533                             }
534                             return null;
535                         });
536             }
537         };
538 
539         final Consumer<CommandExecutor> onReleaseLeadership = exec -> {
540             if (pluginsForLeaderOnly != null) {
541                 logger.info("Stopping plugins on the leader replica ..");
542                 final CompletableFuture<?> future =
543                         pluginsForLeaderOnly
544                                 .stop(cfg, pm, exec, meterRegistry, purgeWorker, projectInitializer,
545                                       mirrorAccessController)
546                                 .handle((unused, cause) -> {
547                                     if (cause == null) {
548                                         logger.info("Stopped plugins on the leader replica.");
549                                     } else {
550                                         logger.error("Failed to stop plugins on the leader replica.", cause);
551                                     }
552                                     return null;
553                                 });
554                 try {
555                     future.get(60, TimeUnit.SECONDS);
556                 } catch (Exception e) {
557                     logger.warn("Failed to stop plugins on the leader replica in 60 seconds.", e);
558                 }
559             }
560         };
561 
562         Consumer<CommandExecutor> onTakeZoneLeadership = null;
563         Consumer<CommandExecutor> onReleaseZoneLeadership = null;
564         // TODO(ikhoon): Deduplicate
565         if (pluginsForZoneLeaderOnly != null) {
566             assert cfg.zone() != null;
567             final String zone = cfg.zone().currentZone();
568             onTakeZoneLeadership = exec -> {
569                 logger.info("Starting plugins on the {} zone leader replica ..", zone);
570                 pluginsForZoneLeaderOnly
571                         .start(cfg, pm, exec, meterRegistry, purgeWorker, projectInitializer,
572                                mirrorAccessController)
573                         .handle((unused, cause) -> {
574                             if (cause == null) {
575                                 logger.info("Started plugins on the {} zone leader replica.", zone);
576                             } else {
577                                 logger.error("Failed to start plugins on the {} zone leader replica..",
578                                              zone, cause);
579                             }
580                             return null;
581                         });
582             };
583             onReleaseZoneLeadership = exec -> {
584                 logger.info("Stopping plugins on the {} zone leader replica ..", zone);
585                 final CompletableFuture<?> future =
586                         pluginsForZoneLeaderOnly
587                                 .stop(cfg, pm, exec, meterRegistry, purgeWorker,
588                                       projectInitializer, mirrorAccessController)
589                                 .handle((unused, cause) -> {
590                                     if (cause == null) {
591                                         logger.info("Stopped plugins on the {} zone leader replica.", zone);
592                                     } else {
593                                         logger.error("Failed to stop plugins on the {} zone leader replica.",
594                                                      zone, cause);
595                                     }
596                                     return null;
597                                 });
598                 try {
599                     future.get(10, TimeUnit.SECONDS);
600                 } catch (Exception e) {
601                     logger.warn("Failed to stop plugins on the {} zone leader replica in 10 seconds", zone, e);
602                 }
603             };
604         }
605 
606         statusManager = new ServerStatusManager(cfg.dataDir());
607         logger.info("Startup mode: {}", statusManager.serverStatus());
608         final CommandExecutor executor;
609         final ReplicationMethod replicationMethod = cfg.replicationConfig().method();
610         switch (replicationMethod) {
611             case ZOOKEEPER:
612                 executor = newZooKeeperCommandExecutor(pm, repositoryWorker, statusManager, meterRegistry,
613                                                        sessionManager, encryptionStorageManager,
614                                                        onTakeLeadership, onReleaseLeadership,
615                                                        onTakeZoneLeadership, onReleaseZoneLeadership);
616                 break;
617             case NONE:
618                 logger.info("No replication mechanism specified; entering standalone");
619                 executor = new StandaloneCommandExecutor(pm, repositoryWorker, statusManager, sessionManager,
620                                                          encryptionStorageManager,
621                                                          onTakeLeadership, onReleaseLeadership,
622                                                          onTakeZoneLeadership, onReleaseZoneLeadership);
623                 break;
624             default:
625                 throw new Error("unknown replication method: " + replicationMethod);
626         }
627         projectInitializer = new InternalProjectInitializer(executor, pm, encryptionStorageManager);
628         mirrorAccessController = new DefaultMirrorAccessController();
629 
630         final ServerStatus initialServerStatus = statusManager.serverStatus();
631         executor.setWritable(initialServerStatus.writable());
632         if (!initialServerStatus.replicating()) {
633             projectInitializer.initializeInReadOnlyMode();
634             setMirrorAccessControllerRepository(pm, executor);
635             return executor;
636         }
637         try {
638             final CompletableFuture<Void> startFuture = executor.start();
639             while (!startFuture.isDone()) {
640                 if (numPendingStopRequests.get() > 0) {
641                     // Stop request has been issued.
642                     executor.stop().get();
643                     break;
644                 }
645 
646                 try {
647                     startFuture.get(100, TimeUnit.MILLISECONDS);
648                 } catch (TimeoutException unused) {
649                     // Taking long time ..
650                 }
651             }
652 
653             // Trigger the exception if any.
654             startFuture.get();
655             projectInitializer.initialize();
656         } catch (Exception e) {
657             logger.warn("Failed to start the command executor. Entering read-only.", e);
658             projectInitializer.initializeInReadOnlyMode();
659         }
660         setMirrorAccessControllerRepository(pm, executor);
661         return executor;
662     }

674         final AuthConfig authCfg = cfg.authConfig();
675         if (authCfg == null) {
676             return null;
677         }
678

681         try {

695                     logger.warn("Failed to close a session manager.", e);
696                 }
697             }
698         }

701     private Server startServer(ProjectManager pm, CommandExecutor executor,
702                                ScheduledExecutorService purgeWorker, MeterRegistry meterRegistry,
703                                @Nullable SessionManager sessionManager,
704                                InternalProjectInitializer projectInitializer,
705                                EncryptionStorageManager encryptionStorageManager) {
706         final ServerBuilder sb = Server.builder();
707         cfg.ports().forEach(sb::port);
708 
709         final boolean needsTls =
710                 cfg.ports().stream().anyMatch(ServerPort::hasTls) ||
711                 (cfg.managementConfig() != null && cfg.managementConfig().protocol().isTls());
712 
713         if (needsTls) {
714             try {
715                 final TlsConfig tlsConfig = cfg.tls();
716                 if (tlsConfig != null) {
717                     try (InputStream keyCertChainInputStream = tlsConfig.keyCertChainInputStream();
718                          InputStream keyInputStream = tlsConfig.keyInputStream()) {
719                         sb.tls(keyCertChainInputStream, keyInputStream, tlsConfig.keyPassword());
720                     }
721                 } else {
722                     logger.warn(
723                             "Missing TLS configuration. Generating a self-signed certificate for TLS support.");
724                     sb.tlsSelfSigned();
725                 }
726             } catch (Exception e) {
727                 Exceptions.throwUnsafely(e);
728             }
729         }
730 
731         sb.clientAddressSources(cfg.clientAddressSourceList());
732         sb.clientAddressTrustedProxyFilter(cfg.trustedProxyAddressPredicate());
733 
734         cfg.numWorkers().ifPresent(
735                 numWorkers -> sb.workerGroup(EventLoopGroups.newEventLoopGroup(numWorkers), true));
736         cfg.maxNumConnections().ifPresent(sb::maxNumConnections);
737         cfg.idleTimeoutMillis().ifPresent(sb::idleTimeoutMillis);
738         cfg.requestTimeoutMillis().ifPresent(sb::requestTimeoutMillis);
739         sb.maxRequestLength(cfg.maxFrameLength().orElse(DEFAULT_MAX_FRAME_LENGTH));
740         final Optional<GracefulShutdownTimeout> gracefulShutdownTimeout = cfg.gracefulShutdownTimeout();
741         final GracefulShutdown gracefulShutdown;
742         if (gracefulShutdownTimeout.isPresent()) {
743             gracefulShutdown =
744                     GracefulShutdown.builder()
745                                     .quietPeriodMillis(gracefulShutdownTimeout.get().quietPeriodMillis())
746                                     .timeoutMillis(gracefulShutdownTimeout.get().timeoutMillis())
747                                     .toExceptionFunction((ctx, req) -> new ShuttingDownException())
748                                     .build();
749         } else {
750             gracefulShutdown = GracefulShutdown.builder()
751                                                .toExceptionFunction((ctx, req) -> new ShuttingDownException())
752                                                .build();
753         }
754         sb.gracefulShutdown(gracefulShutdown);
755 
756         final MetadataService mds = new MetadataService(pm, executor, projectInitializer);
757         final WatchService watchService = new WatchService(meterRegistry);
758         final AuthProvider authProvider =
759                 createAuthProvider(executor, sessionManager, mds, needsTls, encryptionStorageManager);
760         final ProjectApiManager projectApiManager =
761                 new ProjectApiManager(pm, executor, mds, encryptionStorageManager);
762 
763         if (cfg.enableThriftService()) {
764             if (THRIFT_FOUND) {
765                 configureThriftService(cfg, sb, projectApiManager, executor, watchService, mds);
766             } else {
767                 logger.warn("Thrift service is enabled in the configuration, but the Thrift library " +
768                             "is not found. Skip enabling the Thrift service.");
769             }
770         }
771 
772         sb.service("/title", webAppTitleFile(cfg.webAppTitle(), SystemInfo.hostname()).asService());
773 
774         sb.service(HEALTH_CHECK_PATH, HealthCheckService.builder()
775                                                         .checkers(serverHealth)
776                                                         .build());
777         configManagement(sb, config().managementConfig());
778 
779         sb.serviceUnder("/docs/",
780                         DocService.builder()
781                                   .exampleHeaders(CentralDogmaService.class,
782                                                   HttpHeaders.of(HttpHeaderNames.AUTHORIZATION,
783                                                                  "Bearer " + CsrfToken.ANONYMOUS))
784                                   .build());
785         final Function<? super HttpService, AuthService> authService =
786                 authService(authProvider, sessionManager);
787         configureHttpApi(sb, projectApiManager, executor, watchService, mds, authProvider, authService,
788                          meterRegistry, encryptionStorageManager, sessionManager, needsTls);
789 
790         configureMetrics(sb, meterRegistry);
791         // Add the CORS service as the last decorator(executed first) so that the CORS service is applied
792         // before AuthService.
793         configCors(sb, config().corsConfig());
794 
795         // Configure access log format.
796         final String accessLogFormat = cfg.accessLogFormat();
797         if (isNullOrEmpty(accessLogFormat)) {
798             sb.accessLogWriter(AccessLogWriter.disabled(), true);
799         } else if ("common".equals(accessLogFormat)) {
800             sb.accessLogWriter(AccessLogWriter.common(), true);
801         } else if ("combined".equals(accessLogFormat)) {
802             sb.accessLogWriter(AccessLogWriter.combined(), true);
803         } else {
804             sb.accessLogFormat(accessLogFormat);
805         }
806 
807         if (pluginsForAllReplicas != null) {
808             final PluginInitContext pluginInitContext =
809                     new PluginInitContext(config(), pm, executor, meterRegistry, purgeWorker, sb,
810                                           authService, projectInitializer, mirrorAccessController);
811             pluginsForAllReplicas.plugins()
812                                  .forEach(p -> {
813                                      if (!(p instanceof AllReplicasPlugin)) {
814                                          return;
815                                      }
816                                      final AllReplicasPlugin plugin = (AllReplicasPlugin) p;
817                                      plugin.init(pluginInitContext);
818                                  });
819         }
820         // Configure the uncaught exception handler just before starting the server so that override the
821         // default exception handler set by third-party libraries such as NIOServerCnxnFactory.
822         Thread.setDefaultUncaughtExceptionHandler((t, e) -> logger.warn("Uncaught exception: {}", t, e));
823 
824         final Server s = sb.build();
825         s.start().join();
826         return s;
827     }

846     @Nullable
847     private AuthProvider createAuthProvider(
848             CommandExecutor commandExecutor, @Nullable SessionManager sessionManager, MetadataService mds,
849             boolean tlsEnabled, EncryptionStorageManager encryptionStorageManager) {
850         final AuthConfig authCfg = cfg.authConfig();
851         if (authCfg == null) {
852             return null;
853         }
854 
855         checkState(sessionManager != null, "SessionManager is null");
856         final BooleanSupplier sessionPropagatorWritableChecker = commandExecutor::isWritable;
857         final AuthProviderParameters parameters = new AuthProviderParameters(
858                 // Find application first, then find the session token.
859                 new ApplicationTokenAuthorizer(mds::findTokenBySecret)
860                         .orElse(new SessionCookieAuthorizer(
861                                 sessionManager, sessionPropagatorWritableChecker,
862                                 tlsEnabled, encryptionStorageManager,
863                                 authCfg.systemAdministrators())),
864                 cfg,
865                 sessionManager::generateSessionId,
866                 // Propagate login and logout events to the other replicas.
867                 session -> commandExecutor.execute(Command.createSession(session)),
868                 sessionId -> commandExecutor.execute(Command.removeSession(sessionId)),
869                 sessionPropagatorWritableChecker,
870                 sessionManager, tlsEnabled,
871                 encryptionStorageManager);
872         return authCfg.factory().create(parameters);
873     }

925     private void configureHttpApi(ServerBuilder sb,
926                                   ProjectApiManager projectApiManager, CommandExecutor executor,

934                 // Use the default ObjectMapper without any configuration.

938                 new RequiresRepositoryRoleDecoratorFactory(mds),
939                 new RequiresProjectRoleDecoratorFactory(mds)
940         );

942           // TODO(ikhoon): Consider exposing ReflectiveDependencyInjector as a public API via
943           //               DependencyInjector.ofReflective()
944           .dependencyInjector(new ReflectiveDependencyInjector(), false);

946         // Enable content compression for API responses.
947         final Function<? super HttpService, ? extends HttpService> decorator =
948                 authService.andThen(contentEncodingDecorator());
949         for (String path : ImmutableList.of(API_V0_PATH_PREFIX, API_V1_PATH_PREFIX)) {
950             final DecoratingServiceBindingBuilder decoratorBuilder =
951                     sb.routeDecorator().pathPrefix(path);
952             for (Route loginRoute : LOGIN_API_ROUTES) {
953                 decoratorBuilder.exclude(loginRoute);
954             }
955             for (Route logoutRoute : LOGOUT_API_ROUTES) {
956                 decoratorBuilder.exclude(logoutRoute);
957             }
958             decoratorBuilder.build(decorator);

960 
961         assert statusManager != null;
962         final ContextPathServicesBuilder apiV1ServiceBuilder = sb.contextPath(API_V1_PATH_PREFIX);
963         apiV1ServiceBuilder
964                 .annotatedService(new ServerStatusService(executor, statusManager))
965                 .annotatedService(new ProjectServiceV1(projectApiManager, executor))
966                 .annotatedService(new RepositoryServiceV1(executor, mds, encryptionStorageManager))
967                 .annotatedService(new CredentialServiceV1(projectApiManager, executor));

969             apiV1ServiceBuilder.annotatedService(new LoggerService());
970         }
971 
972         if (GIT_MIRROR_ENABLED) {
973             mirrorRunner = new MirrorRunner(projectApiManager, executor, cfg, meterRegistry,
974                                             mirrorAccessController);
975 
976             apiV1ServiceBuilder
977                     .annotatedService(new MirroringServiceV1(projectApiManager, executor, mirrorRunner, cfg,
978                                                              mirrorAccessController))
979                     .annotatedService(new MirrorAccessControlService(executor, mirrorAccessController));
980         }
981 
982         apiV1ServiceBuilder.annotatedService()
983                            .defaultServiceNaming(new ServiceNaming() {
984                                private final String serviceName = ContentServiceV1.class.getName();
985                                private final String watchServiceName =
986                                        serviceName.replace("ContentServiceV1", "WatchContentServiceV1");
987 
988                                @Override
989                                public String serviceName(ServiceRequestContext ctx) {
990                                    if (ctx.request().headers().contains(HttpHeaderNames.IF_NONE_MATCH)) {
991                                        return watchServiceName;
992                                    }
993                                    return serviceName;
994                                }
995                            })
996                            .build(new ContentServiceV1(executor, watchService, meterRegistry));
997 
998         if (authProvider != null) {
999             sb.service("/security_enabled", new AbstractHttpService() {
1000                 @Override
1001                 protected HttpResponse doGet(ServiceRequestContext ctx, HttpRequest req) {
1002                     return HttpResponse.of(HttpStatus.OK);
1003                 }
1004             });
1005 
1006             final AuthConfig authCfg = cfg.authConfig();
1007             assert authCfg != null : "authCfg";
1008             assert sessionManager != null : "sessionManager";
1009             apiV1ServiceBuilder
1010                     .annotatedService(new MetadataApiService(executor, mds, authCfg.loginNameNormalizer()))
1011                     .annotatedService(new TokenService(executor, mds));
1012 
1013             // authentication services:
1014             Optional.ofNullable(authProvider.loginApiService())
1015                     .ifPresent(login -> LOGIN_API_ROUTES.forEach(mapping -> sb.service(mapping, login)));
1016 
1017             // Provide logout API by default.
1018             final HttpService logout =
1019                     Optional.ofNullable(authProvider.logoutApiService())
1020                             .orElseGet(() -> new DefaultLogoutService(
1021                                     authProvider.parameters().logoutSessionPropagator(),
1022                                     authProvider.parameters().sessionPropagatorWritableChecker(),
1023                                     sessionManager, needsTls, encryptionStorageManager));
1024             for (Route route : LOGOUT_API_ROUTES) {
1025                 sb.service(route, logout);
1026             }
1027 
1028             authProvider.moreServices().forEach(sb::service);
1029         }
1030 
1031         if (encryptionStorageManager.enabled()) {
1032             apiV1ServiceBuilder.annotatedService(new KeyManagementService(executor, encryptionStorageManager));
1033         }
1034 
1035         sb.annotatedService()
1036           .decorator(decorator)
1037           .decorator(DecodingService.newDecorator())
1038           .build(new GitHttpService(projectApiManager));
1039         sb.annotatedService(API_V0_PATH_PREFIX, new UserService(sessionManager, needsTls, executor,
1040                                                                 authProvider, encryptionStorageManager));
1041 
1042         if (cfg.isWebAppEnabled()) {
1043             sb.annotatedService(API_V0_PATH_PREFIX, new RepositoryService(projectApiManager, executor));
1044 
1045             if (authProvider != null) {
1046                 // Will redirect to /web/auth/login by default.
1047                 final AllowedUrisConfig allowedUrisConfig = AllowedUrisConfig.of(config().corsConfig());
1048                 sb.service(LOGIN_PATH, authProvider.webLoginService(allowedUrisConfig));
1049                 // Will redirect to /web/auth/logout by default.
1050                 sb.service(LOGOUT_PATH, authProvider.webLogoutService(allowedUrisConfig));
1051             }
1052 
1053             // If the index.html is just returned, Next.js will handle the all remaining process such as
1054             // fetching resources and routes to the target pages.
1055             sb.serviceUnder("/app", HttpFile.of(CentralDogma.class.getClassLoader(),
1056                                                 "com/linecorp/centraldogma/webapp/index.html")
1057                                             .asService());
1058             // Serve all web resources except for '/app'.
1059             sb.route()
1060               .pathPrefix("/")
1061               .exclude("prefix:/app")
1062               .exclude("prefix:/api")
1063               .build(FileService.builder(CentralDogma.class.getClassLoader(),
1064                                          "com/linecorp/centraldogma/webapp")
1065                                 .cacheControl(ServerCacheControl.REVALIDATED)
1066                                 .autoDecompress(true)
1067                                 .serveCompressedFiles(true)
1068                                 .fallbackFileExtensions("html")
1069                                 .build());
1070         }
1071 
1072         sb.errorHandler(new HttpApiExceptionHandler());
1073     }

1093     private static void configManagement(ServerBuilder sb, @Nullable ManagementConfig managementConfig) {
1094         if (managementConfig == null) {
1095             return;
1096         }
1097 
1098         // curl -L https://<address>:<port>/internal/management/jvm/threaddump
1099         // curl -L https://<address>:<port>/internal/management/jvm/heapdump -o heapdump.hprof
1100         final int port = managementConfig.port();

1114             }
1115             sb.virtualHost(port)
1116               .route()
1117               .pathPrefix(managementConfig.path())
1118               .defaultServiceName("management")
1119               .build(ManagementService.of());

1123     private static Function<? super HttpService, EncodingService> contentEncodingDecorator() {
1124         return delegate -> EncodingService
1125                 .builder()
1126                 .encodableContentTypes(contentType -> {
1127                     if ("application".equals(contentType.type())) {
1128                         final String subtype = contentType.subtype();
1129                         switch (subtype) {
1130                             case "json":
1131                             case "xml":
1132                             case "x-git-upload-pack-advertisement":
1133                             case "x-git-upload-pack-result":
1134                                 return true;
1135                             default:
1136                                 return subtype.endsWith("+json") ||
1137                                        subtype.endsWith("+xml");
1138                         }
1139                     }
1140                     return false;
1141                 })
1142                 .build(delegate);

1149         // CompositeMeterRegistry
1150         if (registry instanceof PrometheusMeterRegistry) {

1152             sb.service(METRICS_PATH,
1153                        PrometheusExpositionService.of(prometheusMeterRegistry.getPrometheusRegistry()));
1154         } else if (registry instanceof CompositeMeterRegistry) {
1155             final PrometheusMeterRegistry prometheusMeterRegistry = PrometheusMeterRegistries.newRegistry();

1157             sb.service(METRICS_PATH,
1158                        PrometheusExpositionService.of(prometheusMeterRegistry.getPrometheusRegistry()));
1159             meterRegistryToBeClosed = prometheusMeterRegistry;
1160         } else {
1161             logger.info("Not exposing a prometheus endpoint for the type: {}", registry.getClass());

1167         new FileDescriptorMetrics().bindTo(registry);
1168         new ProcessorMetrics().bindTo(registry);
1169         new ClassLoaderMetrics().bindTo(registry);
1170         new UptimeMetrics().bindTo(registry);

1180     private void doStop() {
1181         if (server == null) {
1182             return;
1183         }
1184 
1185         final Server server = this.server;
1186         final CommandExecutor executor = this.executor;
1187         final ProjectManager pm = this.pm;
1188         final EncryptionStorageManager encryptionStorageManager = this.encryptionStorageManager;
1189         final ExecutorService repositoryWorker = this.repositoryWorker;
1190         final ExecutorService purgeWorker = this.purgeWorker;
1191         final SessionManager sessionManager = this.sessionManager;
1192         final MirrorRunner mirrorRunner = this.mirrorRunner;
1193 
1194         this.server = null;
1195         this.executor = null;
1196         this.encryptionStorageManager = null;
1197         this.pm = null;
1198         this.repositoryWorker = null;
1199         this.sessionManager = null;
1200         this.mirrorRunner = null;
1201         if (meterRegistryToBeClosed != null) {
1202             assert meterRegistry instanceof CompositeMeterRegistry;
1203             ((CompositeMeterRegistry) meterRegistry).remove(meterRegistryToBeClosed);
1204             meterRegistryToBeClosed.close();
1205             meterRegistryToBeClosed = null;
1206         }
1207 
1208         logger.info("Stopping the Central Dogma ..");
1209         if (!doStop(server, executor, pm, repositoryWorker, purgeWorker, sessionManager, mirrorRunner,
1210                     encryptionStorageManager)) {
1211             logger.warn("Stopped the Central Dogma with failure.");
1212         } else {
1213             logger.info("Stopped the Central Dogma successfully.");
1214         }
1215     }

1217     private static boolean doStop(
1218             @Nullable Server server, @Nullable CommandExecutor executor,
1219             @Nullable ProjectManager pm,
1220             @Nullable ExecutorService repositoryWorker, @Nullable ExecutorService purgeWorker,
1221             @Nullable SessionManager sessionManager, @Nullable MirrorRunner mirrorRunner,
1222             @Nullable EncryptionStorageManager encryptionStorageManager) {
1223 
1224         boolean success = true;
1225 
1226         // Stop the server first to reject new requests before stopping other resources.
1227         try {
1228             if (server != null) {
1229                 logger.info("Stopping the RPC server ..");
1230                 server.stop().join();
1231                 logger.info("Stopped the RPC server.");
1232             }
1233         } catch (Throwable t) {
1234             success = false;
1235             logger.warn("Failed to stop the RPC server:", t);
1236         }
1237 
1238         try {
1239             if (sessionManager != null) {
1240                 logger.info("Stopping the session manager ..");
1241                 sessionManager.close();
1242                 logger.info("Stopped the session manager.");
1243             }
1244         } catch (Throwable t) {
1245             success = false;
1246             logger.warn("Failed to stop the session manager:", t);
1247         }
1248 
1249         try {
1250             if (pm != null) {
1251                 logger.info("Stopping the project manager ..");
1252                 pm.close(ShuttingDownException::new);
1253                 logger.info("Stopped the project manager.");
1254             }
1255         } catch (Throwable t) {
1256             success = false;
1257             logger.warn("Failed to stop the project manager:", t);
1258         }
1259 
1260         try {
1261             if (executor != null) {
1262                 logger.info("Stopping the command executor ..");
1263                 executor.stop().get(60, TimeUnit.SECONDS);
1264                 logger.info("Stopped the command executor.");
1265             }
1266         } catch (TimeoutException unused) {
1267             success = false;
1268             logger.warn("Failed to stop the command executor in 60 seconds.");
1269         } catch (InterruptedException unused) {
1270             success = false;
1271             Thread.currentThread().interrupt();
1272             logger.warn("Interrupted while stopping the command executor.");
1273         } catch (Throwable t) {
1274             success = false;
1275             logger.warn("Failed to stop the command executor:", t);
1276         }
1277 
1278         try {
1279             if (encryptionStorageManager != null) {
1280                 logger.info("Stopping the encryption storage manager ..");
1281                 encryptionStorageManager.close();
1282                 logger.info("Stopped the encryption storage manager.");
1283             }
1284         } catch (Throwable t) {
1285             success = false;
1286             logger.warn("Failed to stop the encryption storage manager:", t);
1287         }
1288 
1289         final BiFunction<ExecutorService, String, Boolean> stopWorker = (worker, name) -> {
1290             try {
1291                 if (worker != null && !worker.isTerminated()) {
1292                     logger.info("Stopping the {} worker ..", name);
1293                     boolean interruptLater = false;
1294                     while (!worker.isTerminated()) {
1295                         worker.shutdownNow();
1296                         try {
1297                             worker.awaitTermination(1, TimeUnit.SECONDS);
1298                         } catch (InterruptedException e) {
1299                             // Interrupt later.
1300                             interruptLater = true;
1301                         }
1302                     }
1303                     logger.info("Stopped the {} worker.", name);
1304 
1305                     if (interruptLater) {
1306                         Thread.currentThread().interrupt();
1307                     }
1308                 }
1309                 return true;
1310             } catch (Throwable t) {
1311                 logger.warn("Failed to stop the " + name + " worker:", t);
1312                 return false;
1313             }
1314         };
1315         if (!stopWorker.apply(repositoryWorker, "repository")) {
1316             success = false;
1317         }
1318         if (!stopWorker.apply(purgeWorker, "purge")) {
1319             success = false;
1320         }
1321 
1322         try {
1323             if (mirrorRunner != null) {
1324                 logger.info("Stopping the mirror runner..");
1325                 mirrorRunner.close();
1326                 logger.info("Stopped the mirror runner.");
1327             }
1328         } catch (Throwable t) {
1329             success = false;
1330             logger.warn("Failed to stop the mirror runner:", t);
1331         }
1332 
1333         return success;
1334     }

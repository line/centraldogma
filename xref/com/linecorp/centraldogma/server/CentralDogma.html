278     CentralDogma(CentralDogmaConfig cfg, MeterRegistry meterRegistry, List<Plugin> plugins) {
279         this.cfg = requireNonNull(cfg, "cfg");
280         pluginGroups = PluginGroup.loadPlugins(CentralDogma.class.getClassLoader(), cfg, plugins);
281         pluginsForAllReplicas = pluginGroups.get(PluginTarget.ALL_REPLICAS);
282         pluginsForLeaderOnly = pluginGroups.get(PluginTarget.LEADER_ONLY);
283         pluginsForZoneLeaderOnly = pluginGroups.get(PluginTarget.ZONE_LEADER_ONLY);
284         if (pluginsForZoneLeaderOnly != null) {
285             checkState(cfg.zone() != null,
286                        "zone must be specified when zone leader plugins are enabled.");
287         }
288         startStop = new CentralDogmaStartStop(pluginsForAllReplicas);
289         this.meterRegistry = meterRegistry;
290     }

379     /**
380      * Stops the server. This method does nothing if the server is stopped already.
381      */
382     public CompletableFuture<Void> stop() {
383         serverHealth.setHealthy(false);
384 
385         final Optional<GracefulShutdownTimeout> gracefulTimeoutOpt = cfg.gracefulShutdownTimeout();
386         if (gracefulTimeoutOpt.isPresent()) {
387             try {
388                 // Sleep 1 second so that clients have some time to redirect traffic according
389                 // to the health status
390                 Thread.sleep(1000);
391             } catch (InterruptedException e) {
392                 logger.debug("Interrupted while waiting for quiet period", e);
393                 Thread.currentThread().interrupt();
394             }
395         }
396 
397         numPendingStopRequests.incrementAndGet();
398         return startStop.stop().thenRun(numPendingStopRequests::decrementAndGet);
399     }

406     private boolean doStart() throws Exception {
407         boolean success = false;
408         ExecutorService repositoryWorker = null;
409         ScheduledExecutorService purgeWorker = null;
410         EncryptionStorageManager encryptionStorageManager = null;
411         ProjectManager pm = null;
412         CommandExecutor executor = null;
413         Server server = null;
414         SessionManager sessionManager = null;
415         try {
416             logger.info("Starting the Central Dogma ..");
417 
418             final ThreadPoolExecutor repositoryWorkerImpl = new ThreadPoolExecutor(
419                     cfg.numRepositoryWorkers(), cfg.numRepositoryWorkers(),
420                     // TODO(minwoox): Use LinkedTransferQueue when we upgrade to JDK 21.
421                     60, TimeUnit.SECONDS, new LinkedBlockingQueue<>(),
422                     new DefaultThreadFactory("repository-worker", true));
423             repositoryWorkerImpl.allowCoreThreadTimeOut(true);
424             repositoryWorker = ExecutorServiceMetrics.monitor(meterRegistry, repositoryWorkerImpl,
425                                                               "repositoryWorker");
426 
427             logger.info("Starting the project manager: {}", cfg.dataDir());
428 
429             purgeWorker = Executors.newSingleThreadScheduledExecutor(
430                     new DefaultThreadFactory("purge-worker", true));
431 
432             encryptionStorageManager = EncryptionStorageManager.of(cfg);
433 
434             pm = new DefaultProjectManager(cfg.dataDir(), repositoryWorker, purgeWorker,
435                                            meterRegistry, cfg.repositoryCacheSpec(), encryptionStorageManager);
436 
437             logger.info("Started the project manager: {}", pm);
438 
439             logger.info("Current settings:\n{}", cfg);
440 
441             sessionManager = initializeSessionManager();
442 
443             logger.info("Starting the command executor ..");
444             executor = startCommandExecutor(pm, repositoryWorker, purgeWorker,
445                                             meterRegistry, sessionManager, encryptionStorageManager);
446             // The projectInitializer is set in startCommandExecutor.
447             assert projectInitializer != null;
448             if (executor.isWritable()) {
449                 logger.info("Started the command executor.");
450             }
451 
452             logger.info("Starting the RPC server.");
453             server = startServer(pm, executor, purgeWorker, meterRegistry, sessionManager,
454                                  projectInitializer, encryptionStorageManager);
455             logger.info("Started the RPC server at: {}", server.activePorts());
456             logger.info("Started the Central Dogma successfully.");
457             success = true;
458         } finally {
459             if (success) {
460                 this.repositoryWorker = repositoryWorker;
461                 this.purgeWorker = purgeWorker;
462                 this.encryptionStorageManager = encryptionStorageManager;
463                 this.pm = pm;
464                 this.executor = executor;
465                 this.server = server;
466                 this.sessionManager = sessionManager;
467             } else {
468                 doStop(server, executor, pm, repositoryWorker, purgeWorker, sessionManager, mirrorRunner,
469                        encryptionStorageManager);
470             }
471         }
472         return success;
473     }

481     private CommandExecutor startCommandExecutor(
482             ProjectManager pm, Executor repositoryWorker,
483             ScheduledExecutorService purgeWorker, MeterRegistry meterRegistry,
484             @Nullable SessionManager sessionManager, EncryptionStorageManager encryptionStorageManager) {
485 
486         final Consumer<CommandExecutor> onTakeLeadership = exec -> {
487             if (pluginsForLeaderOnly != null) {
488                 logger.info("Starting plugins on the leader replica ..");
489                 pluginsForLeaderOnly
490                         .start(cfg, pm, exec, meterRegistry, purgeWorker, projectInitializer,
491                                mirrorAccessController)
492                         .handle((unused, cause) -> {
493                             if (cause == null) {
494                                 logger.info("Started plugins on the leader replica.");
495                             } else {
496                                 logger.error("Failed to start plugins on the leader replica..", cause);
497                             }
498                             return null;
499                         });
500             }
501         };
502 
503         final Consumer<CommandExecutor> onReleaseLeadership = exec -> {
504             if (pluginsForLeaderOnly != null) {
505                 logger.info("Stopping plugins on the leader replica ..");
506                 final CompletableFuture<?> future =
507                         pluginsForLeaderOnly
508                                 .stop(cfg, pm, exec, meterRegistry, purgeWorker, projectInitializer,
509                                       mirrorAccessController)
510                                 .handle((unused, cause) -> {
511                                     if (cause == null) {
512                                         logger.info("Stopped plugins on the leader replica.");
513                                     } else {
514                                         logger.error("Failed to stop plugins on the leader replica.", cause);
515                                     }
516                                     return null;
517                                 });
518                 try {
519                     future.get(10, TimeUnit.SECONDS);
520                 } catch (Exception e) {
521                     logger.warn("Failed to stop plugins on the leader replica in 10 seconds.", e);
522                 }
523             }
524         };
525 
526         Consumer<CommandExecutor> onTakeZoneLeadership = null;
527         Consumer<CommandExecutor> onReleaseZoneLeadership = null;
528         // TODO(ikhoon): Deduplicate
529         if (pluginsForZoneLeaderOnly != null) {
530             assert cfg.zone() != null;
531             final String zone = cfg.zone().currentZone();
532             onTakeZoneLeadership = exec -> {
533                 logger.info("Starting plugins on the {} zone leader replica ..", zone);
534                 pluginsForZoneLeaderOnly
535                         .start(cfg, pm, exec, meterRegistry, purgeWorker, projectInitializer,
536                                mirrorAccessController)
537                         .handle((unused, cause) -> {
538                             if (cause == null) {
539                                 logger.info("Started plugins on the {} zone leader replica.", zone);
540                             } else {
541                                 logger.error("Failed to start plugins on the {} zone leader replica..",
542                                              zone, cause);
543                             }
544                             return null;
545                         });
546             };
547             onReleaseZoneLeadership = exec -> {
548                 logger.info("Stopping plugins on the {} zone leader replica ..", zone);
549                 final CompletableFuture<?> future =
550                         pluginsForZoneLeaderOnly
551                                 .stop(cfg, pm, exec, meterRegistry, purgeWorker,
552                                       projectInitializer, mirrorAccessController)
553                                 .handle((unused, cause) -> {
554                                     if (cause == null) {
555                                         logger.info("Stopped plugins on the {} zone leader replica.", zone);
556                                     } else {
557                                         logger.error("Failed to stop plugins on the {} zone leader replica.",
558                                                      zone, cause);
559                                     }
560                                     return null;
561                                 });
562                 try {
563                     future.get(10, TimeUnit.SECONDS);
564                 } catch (Exception e) {
565                     logger.warn("Failed to stop plugins on the {} zone leader replica in 10 seconds", zone, e);
566                 }
567             };
568         }
569 
570         statusManager = new ServerStatusManager(cfg.dataDir());
571         logger.info("Startup mode: {}", statusManager.serverStatus());
572         final CommandExecutor executor;
573         final ReplicationMethod replicationMethod = cfg.replicationConfig().method();
574         switch (replicationMethod) {
575             case ZOOKEEPER:
576                 executor = newZooKeeperCommandExecutor(pm, repositoryWorker, statusManager, meterRegistry,
577                                                        sessionManager, encryptionStorageManager,
578                                                        onTakeLeadership, onReleaseLeadership,
579                                                        onTakeZoneLeadership, onReleaseZoneLeadership);
580                 break;
581             case NONE:
582                 logger.info("No replication mechanism specified; entering standalone");
583                 executor = new StandaloneCommandExecutor(pm, repositoryWorker, statusManager, sessionManager,
584                                                          encryptionStorageManager,
585                                                          onTakeLeadership, onReleaseLeadership,
586                                                          onTakeZoneLeadership, onReleaseZoneLeadership);
587                 break;
588             default:
589                 throw new Error("unknown replication method: " + replicationMethod);
590         }
591         projectInitializer = new InternalProjectInitializer(executor, pm);
592         mirrorAccessController = new DefaultMirrorAccessController();
593 
594         final ServerStatus initialServerStatus = statusManager.serverStatus();
595         executor.setWritable(initialServerStatus.writable());
596         if (!initialServerStatus.replicating()) {
597             projectInitializer.whenInitialized().complete(null);
598             return executor;
599         }
600         try {
601             final CompletableFuture<Void> startFuture = executor.start();
602             while (!startFuture.isDone()) {
603                 if (numPendingStopRequests.get() > 0) {
604                     // Stop request has been issued.
605                     executor.stop().get();
606                     break;
607                 }
608 
609                 try {
610                     startFuture.get(100, TimeUnit.MILLISECONDS);
611                 } catch (TimeoutException unused) {
612                     // Taking long time ..
613                 }
614             }
615 
616             // Trigger the exception if any.
617             startFuture.get();
618             projectInitializer.initialize();
619             final CrudRepository<MirrorAccessControl> accessControlRepository =
620                     new GitCrudRepository<>(MirrorAccessControl.class, executor, pm,
621                                             INTERNAL_PROJECT_DOGMA, Project.REPO_DOGMA,
622                                             MIRROR_ACCESS_CONTROL_PATH);
623             mirrorAccessController.setRepository(accessControlRepository);
624         } catch (Exception e) {
625             projectInitializer.whenInitialized().complete(null);
626             logger.warn("Failed to start the command executor. Entering read-only.", e);
627         }
628 
629         return executor;
630     }

633     private SessionManager initializeSessionManager() throws Exception {
634         final AuthConfig authCfg = cfg.authConfig();
635         if (authCfg == null) {
636             return null;
637         }
638

641         try {

646             success = true;

655                     logger.warn("Failed to close a session manager.", e);
656                 }
657             }
658         }

661     private Server startServer(ProjectManager pm, CommandExecutor executor,
662                                ScheduledExecutorService purgeWorker, MeterRegistry meterRegistry,
663                                @Nullable SessionManager sessionManager,
664                                InternalProjectInitializer projectInitializer,
665                                EncryptionStorageManager encryptionStorageManager) {
666         final ServerBuilder sb = Server.builder();
667         cfg.ports().forEach(sb::port);
668 
669         final boolean needsTls =
670                 cfg.ports().stream().anyMatch(ServerPort::hasTls) ||
671                 (cfg.managementConfig() != null && cfg.managementConfig().protocol().isTls());
672 
673         if (needsTls) {
674             try {
675                 final TlsConfig tlsConfig = cfg.tls();
676                 if (tlsConfig != null) {
677                     try (InputStream keyCertChainInputStream = tlsConfig.keyCertChainInputStream();
678                          InputStream keyInputStream = tlsConfig.keyInputStream()) {
679                         sb.tls(keyCertChainInputStream, keyInputStream, tlsConfig.keyPassword());
680                     }
681                 } else {
682                     logger.warn(
683                             "Missing TLS configuration. Generating a self-signed certificate for TLS support.");
684                     sb.tlsSelfSigned();
685                 }
686             } catch (Exception e) {
687                 Exceptions.throwUnsafely(e);
688             }
689         }
690 
691         sb.clientAddressSources(cfg.clientAddressSourceList());
692         sb.clientAddressTrustedProxyFilter(cfg.trustedProxyAddressPredicate());
693 
694         cfg.numWorkers().ifPresent(
695                 numWorkers -> sb.workerGroup(EventLoopGroups.newEventLoopGroup(numWorkers), true));
696         cfg.maxNumConnections().ifPresent(sb::maxNumConnections);
697         cfg.idleTimeoutMillis().ifPresent(sb::idleTimeoutMillis);
698         cfg.requestTimeoutMillis().ifPresent(sb::requestTimeoutMillis);
699         sb.maxRequestLength(cfg.maxFrameLength().orElse(DEFAULT_MAX_FRAME_LENGTH));
700         cfg.gracefulShutdownTimeout().ifPresent(t -> {
701             final GracefulShutdown gracefulShutdown =
702                     GracefulShutdown.builder()
703                                     .quietPeriodMillis(t.quietPeriodMillis())
704                                     .timeoutMillis(t.timeoutMillis())
705                                     .toExceptionFunction((ctx, req) -> {
706                                         return new ShuttingDownException();
707                                     })
708                                     .build();
709             sb.gracefulShutdown(gracefulShutdown);
710         });
711 
712         final MetadataService mds = new MetadataService(pm, executor, projectInitializer);
713         final WatchService watchService = new WatchService(meterRegistry);
714         final AuthProvider authProvider = createAuthProvider(executor, sessionManager, mds);
715         final ProjectApiManager projectApiManager =
716                 new ProjectApiManager(pm, executor, mds, encryptionStorageManager);
717 
718         configureThriftService(sb, projectApiManager, executor, watchService, mds);
719 
720         sb.service("/title", webAppTitleFile(cfg.webAppTitle(), SystemInfo.hostname()).asService());
721 
722         sb.service(HEALTH_CHECK_PATH, HealthCheckService.builder()
723                                                         .checkers(serverHealth)
724                                                         .build());
725         configManagement(sb, config().managementConfig());
726 
727         sb.serviceUnder("/docs/",
728                         DocService.builder()
729                                   .exampleHeaders(CentralDogmaService.class,
730                                                   HttpHeaders.of(HttpHeaderNames.AUTHORIZATION,
731                                                                  "Bearer " + CsrfToken.ANONYMOUS))
732                                   .build());
733         final Function<? super HttpService, AuthService> authService =
734                 authService(mds, authProvider, sessionManager);
735         configureHttpApi(sb, projectApiManager, executor, watchService, mds, authProvider, authService,
736                          meterRegistry, encryptionStorageManager);
737 
738         configureMetrics(sb, meterRegistry);
739         // Add the CORS service as the last decorator(executed first) so that the CORS service is applied
740         // before AuthService.
741         configCors(sb, config().corsConfig());
742 
743         // Configure access log format.
744         final String accessLogFormat = cfg.accessLogFormat();
745         if (isNullOrEmpty(accessLogFormat)) {
746             sb.accessLogWriter(AccessLogWriter.disabled(), true);
747         } else if ("common".equals(accessLogFormat)) {
748             sb.accessLogWriter(AccessLogWriter.common(), true);
749         } else if ("combined".equals(accessLogFormat)) {
750             sb.accessLogWriter(AccessLogWriter.combined(), true);
751         } else {
752             sb.accessLogFormat(accessLogFormat);
753         }
754 
755         if (pluginsForAllReplicas != null) {
756             final PluginInitContext pluginInitContext =
757                     new PluginInitContext(config(), pm, executor, meterRegistry, purgeWorker, sb,
758                                           authService, projectInitializer, mirrorAccessController);
759             pluginsForAllReplicas.plugins()
760                                  .forEach(p -> {
761                                      if (!(p instanceof AllReplicasPlugin)) {
762                                          return;
763                                      }
764                                      final AllReplicasPlugin plugin = (AllReplicasPlugin) p;
765                                      plugin.init(pluginInitContext);
766                                  });
767         }
768         // Configure the uncaught exception handler just before starting the server so that override the
769         // default exception handler set by third-party libraries such as NIOServerCnxnFactory.
770         Thread.setDefaultUncaughtExceptionHandler((t, e) -> logger.warn("Uncaught exception: {}", t, e));
771 
772         final Server s = sb.build();
773         s.start().join();
774         return s;
775     }

795     private AuthProvider createAuthProvider(
796             CommandExecutor commandExecutor, @Nullable SessionManager sessionManager, MetadataService mds) {
797         final AuthConfig authCfg = cfg.authConfig();
798         if (authCfg == null) {
799             return null;
800         }

803         final AuthProviderParameters parameters = new AuthProviderParameters(
804                 // Find application first, then find the session token.
805                 new ApplicationTokenAuthorizer(mds::findTokenBySecret).orElse(

812         return authCfg.factory().create(parameters);

1074     private static Function<? super HttpService, EncodingService> contentEncodingDecorator() {
1075         return delegate -> EncodingService
1076                 .builder()
1077                 .encodableContentTypes(contentType -> {
1078                     if ("application".equals(contentType.type())) {
1079                         final String subtype = contentType.subtype();
1080                         switch (subtype) {
1081                             case "json":
1082                             case "xml":
1083                             case "x-thrift":
1084                             case "x-git-upload-pack-advertisement":
1085                             case "x-git-upload-pack-result":
1086                                 return true;
1087                             default:
1088                                 return subtype.endsWith("+json") ||
1089                                        subtype.endsWith("+xml") ||
1090                                        subtype.startsWith("vnd.apache.thrift.");
1091                         }
1092                     }
1093                     return false;
1094                 })
1095                 .build(delegate);
1096     }

1133     private void doStop() {
1134         if (server == null) {
1135             return;
1136         }
1137 
1138         final Server server = this.server;
1139         final CommandExecutor executor = this.executor;
1140         final ProjectManager pm = this.pm;
1141         final EncryptionStorageManager encryptionStorageManager = this.encryptionStorageManager;
1142         final ExecutorService repositoryWorker = this.repositoryWorker;
1143         final ExecutorService purgeWorker = this.purgeWorker;

1146

1148         this.executor = null;
1149         this.encryptionStorageManager = null;
1150         this.pm = null;
1151         this.repositoryWorker = null;
1152         this.sessionManager = null;
1153         this.mirrorRunner = null;
1154         if (meterRegistryToBeClosed != null) {
1155             assert meterRegistry instanceof CompositeMeterRegistry;
1156             ((CompositeMeterRegistry) meterRegistry).remove(meterRegistryToBeClosed);
1157             meterRegistryToBeClosed.close();
1158             meterRegistryToBeClosed = null;
1159         }
1160 
1161         logger.info("Stopping the Central Dogma ..");
1162         if (!doStop(server, executor, pm, repositoryWorker, purgeWorker, sessionManager, mirrorRunner,
1163                     encryptionStorageManager)) {
1164             logger.warn("Stopped the Central Dogma with failure.");
1165         } else {
1166             logger.info("Stopped the Central Dogma successfully.");
1167         }
1168     }

1235         final BiFunction<ExecutorService, String, Boolean> stopWorker = (worker, name) -> {
1236             try {
1237                 if (worker != null && !worker.isTerminated()) {
1238                     logger.info("Stopping the {} worker ..", name);
1239                     boolean interruptLater = false;
1240                     while (!worker.isTerminated()) {
1241                         worker.shutdownNow();
1242                         try {
1243                             worker.awaitTermination(1, TimeUnit.SECONDS);
1244                         } catch (InterruptedException e) {
1245                             // Interrupt later.
1246                             interruptLater = true;
1247                         }
1248                     }
1249                     logger.info("Stopped the {} worker.", name);
1250 
1251                     if (interruptLater) {
1252                         Thread.currentThread().interrupt();
1253                     }
1254                 }
1255                 return true;
1256             } catch (Throwable t) {
1257                 logger.warn("Failed to stop the " + name + " worker:", t);
1258                 return false;
1259             }
1260         };
1261         if (!stopWorker.apply(repositoryWorker, "repository")) {
1262             success = false;
1263         }
1264         if (!stopWorker.apply(purgeWorker, "purge")) {
1265             success = false;
1266         }

16  package com.linecorp.centraldogma.server;
17  
18  import static com.google.common.base.MoreObjects.firstNonNull;
19  import static java.util.Objects.requireNonNull;
20  
21  import java.io.File;
22  import java.io.IOException;
23  import java.nio.file.Files;
24  import java.nio.file.Path;
25  
26  import javax.annotation.Nullable;
27  
28  import com.fasterxml.jackson.annotation.JsonCreator;
29  import com.fasterxml.jackson.annotation.JsonProperty;
30  import com.google.common.base.MoreObjects;
31  
32  /**
33   * Encryption configuration.
34   */
35  public final class EncryptionConfig {
36  
37      private static final String ENCRYPTION_MARKER_FILE_NAME = ".encryption-enabled";
38  
39      private final boolean enabled;
40      private final boolean encryptSessionCookie;
41      @Nullable
42      private final String kekId;
43  
44      /**
45       * Creates an instance.
46       */
47      @JsonCreator
48      public EncryptionConfig(@JsonProperty("enabled") @Nullable Boolean enabled,
49                              @JsonProperty("encryptSessionCookie")
50                              @Nullable Boolean encryptSessionCookie,
51                              @JsonProperty("kekId") @Nullable String kekId) {
52          this.enabled = firstNonNull(enabled, false);
53          this.encryptSessionCookie = this.enabled && firstNonNull(encryptSessionCookie, false);
54          if (this.enabled) {
55              requireNonNull(kekId, "kekId");
56          }
57          this.kekId = kekId;
58      }
59  
60      /**
61       * Validates the encryption state with the data directory.
62       * This method checks if encryption was previously enabled and prevents disabling it.
63       *
64       * @param dataDir the data directory where the marker file is stored
65       * @throws IllegalStateException if encryption was previously enabled but is now disabled
66       */
67      void validateEncryptionState(File dataDir) {
68          requireNonNull(dataDir, "dataDir");
69          final Path markerFilePath = dataDir.toPath().resolve(ENCRYPTION_MARKER_FILE_NAME);
70          final boolean wasEncryptionEnabled = Files.exists(markerFilePath);
71  
72          if (wasEncryptionEnabled && !enabled) {
73              throw new IllegalStateException(
74                      "Cannot disable encryption after it has been enabled. " +
75                      "Encryption was previously enabled. To proceed, you must keep encryption enabled.");
76          }
77  
78          if (enabled && !wasEncryptionEnabled) {
79              // Create the marker file to indicate encryption has been enabled
80              try {
81                  Files.createDirectories(dataDir.toPath());
82                  final String markerContent =
83                          "# DO NOT DELETE THIS FILE\n" +
84                          "#\n" +
85                          "# This file indicates that encryption has been enabled for this " +
86                          "Central Dogma server.\n" +
87                          "# Once encryption is enabled, it cannot be disabled.\n" +
88                          "#\n" +
89                          "# Deleting this file may cause the server to malfunction or fail to start.\n";
90                  Files.write(markerFilePath, markerContent.getBytes());
91              } catch (IOException e) {
92                  throw new IllegalStateException(
93                          "Failed to create encryption marker file: " + markerFilePath, e);
94              }
95          }
96      }
97  
98      /**
99       * Returns whether encryption is enabled.
100      */
101     @JsonProperty
102     public boolean enabled() {
103         return enabled;
104     }
105 
106     /**
107      * Returns whether to encrypt session cookies.
108      */
109     @JsonProperty
110     public boolean encryptSessionCookie() {
111         return encryptSessionCookie;
112     }
113 
114     /**
115      * Returns the Key Encryption Key (KEK) ID.
116      */
117     @JsonProperty
118     @Nullable
119     public String kekId() {
120         return kekId;
121     }
122 
123     @Override
124     public String toString() {
125         return MoreObjects.toStringHelper(this)
126                           .add("enabled", enabled)
127                           .add("encryptSessionCookie", encryptSessionCookie)
128                           .add("kekId", kekId)
129                           .toString();
130     }
131 }

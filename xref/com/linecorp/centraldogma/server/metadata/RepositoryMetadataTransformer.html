16  package com.linecorp.centraldogma.server.metadata;
17  
18  import java.util.Map.Entry;
19  import java.util.function.BiFunction;
20  
21  import com.google.common.collect.ImmutableMap;
22  
23  import com.linecorp.centraldogma.common.Revision;
24  
25  final class RepositoryMetadataTransformer extends ProjectMetadataTransformer {
26  
27      RepositoryMetadataTransformer(String repoName,
28                                    BiFunction<Revision, RepositoryMetadata, RepositoryMetadata> transformer) {
29          super((headRevision, projectMetadata) -> {
30              final RepositoryMetadata repositoryMetadata = projectMetadata.repo(repoName);
31              assert repositoryMetadata.name().equals(repoName);
32              final RepositoryMetadata newRepositoryMetadata =
33                      transformer.apply(headRevision, repositoryMetadata);
34              return newProjectMetadata(projectMetadata, newRepositoryMetadata);
35          });
36      }
37  
38      private static ProjectMetadata newProjectMetadata(ProjectMetadata projectMetadata,
39                                                        RepositoryMetadata repositoryMetadata) {
40          final ImmutableMap.Builder<String, RepositoryMetadata> builder =
41                  ImmutableMap.builderWithExpectedSize(projectMetadata.repos().size());
42          for (Entry<String, RepositoryMetadata> entry : projectMetadata.repos().entrySet()) {
43              if (entry.getKey().equals(repositoryMetadata.name())) {
44                  builder.put(entry.getKey(), repositoryMetadata);
45              } else {
46                  builder.put(entry);
47              }
48          }
49          final ImmutableMap<String, RepositoryMetadata> newRepos = builder.build();
50          return new ProjectMetadata(projectMetadata.name(),
51                                     newRepos,
52                                     projectMetadata.members(),
53                                     projectMetadata.tokens(),
54                                     projectMetadata.creation(),
55                                     projectMetadata.removal());
56      }
57  }

105     private CompletableFuture<ProjectMetadata> getOrFetchMetadata(String projectName) {
106         final ProjectMetadata metadata = getMetadata(projectName);
107         final Set<String> reposWithMetadata = metadata.repos().keySet();
108         final Set<String> repos = projectManager.get(projectName).repos().list().keySet();
109 
110         // Make sure all repositories have metadata. If not, create missing metadata.
111         // A repository can have missing metadata when a dev forgot to call `addRepo()`
112         // after creating a new repository.
113         final ImmutableList.Builder<CompletableFuture<Revision>> builder = ImmutableList.builder();
114         for (String repo : repos) {
115             if (reposWithMetadata.contains(repo) || Project.isInternalRepo(repo)) {
116                 continue;
117             }
118 
119             final String projectAndRepositoryName = projectName + '/' + repo;
120             final CompletableFuture<Revision> future = new CompletableFuture<>();
121             final CompletableFuture<Revision> futureInMap =
122                     reposInAddingMetadata.computeIfAbsent(projectAndRepositoryName, key -> future);
123             if (futureInMap != future) { // The metadata is already in adding.
124                 builder.add(futureInMap);
125                 continue;
126             }
127 
128             logger.warn("Adding missing repository metadata: {}/{}", projectName, repo);
129             final Author author = projectManager.get(projectName).repos().get(repo).author();
130             final CompletableFuture<Revision> addRepoFuture = addRepo(author, projectName, repo);
131             addRepoFuture.handle((revision, cause) -> {
132                 if (cause != null) {
133                     future.completeExceptionally(cause);
134                 } else {
135                     future.complete(revision);
136                 }
137                 reposInAddingMetadata.remove(projectAndRepositoryName);
138                 return null;
139             });
140             builder.add(future);
141         }
142 
143         final ImmutableList<CompletableFuture<Revision>> futures = builder.build();
144         if (futures.isEmpty()) {
145             // All repositories have metadata.
146             return CompletableFuture.completedFuture(metadata);
147         }
148 
149         // Some repository did not have metadata and thus will add the missing ones.
150         return CompletableFutures.successfulAsList(futures, cause -> {
151             final Throwable peeled = Exceptions.peel(cause);
152             // The metadata of the repository is added by another worker, so we can ignore the exception.
153             if (peeled instanceof RepositoryExistsException) {
154                 return null;
155             }
156             return Exceptions.throwUnsafely(cause);
157         }).thenCompose(unused -> {
158             logger.info("Fetching {}/{}{} again",
159                         projectName, Project.REPO_DOGMA, METADATA_JSON);
160             return fetchMetadata(projectName);
161         });
162     }

164     private ProjectMetadata getMetadata(String projectName) {
165         final Project project = projectManager.get(projectName);
166         final ProjectMetadata metadata = project.metadata();
167         if (metadata == null) {
168             throw new EntryNotFoundException("project metadata not found: " + projectName);
169         }
170         return metadata;
171     }

184

197                 });
198         return metadataRepo.push(projectName, Project.REPO_DOGMA, author,
199                                  "Remove the project: " + projectName, transformer);

243         requireNonNull(author, "author");
244         requireNonNull(projectName, "projectName");
245         requireNonNull(member, "member");
246         requireNonNull(projectRole, "projectRole");
247 
248         final Member newMember = new Member(member, projectRole, UserAndTimestamp.of(author));

250                 "Add a member '" + newMember.id() + "' to the project '" + projectName + '\'';
251 
252         final ProjectMetadataTransformer transformer =
253                 new ProjectMetadataTransformer((headRevision, projectMetadata) -> {
254                     if (projectMetadata.members().containsKey(newMember.id())) {
255                         throw new ChangeConflictException("Member already exists: " + newMember.id());
256                     }
257                     final ImmutableMap<String, Member> newMembers =
258                             ImmutableMap.<String, Member>builderWithExpectedSize(
259                                                 projectMetadata.members().size() + 1)
260                                         .putAll(projectMetadata.members())
261                                         .put(newMember.id(), newMember)
262                                         .build();
263                     return new ProjectMetadata(projectMetadata.name(),
264                                                projectMetadata.repos(),
265                                                newMembers,
266                                                null,
267                                                projectMetadata.appIds(),
268                                                projectMetadata.creation(),
269                                                projectMetadata.removal());
270                 });
271         return metadataRepo.push(projectName, Project.REPO_DOGMA, author, commitSummary, transformer);
272     }

293                             removeMemberFromRepositories(projectMetadata, memberId);
294                     return new ProjectMetadata(projectMetadata.name(),
295                                                newRepos,
296                                                newMembers,

298                                                projectMetadata.appIds(),

302         return metadataRepo.push(projectName, Project.REPO_DOGMA, author, commitSummary, transformer);

305     private static ImmutableMap<String, RepositoryMetadata> removeMemberFromRepositories(
306             ProjectMetadata projectMetadata, String memberId) {
307         final ImmutableMap.Builder<String, RepositoryMetadata> reposBuilder =
308                 ImmutableMap.builderWithExpectedSize(projectMetadata.repos().size());
309         for (Entry<String, RepositoryMetadata> entry : projectMetadata.repos().entrySet()) {
310             final RepositoryMetadata repositoryMetadata = entry.getValue();
311             final Roles roles = repositoryMetadata.roles();
312             final Map<String, RepositoryRole> users = roles.users();
313             if (users.get(memberId) != null) {
314                 final ImmutableMap<String, RepositoryRole> newUsers = removeFromMap(users, memberId);
315                 final Roles newRoles = new Roles(roles.projectRoles(), newUsers, null, roles.appIds());
316                 reposBuilder.put(entry.getKey(),
317                                  new RepositoryMetadata(repositoryMetadata.name(),
318                                                         newRoles,
319                                                         repositoryMetadata.creation(),
320                                                         repositoryMetadata.removal(),
321                                                         repositoryMetadata.status()));
322             } else {
323                 reposBuilder.put(entry);
324             }
325         }
326         return reposBuilder.build();
327     }

331      */
332     public CompletableFuture<Revision> updateMemberRole(Author author, String projectName,
333                                                         User member, ProjectRole projectRole) {
334         requireNonNull(author, "author");
335         requireNonNull(projectName, "projectName");
336         requireNonNull(member, "member");
337         requireNonNull(projectRole, "projectRole");
338 
339         final String commitSummary = "Updates the role of the member '" + member.id() +
340                                      "' as '" + projectRole + "' for the project '" + projectName + '\'';
341 
342         final ProjectMetadataTransformer transformer =
343                 new ProjectMetadataTransformer((headRevision, projectMetadata) -> {
344                     final Member existingMember = projectMetadata.members().get(member.id());
345                     if (existingMember == null) {
346                         throw new EntryNotFoundException("Member not found: " + member.id());
347                     }
348                     if (existingMember.role() == projectRole) {
349                         throw new RedundantChangeException("Member already has role: " + projectRole);
350                     }
351                     final Member updatedMember = new Member(existingMember.login(), projectRole,
352                                                             existingMember.creation());
353                     final ImmutableMap<String, Member> newMembers =
354                             ImmutableMap.<String, Member>builderWithExpectedSize(
355                                                 projectMetadata.members().size())
356                                         .putAll(projectMetadata.members())
357                                         .put(member.id(), updatedMember)
358                                         .buildKeepingLast();
359                     return new ProjectMetadata(projectMetadata.name(),
360                                                projectMetadata.repos(),
361                                                newMembers,
362                                                null,
363                                                projectMetadata.appIds(),
364                                                projectMetadata.creation(),
365                                                projectMetadata.removal());
366                 });
367         return metadataRepo.push(projectName, Project.REPO_DOGMA, author, commitSummary, transformer);
368     }

407         // TODO(minwoox): Prohibit adding internal repositories after migration is done.
408         requireNonNull(author, "author");
409         requireNonNull(projectName, "projectName");

414                 "Add a repo '" + repositoryMetadata.id() + "' to the project '" + projectName + '\'';
415 
416         final ProjectMetadataTransformer transformer =
417                 new ProjectMetadataTransformer((headRevision, projectMetadata) -> {
418                     if (projectMetadata.repos().containsKey(repoName)) {
419                         throw RepositoryExistsException.of(projectName, repoName);
420                     }
421                     final ImmutableMap<String, RepositoryMetadata> newRepos =
422                             ImmutableMap.<String, RepositoryMetadata>builderWithExpectedSize(
423                                                 projectMetadata.repos().size() + 1)
424                                         .putAll(projectMetadata.repos())
425                                         .put(repoName, repositoryMetadata)
426                                         .build();
427                     return new ProjectMetadata(projectMetadata.name(),
428                                                newRepos,
429                                                projectMetadata.members(),
430                                                null,
431                                                projectMetadata.appIds(),
432                                                projectMetadata.creation(),
433                                                projectMetadata.removal());
434                 });
435         return metadataRepo.push(projectName, Project.REPO_DOGMA, author, commitSummary, transformer);

666                                                                    repositoryMetadata.removal(),

1016     @Nullable
1017     private static RepositoryRole repositoryRole(Roles roles, @Nullable RepositoryRole repositoryRole,
1018                                                  ProjectRole projectRole) {
1019         if (projectRole == ProjectRole.OWNER) {
1020             return RepositoryRole.ADMIN;
1021         }
1022 
1023         final RepositoryRole memberOrGuestRole;
1024         if (projectRole == ProjectRole.MEMBER) {
1025             memberOrGuestRole = roles.projectRoles().member();
1026         } else {
1027             assert projectRole == ProjectRole.GUEST;
1028             memberOrGuestRole = roles.projectRoles().guest();
1029         }
1030 
1031         if (repositoryRole == RepositoryRole.ADMIN || memberOrGuestRole == RepositoryRole.ADMIN) {
1032             return RepositoryRole.ADMIN;
1033         }
1034 
1035         if (repositoryRole == RepositoryRole.WRITE || memberOrGuestRole == RepositoryRole.WRITE) {
1036             return RepositoryRole.WRITE;
1037         }
1038 
1039         if (repositoryRole == RepositoryRole.READ || memberOrGuestRole == RepositoryRole.READ) {
1040             return RepositoryRole.READ;
1041         }
1042 
1043         return null;
1044     }

1187         // Remove the token from projects that only have the token.
1188         for (Project project : projects) {
1189             // Fetch the metadata to get the latest information.
1190             final ProjectMetadata projectMetadata = fetchMetadata(project.name()).join();
1191             final boolean containsTargetTokenInTheProject =
1192                     projectMetadata.appIds().values()
1193                                    .stream()
1194                                    .anyMatch(token -> token.appId().equals(appId));
1195 
1196             if (containsTargetTokenInTheProject) {
1197                 removeToken(project.name(), author, appId, true).join();
1198             }
1199         }
1200 
1201         final String commitSummary = "Remove the token: " + appId;
1202 
1203         final TokensTransformer transformer = new TokensTransformer((headRevision, tokens) -> {
1204             final Token token = tokens.get(appId);
1205             final Map<String, AppIdentity> newAppIds = removeFromMap(tokens.appIds(), appId);
1206             final String secret = token.secret();
1207             assert secret != null;
1208             final Map<String, String> newSecrets = removeFromMap(tokens.secrets(), secret);
1209             return new Tokens(newAppIds, newSecrets, tokens.certificateIds());
1210         });
1211         return tokenRepo.push(INTERNAL_PROJECT_DOGMA, Project.REPO_DOGMA, author, commitSummary, transformer)
1212                         .join();
1213     }

1223

1333     private static <T> ImmutableMap<String, T> addToMap(Map<String, T> map, String key, T value) {
1334         return ImmutableMap.<String, T>builderWithExpectedSize(map.size() + 1)
1335                            .putAll(map)
1336                            .put(key, value)
1337                            .build();
1338     }
1339 
1340     private static <T> Map<String, T> updateMap(Map<String, T> map, String key, T value) {
1341         final ImmutableMap.Builder<String, T> builder = ImmutableMap.builderWithExpectedSize(map.size());
1342         for (Entry<String, T> entry : map.entrySet()) {
1343             if (entry.getKey().equals(key)) {
1344                 builder.put(key, value);
1345             } else {
1346                 builder.put(entry);
1347             }
1348         }
1349         return builder.build();
1350     }
1351 
1352     private static <T> ImmutableMap<String, T> removeFromMap(Map<String, T> map, String id) {
1353         return map.entrySet().stream()
1354                   .filter(e -> !e.getKey().equals(id))
1355                   .collect(toImmutableMap(Entry::getKey, Entry::getValue));
1356     }

1358     /**
1359      * Updates the {@link ServerStatus} of the specified {@code repoName}.
1360      */
1361     public CompletableFuture<Revision> updateRepositoryStatus(
1362             Author author, String projectName, String repoName, RepositoryStatus repositoryStatus) {
1363         requireNonNull(author, "author");
1364         requireNonNull(projectName, "projectName");
1365         requireNonNull(repoName, "repoName");
1366         requireNonNull(repositoryStatus, "repositoryStatus");
1367         final String newRepoName;
1368         if (Project.REPO_META.equals(repoName)) {
1369             newRepoName = Project.REPO_DOGMA; // Use dogma repository because meta repository will be removed.
1370         } else {
1371             newRepoName = repoName;
1372         }
1373 
1374         final ProjectMetadataTransformer transformer;
1375         if (Project.REPO_DOGMA.equals(newRepoName)) {
1376             // Have to use ProjectMetadataTransformer because the repository metadata of dogma repository
1377             // might not exist.
1378             transformer = new ProjectMetadataTransformer((headRevision, projectMetadata) -> {
1379                 final RepositoryMetadata repositoryMetadata = projectMetadata.repos().get(Project.REPO_DOGMA);
1380                 if (repositoryMetadata != null) {
1381                     throwIfRedundant(repositoryStatus, headRevision, repositoryMetadata, Project.REPO_DOGMA);
1382                 }
1383                 final RepositoryMetadata newRepositoryMetadata = RepositoryMetadata.ofDogma(repositoryStatus);
1384                 final Builder<String, RepositoryMetadata> builder = ImmutableMap.builder();
1385                 builder.put(Project.REPO_DOGMA, newRepositoryMetadata);
1386                 projectMetadata.repos().forEach((name, metadata) -> {
1387                     if (!Project.REPO_DOGMA.equals(name)) {
1388                         builder.put(name, metadata);
1389                     }
1390                 });
1391                 return new ProjectMetadata(projectMetadata.name(),
1392                                            builder.build(),
1393                                            projectMetadata.members(),
1394                                            null,
1395                                            projectMetadata.appIds(),
1396                                            projectMetadata.creation(),
1397                                            projectMetadata.removal());
1398             });
1399         } else {
1400             transformer = new RepositoryMetadataTransformer(
1401                     newRepoName, (headRevision, repositoryMetadata) -> {
1402                 throwIfRedundant(repositoryStatus, headRevision, repositoryMetadata, newRepoName);
1403 
1404                 return new RepositoryMetadata(repositoryMetadata.name(),
1405                                               repositoryMetadata.roles(),
1406                                               repositoryMetadata.creation(),
1407                                               repositoryMetadata.removal(),
1408                                               repositoryStatus);
1409             });
1410         }
1411 
1412         final String commitSummary = "Update the status of '" + projectName + '/' + newRepoName +
1413                                      "'. status: " + repositoryStatus;
1414         return metadataRepo.push(projectName, Project.REPO_DOGMA, author, commitSummary, transformer, true);
1415     }

1417     private static void throwIfRedundant(RepositoryStatus repositoryStatus, Revision headRevision,
1418                                          RepositoryMetadata repositoryMetadata, String newRepoName) {
1419         if (repositoryMetadata.status() == repositoryStatus) {
1420             throw new RedundantChangeException(
1421                     headRevision,
1422                     "the status of '" + newRepoName + "' isn't changed. status: " + repositoryStatus);
1423         }
1424     }

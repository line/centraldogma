117     private CompletableFuture<HolderWithRevision<ProjectMetadata>> fetchMetadata(String projectName) {
118         return fetchMetadata0(projectName).thenCompose(holder -> {
119             final Set<String> repos = projectManager.get(projectName).repos().list().keySet();
120             final Set<String> reposWithMetadata = holder.object().repos().keySet();
121 
122             // Make sure all repositories have metadata. If not, create missing metadata.
123             // A repository can have missing metadata when a dev forgot to call `addRepo()`
124             // after creating a new repository.
125             final ImmutableList.Builder<CompletableFuture<Revision>> builder = ImmutableList.builder();
126             for (String repo : repos) {
127                 if (reposWithMetadata.contains(repo) ||
128                     repo.equals(Project.REPO_DOGMA)) {
129                     continue;
130                 }
131 
132                 final String projectAndRepositoryName = projectName + '/' + repo;
133                 final CompletableFuture<Revision> future = new CompletableFuture<>();
134                 final CompletableFuture<Revision> futureInMap =
135                         reposInAddingMetadata.computeIfAbsent(projectAndRepositoryName, key -> future);
136                 if (futureInMap != future) { // The metadata is already in adding.
137                     builder.add(futureInMap);
138                     continue;
139                 }
140 
141                 logger.warn("Adding missing repository metadata: {}/{}", projectName, repo);
142                 final Author author = projectManager.get(projectName).repos().get(repo).author();
143                 final CompletableFuture<Revision> addRepoFuture = addRepo(author, projectName, repo);
144                 addRepoFuture.handle((revision, cause) -> {
145                     if (cause != null) {
146                         future.completeExceptionally(cause);
147                     } else {
148                         future.complete(revision);
149                     }
150                     reposInAddingMetadata.remove(projectAndRepositoryName);
151                     return null;
152                 });
153                 builder.add(future);
154             }
155 
156             final ImmutableList<CompletableFuture<Revision>> futures = builder.build();
157             if (futures.isEmpty()) {
158                 // All repositories have metadata.
159                 return CompletableFuture.completedFuture(holder);
160             }
161 
162             // Some repository did not have metadata and thus will add the missing ones.
163             return CompletableFutures.successfulAsList(futures, cause -> {
164                 final Throwable peeled = Exceptions.peel(cause);
165                 // The metadata of the repository is added by another worker, so we can ignore the exception.
166                 if (peeled instanceof RepositoryExistsException) {
167                     return null;
168                 }
169                 return Exceptions.throwUnsafely(cause);
170             }).thenCompose(unused -> {
171                 logger.info("Fetching {}/{}{} again",
172                             projectName, Project.REPO_DOGMA, METADATA_JSON);
173                 return fetchMetadata0(projectName);
174             });
175         });
176     }

246      * {@code projectName}. It also removes the {@link RepositoryRole} of the specified {@code user} from every

251         requireNonNull(projectName, "projectName");
252         requireNonNull(user, "user");
253

255         final String commitSummary =
256                 "Remove the member '" + memberId + "' from the project '" + projectName + '\'';
257 
258         final ProjectMetadataTransformer transformer =
259                 new ProjectMetadataTransformer((headRevision, projectMetadata) -> {
260                     projectMetadata.member(memberId); // Raises an exception if the member does not exist.
261                     final Map<String, Member> newMembers = removeFromMap(projectMetadata.members(), memberId);
262                     final ImmutableMap<String, RepositoryMetadata> newRepos =
263                             removeMemberFromRepositories(projectMetadata, memberId);
264                     return new ProjectMetadata(projectMetadata.name(),
265                                                newRepos,
266                                                newMembers,
267                                                projectMetadata.tokens(),
268                                                projectMetadata.creation(),
269                                                projectMetadata.removal());
270                 });
271         return metadataRepo.push(projectName, Project.REPO_DOGMA, author, commitSummary, transformer);
272     }

274     private static ImmutableMap<String, RepositoryMetadata> removeMemberFromRepositories(
275             ProjectMetadata projectMetadata, String memberId) {
276         final ImmutableMap.Builder<String, RepositoryMetadata> reposBuilder =
277                 ImmutableMap.builderWithExpectedSize(projectMetadata.repos().size());
278         for (Entry<String, RepositoryMetadata> entry : projectMetadata.repos().entrySet()) {
279             final RepositoryMetadata repositoryMetadata = entry.getValue();
280             final Roles roles = repositoryMetadata.roles();
281             final Map<String, RepositoryRole> users = roles.users();
282             if (users.get(memberId) != null) {
283                 final ImmutableMap<String, RepositoryRole> newUsers = removeFromMap(users, memberId);
284                 final Roles newRoles = new Roles(roles.projectRoles(), newUsers, roles.tokens());
285                 reposBuilder.put(entry.getKey(),
286                                  new RepositoryMetadata(repositoryMetadata.name(),
287                                                         newRoles,
288                                                         repositoryMetadata.creation(),
289                                                         repositoryMetadata.removal(),
290                                                         repositoryMetadata.writeQuota()));
291             } else {
292                 reposBuilder.put(entry);
293             }
294         }
295         return reposBuilder.build();
296     }

425     /**
426      * Updates the member and guest {@link RepositoryRole} of the specified {@code repoName} in the specified
427      * {@code projectName}.
428      */
429     public CompletableFuture<Revision> updateRepositoryProjectRoles(Author author,
430                                                                     String projectName, String repoName,
431                                                                     ProjectRoles projectRoles) {
432         requireNonNull(author, "author");
433         requireNonNull(projectName, "projectName");
434         requireNonNull(repoName, "repoName");
435 
436         if (Project.REPO_DOGMA.equals(repoName)) {
437             throw new UnsupportedOperationException(
438                     "Can't update role for internal repository: " + repoName);
439         }
440 
441         if (Project.REPO_META.equals(repoName)) {
442             if (projectRoles.guest() != null) {
443                 throw new UnsupportedOperationException(
444                         "Can't give a role to guest for internal repository: " + repoName);
445             }
446         }
447         final String commitSummary =
448                 "Update the project roles of the '" + repoName + "' in the project '" + projectName + '\'';
449         final RepositoryMetadataTransformer transformer = new RepositoryMetadataTransformer(
450                 repoName, (headRevision, repositoryMetadata) -> {
451             if (repositoryMetadata.roles().projectRoles().equals(projectRoles)) {
452                 throw new RedundantChangeException(
453                         headRevision,
454                         "the project roles of '" + projectName + '/' + repoName + "' isn't changed.");
455             }
456             final Roles newRoles = new Roles(projectRoles, repositoryMetadata.roles().users(),
457                                              repositoryMetadata.roles().tokens());
458             return new RepositoryMetadata(repositoryMetadata.name(),
459                                           newRoles,
460                                           repositoryMetadata.creation(),
461                                           repositoryMetadata.removal(),
462                                           repositoryMetadata.writeQuota());
463         });
464         return metadataRepo.push(projectName, Project.REPO_DOGMA, author, commitSummary, transformer);
465     }

522     private CompletableFuture<Revision> removeToken(String projectName, Author author, String appId,
523                                                     boolean quiet) {
524         final String commitSummary = "Remove the token '" + appId + "' from the project '" + projectName + '\'';
525         final ProjectMetadataTransformer transformer =
526                 new ProjectMetadataTransformer((headRevision, projectMetadata) -> {
527             final Map<String, TokenRegistration> tokens = projectMetadata.tokens();
528             final Map<String, TokenRegistration> newTokens;
529             if (tokens.get(appId) == null) {
530                 if (!quiet) {
531                     throw new TokenNotFoundException(
532                             "failed to find the token " + appId + " in project " + projectName);
533                 }
534                 newTokens = tokens;
535             } else {
536                 newTokens = tokens.entrySet()
537                                   .stream()
538                                   .filter(entry -> !entry.getKey().equals(appId))
539                                   .collect(toImmutableMap(Entry::getKey, Entry::getValue));
540             }
541 
542             final ImmutableMap<String, RepositoryMetadata> newRepos =
543                     removeTokenFromRepositories(appId, projectMetadata);
544             return new ProjectMetadata(projectMetadata.name(),
545                                        newRepos,
546                                        projectMetadata.members(),
547                                        newTokens,
548                                        projectMetadata.creation(),
549                                        projectMetadata.removal());
550         });
551         return metadataRepo.push(projectName, Project.REPO_DOGMA, author, commitSummary, transformer);
552     }

554     private static ImmutableMap<String, RepositoryMetadata> removeTokenFromRepositories(
555             String appId, ProjectMetadata projectMetadata) {
556         final ImmutableMap.Builder<String, RepositoryMetadata> builder =
557                 ImmutableMap.builderWithExpectedSize(projectMetadata.repos().size());
558         for (Entry<String, RepositoryMetadata> entry : projectMetadata.repos().entrySet()) {
559             final RepositoryMetadata repositoryMetadata = entry.getValue();
560             final Roles roles = repositoryMetadata.roles();
561             if (roles.tokens().get(appId) != null) {
562                 final Map<String, RepositoryRole> newTokens = removeFromMap(roles.tokens(), appId);
563                 final Roles newRoles = new Roles(roles.projectRoles(), roles.users(), newTokens);
564                 builder.put(entry.getKey(), new RepositoryMetadata(repositoryMetadata.name(),
565                                                                    newRoles,
566                                                                    repositoryMetadata.creation(),
567                                                                    repositoryMetadata.removal(),
568                                                                    repositoryMetadata.writeQuota()));
569             } else {
570                 builder.put(entry);
571             }
572         }
573         return builder.build();
574     }

903     @Nullable
904     private static RepositoryRole repositoryRole(Roles roles, @Nullable RepositoryRole repositoryRole,
905                                                  ProjectRole projectRole) {
906         if (projectRole == ProjectRole.OWNER) {
907             return RepositoryRole.ADMIN;
908         }
909 
910         final RepositoryRole memberOrGuestRole;
911         if (projectRole == ProjectRole.MEMBER) {
912             memberOrGuestRole = roles.projectRoles().member();
913         } else {
914             assert projectRole == ProjectRole.GUEST;
915             memberOrGuestRole = roles.projectRoles().guest();
916         }
917 
918         if (repositoryRole == RepositoryRole.ADMIN || memberOrGuestRole == RepositoryRole.ADMIN) {
919             return RepositoryRole.ADMIN;
920         }
921 
922         if (repositoryRole == RepositoryRole.WRITE || memberOrGuestRole == RepositoryRole.WRITE) {
923             return RepositoryRole.WRITE;
924         }
925 
926         if (repositoryRole == RepositoryRole.READ || memberOrGuestRole == RepositoryRole.READ) {
927             return RepositoryRole.READ;
928         }
929 
930         return null;
931     }

1137             final Token token = tokens.get(appId); // Raise an exception if not found.
1138             if (toBeSystemAdmin == token.isSystemAdmin()) {
1139                 throw new RedundantChangeException(
1140                         headRevision,
1141                         "The token is already " + (toBeSystemAdmin ? "admin" : "user"));
1142             }
1143 
1144             final Token newToken = token.withSystemAdmin(toBeSystemAdmin);
1145             return new Tokens(updateMap(tokens.appIds(), appId, newToken), tokens.secrets());
1146         });
1147         return tokenRepo.push(INTERNAL_PROJECT_DOGMA, Project.REPO_DOGMA, author, commitSummary, transformer);
1148     }

1201     private static <T> Map<String, T> updateMap(Map<String, T> map, String key, T value) {
1202         final ImmutableMap.Builder<String, T> builder = ImmutableMap.builderWithExpectedSize(map.size());
1203         for (Entry<String, T> entry : map.entrySet()) {
1204             if (entry.getKey().equals(key)) {
1205                 builder.put(key, value);
1206             } else {
1207                 builder.put(entry);
1208             }
1209         }
1210         return builder.build();
1211     }

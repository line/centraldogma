16  package com.linecorp.centraldogma.server.metadata;
17  
18  import java.io.IOException;
19  import java.util.Collection;
20  import java.util.Map;
21  import java.util.Map.Entry;
22  
23  import javax.annotation.Nullable;
24  
25  import com.fasterxml.jackson.core.JsonParser;
26  import com.fasterxml.jackson.core.type.TypeReference;
27  import com.fasterxml.jackson.databind.DeserializationContext;
28  import com.fasterxml.jackson.databind.JsonNode;
29  import com.fasterxml.jackson.databind.deser.std.StdDeserializer;
30  import com.google.common.collect.ImmutableMap;
31  import com.google.common.collect.ImmutableMap.Builder;
32  
33  import com.linecorp.centraldogma.common.RepositoryRole;
34  import com.linecorp.centraldogma.internal.Jackson;
35  import com.linecorp.centraldogma.server.QuotaConfig;
36  
37  final class RepositoryMetadataDeserializer extends StdDeserializer<RepositoryMetadata> {
38  
39      private static final long serialVersionUID = 1173216371065909688L;
40  
41      private static final TypeReference<Map<String, Collection<Permission>>> PER_PERMISSIONS_TYPE =
42              new TypeReference<Map<String, Collection<Permission>>>() {};
43  
44      RepositoryMetadataDeserializer() {
45          super(RepositoryMetadata.class);
46      }
47  
48      @Override
49      public RepositoryMetadata deserialize(JsonParser p, DeserializationContext ctxt)
50              throws IOException {
51          final JsonNode jsonNode = p.readValueAsTree();
52          final String name = jsonNode.get("name").textValue();
53          final JsonNode perRolePermissionsNode = jsonNode.get("perRolePermissions");
54  
55          final Roles roles;
56          if (perRolePermissionsNode != null)  {
57              assert jsonNode.get("roles") == null;
58              // legacy format
59              final PerRolePermissions perRolePermissions =
60                      Jackson.treeToValue(perRolePermissionsNode, PerRolePermissions.class);
61              final Map<String, Collection<Permission>> perUserPermissions =
62                      Jackson.readValue(jsonNode.get("perUserPermissions").traverse(), PER_PERMISSIONS_TYPE);
63              final Map<String, Collection<Permission>> perTokenPermissions =
64                      Jackson.readValue(jsonNode.get("perTokenPermissions").traverse(), PER_PERMISSIONS_TYPE);
65              roles = new Roles(ProjectRoles.of(repositoryRole(perRolePermissions.member()),
66                                                repositoryRole(perRolePermissions.guest())),
67                                convert(perUserPermissions), convert(perTokenPermissions));
68          } else {
69              // new format
70              roles = Jackson.treeToValue(jsonNode.get("roles"), Roles.class);
71          }
72  
73          final UserAndTimestamp creation = Jackson.treeToValue(jsonNode.get("creation"), UserAndTimestamp.class);
74          final JsonNode removalNode = jsonNode.get("removal");
75          final UserAndTimestamp removal =
76                  removalNode == null ? null : Jackson.treeToValue(removalNode, UserAndTimestamp.class);
77  
78          final JsonNode writeQuotaNode = jsonNode.get("writeQuota");
79          final QuotaConfig writeQuota =
80                  writeQuotaNode == null ? null : Jackson.treeToValue(writeQuotaNode, QuotaConfig.class);
81  
82          return new RepositoryMetadata(name, roles, creation, removal, writeQuota);
83      }
84  
85      @Nullable
86      private static RepositoryRole repositoryRole(Collection<Permission> permissions) {
87          if (permissions.isEmpty()) {
88              return null;
89          }
90          if (permissions.contains(Permission.WRITE)) {
91              return RepositoryRole.WRITE;
92          }
93          return RepositoryRole.READ;
94      }
95  
96      private static Map<String, RepositoryRole> convert(Map<String, Collection<Permission>> permissions) {
97          final Builder<String, RepositoryRole> builder = ImmutableMap.builder();
98          for (Entry<String, Collection<Permission>> entry : permissions.entrySet()) {
99              final RepositoryRole repositoryRole = repositoryRole(entry.getValue());
100             if (repositoryRole != null) {
101                 builder.put(entry.getKey(), repositoryRole);
102             }
103         }
104 
105         return builder.build();
106     }
107 }

16  package com.linecorp.centraldogma.server.metadata;
17  
18  import static com.google.common.collect.ImmutableMap.toImmutableMap;
19  import static com.linecorp.centraldogma.server.metadata.PerRolePermissions.READ_WRITE;
20  
21  import java.io.IOException;
22  import java.util.Collection;
23  import java.util.Map;
24  import java.util.Map.Entry;
25  
26  import javax.annotation.Nullable;
27  
28  import com.fasterxml.jackson.core.JsonParser;
29  import com.fasterxml.jackson.core.type.TypeReference;
30  import com.fasterxml.jackson.databind.DeserializationContext;
31  import com.fasterxml.jackson.databind.JsonNode;
32  import com.fasterxml.jackson.databind.deser.std.StdDeserializer;
33  import com.google.common.collect.ImmutableList;
34  
35  import com.linecorp.centraldogma.common.RepositoryRole;
36  import com.linecorp.centraldogma.internal.Jackson;
37  import com.linecorp.centraldogma.server.QuotaConfig;
38  
39  final class RepositoryMetadataDeserializer extends StdDeserializer<RepositoryMetadata> {
40  
41      private static final long serialVersionUID = 1173216371065909688L;
42  
43      private static final TypeReference<Map<String, Collection<Permission>>> PER_PERMISSIONS_TYPE =
44              new TypeReference<Map<String, Collection<Permission>>>() {};
45  
46      RepositoryMetadataDeserializer() {
47          super(RepositoryMetadata.class);
48      }
49  
50      @Override
51      public RepositoryMetadata deserialize(JsonParser p, DeserializationContext ctxt)
52              throws IOException {
53          final JsonNode jsonNode = p.readValueAsTree();
54          final String name = jsonNode.get("name").textValue();
55          final JsonNode perRolePermissionsNode = jsonNode.get("perRolePermissions");
56  
57          final PerRolePermissions perRolePermissions;
58          final Map<String, Collection<Permission>> perUserPermissions;
59          final Map<String, Collection<Permission>> perTokenPermissions;
60          if (perRolePermissionsNode != null)  {
61              assert jsonNode.get("roles") == null;
62              // legacy format
63              perRolePermissions = Jackson.treeToValue(perRolePermissionsNode, PerRolePermissions.class);
64              perUserPermissions = Jackson.readValue(jsonNode.get("perUserPermissions").traverse(),
65                                                     PER_PERMISSIONS_TYPE);
66              perTokenPermissions = Jackson.readValue(jsonNode.get("perTokenPermissions").traverse(),
67                                                      PER_PERMISSIONS_TYPE);
68          } else {
69              // new format
70              final Roles roles = Jackson.treeToValue(jsonNode.get("roles"), Roles.class);
71              perRolePermissions = new PerRolePermissions(READ_WRITE,
72                                                          getPermissions(roles.projectRoles().member()),
73                                                          getPermissions(roles.projectRoles().guest()), null);
74              perUserPermissions = convert(roles.users());
75              perTokenPermissions = convert(roles.tokens());
76          }
77  
78          final UserAndTimestamp creation = Jackson.treeToValue(jsonNode.get("creation"), UserAndTimestamp.class);
79          final JsonNode removalNode = jsonNode.get("removal");
80          final UserAndTimestamp removal =
81                  removalNode == null ? null : Jackson.treeToValue(removalNode, UserAndTimestamp.class);
82  
83          final JsonNode writeQuotaNode = jsonNode.get("writeQuota");
84          final QuotaConfig writeQuota =
85                  writeQuotaNode == null ? null : Jackson.treeToValue(writeQuotaNode, QuotaConfig.class);
86  
87          return new RepositoryMetadata(name, perRolePermissions, perUserPermissions, perTokenPermissions,
88                                        creation, removal, writeQuota);
89      }
90  
91      private static Map<String, Collection<Permission>> convert(
92              Map<String, RepositoryRole> roles) {
93          return roles.entrySet().stream()
94                      .collect(toImmutableMap(Entry::getKey, entry -> getPermissions(entry.getValue())));
95      }
96  
97      static Collection<Permission> getPermissions(@Nullable RepositoryRole repositoryRole) {
98          if (repositoryRole == null) {
99              return ImmutableList.of();
100         }
101         if (repositoryRole == RepositoryRole.READ) {
102             return ImmutableList.of(Permission.READ);
103         }
104         // WRITE or ADMIN RepositoryRole.
105         return ImmutableList.of(Permission.READ, Permission.WRITE);
106     }
107 }

16  package com.linecorp.centraldogma.server.command;
17  
18  import static java.util.Objects.requireNonNull;
19  
20  import java.util.Objects;
21  
22  import javax.annotation.Nullable;
23  
24  import com.fasterxml.jackson.annotation.JsonCreator;
25  import com.fasterxml.jackson.annotation.JsonProperty;
26  import com.google.common.base.MoreObjects.ToStringHelper;
27  
28  import com.linecorp.centraldogma.common.Author;
29  import com.linecorp.centraldogma.server.storage.encryption.WrappedDekDetails;
30  
31  /**
32   * A {@link Command} that rotates the wrapped data encryption key (WDEK) for a repository.
33   */
34  public final class RotateWdekCommand extends RepositoryCommand<Void> {
35  
36      private final WrappedDekDetails wdekDetails;
37      private final boolean reencrypt;
38  
39      @JsonCreator
40      RotateWdekCommand(@JsonProperty("timestamp") @Nullable Long timestamp,
41                        @JsonProperty("author") @Nullable Author author,
42                        @JsonProperty("projectName") String projectName,
43                        @JsonProperty("repositoryName") String repositoryName,
44                        @JsonProperty("wdekDetails") WrappedDekDetails wdekDetails,
45                        @JsonProperty("reencrypt") boolean reencrypt) {
46          super(CommandType.ROTATE_WDEK, timestamp, author, projectName, repositoryName);
47          this.wdekDetails = requireNonNull(wdekDetails, "wdekDetails");
48          this.reencrypt = reencrypt;
49      }
50  
51      /**
52       * Returns the details of the new wrapped data encryption key (WDEK).
53       */
54      @JsonProperty
55      public WrappedDekDetails wdekDetails() {
56          return wdekDetails;
57      }
58  
59      /**
60       * Returns whether to re-encrypt all existing data with the new WDEK.
61       */
62      @JsonProperty
63      public boolean reencrypt() {
64          return reencrypt;
65      }
66  
67      @Override
68      public boolean equals(Object obj) {
69          if (this == obj) {
70              return true;
71          }
72  
73          if (!(obj instanceof RotateWdekCommand)) {
74              return false;
75          }
76  
77          final RotateWdekCommand that = (RotateWdekCommand) obj;
78          return super.equals(that) &&
79                 reencrypt == that.reencrypt &&
80                 wdekDetails.equals(that.wdekDetails);
81      }
82  
83      @Override
84      public int hashCode() {
85          return Objects.hash(wdekDetails, reencrypt) * 31 + super.hashCode();
86      }
87  
88      @Override
89      ToStringHelper toStringHelper() {
90          return super.toStringHelper()
91                      .add("wdekDetails", wdekDetails)
92                      .add("reencrypt", reencrypt);
93      }
94  }

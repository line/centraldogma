117     private StandaloneCommandExecutor(ProjectManager projectManager,
118                                       Executor repositoryWorker,
119                                       ServerStatusManager serverStatusManager,
120                                       @Nullable SessionManager sessionManager,
121                                       double permitsPerSecond,
122                                       @Nullable Consumer<CommandExecutor> onTakeLeadership,
123                                       @Nullable Consumer<CommandExecutor> onReleaseLeadership,
124                                       @Nullable Consumer<CommandExecutor> onTakeZoneLeadership,
125                                       @Nullable Consumer<CommandExecutor> onReleaseZoneLeadership) {
126         super(onTakeLeadership, onReleaseLeadership, onTakeZoneLeadership, onReleaseZoneLeadership);
127         this.projectManager = requireNonNull(projectManager, "projectManager");
128         this.repositoryWorker = requireNonNull(repositoryWorker, "repositoryWorker");
129         this.serverStatusManager = requireNonNull(serverStatusManager, "serverStatusManager");
130         this.sessionManager = sessionManager;
131         this.permitsPerSecond = permitsPerSecond;
132         writeRateLimiters = new ConcurrentHashMap<>();
133     }

140     @Override
141     protected void doStart(@Nullable Runnable onTakeLeadership,
142                            @Nullable Runnable onReleaseLeadership,
143                            @Nullable Runnable onTakeZoneLeadership,
144                            @Nullable Runnable onReleaseZoneLeadership) {
145         if (onTakeLeadership != null) {
146             onTakeLeadership.run();
147         }
148         if (onTakeZoneLeadership != null) {
149             onTakeZoneLeadership.run();
150         }
151     }

155         if (onReleaseLeadership != null) {

157         }
158         if (onReleaseZoneLeadership != null) {
159             onReleaseZoneLeadership.run();
160         }

163     @Override
164     @SuppressWarnings("unchecked")
165     protected <T> CompletableFuture<T> doExecute(Command<T> command) throws Exception {
166         if (command instanceof CreateProjectCommand) {
167             return (CompletableFuture<T>) createProject((CreateProjectCommand) command);
168         }
169 
170         if (command instanceof RemoveProjectCommand) {
171             return (CompletableFuture<T>) removeProject((RemoveProjectCommand) command);
172         }
173 
174         if (command instanceof UnremoveProjectCommand) {
175             return (CompletableFuture<T>) unremoveProject((UnremoveProjectCommand) command);
176         }
177 
178         if (command instanceof PurgeProjectCommand) {
179             return (CompletableFuture<T>) purgeProject((PurgeProjectCommand) command);
180         }
181 
182         if (command instanceof CreateRepositoryCommand) {
183             return (CompletableFuture<T>) createRepository((CreateRepositoryCommand) command);
184         }
185 
186         if (command instanceof RemoveRepositoryCommand) {
187             return (CompletableFuture<T>) removeRepository((RemoveRepositoryCommand) command);
188         }
189 
190         if (command instanceof UnremoveRepositoryCommand) {
191             return (CompletableFuture<T>) unremoveRepository((UnremoveRepositoryCommand) command);
192         }
193 
194         if (command instanceof PurgeRepositoryCommand) {
195             return (CompletableFuture<T>) purgeRepository((PurgeRepositoryCommand) command);
196         }
197 
198         if (command instanceof NormalizingPushCommand) {
199             return (CompletableFuture<T>) push((NormalizingPushCommand) command, true);
200         }
201 
202         if (command instanceof PushAsIsCommand) {
203             return (CompletableFuture<T>) push((PushAsIsCommand) command, false)
204                     .thenApply(CommitResult::revision);
205         }
206 
207         if (command instanceof TransformCommand) {
208             return (CompletableFuture<T>) push((TransformCommand) command, true);
209         }
210 
211         if (command instanceof CreateSessionCommand) {
212             return (CompletableFuture<T>) createSession((CreateSessionCommand) command);
213         }
214 
215         if (command instanceof RemoveSessionCommand) {
216             return (CompletableFuture<T>) removeSession((RemoveSessionCommand) command);
217         }
218 
219         if (command instanceof UpdateServerStatusCommand) {
220             return (CompletableFuture<T>) updateServerStatus((UpdateServerStatusCommand) command);
221         }
222 
223         if (command instanceof ForcePushCommand) {
224             //noinspection TailRecursion
225             return doExecute(((ForcePushCommand<T>) command).delegate());
226         }
227 
228         throw new UnsupportedOperationException(command.toString());
229     }

294     private CompletableFuture<CommitResult> push(RepositoryCommand<?> c, boolean normalizing) {
295         if (c.projectName().equals(INTERNAL_PROJECT_DOGMA) || c.repositoryName().equals(Project.REPO_DOGMA) ||
296             !writeQuotaEnabled()) {
297             return push0(c, normalizing);
298         }
299 
300         final RateLimiter rateLimiter =
301                 writeRateLimiters.get(rateLimiterKey(c.projectName(), c.repositoryName()));
302         if (rateLimiter != null) {
303             return tryPush(c, normalizing, rateLimiter);
304         }
305 
306         return getRateLimiter(c.projectName(), c.repositoryName()).thenCompose(
307                 limiter -> tryPush(c, normalizing, limiter));
308     }
309 
310     private CompletableFuture<CommitResult> tryPush(
311             RepositoryCommand<?> c, boolean normalizing, @Nullable RateLimiter rateLimiter) {
312         if (rateLimiter == null || rateLimiter == UNLIMITED || rateLimiter.tryAcquire()) {
313             return push0(c, normalizing);
314         } else {
315             return CompletableFutures.exceptionallyCompletedFuture(
316                     new TooManyRequestsException("commits", c.executionPath(), rateLimiter.getRate()));
317         }
318     }
319 
320     private CompletableFuture<CommitResult> push0(RepositoryCommand<?> c, boolean normalizing) {
321         if (c instanceof TransformCommand) {
322             final TransformCommand transformCommand = (TransformCommand) c;
323             return repo(c).commit(transformCommand.baseRevision(), transformCommand.timestamp(),
324                                   transformCommand.author(), transformCommand.summary(),
325                                   transformCommand.detail(), transformCommand.markup(),
326                                   transformCommand.transformer());
327         }
328         assert c instanceof AbstractPushCommand;
329         final AbstractPushCommand<?> pushCommand = (AbstractPushCommand<?>) c;
330         return repo(c).commit(pushCommand.baseRevision(), pushCommand.timestamp(), pushCommand.author(),
331                               pushCommand.summary(), pushCommand.detail(), pushCommand.markup(),
332                               pushCommand.changes(), normalizing);
333     }

335     private CompletableFuture<RateLimiter> getRateLimiter(String projectName, String repoName) {
336         final CompletableFuture<RepositoryMetadata> future = repositoryMetadata(projectName, repoName);
337         if (future == null) {
338             // metadata is not available yet.
339             return UnmodifiableFuture.completedFuture(RateLimiter.create(permitsPerSecond));
340         }
341         return future.thenApply(meta -> {
342             setWriteQuota(projectName, repoName, meta.writeQuota());
343             return writeRateLimiters.get(rateLimiterKey(projectName, repoName));
344         });
345     }
346 
347     @Override
348     public final void setWriteQuota(String projectName, String repoName, @Nullable QuotaConfig writeQuota) {
349         if (!writeQuotaEnabled()) {
350             // This method should be called only when a write quota is enabled.
351             return;
352         }
353         final double permitsForRepo = writeQuota != null ? writeQuota.permitsPerSecond() : 0;
354         final double permitsPerSecond = permitsForRepo != 0 ? permitsForRepo : this.permitsPerSecond;
355 
356         writeRateLimiters.compute(rateLimiterKey(projectName, repoName), (key, rateLimiter) -> {
357             if (permitsPerSecond == 0) {
358                 return UNLIMITED;
359             }
360 
361             if (rateLimiter == null) {
362                 return RateLimiter.create(permitsPerSecond);
363             } else {
364                 rateLimiter.setRate(permitsPerSecond);
365                 return rateLimiter;
366             }
367         });
368     }
369 
370     private static String rateLimiterKey(String projectName, String repoName) {
371         return projectName + '/' + repoName;
372     }
373 
374     private boolean writeQuotaEnabled() {
375         return Double.compare(permitsPerSecond, -1) > 0;
376     }
377 
378     private Repository repo(RepositoryCommand<?> c) {
379         return projectManager.get(c.projectName()).repos().get(c.repositoryName());
380     }
381 
382     private CompletableFuture<Void> createSession(CreateSessionCommand c) {
383         if (sessionManager == null) {
384             // Security has been disabled for this replica.
385             return CompletableFuture.completedFuture(null);
386         }
387 
388         final Session session = c.session();
389         return sessionManager.create(session).exceptionally(cause -> {
390             logger.warn("Failed to replicate a session creation: {}", session, cause);
391             return null;
392         });
393     }
394 
395     private CompletableFuture<Void> removeSession(RemoveSessionCommand c) {
396         if (sessionManager == null) {
397             return CompletableFuture.completedFuture(null);
398         }
399 
400         final String sessionId = c.sessionId();
401         return sessionManager.delete(sessionId).exceptionally(cause -> {
402             logger.warn("Failed to replicate a session removal: {}", sessionId, cause);
403             return null;
404         });
405     }
406 
407     private CompletableFuture<Void> updateServerStatus(UpdateServerStatusCommand c) {
408         return CompletableFuture.supplyAsync(() -> {
409             serverStatusManager.updateStatus(c.serverStatus());
410             statusManager().updateStatus(c);
411             return null;
412         }, serverStatusManager.sequentialExecutor());
413     }

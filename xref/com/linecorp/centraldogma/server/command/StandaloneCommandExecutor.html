60      /**
61       * Creates a new instance.
62       *
63       * @param projectManager the project manager for accessing the storage
64       * @param repositoryWorker the executor which is used for performing storage operations
65       * @param sessionManager the session manager for creating/removing a session
66       * @param onTakeLeadership the callback to be invoked after the replica has taken the leadership
67       * @param onReleaseLeadership the callback to be invoked before the replica releases the leadership
68       * @param onTakeZoneLeadership the callback to be invoked after the replica has taken the zone leadership
69       * @param onReleaseZoneLeadership the callback to be invoked before the replica releases the zone leadership
70       */
71      public StandaloneCommandExecutor(ProjectManager projectManager,
72                                       Executor repositoryWorker,
73                                       ServerStatusManager serverStatusManager,
74                                       @Nullable SessionManager sessionManager,
75                                       EncryptionStorageManager encryptionStorageManager,
76                                       @Nullable Consumer<CommandExecutor> onTakeLeadership,
77                                       @Nullable Consumer<CommandExecutor> onReleaseLeadership,
78                                       @Nullable Consumer<CommandExecutor> onTakeZoneLeadership,
79                                       @Nullable Consumer<CommandExecutor> onReleaseZoneLeadership) {
80          super(onTakeLeadership, onReleaseLeadership, onTakeZoneLeadership, onReleaseZoneLeadership);
81          this.projectManager = requireNonNull(projectManager, "projectManager");
82          this.repositoryWorker = requireNonNull(repositoryWorker, "repositoryWorker");
83          this.serverStatusManager = requireNonNull(serverStatusManager, "serverStatusManager");
84          this.sessionManager = sessionManager;
85          this.encryptionStorageManager = requireNonNull(encryptionStorageManager, "encryptionStorageManager");
86      }

93      @Override
94      protected void doStart(@Nullable Runnable onTakeLeadership,
95                             @Nullable Runnable onReleaseLeadership,
96                             @Nullable Runnable onTakeZoneLeadership,
97                             @Nullable Runnable onReleaseZoneLeadership) {
98          if (onTakeLeadership != null) {
99              onTakeLeadership.run();
100         }
101         if (onTakeZoneLeadership != null) {
102             onTakeZoneLeadership.run();
103         }
104     }

106     @Override
107     protected void doStop(@Nullable Runnable onReleaseLeadership, @Nullable Runnable onReleaseZoneLeadership) {
108         if (onReleaseLeadership != null) {

110         }
111         if (onReleaseZoneLeadership != null) {
112             onReleaseZoneLeadership.run();
113         }

116     @Override
117     protected <T> CompletableFuture<T> doExecute(Command<T> command) throws Exception {
118         throwExceptionIfRepositoryNotWritable(command);
119         return doExecute0(command);
120     }
121 
122     private void throwExceptionIfRepositoryNotWritable(Command<?> command) throws Exception {
123         if (command instanceof NormalizableCommit) {
124             assert command instanceof RepositoryCommand;
125             final RepositoryCommand<?> repositoryCommand = (RepositoryCommand<?>) command;
126             if (InternalProjectInitializer.INTERNAL_PROJECT_DOGMA.equals(repositoryCommand.projectName())) {
127                 return;
128             }
129             String repoName = repositoryCommand.repositoryName();
130             if (Project.REPO_META.equals(repoName)) {
131                 // Use REPO_DOGMA for the meta repository because the meta repository will be removed.
132                 repoName = Project.REPO_DOGMA;
133             }
134 
135             final ProjectMetadata metadata = projectManager.get(repositoryCommand.projectName()).metadata();
136             assert metadata != null;
137             final RepositoryMetadata repositoryMetadata = metadata.repos().get(repoName);
138             if (repositoryMetadata == null) {
139                 // The repository metadata is not found, so it is writable.
140                 return;
141             }
142             if (repositoryMetadata.status() == RepositoryStatus.READ_ONLY) {
143                 throw new ReadOnlyException(
144                         "The repository is in read-only. command: " + repositoryCommand);
145             }
146         }
147     }

149     @SuppressWarnings("unchecked")
150     private <T> CompletableFuture<T> doExecute0(Command<T> command) throws Exception {
151         if (command instanceof CreateProjectCommand) {
152             return (CompletableFuture<T>) createProject((CreateProjectCommand) command);
153         }
154 
155         if (command instanceof RemoveProjectCommand) {
156             return (CompletableFuture<T>) removeProject((RemoveProjectCommand) command);
157         }
158 
159         if (command instanceof UnremoveProjectCommand) {
160             return (CompletableFuture<T>) unremoveProject((UnremoveProjectCommand) command);
161         }
162 
163         if (command instanceof PurgeProjectCommand) {
164             return (CompletableFuture<T>) purgeProject((PurgeProjectCommand) command);
165         }
166 
167         if (command instanceof CreateRepositoryCommand) {
168             return (CompletableFuture<T>) createRepository((CreateRepositoryCommand) command);
169         }
170 
171         if (command instanceof RemoveRepositoryCommand) {
172             return (CompletableFuture<T>) removeRepository((RemoveRepositoryCommand) command);
173         }
174 
175         if (command instanceof UnremoveRepositoryCommand) {
176             return (CompletableFuture<T>) unremoveRepository((UnremoveRepositoryCommand) command);
177         }
178 
179         if (command instanceof PurgeRepositoryCommand) {
180             return (CompletableFuture<T>) purgeRepository((PurgeRepositoryCommand) command);
181         }
182 
183         if (command instanceof MigrateToEncryptedRepositoryCommand) {
184             if (!encryptionStorageManager.enabled()) {
185                 throw new IllegalStateException(
186                         "Encryption is not enabled. command: " + command);
187             }
188             return (CompletableFuture<T>) migrateToEncryptedRepository(
189                     (MigrateToEncryptedRepositoryCommand) command);
190         }
191 
192         if (command instanceof NormalizingPushCommand) {
193             return (CompletableFuture<T>) push((NormalizingPushCommand) command, true);
194         }
195 
196         if (command instanceof PushAsIsCommand) {
197             return (CompletableFuture<T>) push((PushAsIsCommand) command, false)
198                     .thenApply(CommitResult::revision);
199         }
200 
201         if (command instanceof TransformCommand) {
202             return (CompletableFuture<T>) push((TransformCommand) command, true);
203         }
204 
205         if (command instanceof RewrapAllKeysCommand) {
206             return (CompletableFuture<T>) rewrapAllKeys();
207         }
208 
209         if (command instanceof RotateWdekCommand) {
210             return (CompletableFuture<T>) rotateWdek((RotateWdekCommand) command);
211         }
212 
213         if (command instanceof RotateSessionMasterKeyCommand) {
214             return (CompletableFuture<T>) rotateSessionMasterKey((RotateSessionMasterKeyCommand) command);
215         }
216 
217         if (command instanceof CreateSessionCommand) {
218             return (CompletableFuture<T>) createSession((CreateSessionCommand) command);
219         }
220 
221         if (command instanceof RemoveSessionCommand) {
222             return (CompletableFuture<T>) removeSession((RemoveSessionCommand) command);
223         }
224 
225         if (command instanceof CreateSessionMasterKeyCommand) {
226             return (CompletableFuture<T>) createSessionMasterKey((CreateSessionMasterKeyCommand) command);
227         }
228 
229         if (command instanceof UpdateServerStatusCommand) {
230             return (CompletableFuture<T>) updateServerStatus((UpdateServerStatusCommand) command);
231         }
232 
233         if (command instanceof ForcePushCommand) {
234             // TODO(minwoox): Should we prevent executing when the replication status is READ_ONLY?
235             //noinspection TailRecursion
236             return doExecute0(((ForcePushCommand<T>) command).delegate());
237         }
238 
239         throw new UnsupportedOperationException(command.toString());
240     }

244     private CompletableFuture<Void> createProject(CreateProjectCommand c) {
245         return CompletableFuture.supplyAsync(() -> {
246             final WrappedDekDetails wdekDetails = c.wdekDetails();
247             final boolean encrypt = wdekDetails != null;
248             if (encrypt) {
249                 encryptionStorageManager.storeWdek(wdekDetails);
250             }
251 
252             try {
253                 projectManager.create(c.projectName(), c.timestamp(), c.author(), encrypt);
254             } catch (Throwable t) {
255                 if (encrypt) {
256                     try {
257                         encryptionStorageManager.removeWdek(c.projectName(), Project.REPO_DOGMA,
258                                                             c.wdekDetails().dekVersion(), true);
259                     } catch (Throwable t2) {
260                         logger.warn("Failed to remove the WDEK of {}/{}",
261                                     c.projectName(), Project.REPO_DOGMA, t2);
262                     }
263                 }
264                 Exceptions.throwUnsafely(t);
265             }
266 
267             return null;
268         }, repositoryWorker);
269     }

294     private CompletableFuture<Void> createRepository(CreateRepositoryCommand c) {
295         return CompletableFuture.supplyAsync(() -> {
296             final WrappedDekDetails wdekDetails = c.wdekDetails();
297             final boolean encrypt = wdekDetails != null;
298             if (encrypt) {
299                 encryptionStorageManager.storeWdek(wdekDetails);
300             }
301 
302             try {
303                 projectManager.get(c.projectName()).repos().create(c.repositoryName(), c.timestamp(),
304                                                                    c.author(), encrypt);
305             } catch (Throwable t) {
306                 if (encrypt) {
307                     try {
308                         encryptionStorageManager.removeWdek(c.projectName(), c.repositoryName(),
309                                                             c.wdekDetails().dekVersion(), true);
310                     } catch (Throwable t2) {
311                         logger.warn("Failed to remove the WDEK of {}/{}",
312                                     c.projectName(), c.repositoryName(), t2);
313                     }
314                 }
315                 Exceptions.throwUnsafely(t);
316             }
317 
318             return null;
319         }, repositoryWorker);
320     }

343     private CompletableFuture<Void> migrateToEncryptedRepository(MigrateToEncryptedRepositoryCommand c) {
344         final RepositoryManager repositoryManager = projectManager.get(c.projectName()).repos();
345         final Repository repository = repositoryManager.get(c.repositoryName());
346         if (repository.isEncrypted()) {
347             throw new IllegalStateException(
348                     "The repository is already encrypted: " + c.projectName() + '/' + c.repositoryName());
349         }
350         return CompletableFuture.supplyAsync(() -> {
351             encryptionStorageManager.storeWdek(c.wdekDetails());
352             try {
353                 repositoryManager.migrateToEncryptedRepository(c.repositoryName());
354             } catch (Throwable t) {
355                 try {
356                     encryptionStorageManager.removeWdek(c.projectName(), c.repositoryName(),
357                                                         c.wdekDetails().dekVersion(), true);
358                 } catch (Throwable t2) {
359                     logger.warn("Failed to remove the WDEK of {}/{}",
360                                 c.projectName(), c.repositoryName(), t2);
361                 }
362                 throw t;
363             }
364             return null;
365         }, repositoryWorker);
366     }

368     private CompletableFuture<CommitResult> push(RepositoryCommand<?> c, boolean normalizing) {
369         if (c instanceof TransformCommand) {
370             final TransformCommand transformCommand = (TransformCommand) c;
371             return repo(c).commit(transformCommand.baseRevision(), transformCommand.timestamp(),
372                                   transformCommand.author(), transformCommand.summary(),
373                                   transformCommand.detail(), transformCommand.markup(),
374                                   transformCommand.transformer());
375         }
376         assert c instanceof AbstractPushCommand;
377         final AbstractPushCommand<?> pushCommand = (AbstractPushCommand<?>) c;
378         return repo(c).commit(pushCommand.baseRevision(), pushCommand.timestamp(), pushCommand.author(),
379                               pushCommand.summary(), pushCommand.detail(), pushCommand.markup(),
380                               pushCommand.changes(), normalizing);
381     }
382 
383     private Repository repo(RepositoryCommand<?> c) {
384         return projectManager.get(c.projectName()).repos().get(c.repositoryName());
385     }

387     private CompletableFuture<Void> rewrapAllKeys() {
388         if (!encryptionStorageManager.enabled()) {
389             throw new IllegalStateException("Encryption is not enabled.");
390         }
391 
392         logger.info("Rewrapping all keys with kek: {}", encryptionStorageManager.kekId());
393         return CompletableFuture.supplyAsync(() -> encryptionStorageManager.rewrapAllKeys(repositoryWorker),
394                                              repositoryWorker)
395                                 .thenCompose(Function.identity())
396                                 .handle((unused, cause) -> {
397                                     if (cause != null) {
398                                         logger.warn("Failed to rewrap all keys with kek: {}",
399                                                     encryptionStorageManager.kekId(),
400                                                     cause);
401                                         Exceptions.throwUnsafely(cause);
402                                     } else {
403                                         logger.info("All keys rewrapped.");
404                                     }
405                                     return null;
406                                 });
407     }

409     private CompletableFuture<Void> rotateWdek(RotateWdekCommand c) {
410         if (!encryptionStorageManager.enabled()) {
411             throw new IllegalStateException("Encryption is not enabled. command: " + c);
412         }
413 
414         return CompletableFuture.supplyAsync(() -> {
415             logger.info("Rotating WDEK for {}/{} to version {}",
416                         c.projectName(), c.repositoryName(), c.wdekDetails().dekVersion());
417             try {
418                 encryptionStorageManager.rotateWdek(c.wdekDetails());
419                 logger.info("WDEK rotated for {}/{}.", c.projectName(), c.repositoryName());
420             } catch (Throwable t) {
421                 logger.warn("Failed to rotate WDEK for {}/{} to version {}",
422                             c.projectName(), c.repositoryName(), c.wdekDetails().dekVersion(), t);
423                 Exceptions.throwUnsafely(t);
424             }
425             if (c.reencrypt()) {
426                 try {
427                     logger.info("Re-encrypting all data in {}/{} with the new WDEK.",
428                                 c.projectName(), c.repositoryName());
429                     encryptionStorageManager.reencryptRepositoryData(c.projectName(), c.repositoryName());
430                     logger.info("All data re-encrypted in {}/{}.", c.projectName(), c.repositoryName());
431                 } catch (Throwable t) {
432                     logger.warn("Failed to re-encrypt all data in {}/{} with the new WDEK.",
433                                 c.projectName(), c.repositoryName(), t);
434                     Exceptions.throwUnsafely(t);
435                 }
436             }
437 
438             return null;
439         }, repositoryWorker);
440     }

442     private CompletableFuture<Void> rotateSessionMasterKey(RotateSessionMasterKeyCommand c) {
443         if (!encryptionStorageManager.encryptSessionCookie()) {
444             throw new IllegalStateException("Session cookie encryption is not enabled. command: " + c);
445         }
446 
447         return CompletableFuture.supplyAsync(() -> {
448             final int version = c.sessionMasterKey().version();
449             logger.info("Rotating session master key to version {}", version);
450             try {
451                 encryptionStorageManager.rotateSessionMasterKey(c.sessionMasterKey());
452                 logger.info("Session master key rotated.");
453             } catch (Throwable t) {
454                 logger.warn("Failed to rotate session master key to version {}",
455                             version, t);
456                 Exceptions.throwUnsafely(t);
457             }
458             return null;
459         }, repositoryWorker);
460     }

462     private CompletableFuture<Void> createSession(CreateSessionCommand c) {
463         if (sessionManager == null) {
464             // Security has been disabled for this replica.
465             return CompletableFuture.completedFuture(null);
466         }
467 
468         final Session session = c.session();
469         return sessionManager.create(session).exceptionally(cause -> {
470             logger.warn("Failed to replicate a session creation: {}", session, cause);
471             return null;
472         });
473     }
474 
475     private CompletableFuture<Void> removeSession(RemoveSessionCommand c) {
476         if (sessionManager == null) {
477             return CompletableFuture.completedFuture(null);
478         }
479 
480         final String sessionId = c.sessionId();
481         return sessionManager.delete(sessionId).exceptionally(cause -> {
482             logger.warn("Failed to replicate a session removal: {}", sessionId, cause);
483             return null;
484         });
485     }
486 
487     private CompletableFuture<Void> createSessionMasterKey(CreateSessionMasterKeyCommand c) {
488         if (!encryptionStorageManager.enabled()) {
489             throw new IllegalStateException("Encryption is not enabled. command: " + c);
490         }
491 
492         return CompletableFuture.supplyAsync(() -> {
493             encryptionStorageManager.storeSessionMasterKey(c.sessionMasterKey());
494             return null;
495         }, repositoryWorker);
496     }

498     private CompletableFuture<Void> updateServerStatus(UpdateServerStatusCommand c) {
499         return CompletableFuture.supplyAsync(() -> {
500             serverStatusManager.updateStatus(c.serverStatus());
501             statusManager().updateStatus(c);
502             return null;
503         }, serverStatusManager.sequentialExecutor());
504     }

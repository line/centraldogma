38  /**
39   * A Central Dogma command which is used to mutate projects and repositories.
40   *
41   * @param <T> the result type of a {@link Command}
42   */
43  @JsonTypeInfo(use = JsonTypeInfo.Id.NAME, property = "type")
44  @JsonSubTypes({
45          @Type(value = CreateProjectCommand.class, name = "CREATE_PROJECT"),
46          @Type(value = ResetMetaRepositoryCommand.class, name = "RESET_META_REPOSITORY"),
47          @Type(value = RemoveProjectCommand.class, name = "REMOVE_PROJECT"),
48          @Type(value = PurgeProjectCommand.class, name = "PURGE_PROJECT"),
49          @Type(value = UnremoveProjectCommand.class, name = "UNREMOVE_PROJECT"),
50          @Type(value = CreateRepositoryCommand.class, name = "CREATE_REPOSITORY"),
51          @Type(value = RemoveRepositoryCommand.class, name = "REMOVE_REPOSITORY"),
52          @Type(value = PurgeRepositoryCommand.class, name = "PURGE_REPOSITORY"),
53          @Type(value = UnremoveRepositoryCommand.class, name = "UNREMOVE_REPOSITORY"),
54          @Type(value = MigrateToEncryptedRepositoryCommand.class, name = "MIGRATE_TO_ENCRYPTED_REPOSITORY"),
55          @Type(value = NormalizingPushCommand.class, name = "NORMALIZING_PUSH"),
56          @Type(value = PushAsIsCommand.class, name = "PUSH"),
57          @Type(value = CreateSessionCommand.class, name = "CREATE_SESSIONS"),
58          @Type(value = RemoveSessionCommand.class, name = "REMOVE_SESSIONS"),
59          @Type(value = UpdateServerStatusCommand.class, name = "UPDATE_SERVER_STATUS"),
60          @Type(value = ForcePushCommand.class, name = "FORCE_PUSH_COMMAND"),
61  })

432     /**
433      * Returns a new {@link Command} which is used to force-push {@link Command} even the server is in
434      * read-only mode. This command is useful for migrating the repository content during maintenance mode.
435      */
436     static <T> Command<T> forcePush(Command<T> delegate) {
437         requireNonNull(delegate, "delegate");
438         checkArgument(delegate.type() == CommandType.CREATE_PROJECT ||
439                       delegate.type() == CommandType.CREATE_REPOSITORY ||
440                       delegate.type() == CommandType.NORMALIZING_PUSH ||
441                       delegate.type() == CommandType.TRANSFORM ||
442                       delegate.type() == CommandType.PUSH,
443                       "delegate: %s " +
444                       "(expected: CREATE_PROJECT, CREATE_REPOSITORY, NORMALIZING_PUSH, TRANSFORM or PUSH)",
445                       delegate);
446         return new ForcePushCommand<>(delegate);
447     }

41  /**
42   * A Central Dogma command which is used to mutate projects and repositories.
43   *
44   * @param <T> the result type of a {@link Command}
45   */

49          @Type(value = RemoveProjectCommand.class, name = "REMOVE_PROJECT"),
50          @Type(value = PurgeProjectCommand.class, name = "PURGE_PROJECT"),
51          @Type(value = UnremoveProjectCommand.class, name = "UNREMOVE_PROJECT"),
52          @Type(value = CreateRepositoryCommand.class, name = "CREATE_REPOSITORY"),
53          @Type(value = RemoveRepositoryCommand.class, name = "REMOVE_REPOSITORY"),
54          @Type(value = PurgeRepositoryCommand.class, name = "PURGE_REPOSITORY"),
55          @Type(value = UnremoveRepositoryCommand.class, name = "UNREMOVE_REPOSITORY"),
56          @Type(value = MigrateToEncryptedRepositoryCommand.class, name = "MIGRATE_TO_ENCRYPTED_REPOSITORY"),
57          @Type(value = NormalizingPushCommand.class, name = "NORMALIZING_PUSH"),

471     /**
472      * Returns a new {@link Command} which is used to force-push {@link Command} even the server is in
473      * read-only mode. This command is useful for migrating the repository content during maintenance mode.
474      */
475     static <T> Command<T> forcePush(Command<T> delegate) {
476         requireNonNull(delegate, "delegate");
477         checkArgument(delegate.type() == CommandType.CREATE_PROJECT ||
478                       delegate.type() == CommandType.CREATE_REPOSITORY ||
479                       delegate.type() == CommandType.NORMALIZING_PUSH ||
480                       delegate.type() == CommandType.TRANSFORM ||
481                       delegate.type() == CommandType.PUSH,
482                       "delegate: %s " +
483                       "(expected: CREATE_PROJECT, CREATE_REPOSITORY, NORMALIZING_PUSH, TRANSFORM or PUSH)",
484                       delegate);
485         return new ForcePushCommand<>(delegate);
486     }
